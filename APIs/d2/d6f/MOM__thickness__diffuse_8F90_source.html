25 !* *
26 !* The subroutine in this file implements horizontal interface *
27 !* depth diffusion. In a Z-coordinate model, this would be described *
28 !* as Gent-McWilliams diffusion, with an identical interpretation *
29 !* of the diffusion coefficient. The coefficient is locally limited *
30 !* to guarantee numerical stability, and are vertically uniform. *
31 !* *
32 !* Macros written all in capital letters are defined in MOM_memory.h. *
33 !* *
34 !* A small fragment of the grid is shown below: *
35 !* *
36 !* j+1 x ^ x ^ x At x: q *
37 !* j+1 > o > o > At ^: v, vh, vav *
38 !* j x ^ x ^ x At >: u, uh, uav *
39 !* j > o > o > At o: h *
40 !* j-1 x ^ x ^ x *
41 !* i-1 i i+1 At x & ^: *
42 !* i i+1 At > & o: *
43 !* *
44 !* The boundaries always run through q grid points (x). *
45 !* *

100 subroutine thickness_diffuse(h, uhtr, vhtr, tv, dt, G, GV, MEKE, VarMix, CDp, CS)
101  type(ocean_grid_type), intent(in) :: G
102  type(verticalgrid_type), intent(in) :: GV
103  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(inout) :: h
104  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(inout) :: uhtr
105  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(inout) :: vhtr
106  type(thermo_var_ptrs), intent(in) :: tv
107  real, intent(in) :: dt
108  type(meke_type), intent(inout) :: MEKE
109  type(varmix_cs), pointer :: VarMix
110  type(cont_diag_ptrs), intent(inout) :: CDp
111  type(thickness_diffuse_cs), pointer :: CS
112 
113 ! This subroutine does interface depth diffusion. The fluxes are
114 ! limited to give positive definiteness, and the diffusivities are
115 ! limited to guarantee stability.
116 
117 ! Arguments:
118 ! (in/out) h - layer thickness (meter)
119 ! (in/out) uhtr - accumulated zonal mass fluxes (m2 H)
120 ! (in/out) vhtr - accumulated meridional mass fluxes (m2 H)
121 ! (in) tv - A structure containing pointers to available
122 ! thermodynamic fields. Absent fields have NULL ptrs.
123 ! (in) dt - time increment (seconds)
124 ! (in) G - ocean grid structure
125 ! (in) GV - The ocean's vertical grid structure.
126 ! (in) VarMix - A structure containing fields related to
127 ! variable lateral mixing.
128 ! (inout) MEKE - A structure containing information about the Mesoscale Eddy
129 ! Kinetic Energy parameterization; this might be unassociated.
130 ! (inout) CDp - structure pointing to terms in continuity equations
131 ! (in) CS - control structure from previous call to thickness_diffuse_init
132 
133  real :: e(szi_(g), szj_(g), szk_(g)+1) ! heights of interfaces, relative to mean
134  ! sea level,in H units, positive up.
135  real :: uhD(szib_(g), szj_(g), szk_(g)) ! uhD & vhD are the diffusive u*h &
136  real :: vhD(szi_(g), szjb_(g), szk_(g)) ! v*h fluxes (m2 H s-1)
137 
138  real, dimension(SZIB_(G), SZJ_(G), SZK_(G)+1) :: &
139  KH_u, & ! interface height diffusivities in u-columns (m2 s-1)
140  int_slope_u ! A nondimensional ratio from 0 to 1 that gives the relative
141  ! weighting of the interface slopes to that calculated also
142  ! using density gradients at u points. The physically correct
143  ! slopes occur at 0, while 1 is used for numerical closures.
144  real, dimension(SZI_(G), SZJB_(G), SZK_(G)+1) :: &
145  KH_v, & ! interface height diffusivities in v-columns (m2 s-1)
146  int_slope_v ! A nondimensional ratio from 0 to 1 that gives the relative
147  ! weighting of the interface slopes to that calculated also
148  ! using density gradients at v points. The physically correct
149  ! slopes occur at 0, while 1 is used for numerical closures.
150  real, dimension(SZI_(G), SZJ_(G), SZK_(G)) :: &
151  KH_t ! diagnosed diffusivity at tracer points (m^2/s)
152 
153  real, dimension(SZIB_(G), SZJ_(G)) :: &
154  KH_u_CFL ! The maximum stable interface height diffusivity at u grid points (m2 s-1)
155  real, dimension(SZI_(G), SZJB_(G)) :: &
156  KH_v_CFL ! The maximum stable interface height diffusivity at v grid points (m2 s-1)
157  real :: Khth_Loc_u(szib_(g), szj_(g))
158  real :: Khth_Loc(szib_(g), szjb_(g)) ! locally calculated thickness diffusivity (m2/s)
159  real :: H_to_m, m_to_H ! Local copies of unit conversion factors.
160  real :: h_neglect ! A thickness that is so small it is usually lost
161  ! in roundoff and can be neglected, in H.
162  logical :: use_VarMix, Resoln_scaled, use_stored_slopes
163  integer :: i, j, k, is, ie, js, je, nz
164  logical :: MEKE_not_null
165  real :: hu(szi_(g), szj_(g)) ! u-thickness (H)
166  real :: hv(szi_(g), szj_(g)) ! v-thickness (H)
167  real :: KH_u_lay(szi_(g), szj_(g)) ! layer ave thickness diffusivities (m2/sec)
168  real :: KH_v_lay(szi_(g), szj_(g)) ! layer ave thickness diffusivities (m2/sec)
169 
170  if (.not. ASSOCIATED(cs)) call mom_error(fatal, "MOM_thickness_diffuse:"// &
171  "Module must be initialized before it is used.")
172  meke_not_null = (loc(meke) .NE. 0)
173 
174  if ((.not.cs%thickness_diffuse) .or. &
175  .not.( cs%Khth > 0.0 .or. associated(varmix) .or. meke_not_null ) ) return
176 
177  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
178  h_neglect = gv%H_subroundoff
179  h_to_m = gv%H_to_m ; m_to_h = gv%m_to_H
180 
181  if (meke_not_null) then
182  if (ASSOCIATED(meke%GM_src)) then
183  do j=js,je ; do i=is,ie ; meke%GM_src(i,j) = 0. ; enddo ; enddo
184  endif
185  endif
186 
187  use_varmix = .false. ; resoln_scaled = .false. ; use_stored_slopes = .false.
188  if (Associated(varmix)) then
189  use_varmix = varmix%use_variable_mixing
190  resoln_scaled = varmix%Resoln_scaled_KhTh
191  use_stored_slopes = varmix%use_stored_slopes
192  endif
193 
194 !$OMP parallel do default(none) shared(is,ie,js,je,KH_u_CFL,dt,G,CS)
195  do j=js,je ; do i=is-1,ie
196  kh_u_cfl(i,j) = (0.25*cs%max_Khth_CFL) / &
197  (dt*(g%IdxCu(i,j)*g%IdxCu(i,j) + g%IdyCu(i,j)*g%IdyCu(i,j)))
198  enddo ; enddo
199 !$OMP parallel do default(none) shared(is,ie,js,je,KH_v_CFL,dt,G,CS)
200  do j=js-1,je ; do i=is,ie
201  kh_v_cfl(i,j) = (0.25*cs%max_Khth_CFL) / &
202  (dt*(g%IdxCv(i,j)*g%IdxCv(i,j) + g%IdyCv(i,j)*g%IdyCv(i,j)))
203  enddo ; enddo
204 
205  call find_eta(h, tv, g%g_Earth, g, gv, e, halo_size=1)
206 
207  ! Set the diffusivities.
208 !$OMP parallel default(none) shared(is,ie,js,je,Khth_Loc_u,CS,use_VarMix,VarMix, &
209 !$OMP MEKE_not_null,MEKE,Resoln_scaled,KH_u, &
210 !$OMP KH_u_CFL,nz,Khth_Loc,KH_v,KH_v_CFL,int_slope_u, &
211 !$OMP int_slope_v )
212 !$OMP do
213  do j=js,je; do i=is-1,ie
214  khth_loc_u(i,j) = cs%Khth
215  enddo ; enddo
216 
217  if (use_varmix) then
218 !$OMP do
219  do j=js,je ; do i=is-1,ie
220  khth_loc_u(i,j) = khth_loc_u(i,j) + cs%KHTH_Slope_Cff*varmix%L2u(i,j)*varmix%SN_u(i,j)
221  enddo ; enddo
222  endif
223 
224  if (meke_not_null) then ; if (associated(meke%Kh)) then
225 !$OMP do
226  do j=js,je ; do i=is-1,ie
227  khth_loc_u(i,j) = khth_loc_u(i,j) + meke%KhTh_fac*sqrt(meke%Kh(i,j)*meke%Kh(i+1,j))
228  enddo ; enddo
229  endif ; endif
230 
231  if (resoln_scaled) then
232 !$OMP do
233  do j=js,je; do i=is-1,ie
234  khth_loc_u(i,j) = khth_loc_u(i,j) * varmix%Res_fn_u(i,j)
235  enddo ; enddo
236  endif
237 
238  if (cs%Khth_Max > 0) then
239 !$OMP do
240  do j=js,je; do i=is-1,ie
241  khth_loc_u(i,j) = max(cs%Khth_min, min(khth_loc_u(i,j),cs%Khth_Max))
242  enddo ; enddo
243  else
244 !$OMP do
245  do j=js,je; do i=is-1,ie
246  khth_loc_u(i,j) = max(cs%Khth_min, khth_loc_u(i,j))
247  enddo ; enddo
248  endif
249 !$OMP do
250  do j=js,je; do i=is-1,ie
251  kh_u(i,j,1) = min(kh_u_cfl(i,j), khth_loc_u(i,j))
252  enddo ; enddo
253 
254 !$OMP do
255  do k=2,nz+1 ; do j=js,je ; do i=is-1,ie
256  kh_u(i,j,k) = kh_u(i,j,1)
257  enddo ; enddo ; enddo
258 
259 !$OMP do
260  do j=js-1,je ; do i=is,ie
261  khth_loc(i,j) = cs%Khth
262  enddo ; enddo
263 
264  if (use_varmix) then
265 !$OMP do
266  do j=js-1,je ; do i=is,ie
267  khth_loc(i,j) = khth_loc(i,j) + cs%KHTH_Slope_Cff*varmix%L2v(i,j)*varmix%SN_v(i,j)
268  enddo ; enddo
269  endif
270  if (meke_not_null) then ; if (associated(meke%Kh)) then
271 !$OMP do
272  do j=js-1,je ; do i=is,ie
273  khth_loc(i,j) = khth_loc(i,j) + meke%KhTh_fac*sqrt(meke%Kh(i,j)*meke%Kh(i,j+1))
274  enddo ; enddo
275  endif ; endif
276 
277  if (resoln_scaled) then
278 !$OMP do
279  do j=js-1,je ; do i=is,ie
280  khth_loc(i,j) = khth_loc(i,j) * varmix%Res_fn_v(i,j)
281  enddo ; enddo
282  endif
283 
284  if (cs%Khth_Max > 0) then
285 !$OMP do
286  do j=js-1,je ; do i=is,ie
287  khth_loc(i,j) = max(cs%Khth_min, min(khth_loc(i,j),cs%Khth_Max))
288  enddo ; enddo
289  else
290 !$OMP do
291  do j=js-1,je ; do i=is,ie
292  khth_loc(i,j) = max(cs%Khth_min, khth_loc(i,j))
293  enddo ; enddo
294  endif
295 
296  if (cs%max_Khth_CFL > 0.0) then
297 !$OMP do
298  do j=js-1,je ; do i=is,ie
299  kh_v(i,j,1) = min(kh_v_cfl(i,j), khth_loc(i,j))
300  enddo ; enddo
301  endif
302 !$OMP do
303  do k=2,nz+1 ; do j=js-1,je ; do i=is,ie
304  kh_v(i,j,k) = kh_v(i,j,1)
305  enddo ; enddo ; enddo
306 !$OMP do
307  do k=1,nz+1 ; do j=js,je ; do i=is-1,ie ; int_slope_u(i,j,k) = 0.0 ; enddo ; enddo ; enddo
308 !$OMP do
309  do k=1,nz+1 ; do j=js-1,je ; do i=is,ie ; int_slope_v(i,j,k) = 0.0 ; enddo ; enddo ; enddo
310 !$OMP end parallel
311 
312  if (cs%detangle_interfaces) then
313  call add_detangling_kh(h, e, kh_u, kh_v, kh_u_cfl, kh_v_cfl, tv, dt, g, gv, &
314  cs, int_slope_u, int_slope_v)
315  endif
316 
317  if (cs%debug) then
318  call uchksum(kh_u(:,:,:),"Kh_u",g,haloshift=0)
319  call vchksum(kh_v(:,:,:),"Kh_v",g,haloshift=0)
320  call uchksum(int_slope_u(:,:,:),"int_slope_u",g,haloshift=0)
321  call vchksum(int_slope_v(:,:,:),"int_slope_v",g,haloshift=0)
322  call hchksum(h(:,:,:)*h_to_m,"thickness_diffuse_1 h",g,haloshift=1)
323  call hchksum(e(:,:,:),"thickness_diffuse_1 e",g,haloshift=1)
324  if (use_stored_slopes) then
325  call uchksum(varmix%slope_x(:,:,:),"VarMix%slope_x",g,haloshift=0)
326  call vchksum(varmix%slope_y(:,:,:),"VarMix%slope_y",g,haloshift=0)
327  endif
328  if (associated(tv%eqn_of_state)) then
329  call hchksum(tv%T(:,:,:),"thickness_diffuse T",g,haloshift=1)
330  call hchksum(tv%S(:,:,:),"thickness_diffuse S",g,haloshift=1)
331  endif
332  endif
333 
334  ! Calculate uhD, vhD from h, e, KH_u, KH_v, tv%T/S
335  if (use_stored_slopes) then
336  call thickness_diffuse_full(h, e, kh_u, kh_v, tv, uhd, vhd, dt, g, gv, meke, cs, &
337  int_slope_u, int_slope_v, varmix%slope_x, varmix%slope_y)
338  else
339  call thickness_diffuse_full(h, e, kh_u, kh_v, tv, uhd, vhd, dt, g, gv, meke, cs, &
340  int_slope_u, int_slope_v)
341  endif
342 !$OMP parallel do default(none) shared(is,ie,js,je,nz,uhtr,uhD,dt,vhtr,CDp,vhD,h,G,GV)
343  do k=1,nz
344  do j=js,je ; do i=is-1,ie
345  uhtr(i,j,k) = uhtr(i,j,k) + uhd(i,j,k)*dt
346  if (ASSOCIATED(cdp%uhGM)) cdp%uhGM(i,j,k) = uhd(i,j,k)
347  enddo ; enddo
348  do j=js-1,je ; do i=is,ie
349  vhtr(i,j,k) = vhtr(i,j,k) + vhd(i,j,k)*dt
350  if (ASSOCIATED(cdp%vhGM)) cdp%vhGM(i,j,k) = vhd(i,j,k)
351  enddo ; enddo
352  do j=js,je ; do i=is,ie
353  h(i,j,k) = h(i,j,k) - dt * g%IareaT(i,j) * &
354  ((uhd(i,j,k) - uhd(i-1,j,k)) + (vhd(i,j,k) - vhd(i,j-1,k)))
355  if (h(i,j,k) < gv%Angstrom) h(i,j,k) = gv%Angstrom
356  enddo ; enddo
357  enddo
358 
359  ! Whenever thickness changes let the diag manager know, target grids
360  ! for vertical remapping may need to be regenerated.
361  ! This needs to happen after the H update and before the next post_data.
362  call diag_update_target_grids(cs%diag)
363 
364  if (meke_not_null .AND. ASSOCIATED(varmix)) then
365  if (ASSOCIATED(meke%Rd_dx_h) .and. ASSOCIATED(varmix%Rd_dx_h)) then
366 !$OMP parallel do default(none) shared(is,ie,js,je,MEKE,VarMix)
367  do j=js,je ; do i=is,ie
368  meke%Rd_dx_h(i,j) = varmix%Rd_dx_h(i,j)
369  enddo ; enddo
370  endif
371  endif
372 
373  if (cs%debug) then
374  call uchksum(uhd(:,:,:)*h_to_m,"thickness_diffuse uhD",g,haloshift=0)
375  call vchksum(vhd(:,:,:)*h_to_m,"thickness_diffuse vhD",g,haloshift=0)
376  call uchksum(uhtr(:,:,:)*h_to_m,"thickness_diffuse uhtr",g,haloshift=0)
377  call vchksum(vhtr(:,:,:)*h_to_m,"thickness_diffuse vhtr",g,haloshift=0)
378  call hchksum(h(:,:,:)*h_to_m,"thickness_diffuse h",g,haloshift=0)
379  endif
380 
381  ! offer diagnostic fields for averaging
382  if (query_averaging_enabled(cs%diag)) then
383  if (cs%id_uhGM > 0) call post_data(cs%id_uhGM, cdp%uhGM, cs%diag)
384  if (cs%id_vhGM > 0) call post_data(cs%id_vhGM, cdp%vhGM, cs%diag)
385  if (cs%id_GMwork > 0) call post_data(cs%id_GMwork, cs%GMwork, cs%diag)
386  if (cs%id_KH_u > 0) call post_data(cs%id_KH_u, kh_u, cs%diag)
387  if (cs%id_KH_v > 0) call post_data(cs%id_KH_v, kh_v, cs%diag)
388  if (cs%id_KH_u1 > 0) call post_data(cs%id_KH_u1, kh_u(:,:,1), cs%diag)
389  if (cs%id_KH_v1 > 0) call post_data(cs%id_KH_v1, kh_v(:,:,1), cs%diag)
390 
391  ! Diagnose diffusivity at T-cell point. Do simple average, rather than
392  ! thickness-weighted average, in order that KH_t is depth-independent
393  ! in the case where KH_u and KH_v are depth independent. Otherwise,
394  ! if use thickess weighted average, the variations of thickness with
395  ! depth will place a spurious depth dependence to the diagnosed KH_t.
396  if (cs%id_KH_t > 0 .or. cs%id_KH_t1 > 0) then
397  do k=1,nz
398  ! thicknesses across u and v faces, converted to 0/1 mask;
399  ! layer average of the interface diffusivities KH_u and KH_v
400  do j=js,je ; do i=is-1,ie
401  hu(i,j) = 2.0*h(i,j,k)*h(i+1,j,k)/(h(i,j,k)+h(i+1,j,k)+h_neglect)
402  if(hu(i,j) /= 0.0) hu(i,j) = 1.0
403  kh_u_lay(i,j) = 0.5*(kh_u(i,j,k)+kh_u(i,j,k+1))
404  enddo ; enddo
405  do j=js-1,je ; do i=is,ie
406  hv(i,j) = 2.0*h(i,j,k)*h(i,j+1,k)/(h(i,j,k)+h(i,j+1,k)+h_neglect)
407  if(hv(i,j) /= 0.0) hv(i,j) = 1.0
408  kh_v_lay(i,j) = 0.5*(kh_v(i,j,k)+kh_v(i,j,k+1))
409  enddo ; enddo
410  ! diagnose diffusivity at T-point
411  do j=js,je ; do i=is,ie
412  kh_t(i,j,k) = ((hu(i-1,j)*kh_u_lay(i-1,j)+hu(i,j)*kh_u_lay(i,j)) &

414  / (hu(i-1,j)+hu(i,j)+hv(i,j-1)+hv(i,j)+h_neglect)
415  enddo ; enddo
416  enddo

418  if(cs%id_KH_t1 > 0) call post_data(cs%id_KH_t1, kh_t(:,:,1), cs%diag)

421  endif

531  ! equation of state.

534  ! goes to 0.
535  real :: H_to_m, m_to_H ! Local copies of unit conversion factors.
536 
537 ! Diagnostics that should be eliminated altogether later...
538  ! real, dimension(SZIB_(G), SZJ_(G), SZK_(G)+1) :: sfn_x, sfn_slope_x
539  ! real, dimension(SZI_(G), SZJB_(G), SZK_(G)+1) :: sfn_y, sfn_slope_y
540  logical :: MEKE_not_null, present_int_slope_u, present_int_slope_v
541  logical :: present_slope_x, present_slope_y, calc_derivatives
542  integer :: is, ie, js, je, nz, IsdB
543  integer :: i, j, k
544  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke ; isdb = g%IsdB
545 
546  h_to_m = gv%H_to_m ; m_to_h = gv%m_to_H
547  i4dt = 0.25 / dt
548  i_slope_max2 = 1.0 / (cs%slope_max**2)
549  g_scale = g%g_Earth * h_to_m
550  h_neglect = gv%H_subroundoff ; h_neglect2 = h_neglect**2
551  dz_neglect = gv%H_subroundoff*h_to_m
552 
553  use_eos = associated(tv%eqn_of_state)
554  meke_not_null = (loc(meke) .NE. 0)
555  present_int_slope_u = PRESENT(int_slope_u)
556  present_int_slope_v = PRESENT(int_slope_v)
557  present_slope_x = PRESENT(slope_x)
558  present_slope_y = PRESENT(slope_y)
559 
560  nk_linear = max(gv%nkml, 1)
561 
562  find_work = .false.
563  if (meke_not_null) find_work = ASSOCIATED(meke%GM_src)
564  find_work = (ASSOCIATED(cs%GMwork) .or. find_work)
565 
566  if (use_eos) then
567  call vert_fill_ts(h, tv%T, tv%S, cs%kappa_smooth, dt, t, s, g, gv, 1)
568  endif

570 !$OMP parallel default(none) shared(is,ie,js,je,h_avail_rsum,pres,h_avail,I4dt, &
571 !$OMP G,GV,h,h_frac,nz,uhtot,Work_u,vhtot,Work_v )
572  ! Find the maximum and minimum permitted streamfunction.
573 !$OMP do
574  do j=js-1,je+1 ; do i=is-1,ie+1
575  h_avail_rsum(i,j,1) = 0.0
576  pres(i,j,1) = 0.0 ! ### This should be atmospheric pressure.
577 
578  h_avail(i,j,1) = max(i4dt*g%areaT(i,j)*(h(i,j,1)-gv%Angstrom),0.0)
579  h_avail_rsum(i,j,2) = h_avail(i,j,1)
580  h_frac(i,j,1) = 1.0
581  pres(i,j,2) = pres(i,j,1) + gv%H_to_Pa*h(i,j,1)
582  enddo ; enddo
583 !$OMP do
584  do j=js-1,je+1
585  do k=2,nz ; do i=is-1,ie+1
586  h_avail(i,j,k) = max(i4dt*g%areaT(i,j)*(h(i,j,k)-gv%Angstrom),0.0)
587  h_avail_rsum(i,j,k+1) = h_avail_rsum(i,j,k) + h_avail(i,j,k)
588  h_frac(i,j,k) = 0.0 ; if (h_avail(i,j,k) > 0.0) &
589  h_frac(i,j,k) = h_avail(i,j,k) / h_avail_rsum(i,j,k+1)
590  pres(i,j,k+1) = pres(i,j,k) + gv%H_to_Pa*h(i,j,k)
591  enddo ; enddo
592  enddo
593 !$OMP do
594  do j=js,je ; do i=is-1,ie
595  uhtot(i,j) = 0.0 ; work_u(i,j) = 0.0
596  ! sfn_x(I,j,1) = 0.0 ; sfn_slope_x(I,j,1) = 0.0
597  ! sfn_x(I,j,nz+1) = 0.0 ; sfn_slope_x(I,j,nz+1) = 0.0
598  enddo ; enddo
599 !$OMP do
600  do j=js-1,je ; do i=is,ie
601  vhtot(i,j) = 0.0 ; work_v(i,j) = 0.0
602  ! sfn_y(i,J,1) = 0.0 ; sfn_slope_y(i,J,1) = 0.0
603  ! sfn_y(i,J,nz+1) = 0.0 ; sfn_slope_y(i,J,nz+1) = 0.0
604  enddo ; enddo
605 !$OMP end parallel
606 
607 !$OMP parallel do default(none) shared(nz,is,ie,js,je,find_work,use_EOS,G,GV,pres,T,S, &
608 !$OMP nk_linear,IsdB,tv,h,h_neglect,e,dz_neglect, &
609 !$OMP I_slope_max2,h_neglect2,present_int_slope_u, &
610 !$OMP int_slope_u,KH_u,uhtot,h_frac,h_avail_rsum, &
611 !$OMP uhD,h_avail,G_scale,work_u,CS,slope_x, &
612 !$OMP slope_y,present_slope_x,H_to_m,m_to_H) &
613 !$OMP private(drdiA,drdiB,drdkL,drdkR,pres_u,T_u,S_u, &
614 !$OMP drho_dT_u,drho_dS_u,hg2A,hg2B,hg2L,hg2R,haA, &
615 !$OMP haB,haL,haR,dzaL,dzaR,wtA,wtB,wtL,wtR,drdz, &
616 !$OMP drdx,mag_grad2,Slope,slope2_Ratio,Sfn_unlim, &
617 !$OMP Sfn_safe,Sfn_est,Sfn,calc_derivatives)
618  do j=js,je ; do k=nz,2,-1
619  if (find_work .and. .not.(use_eos)) then
620  drdia = 0.0 ; drdib = 0.0
621 ! drdkL = GV%g_prime(k) ; drdkR = GV%g_prime(k)
622  drdkl = gv%Rlay(k)-gv%Rlay(k-1) ; drdkr = gv%Rlay(k)-gv%Rlay(k-1)
623  endif
624 
625  calc_derivatives = use_eos .and. (k >= nk_linear) .and. &
626  (find_work .or. .not. present_slope_x)
627 
628  ! Calculate the zonal fluxes and gradients.
629  if (calc_derivatives) then
630  do i=is-1,ie
631  pres_u(i) = 0.5*(pres(i,j,k) + pres(i+1,j,k))
632  t_u(i) = 0.25*((t(i,j,k) + t(i+1,j,k)) + (t(i,j,k-1) + t(i+1,j,k-1)))
633  s_u(i) = 0.25*((s(i,j,k) + s(i+1,j,k)) + (s(i,j,k-1) + s(i+1,j,k-1)))
634  enddo
635  call calculate_density_derivs(t_u, s_u, pres_u, drho_dt_u, &
636  drho_ds_u, (is-isdb+1)-1, ie-is+2, tv%eqn_of_state)
637  endif
638 
639  do i=is-1,ie
640  if (calc_derivatives) then
641  ! Estimate the horizontal density gradients along layers.
642  drdia = drho_dt_u(i) * (t(i+1,j,k-1)-t(i,j,k-1)) + &
643  drho_ds_u(i) * (s(i+1,j,k-1)-s(i,j,k-1))
644  drdib = drho_dt_u(i) * (t(i+1,j,k)-t(i,j,k)) + &
645  drho_ds_u(i) * (s(i+1,j,k)-s(i,j,k))
646 
647  ! Estimate the vertical density gradients times the grid spacing.
648  drdkl = (drho_dt_u(i) * (t(i,j,k)-t(i,j,k-1)) + &
649  drho_ds_u(i) * (s(i,j,k)-s(i,j,k-1)))
650  drdkr = (drho_dt_u(i) * (t(i+1,j,k)-t(i+1,j,k-1)) + &
651  drho_ds_u(i) * (s(i+1,j,k)-s(i+1,j,k-1)))
652  endif
653 
654  if (k > nk_linear) then
655  if (use_eos) then
656  if (present_slope_x) then
657  slope = slope_x(i,j,k)
658  slope2_ratio = slope**2 * i_slope_max2
659  else
660  hg2a = h(i,j,k-1)*h(i+1,j,k-1) + h_neglect2
661  hg2b = h(i,j,k)*h(i+1,j,k) + h_neglect2
662  hg2l = h(i,j,k-1)*h(i,j,k) + h_neglect2
663  hg2r = h(i+1,j,k-1)*h(i+1,j,k) + h_neglect2
664  haa = 0.5*(h(i,j,k-1) + h(i+1,j,k-1))
665  hab = 0.5*(h(i,j,k) + h(i+1,j,k)) + h_neglect
666  hal = 0.5*(h(i,j,k-1) + h(i,j,k)) + h_neglect
667  har = 0.5*(h(i+1,j,k-1) + h(i+1,j,k)) + h_neglect
668  if (gv%Boussinesq) then
669  dzal = hal * h_to_m ; dzar = har * h_to_m
670  else
671  dzal = 0.5*(e(i,j,k-1) - e(i,j,k+1)) + dz_neglect
672  dzar = 0.5*(e(i+1,j,k-1) - e(i+1,j,k+1)) + dz_neglect
673  endif
674  ! Use the harmonic mean thicknesses to weight the horizontal gradients.
675  ! These unnormalized weights have been rearranged to minimize divisions.
676  wta = hg2a*hab ; wtb = hg2b*haa
677  wtl = hg2l*(har*dzar) ; wtr = hg2r*(hal*dzal)
678 
679  drdz = (wtl * drdkl + wtr * drdkr) / (dzal*wtl + dzar*wtr)
680  ! The expression for drdz above is mathematically equivalent to:
681  ! drdz = ((hg2L/haL) * drdkL/dzaL + (hg2R/haR) * drdkR/dzaR) / &
682  ! ((hg2L/haL) + (hg2R/haR))
683  ! This is the gradient of density along geopotentials.
684  drdx = ((wta * drdia + wtb * drdib) / (wta + wtb) - &
685  drdz * (e(i,j,k)-e(i+1,j,k))) * g%IdxCu(i,j)
686 
687  ! This estimate of slope is accurate for small slopes, but bounded
688  ! to be between -1 and 1.
689  mag_grad2 = drdx**2 + drdz**2
690  if (mag_grad2 > 0.0) then
691  slope = drdx / sqrt(mag_grad2)
692  slope2_ratio = slope**2 * i_slope_max2
693  else ! Just in case mag_grad2 = 0 ever.
694  slope = 0.0
695  slope2_ratio = 1.0e20 ! Force the use of the safe streamfunction.
696  endif
697  endif
698 
699  ! Adjust real slope by weights that bias towards slope of interfaces
700  ! that ignore density gradients along layers.
701  if (present_int_slope_u) then
702  slope = (1.0 - int_slope_u(i,j,k)) * slope + &
703  int_slope_u(i,j,k) * ((e(i+1,j,k)-e(i,j,k)) * g%IdxCu(i,j))
704  slope2_ratio = (1.0 - int_slope_u(i,j,k)) * slope2_ratio
705  endif
706  if (cs%id_slope_x > 0) cs%diagSlopeX(i,j,k) = slope
707 
708  ! Estimate the streamfunction at each interface.
709  sfn_unlim = -((kh_u(i,j,k)*g%dy_Cu(i,j))*slope) * m_to_h
710  if (uhtot(i,j) <= 0.0) then
711  ! The transport that must balance the transport below is positive.
712  sfn_safe = uhtot(i,j) * (1.0 - h_frac(i,j,k))
713  else ! (uhtot(I,j) > 0.0)
714  sfn_safe = uhtot(i,j) * (1.0 - h_frac(i+1,j,k))
715  endif
716 
717  ! Avoid moving dense water upslope from below the level of
718  ! the bottom on the receiving side.
719  if (sfn_unlim > 0.0) then ! The flow below this interface is positive.
720  if (e(i,j,k) < e(i+1,j,nz+1)) then
721  sfn_unlim = 0.0 ! This is not uhtot, because it may compensate for
722  ! deeper flow in very unusual cases.
723  elseif (e(i+1,j,nz+1) > e(i,j,k+1)) then
724  ! Scale the transport with the fraction of the donor layer above
725  ! the bottom on the receiving side.
726  sfn_unlim = sfn_unlim * ((e(i,j,k) - e(i+1,j,nz+1)) / &
727  ((e(i,j,k) - e(i,j,k+1)) + dz_neglect))
728  endif
729  else
730  if (e(i+1,j,k) < e(i,j,nz+1)) then ; sfn_unlim = 0.0
731  elseif (e(i,j,nz+1) > e(i+1,j,k+1)) then
732  sfn_unlim = sfn_unlim * ((e(i+1,j,k) - e(i,j,nz+1)) / &
733  ((e(i+1,j,k) - e(i+1,j,k+1)) + dz_neglect))
734  endif
735  endif
736 
737  sfn_est = (sfn_unlim + slope2_ratio*sfn_safe) / (1.0 + slope2_ratio)
738  else ! With .not.use_EOS, the layers are constant density.
739  if (present_slope_x) then
740  slope = slope_x(i,j,k)
741  else
742  slope = ((e(i,j,k)-e(i+1,j,k))*g%IdxCu(i,j)) * m_to_h
743  endif
744  sfn_est = (kh_u(i,j,k)*g%dy_Cu(i,j)) * slope
745  ! ((e(i,j,K)-e(i+1,j,K))*G%IdxCu(I,j))) * m_to_H
746  if (cs%id_slope_x > 0) cs%diagSlopeX(i,j,k) = slope
747  endif
748 
749  ! Make sure that there is enough mass above to allow the streamfunction
750  ! to satisfy the boundary condition of 0 at the surface.
751  sfn = min(max(sfn_est, -h_avail_rsum(i,j,k)), h_avail_rsum(i+1,j,k))
752 
753  ! The actual transport is limited by the mass available in the two
754  ! neighboring grid cells.
755  uhd(i,j,k) = max(min((sfn - uhtot(i,j)), h_avail(i,j,k)), &
756  -h_avail(i+1,j,k))
757 
758  ! sfn_x(I,j,K) = max(min(Sfn, uhtot(I,j)+h_avail(i,j,k)), &
759  ! uhtot(I,j)-h_avail(i+1,j,K))
760  ! sfn_slope_x(I,j,K) = max(uhtot(I,j)-h_avail(i+1,j,k), &
761  ! min(uhtot(I,j)+h_avail(i,j,k), &
762  ! min(h_avail_rsum(i+1,j,K), max(-h_avail_rsum(i,j,K), &
763  ! (KH_u(I,j,K)*G%dy_Cu(I,j)) * ((e(i,j,K)-e(i+1,j,K))*G%IdxCu(I,j)) )) ))
764  else ! k <= nk_linear
765  ! Balance the deeper flow with a return flow uniformly distributed
766  ! though the remaining near-surface layers. This is the same as
767  ! using Sfn_safe above. There is no need to apply the limiters in
768  ! this case.
769  if (uhtot(i,j) <= 0.0) then
770  uhd(i,j,k) = -uhtot(i,j) * h_frac(i,j,k)
771  else ! (uhtot(I,j) > 0.0)
772  uhd(i,j,k) = -uhtot(i,j) * h_frac(i+1,j,k)
773  endif
774 
775  ! sfn_x(I,j,K) = sfn_x(I,j,K+1) + uhD(I,j,k)
776  ! if (sfn_slope_x(I,j,K+1) <= 0.0) then
777  ! sfn_slope_x(I,j,K) = sfn_slope_x(I,j,K+1) * (1.0 - h_frac(i,j,k))
778  ! else
779  ! sfn_slope_x(I,j,K) = sfn_slope_x(I,j,K+1) * (1.0 - h_frac(i+1,j,k))
780  ! endif
781  endif
782 
783  uhtot(i,j) = uhtot(i,j) + uhd(i,j,k)
784 
785  if (find_work) then
786  ! This is the energy tendency based on the original profiles, and does
787  ! not include any nonlinear terms due to a finite time step (which would
788  ! involve interactions between the fluxes through the different faces.
789  ! A second order centered estimate is used for the density transfered
790  ! between water columns.
791 
792  work_u(i,j) = work_u(i,j) + g_scale * &
793  ( uhtot(i,j) * (drdkr * e(i+1,j,k) - drdkl * e(i,j,k)) - &
794  (uhd(i,j,k) * drdib) * 0.25 * &
795  ((e(i,j,k) + e(i,j,k+1)) + (e(i+1,j,k) + e(i+1,j,k+1))) )
796  endif
797 
798  enddo
799  enddo ; enddo ! end of j-loop
800 
801  ! Calculate the meridional fluxes and gradients.
802 !$OMP parallel do default(none) shared(nz,is,ie,js,je,find_work,use_EOS,G,GV,pres,T,S, &
803 !$OMP nk_linear,IsdB,tv,h,h_neglect,e,dz_neglect, &
804 !$OMP I_slope_max2,h_neglect2,present_int_slope_v, &
805 !$OMP int_slope_v,KH_v,vhtot,h_frac,h_avail_rsum, &
806 !$OMP vhD,h_avail,G_scale,Work_v,CS,slope_y, &
807 !$OMP present_slope_y,present_slope_x,m_to_H,H_to_m) &
808 !$OMP private(drdjA,drdjB,drdkL,drdkR,pres_v,T_v,S_v, &
809 !$OMP drho_dT_v,drho_dS_v,hg2A,hg2B,hg2L,hg2R,haA, &
810 !$OMP haB,haL,haR,dzaL,dzaR,wtA,wtB,wtL,wtR,drdz, &
811 !$OMP drdy,mag_grad2,Slope,slope2_Ratio,Sfn_unlim, &
812 !$OMP Sfn_safe,Sfn_est,Sfn,calc_derivatives)
813  do j=js-1,je ; do k=nz,2,-1
814  if (find_work .and. .not.(use_eos)) then
815  drdja = 0.0 ; drdjb = 0.0
816 ! drdkL = GV%g_prime(k) ; drdkR = GV%g_prime(k)
817  drdkl = gv%Rlay(k)-gv%Rlay(k-1) ; drdkr = gv%Rlay(k)-gv%Rlay(k-1)
818  endif
819 
820  calc_derivatives = use_eos .and. (k >= nk_linear) .and. &
821  (find_work .or. .not. present_slope_x)
822 
823  if (calc_derivatives) then
824  do i=is,ie
825  pres_v(i) = 0.5*(pres(i,j,k) + pres(i,j+1,k))
826  t_v(i) = 0.25*((t(i,j,k) + t(i,j+1,k)) + (t(i,j,k-1) + t(i,j+1,k-1)))
827  s_v(i) = 0.25*((s(i,j,k) + s(i,j+1,k)) + (s(i,j,k-1) + s(i,j+1,k-1)))
828  enddo
829  call calculate_density_derivs(t_v, s_v, pres_v, drho_dt_v, &
830  drho_ds_v, is, ie-is+1, tv%eqn_of_state)
831  endif
832  do i=is,ie
833  if (calc_derivatives) then
834  ! Estimate the horizontal density gradients along layers.
835  drdja = drho_dt_v(i) * (t(i,j+1,k-1)-t(i,j,k-1)) + &
836  drho_ds_v(i) * (s(i,j+1,k-1)-s(i,j,k-1))
837  drdjb = drho_dt_v(i) * (t(i,j+1,k)-t(i,j,k)) + &
838  drho_ds_v(i) * (s(i,j+1,k)-s(i,j,k))
839 
840  ! Estimate the vertical density gradients times the grid spacing.
841  drdkl = (drho_dt_v(i) * (t(i,j,k)-t(i,j,k-1)) + &
842  drho_ds_v(i) * (s(i,j,k)-s(i,j,k-1)))
843  drdkr = (drho_dt_v(i) * (t(i,j+1,k)-t(i,j+1,k-1)) + &
844  drho_ds_v(i) * (s(i,j+1,k)-s(i,j+1,k-1)))
845  endif
846 
847  if (k > nk_linear) then
848  if (use_eos) then
849  if (present_slope_y) then
850  slope = slope_y(i,j,k)
851  slope2_ratio = slope**2 * i_slope_max2
852  else
853  hg2a = h(i,j,k-1)*h(i,j+1,k-1) + h_neglect2
854  hg2b = h(i,j,k)*h(i,j+1,k) + h_neglect2
855  hg2l = h(i,j,k-1)*h(i,j,k) + h_neglect2
856  hg2r = h(i,j+1,k-1)*h(i,j+1,k) + h_neglect2
857  haa = 0.5*(h(i,j,k-1) + h(i,j+1,k-1)) + h_neglect
858  hab = 0.5*(h(i,j,k) + h(i,j+1,k)) + h_neglect
859  hal = 0.5*(h(i,j,k-1) + h(i,j,k)) + h_neglect
860  har = 0.5*(h(i,j+1,k-1) + h(i,j+1,k)) + h_neglect
861  if (gv%Boussinesq) then
862  dzal = hal * h_to_m ; dzar = har * h_to_m
863  else
864  dzal = 0.5*(e(i,j,k-1) - e(i,j,k+1)) + dz_neglect
865  dzar = 0.5*(e(i,j+1,k-1) - e(i,j+1,k+1)) + dz_neglect
866  endif
867  ! Use the harmonic mean thicknesses to weight the horizontal gradients.
868  ! These unnormalized weights have been rearranged to minimize divisions.
869  wta = hg2a*hab ; wtb = hg2b*haa
870  wtl = hg2l*(har*dzar) ; wtr = hg2r*(hal*dzal)
871 
872  drdz = (wtl * drdkl + wtr * drdkr) / (dzal*wtl + dzar*wtr)
873  ! The expression for drdz above is mathematically equivalent to:
874  ! drdz = ((hg2L/haL) * drdkL/dzaL + (hg2R/haR) * drdkR/dzaR) / &
875  ! ((hg2L/haL) + (hg2R/haR))
876  ! This is the gradient of density along geopotentials.
877  drdy = ((wta * drdja + wtb * drdjb) / (wta + wtb) - &
878  drdz * (e(i,j,k)-e(i,j+1,k))) * g%IdyCv(i,j)
879 
880  ! This estimate of slope is accurate for small slopes, but bounded
881  ! to be between -1 and 1.
882  mag_grad2 = drdy**2 + drdz**2
883  if (mag_grad2 > 0.0) then
884  slope = drdy / sqrt(mag_grad2)
885  slope2_ratio = slope**2 * i_slope_max2
886  else ! Just in case mag_grad2 = 0 ever.
887  slope = 0.0
888  slope2_ratio = 1.0e20 ! Force the use of the safe streamfunction.
889  endif
890  endif
891 
892  ! Adjust real slope by weights that bias towards slope of interfaces
893  ! that ignore density gradients along layers.
894  if (present_int_slope_v) then
895  slope = (1.0 - int_slope_v(i,j,k)) * slope + &
896  int_slope_v(i,j,k) * ((e(i,j+1,k)-e(i,j,k)) * g%IdyCv(i,j))
897  slope2_ratio = (1.0 - int_slope_v(i,j,k)) * slope2_ratio
898  endif
899  if (cs%id_slope_y > 0) cs%diagSlopeY(i,j,k) = slope
900 
901  ! Estimate the streamfunction at each interface.
902  sfn_unlim = -((kh_v(i,j,k)*g%dx_Cv(i,j))*slope) * m_to_h
903  if (vhtot(i,j) <= 0.0) then
904  sfn_safe = vhtot(i,j) * (1.0 - h_frac(i,j,k))
905  else ! (vhtot(I,j) > 0.0)
906  sfn_safe = vhtot(i,j) * (1.0 - h_frac(i,j+1,k))
907  endif
908 
909  ! Avoid moving dense water upslope from below the level of
910  ! the bottom on the receiving side.
911  if (sfn_unlim > 0.0) then ! The flow below this interface is positive.
912  if (e(i,j,k) < e(i,j+1,nz+1)) then
913  sfn_unlim = 0.0 ! This is not vhtot, because it may compensate for
914  ! deeper flow in very unusual cases.
915  elseif (e(i,j+1,nz+1) > e(i,j,k+1)) then
916  ! Scale the transport with the fraction of the donor layer above
917  ! the bottom on the receiving side.
918  sfn_unlim = sfn_unlim * ((e(i,j,k) - e(i,j+1,nz+1)) / &
919  ((e(i,j,k) - e(i,j,k+1)) + dz_neglect))
920  endif
921  else
922  if (e(i,j+1,k) < e(i,j,nz+1)) then ; sfn_unlim = 0.0
923  elseif (e(i,j,nz+1) > e(i,j+1,k+1)) then
924  sfn_unlim = sfn_unlim * ((e(i,j+1,k) - e(i,j,nz+1)) / &
925  ((e(i,j+1,k) - e(i,j+1,k+1)) + dz_neglect))
926  endif
927  endif
928 
929  ! Estimate the streamfunction at each interface.
930  sfn_est = (sfn_unlim + slope2_ratio*sfn_safe) / (1.0 + slope2_ratio)
931  else ! With .not.use_EOS, the layers are constant density.
932  if (present_slope_y) then
933  slope = slope_y(i,j,k)
934  else
935  slope = ((e(i,j,k)-e(i,j+1,k))*g%IdyCv(i,j)) * m_to_h
936  endif
937  sfn_est = (kh_v(i,j,k)*g%dx_Cv(i,j)) * slope
938  ! ((e(i,j,K)-e(i,j+1,K))*G%IdyCv(i,J))) * m_to_H
939  if (cs%id_slope_y > 0) cs%diagSlopeY(i,j,k) = slope
940  endif
941 
942  ! Make sure that there is enough mass above to allow the streamfunction
943  ! to satisfy the boundary condition of 0 at the surface.
944  sfn = min(max(sfn_est, -h_avail_rsum(i,j,k)), h_avail_rsum(i,j+1,k))
945 
946  ! The actual transport is limited by the mass available in the two
947  ! neighboring grid cells.
948  vhd(i,j,k) = max(min((sfn - vhtot(i,j)), h_avail(i,j,k)), &
949  -h_avail(i,j+1,k))
950 
951  ! sfn_y(i,J,K) = max(min(Sfn, vhtot(i,J)+h_avail(i,j,k)), &
952  ! vhtot(i,J)-h_avail(i,j+1,k))
953  ! sfn_slope_y(i,J,K) = max(vhtot(i,J)-h_avail(i,j+1,k), &
954  ! min(vhtot(i,J)+h_avail(i,j,k), &
955  ! min(h_avail_rsum(i,j+1,K), max(-h_avail_rsum(i,j,K), &
956  ! (KH_v(i,J,K)*G%dx_Cv(i,J)) * ((e(i,j,K)-e(i,j+1,K))*G%IdyCv(i,J)) )) ))
957  else ! k <= nk_linear
958  ! Balance the deeper flow with a return flow uniformly distributed
959  ! though the remaining near-surface layers.
960  if (vhtot(i,j) <= 0.0) then
961  vhd(i,j,k) = -vhtot(i,j) * h_frac(i,j,k)
962  else ! (vhtot(i,J) > 0.0)
963  vhd(i,j,k) = -vhtot(i,j) * h_frac(i,j+1,k)
964  endif
965 
966  ! sfn_y(i,J,K) = sfn_y(i,J,K+1) + vhD(i,J,k)
967  ! if (sfn_slope_y(i,J,K+1) <= 0.0) then
968  ! sfn_slope_y(i,J,K) = sfn_slope_y(i,J,K+1) * (1.0 - h_frac(i,j,k))
969  ! else
970  ! sfn_slope_y(i,J,K) = sfn_slope_y(i,J,K+1) * (1.0 - h_frac(i,j+1,k))
971  ! endif
972  endif
973 
974  vhtot(i,j) = vhtot(i,j) + vhd(i,j,k)
975 
976  if (find_work) then
977  ! This is the energy tendency based on the original profiles, and does
978  ! not include any nonlinear terms due to a finite time step (which would
979  ! involve interactions between the fluxes through the different faces.
980  ! A second order centered estimate is used for the density transfered
981  ! between water columns.
982 
983  work_v(i,j) = work_v(i,j) + g_scale * &
984  ( vhtot(i,j) * (drdkr * e(i,j+1,k) - drdkl * e(i,j,k)) - &
985  (vhd(i,j,k) * drdjb) * 0.25 * &
986  ((e(i,j,k) + e(i,j,k+1)) + (e(i,j+1,k) + e(i+1,j,k+1))) )
987  endif
988  enddo
989  enddo ; enddo! j-loop
990 
991  ! In layer 1, enforce the boundary conditions that Sfn(z=0) = 0.0
992  if (.not.find_work .or. .not.(use_eos)) then
993  do j=js,je ; do i=is-1,ie ; uhd(i,j,1) = -uhtot(i,j) ; enddo ; enddo
994  do j=js-1,je ; do i=is,ie ; vhd(i,j,1) = -vhtot(i,j) ; enddo ; enddo
995  else
996 !$OMP parallel do default(none) shared(is,ie,js,je,pres,T,S,IsdB,tv,uhD,uhtot, &
997 !$OMP Work_u,G_scale,use_EOS,e) &
998 !$OMP private(pres_u,T_u,S_u,drho_dT_u,drho_dS_u,drdiB)
999  do j=js,je
1000  if (use_eos) then
1001  do i=is-1,ie
1002  pres_u(i) = 0.5*(pres(i,j,1) + pres(i+1,j,1))
1003  t_u(i) = 0.5*(t(i,j,1) + t(i+1,j,1))
1004  s_u(i) = 0.5*(s(i,j,1) + s(i+1,j,1))
1005  enddo
1006  call calculate_density_derivs(t_u, s_u, pres_u, drho_dt_u, &
1007  drho_ds_u, (is-isdb+1)-1, ie-is+2, tv%eqn_of_state)
1008  endif
1009  do i=is-1,ie
1010  uhd(i,j,1) = -uhtot(i,j)
1011 
1012  if (use_eos) then
1013  drdib = drho_dt_u(i) * (t(i+1,j,1)-t(i,j,1)) + &
1014  drho_ds_u(i) * (s(i+1,j,1)-s(i,j,1))
1015  endif
1016  work_u(i,j) = work_u(i,j) + g_scale * ( (uhd(i,j,1) * drdib) * 0.25 * &
1017  ((e(i,j,1) + e(i,j,2)) + (e(i+1,j,1) + e(i+1,j,2))) )
1018 
1019  enddo
1020  enddo
1021 
1022  do j=js-1,je
1023  if (use_eos) then
1024  do i=is,ie
1025  pres_v(i) = 0.5*(pres(i,j,1) + pres(i,j+1,1))
1026  t_v(i) = 0.5*(t(i,j,1) + t(i,j+1,1))
1027  s_v(i) = 0.5*(s(i,j,1) + s(i,j+1,1))
1028  enddo
1029  call calculate_density_derivs(t_v, s_v, pres_v, drho_dt_v, &
1030  drho_ds_v, is, ie-is+1, tv%eqn_of_state)
1031  endif
1032  do i=is,ie
1033  vhd(i,j,1) = -vhtot(i,j)
1034 
1035  if (use_eos) then
1036  drdjb = drho_dt_v(i) * (t(i,j+1,1)-t(i,j,1)) + &
1037  drho_ds_v(i) * (s(i,j+1,1)-s(i,j,1))
1038  endif
1039  work_v(i,j) = work_v(i,j) - g_scale * ( (vhd(i,j,1) * drdjb) * 0.25 * &
1040  ((e(i,j,1) + e(i,j,2)) + (e(i,j+1,1) + e(i,j+1,2))) )
1041  enddo
1042  enddo
1043  endif
1044 
1045  if (find_work) then ; do j=js,je ; do i=is,ie
1046  ! Note that the units of Work_v and Work_u are W, while Work_h is W m-2.
1047  work_h = 0.5 * g%IareaT(i,j) * &
1048  ((work_u(i-1,j) + work_u(i,j)) + (work_v(i,j-1) + work_v(i,j)))
1049  if (ASSOCIATED(cs%GMwork)) cs%GMwork(i,j) = work_h
1050  if (meke_not_null) then ; if (ASSOCIATED(meke%GM_src)) then
1051  meke%GM_src(i,j) = meke%GM_src(i,j) + work_h
1052  endif ; endif
1053  enddo ; enddo ; endif
1054 
1055  if (cs%id_slope_x > 0) call post_data(cs%id_slope_x, cs%diagSlopeX, cs%diag)
1056  if (cs%id_slope_y > 0) call post_data(cs%id_slope_y, cs%diagSlopeY, cs%diag)
1057  ! if (CS%id_sfn_x > 0) call post_data(CS%id_sfn_x, sfn_x, CS%diag)
1058  ! if (CS%id_sfn_y > 0) call post_data(CS%id_sfn_y, sfn_y, CS%diag)
1059  ! if (CS%id_sfn_slope_x > 0) call post_data(CS%id_sfn_slope_x, sfn_slope_x, CS%diag)
1060  ! if (CS%id_sfn_slope_y > 0) call post_data(CS%id_sfn_slope_y, sfn_slope_y, CS%diag)
1061 
1062 end subroutine thickness_diffuse_full

1523  ! lost in roundoff and can be neglected

1525  ! 0 < h_neglect << h0.

1537  h0 = 1.0e-16*sqrt(kappa*dt)*gv%m_to_H
1538

1563  s_f(i,j,k) = b1(i) * (h_tr*s_in(i,j,k) + ent(i,k)*s_f(i,j,k-1))
1564  enddo ; enddo
1565  do i=is,ie
1566  c1(i,nz) = ent(i,nz) * b1(i)
1567  h_tr = h(i,j,nz) + h_neglect

1569  t_f(i,j,nz) = b1(i) * (h_tr*t_in(i,j,nz) + ent(i,nz)*t_f(i,j,nz-1))
1570  s_f(i,j,nz) = b1(i) * (h_tr*s_in(i,j,nz) + ent(i,nz)*s_f(i,j,nz-1))
1571  enddo

1574  s_f(i,j,k) = s_f(i,j,k) + c1(i,k+1)*s_f(i,j,k+1)
1575  enddo ; enddo
1576  enddo
1577  endif
