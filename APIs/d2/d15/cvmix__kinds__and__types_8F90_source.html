57  ! cvmix_data_type contains variables for time-dependent and column-specific
58  ! mixing. Time-independent physical parameters should be stored in
59  ! cvmix_global_params_type and *-mixing specific parameters should be
60  ! stored in cvmix_*_params_type (found in the cvmix_* module).
61  type, public :: cvmix_data_type

109  ! diffusivity coefficients at interfaces
110  ! different coefficients for momentum (Mdiff), temperature (Tdiff), and
111  ! salinity / non-temp tracers (Sdiff)
112  real(cvmix_r8), dimension(:), pointer :: mdiff_iface => null()
113  real(cvmix_r8), dimension(:), pointer :: tdiff_iface => null()
114  real(cvmix_r8), dimension(:), pointer :: sdiff_iface => null()
115  ! units: m^2/s
116 
117  ! shear Richardson number at column interfaces
118  real(cvmix_r8), dimension(:), pointer :: shearrichardson_iface => null()
119  ! units: unitless
120 
121  ! For tidal mixing, we need the squared buoyancy frequency
122  real(cvmix_r8), dimension(:), pointer :: sqrbuoyancyfreq_iface => null()
123  ! units: s^-2
124 
125  ! For KPP, need to store non-local transport term
126  real(cvmix_r8), dimension(:), pointer :: kpp_tnonlocal_iface => null()
127  real(cvmix_r8), dimension(:), pointer :: kpp_snonlocal_iface => null()
128  ! units: unitless (see note below)
129  ! Note that kpp_transport_iface is the value of K_x*gamma_x/flux_x: in
130  ! other words, the user must multiply this value by either the freshwater
131  ! flux or the penetrative shortwave heat flux to come the values in Eqs.
132  ! (7.128) and (7.129) of the CVMix manual.
133  ! Currently only provide nonlocal term for temperature tracer and salinity
134  ! (non-temperature) tracers. Eventually may add support for momentum terms
135  ! (would be 2D for x- and y-, respectively) but current implementation
136  ! assumes momentum term is 0 everywhere.

158 
159  ! For double diffusion mixing, we need to calculate the stratification
