26 !* This file contains the driver routine which selects which *
27 !* continuity solver will be called, based on run-time input. *
28 !* *
29 !* A small fragment of the grid is shown below: *
30 !* *
31 !* j+1 x ^ x ^ x At x: q *
32 !* j+1 > o > o > At ^: v, vh *
33 !* j x ^ x ^ x At >: u, uh *
34 !* j > o > o > At o: h, hin *
35 !* j-1 x ^ x ^ x *
36 !* i-1 i i+1 At x & ^: *
37 !* i i+1 At > & o: *
38 !* *
39 !* The boundaries always run through q grid points (x). *
40 !* *

100  type(bt_cont_type), pointer, optional :: BT_cont
101 ! This subroutine time steps the layer thicknesses, using a monotonically
102 ! limit, directionally split PPM scheme, based on Lin (1994).
103 
104 ! Arguments: u - Zonal velocity, in m s-1.
105 ! (in) v - Meridional velocity, in m s-1.
106 ! (in) hin - Initial layer thickness, in m.
107 ! (out) h - Final layer thickness, in m.
108 ! (out) uh - Volume flux through zonal faces = u*h*dy, m3 s-1.
109 ! (out) vh - Volume flux through meridional faces = v*h*dx,
110 ! in m3 s-1.
111 ! (in) dt - Time increment in s.
112 ! (in) G - The ocean's grid structure.
113 ! (in) GV - The ocean's vertical grid structure.
114 ! (in) CS - The control structure returned by a previous call to
115 ! continuity_init.
116 ! (in, opt) uhbt - The summed volume flux through zonal faces, m3 s-1.
117 ! (in, opt) vhbt - The summed volume flux through meridional faces, m3 s-1.
118 ! (in, opt) OBC - This open boundary condition type specifies whether, where,
119 ! and what open boundary conditions are used.
120 ! (in, opt) visc_rem_u - Both the fraction of the momentum originally in a
121 ! (in, opt) visc_rem_v - layer that remains after a time-step of viscosity,
122 ! and the fraction of a time-step's worth of a
123 ! barotropic acceleration that a layer experiences
124 ! after viscosity is applied, in the zonal (_u) and
125 ! meridional (_v) directions. Nondimensional between
126 ! 0 (at the bottom) and 1 (far above the bottom).
127 ! (out, opt) u_cor - The zonal velocities that give uhbt as the depth-
128 ! integrated transport, in m s-1.
129 ! (out, opt) v_cor - The meridional velocities that give vhbt as the
130 ! depth-integrated transport, in m s-1.
131 ! (in, opt) uhbt_aux - A second set of summed volume fluxes through zonal
132 ! (in, opt) vhbt_aux - and meridional faces, both in m3 s-1.
133 ! (out, opt) u_cor_aux - The zonal and meridional velocities that give uhbt_aux
134 ! (out, opt) v_cor_aux - and vhbt_aux as the depth-integrated transports,
135 ! both in m s-1.
136 ! (out, opt) BT_cont - A structure with elements that describe the effective
137 ! open face areas as a function of barotropic flow.
138  if (present(visc_rem_u) .neqv. present(visc_rem_v)) call mom_error(fatal, &
139  "MOM_continuity: Either both visc_rem_u and visc_rem_v or neither"// &
140  " one must be present in call to continuity.")
141  if (present(u_cor) .neqv. present(v_cor)) call mom_error(fatal, &
142  "MOM_continuity: Either both u_cor and v_cor or neither"// &
143  " one must be present in call to continuity.")
144  if (present(uhbt_aux) .neqv. present(vhbt_aux)) call mom_error(fatal, &
145  "MOM_continuity: Either both uhbt_aux and uhbt_aux or neither"// &
146  " one must be present in call to continuity.")
147  if (present(u_cor_aux) .neqv. present(v_cor_aux)) call mom_error(fatal, &
148  "MOM_continuity: Either both u_cor_aux and v_cor_aux or neither"// &
149  " one must be present in call to continuity.")
150  if (present(u_cor_aux) .neqv. present(uhbt_aux)) call mom_error(fatal, &
151  "MOM_continuity: u_cor_aux can only be calculated if uhbt_aux is"// &
152  " provided, and uhbt_aux has no other purpose. Include both arguments"//&
153  " or neither.")
154 
155  if (cs%continuity_scheme == ppm_scheme) then
156  call continuity_ppm(u, v, hin, h, uh, vh, dt, g, gv, cs%PPM_CSp, uhbt, vhbt, obc, &
157  visc_rem_u, visc_rem_v, u_cor, v_cor, &
158  uhbt_aux, vhbt_aux, u_cor_aux, v_cor_aux, bt_cont)
159  else
160  call mom_error(fatal, "continuity: Unrecognized value of continuity_scheme")
161  endif
162 
163 end subroutine continuity

165 subroutine continuity_init(Time, G, GV, param_file, diag, CS)
166  type(time_type), target, intent(in) :: Time

169  type(param_file_type), intent(in) :: param_file
170  type(diag_ctrl), target, intent(inout) :: diag
171  type(continuity_cs), pointer :: CS
172 ! Arguments: Time - The current model time.
173 ! (in) G - The ocean's grid structure.
174 ! (in) GV - The ocean's vertical grid structure.
175 ! (in) param_file - A structure indicating the open file to parse for
176 ! model parameter values.
177 ! (in) diag - A structure that is used to regulate diagnostic output.
178 ! (in/out) CS - A pointer that is set to point to the control structure
179 ! for this module
180 ! This include declares and sets the variable "version".
181 #include "version_variable.h"
182  character(len=40) :: mod = "MOM_continuity" ! This module's name.
183  character(len=20) :: tmpstr
184 
185  if (associated(cs)) then
186  call mom_error(warning, "continuity_init called with associated control structure.")
187  return
188  endif
189  allocate(cs)
190 
191  ! Read all relevant parameters and write them to the model log.
192  call log_version(param_file, mod, version)
193  call get_param(param_file, mod, "CONTINUITY_SCHEME", tmpstr, &
194  "CONTINUITY_SCHEME selects the discretization for the \n"//&
195  "continuity solver. The only valid value currently is: \n"//&
196  "\t PPM - use a positive-definite (or monotonic) \n"//&
197  "\t piecewise parabolic reconstruction solver.", &
198  default=ppm_string)
199 
200  tmpstr = uppercase(tmpstr) ; cs%continuity_scheme = 0
201  select case (trim(tmpstr))
202  case (ppm_string) ; cs%continuity_scheme = ppm_scheme
203  case default
204  call mom_mesg('continuity_init: CONTINUITY_SCHEME ="'//trim(tmpstr)//'"', 0)
205  call mom_mesg("continuity_init: The only valid value is currently "// &
206  trim(ppm_string), 0)
207  call mom_error(fatal, "continuity_init: Unrecognized setting "// &
208  "#define CONTINUITY_SCHEME "//trim(tmpstr)//" found in input file.")
209  end select
210 
211  if (cs%continuity_scheme == ppm_scheme) then
212  call continuity_ppm_init(time, g, gv, param_file, diag, cs%PPM_CSp)
213  endif
214 
215 end subroutine continuity_init
