234 ! !DESCRIPTION:
235 ! Computes vertical diffusion coefficients for convective mixing.
236 !\\
237 !\\
238 
239 ! !USES:
240 ! Only those used by entire module.
241 
242 ! !INPUT PARAMETERS:
243 
244  integer, intent(in) :: nlev, max_nlev
245  ! max_nlev+1
246  real(cvmix_r8), dimension(max_nlev+1), intent(in) :: Nsqr
247  ! max_nlev
248  real(cvmix_r8), dimension(max_nlev), intent(in) :: dens, dens_lwr
249  type(cvmix_conv_params_type), optional, target, intent(in) :: &
250  cvmix_conv_params_user
251 
252 ! !INPUT/OUTPUT PARAMETERS:
253  ! nlev+1
254  real(cvmix_r8), dimension(max_nlev+1), intent(inout) :: Mdiff_out, &
255  Tdiff_out
256 
257 !EOP
258 !BOC
259 
260 !-----------------------------------------------------------------------
261 !
262 ! local variables
263 !
264 !-----------------------------------------------------------------------
265 
266  real(cvmix_r8) :: vvconv, wgt
267  integer :: kw
268  type(cvmix_conv_params_type), pointer :: cvmix_conv_params_in
269 
270  if (present(cvmix_conv_params_user)) then
271  cvmix_conv_params_in => cvmix_conv_params_user
272  else
273  cvmix_conv_params_in => cvmix_conv_params_saved
274  end if
275 
276 !-----------------------------------------------------------------------
277 !
278 ! enhance the vertical mixing coefficients if gravitationally unstable
279 !
280 !-----------------------------------------------------------------------
281  if (cvmix_conv_params_in%lBruntVaisala) then
282  ! Brunt-Vaisala mixing based on buoyancy
283  ! Based on parameter BVsqr_convect
284  ! diffusivity = convect_diff * wgt
285  ! viscosity = convect_visc * wgt
286 
287  ! For BVsqr_convect < 0:
288  ! wgt = 0 for N^2 > 0
289  ! wgt = 1 for N^2 < BVsqr_convect
290  ! wgt = [1 - (1-N^2/BVsqr_convect)^2]^3 otherwise
291 
292  ! If BVsqr_convect >= 0:
293  ! wgt = 0 for N^2 > 0
294  ! wgt = 1 for N^2 <= 0
295 
296  ! Compute wgt
297  if (cvmix_conv_params_in%BVsqr_convect.lt.0) then
298  do kw=1, nlev
299  wgt = cvmix_zero
300  if (nsqr(kw).le.0) then
301  if (nsqr(kw).gt.cvmix_conv_params_in%BVsqr_convect) then
302  wgt = cvmix_one - nsqr(kw) / cvmix_conv_params_in%BVsqr_convect
303  wgt = (cvmix_one - wgt**2)**3
304  else
305  wgt = cvmix_one
306  end if
307  end if
308  mdiff_out(kw) = wgt*cvmix_get_conv_real('convect_visc', &
309  cvmix_conv_params_in)
310  tdiff_out(kw) = wgt*cvmix_get_conv_real('convect_diff', &
311  cvmix_conv_params_in)
312  end do
313  else ! BVsqr_convect >= 0 => step function
314  do kw=1,nlev-1
315  if (nsqr(kw).le.0) then
316  mdiff_out(kw) = cvmix_get_conv_real('convect_visc', &
317  cvmix_conv_params_in)
318  tdiff_out(kw) = cvmix_get_conv_real('convect_diff', &
319  cvmix_conv_params_in)
320  else
321  mdiff_out(kw) = cvmix_zero
322  tdiff_out(kw) = cvmix_zero
323  end if
324  end do
325  mdiff_out(nlev) = cvmix_zero
326  tdiff_out(nlev) = cvmix_zero
327  end if
328  mdiff_out(nlev+1) = cvmix_zero
329  tdiff_out(nlev+1) = cvmix_zero
330  else
331  ! Default convection mixing based on density
332  do kw=1,nlev-1
333  if (cvmix_conv_params_in%convect_visc.ne.cvmix_zero) then
334  vvconv = cvmix_get_conv_real('convect_visc', cvmix_conv_params_in)
335  else
336  ! convection only affects tracers
337  vvconv = mdiff_out(kw)
338  end if
339 
340 ! if (CVmix_vars%WaterDensity_cntr(kw).gt. &
341 ! CVmix_vars%AdiabWaterDensity_cntr(kw)) then
342  if (dens(kw).gt.dens_lwr(kw)) then
343  mdiff_out(kw+1) = vvconv
344  tdiff_out(kw+1) = cvmix_get_conv_real('convect_diff', &
345  cvmix_conv_params_in)
346  end if
347  end do
348  end if
349

351 
352  end subroutine cvmix_coeffs_conv_low
