45  subroutine cvmix_update_wrap(old_vals, nlev, Mdiff_out, new_Mdiff, &
46  tdiff_out, new_tdiff, sdiff_out, new_sdiff)
47 
48 ! !DESCRIPTION:
49 ! Update diffusivity values based on \verb|old_vals| (either overwrite, sum, or find
50 ! the level-by-level max)
51 !\\
52 !\\
53 
54 ! !USES:
55 ! Only those used by entire module.
56 
57 ! !INPUT PARAMETERS:
58  integer, intent(in) :: old_vals, nlev
59  real(cvmix_r8), dimension(nlev+1), optional, intent(in) :: new_Mdiff, &
60  new_Tdiff, &
61  new_Sdiff
62 
63 ! !OUTPUT PARAMETERS:
64  real(cvmix_r8), dimension(nlev+1), optional, intent(inout) :: Mdiff_out, &
65  Tdiff_out, &
66  Sdiff_out
67 
68 !EOP
69 !BOC
70 
71  integer :: kw
72 
73  select case (old_vals)
74  case (cvmix_sum_old_and_new_vals)
75  if ((present(mdiff_out)).and.(present(new_mdiff))) &
76  mdiff_out = mdiff_out + new_mdiff
77  if ((present(tdiff_out)).and.(present(new_tdiff))) &
78  tdiff_out = tdiff_out + new_tdiff
79  if ((present(sdiff_out)).and.(present(new_sdiff))) &
80  sdiff_out = sdiff_out + new_sdiff
81  case (cvmix_max_old_and_new_vals)
82  do kw=1,nlev+1
83  if ((present(mdiff_out)).and.(present(new_mdiff))) &
84  mdiff_out(kw) = max(mdiff_out(kw), new_mdiff(kw))
85  if ((present(tdiff_out)).and.(present(new_tdiff))) &
86  tdiff_out(kw) = max(tdiff_out(kw), new_tdiff(kw))
87  if ((present(sdiff_out)).and.(present(new_sdiff))) &
88  sdiff_out(kw) = max(sdiff_out(kw), new_sdiff(kw))
89  end do
90  case (cvmix_overwrite_old_val)
91  if ((present(mdiff_out)).and.(present(new_mdiff))) &
92  mdiff_out = new_mdiff
93  if ((present(tdiff_out)).and.(present(new_tdiff))) &
94  tdiff_out = new_tdiff
95  if ((present(sdiff_out)).and.(present(new_sdiff))) &
96  sdiff_out = new_sdiff
97  case DEFAULT
98  print*, "ERROR: do not know how to handle old values!"
99  stop 1
100  end select
101 
102 !EOC
103 
104  end subroutine cvmix_update_wrap
