55 subroutine mom_state_chksum_5arg(mesg, u, v, h, uh, vh, G, GV, haloshift)
56  character(len=*), intent(in) :: mesg
57  type(ocean_grid_type), intent(in) :: G
58  type(verticalgrid_type), intent(in) :: GV
59  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in) :: u
60  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(in) :: v
61  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
62  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in) :: uh
63  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(in) :: vh
64  integer, optional, intent(in) :: haloshift
65 ! This subroutine writes out chksums for the model's basic state variables.
66 ! Arguments: mesg - A message that appears on the chksum lines.
67 ! (in) u - Zonal velocity, in m s-1.
68 ! (in) v - Meridional velocity, in m s-1.
69 ! (in) h - Layer thickness, in m.
70 ! (in) uh - Volume flux through zonal faces = u*h*dy, m3 s-1.
71 ! (in) vh - Volume flux through meridional faces = v*h*dx, in m3 s-1.
72 ! (in) G - The ocean's grid structure.
73 ! (in) GV - The ocean's vertical grid structure.
74  integer :: is, ie, js, je, nz, hs
75  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
76 
77  ! Note that for the chksum calls to be useful for reproducing across PE
78  ! counts, there must be no redundant points, so all variables use is..ie
79  ! and js...je as their extent.
80  hs=1; if (present(haloshift)) hs=haloshift
81  call uchksum(u, mesg//" u",g,haloshift=hs)
82  call vchksum(v, mesg//" v",g,haloshift=hs)
83  call hchksum(gv%H_to_kg_m2*h, mesg//" h",g,haloshift=hs)
84  call uchksum(gv%H_to_kg_m2*uh, mesg//" uh",g,haloshift=hs)
85  call vchksum(gv%H_to_kg_m2*vh, mesg//" vh",g,haloshift=hs)
86 end subroutine mom_state_chksum_5arg

126 ! This subroutine writes out chksums for the model's thermodynamic state

145 subroutine mom_accel_chksum(mesg, CAu, CAv, PFu, PFv, diffu, diffv, G, GV, pbce, &
146  u_accel_bt, v_accel_bt)
147  character(len=*), intent(in) :: mesg
148  type(ocean_grid_type), intent(in) :: G
149  type(verticalgrid_type), intent(in) :: GV
150  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in) :: CAu
151  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(in) :: CAv
152  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in) :: PFu
153  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(in) :: PFv
154  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in) :: diffu
155  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(in) :: diffv
156  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), optional, intent(in) :: pbce
157  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), optional, intent(in) :: u_accel_bt
158  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), optional, intent(in) :: v_accel_bt
159 ! This subroutine writes out chksums for the model's accelerations.
160 ! Arguments: mesg - A message that appears on the chksum lines.
161 ! (in) CAu - Zonal acceleration due to Coriolis and momentum
162 ! advection terms, in m s-2.
163 ! (in) CAv - Meridional acceleration due to Coriolis and
164 ! momentum advection terms, in m s-2.
165 ! (in) PFu - Zonal acceleration due to pressure gradients
166 ! (equal to -dM/dx) in m s-2.
167 ! (in) PFv - Meridional acceleration due to pressure
168 ! gradients (equal to -dM/dy) in m s-2.
169 ! (in) diffu - Zonal acceleration due to convergence of the
170 ! along-isopycnal stress tensor, in m s-2.
171 ! (in) diffv - Meridional acceleration due to convergence of
172 ! the along-isopycnal stress tensor, in m s-2.
173 ! (in) G - The ocean's grid structure.
174 ! (in) GV - The ocean's vertical grid structure.
175 ! (in) pbce - the baroclinic pressure anomaly in each layer
176 ! due to free surface height anomalies, in m s-2.
177 ! pbce points to a space with nz layers or NULL.
178 ! (in) u_accel_bt - The zonal acceleration from terms in the barotropic
179 ! solver, in m s-2.
180 ! (in) v_accel_bt - The meridional acceleration from terms in the
181 ! barotropic solver, in m s-2.
182  integer :: is, ie, js, je, nz
183  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
184 
185  ! Note that for the chksum calls to be useful for reproducing across PE
186  ! counts, there must be no redundant points, so all variables use is..ie
187  ! and js...je as their extent.
188  call uchksum(cau, mesg//" CAu",g,haloshift=0)
189  call vchksum(cav, mesg//" CAv",g,haloshift=0)
190  call uchksum(pfu, mesg//" PFu",g,haloshift=0)
191  call vchksum(pfv, mesg//" PFv",g,haloshift=0)
192  call uchksum(diffu, mesg//" diffu",g,haloshift=0)
193  call vchksum(diffv, mesg//" diffv",g,haloshift=0)
194  if (present(pbce)) &
195  call hchksum(gv%kg_m2_to_H*pbce, mesg//" pbce",g,haloshift=0)
196  if (present(u_accel_bt)) &
197  call uchksum(u_accel_bt, mesg//" u_accel_bt",g,haloshift=0)
198  if (present(v_accel_bt)) &
199  call vchksum(v_accel_bt, mesg//" v_accel_bt",g,haloshift=0)
200 end subroutine mom_accel_chksum

204 subroutine mom_state_stats(mesg, u, v, h, Temp, Salt, G, allowChange, permitDiminishing)
205  type(ocean_grid_type), intent(in) :: G
206  character(len=*), intent(in) :: mesg
207  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in) :: u
208  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(in) :: v
209  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
210  real, pointer, dimension(:,:,:), intent(in) :: Temp, Salt
211  logical, optional, intent(in) :: allowChange, permitDiminishing
212 ! This subroutine monitors statistics for the model's state variables.
213 ! Arguments: mesg - A message that appears on the chksum lines.
214 ! (in) u - Zonal velocity, in m s-1.
215 ! (in) v - Meridional velocity, in m s-1.
216 ! (in) h - Layer thickness, in m.
217 ! (in) T - Temperature, in degree C.
218 ! (in) S - Salinity, in ppt.
219 ! (in) G - The ocean's grid structure.
220 ! (in) allowChange - do not flag an error if the statistics change
221 ! (in) permitDiminishing - do not flag an error if the extrema are diminishing
222  integer :: is, ie, js, je, nz, i, j, k
223  real :: Vol, dV, Area, h_minimum
224  type(stats) :: T, S, delT, delS
225  type(stats), save :: oldT, oldS ! NOTE: save data is not normally allowed but
226  logical, save :: firstCall = .true. ! we use it for debugging purposes here on the
227  logical :: do_TS
228  real, save :: oldVol ! assumption we will not turn this on with threads
229  character(len=80) :: lMsg
230  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
231 
232  do_ts = associated(temp) .and. associated(salt)
233 
234  ! First collect local stats
235  area = 0. ; vol = 0.
236  do j = js, je ; do i = is, ie
237  area = area + g%areaT(i,j)
238  enddo ; enddo
239  t%minimum = 1.e34 ; t%maximum = -1.e34 ; t%average = 0.
240  s%minimum = 1.e34 ; s%maximum = -1.e34 ; s%average = 0.
241  h_minimum = 1.e34
242  do k = 1, nz ; do j = js, je ; do i = is, ie
243  if (g%mask2dT(i,j)>0.) then
244  dv = g%areaT(i,j)*h(i,j,k) ; vol = vol + dv
245  if (do_ts .and. h(i,j,k)>0.) then
246  t%minimum = min( t%minimum, temp(i,j,k) ) ; t%maximum = max( t%maximum, temp(i,j,k) )
247  t%average = t%average + dv*temp(i,j,k)
248  s%minimum = min( s%minimum, salt(i,j,k) ) ; s%maximum = max( s%maximum, salt(i,j,k) )
249  s%average = s%average + dv*salt(i,j,k)
250  endif
251  if (h_minimum > h(i,j,k)) h_minimum = h(i,j,k)
252  endif
253  enddo ; enddo ; enddo
254  call sum_across_pes( area ) ; call sum_across_pes( vol )
255  if (do_ts) then
256  call min_across_pes( t%minimum ) ; call max_across_pes( t%maximum ) ; call sum_across_pes( t%average )
257  call min_across_pes( s%minimum ) ; call max_across_pes( s%maximum ) ; call sum_across_pes( s%average )
258  t%average = t%average / vol ; s%average = s%average / vol
259  endif
260  if (is_root_pe()) then
261  if (.not.firstcall) then
262  dv = vol - oldvol
263  delt%minimum = t%minimum - oldt%minimum ; delt%maximum = t%maximum - oldt%maximum
264  delt%average = t%average - oldt%average
265  dels%minimum = s%minimum - olds%minimum ; dels%maximum = s%maximum - olds%maximum
266  dels%average = s%average - olds%average
267  write(lmsg(1:80),'(2(a,es12.4))') 'Mean thickness =',vol/area,' frac. delta=',dv/vol
268  call mom_mesg(lmsg//trim(mesg))
269  if (do_ts) then
270  write(lmsg(1:80),'(a,3es12.4)') 'Temp min/mean/max =',t%minimum,t%average,t%maximum
271  call mom_mesg(lmsg//trim(mesg))
272  write(lmsg(1:80),'(a,3es12.4)') 'delT min/mean/max =',delt%minimum,delt%average,delt%maximum
273  call mom_mesg(lmsg//trim(mesg))
274  write(lmsg(1:80),'(a,3es12.4)') 'Salt min/mean/max =',s%minimum,s%average,s%maximum
275  call mom_mesg(lmsg//trim(mesg))
276  write(lmsg(1:80),'(a,3es12.4)') 'delS min/mean/max =',dels%minimum,dels%average,dels%maximum
277  call mom_mesg(lmsg//trim(mesg))
278  endif
279  else
280  write(lmsg(1:80),'(a,es12.4)') 'Mean thickness =',vol/area
281  call mom_mesg(lmsg//trim(mesg))
282  if (do_ts) then
283  write(lmsg(1:80),'(a,3es12.4)') 'Temp min/mean/max =',t%minimum,t%average,t%maximum
284  call mom_mesg(lmsg//trim(mesg))
285  write(lmsg(1:80),'(a,3es12.4)') 'Salt min/mean/max =',s%minimum,s%average,s%maximum
286  call mom_mesg(lmsg//trim(mesg))
287  endif
288  endif
289  endif
290  firstcall = .false. ; oldvol = vol
291  oldt%minimum = t%minimum ; oldt%maximum = t%maximum ; oldt%average = t%average
292  olds%minimum = s%minimum ; olds%maximum = s%maximum ; olds%average = s%average
293 
294  if (do_ts .and. t%minimum<-5.0) then
295  do j = js, je ; do i = is, ie
296  if (minval(temp(i,j,:)) == t%minimum) then
297  write(0,'(a,2f12.5)') 'x,y=',g%geoLonT(i,j),g%geoLatT(i,j)
298  write(0,'(a3,3a12)') 'k','h','Temp','Salt'
299  do k = 1, nz
300  write(0,'(i3,3es12.4)') k,h(i,j,k),temp(i,j,k),salt(i,j,k)
301  enddo
302  stop 'Extremum detected'
303  endif
304  enddo ; enddo
305  endif
306 
307  if (h_minimum<0.0) then
308  do j = js, je ; do i = is, ie
309  if (minval(h(i,j,:)) == h_minimum) then
310  write(0,'(a,2f12.5)') 'x,y=',g%geoLonT(i,j),g%geoLatT(i,j)
311  write(0,'(a3,3a12)') 'k','h','Temp','Salt'
312  do k = 1, nz
313  write(0,'(i3,3es12.4)') k,h(i,j,k),temp(i,j,k),salt(i,j,k)
314  enddo
315  stop 'Negative thickness detected'
316  endif
317  enddo ; enddo
318  endif
319 
320 end subroutine mom_state_stats
