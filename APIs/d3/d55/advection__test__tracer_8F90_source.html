24 !* By Robert Hallberg, 2002 *
25 !* *
26 !* This file contains an example of the code that is needed to set *
27 !* up and use a set (in this case eleven) of dynamically passive *
28 !* tracers. These tracers dye the inflowing water or water initially *
29 !* within a range of latitudes or water initially in a range of *
30 !* depths. *
31 !* *
32 !* A single subroutine is called from within each file to register *
33 !* each of the tracers for reinitialization and advection and to *
34 !* register the subroutine that initializes the tracers and set up *
35 !* their output and the subroutine that does any tracer physics or *
36 !* chemistry along with diapycnal mixing (included here because some *
37 !* tracers may float or swim vertically or dye diapycnal processes). *
38 !* *
39 !* *
40 !* Macros written all in capital letters are defined in MOM_memory.h. *
41 !* *
42 !* A small fragment of the grid is shown below: *
43 !* *
44 !* j+1 x ^ x ^ x At x: q *
45 !* j+1 > o > o > At ^: v *
46 !* j x ^ x ^ x At >: u *
47 !* j > o > o > At o: h, tr *
48 !* j-1 x ^ x ^ x *
49 !* i-1 i i+1 At x & ^: *
50 !* i i+1 At > & o: *
51 !* *
52 !* The boundaries always run through q grid points (x). *
53 !* *
54 !********+*********+*********+*********+*********+*********+*********+**

129 function register_advection_test_tracer(HI, GV, param_file, CS, tr_Reg, restart_CS)
130  type(hor_index_type), intent(in) :: HI
131  type(verticalgrid_type), intent(in) :: GV
132  type(param_file_type), intent(in) :: param_file
133  type(advection_test_tracer_cs), pointer :: CS
134  type(tracer_registry_type), pointer :: tr_Reg
135  type(mom_restart_cs), pointer :: restart_CS
136 ! This subroutine is used to register tracer fields and subroutines
137 ! to be used with MOM.
138 ! Arguments: HI - A horizontal index type structure.
139 ! (in) GV - The ocean's vertical grid structure.
140 ! (in) param_file - A structure indicating the open file to parse for
141 ! model parameter values.
142 ! (in/out) CS - A pointer that is set to point to the control structure
143 ! for this module
144 ! (in/out) tr_Reg - A pointer that is set to point to the control structure
145 ! for the tracer advection and diffusion module.
146 ! (in) restart_CS - A pointer to the restart control structure.
147  character(len=80) :: name, longname
148 ! This include declares and sets the variable "version".
149 #include "version_variable.h"
150  character(len=40) :: mod = "advection_test_tracer" ! This module's name.
151  character(len=200) :: inputdir
152  real, pointer :: tr_ptr(:,:,:) => null()
153  logical :: register_advection_test_tracer
154  integer :: isd, ied, jsd, jed, nz, m
155  isd = hi%isd ; ied = hi%ied ; jsd = hi%jsd ; jed = hi%jed ; nz = gv%ke
156 
157  if (associated(cs)) then
158  call mom_error(warning, "register_advection_test_tracer called with an "// &
159  "associated control structure.")
160  return
161  endif
162  allocate(cs)
163 
164  ! Read all relevant parameters and write them to the model log.
165  call log_version(param_file, mod, version, "")
166 
167  call get_param(param_file, mod, "ADVECTION_TEST_X_ORIGIN", cs%x_origin, &
168  "The x-coorindate of the center of the test-functions.\n", default=0.)
169  call get_param(param_file, mod, "ADVECTION_TEST_Y_ORIGIN", cs%y_origin, &
170  "The y-coorindate of the center of the test-functions.\n", default=0.)
171  call get_param(param_file, mod, "ADVECTION_TEST_X_WIDTH", cs%x_width, &
172  "The x-width of the test-functions.\n", default=0.)
173  call get_param(param_file, mod, "ADVECTION_TEST_Y_WIDTH", cs%y_width, &
174  "The y-width of the test-functions.\n", default=0.)
175  call get_param(param_file, mod, "ADVECTION_TEST_TRACER_IC_FILE", cs%tracer_IC_file, &
176  "The name of a file from which to read the initial \n"//&
177  "conditions for the tracers, or blank to initialize \n"//&
178  "them internally.", default=" ")
179 
180  if (len_trim(cs%tracer_IC_file) >= 1) then
181  call get_param(param_file, mod, "INPUTDIR", inputdir, default=".")
182  cs%tracer_IC_file = trim(slasher(inputdir))//trim(cs%tracer_IC_file)
183  call log_param(param_file, mod, "INPUTDIR/ADVECTION_TEST_TRACER_IC_FILE", &
184  cs%tracer_IC_file)
185  endif
186  call get_param(param_file, mod, "SPONGE", cs%use_sponge, &
187  "If true, sponges may be applied anywhere in the domain. \n"//&
188  "The exact location and properties of those sponges are \n"//&
189  "specified from MOM_initialization.F90.", default=.false.)
190 
191  call get_param(param_file, mod, "MASK_TRACERS_IN_MASSLESS_LAYERS", cs%mask_tracers, &
192  "If true, tracers will be masked out in massless layers. \n", &
193  default=.false.)
194 
195  allocate(cs%tr(isd:ied,jsd:jed,nz,ntr)) ; cs%tr(:,:,:,:) = 0.0
196  if (cs%mask_tracers) then
197  allocate(cs%tr_aux(isd:ied,jsd:jed,nz,ntr)) ; cs%tr_aux(:,:,:,:) = 0.0
198  endif
199 
200  do m=1,ntr
201  if (m < 10) then ; write(name,'("tr",I1.1)') m
202  else ; write(name,'("tr",I2.2)') m ; endif
203  write(longname,'("Concentration of Tracer ",I2.2)') m
204  cs%tr_desc(m) = var_desc(name, units="kg kg-1", longname=longname, caller=mod)
205 
206  ! This is needed to force the compiler not to do a copy in the registration
207  ! calls. Curses on the designers and implementers of Fortran90.
208  tr_ptr => cs%tr(:,:,:,m)
209  ! Register the tracer for the restart file.
210  call register_restart_field(tr_ptr, cs%tr_desc(m), .true., restart_cs)
211  ! Register the tracer for horizontal advection & diffusion.
212  call register_tracer(tr_ptr, cs%tr_desc(m), param_file, hi, gv, tr_reg, &
213  tr_desc_ptr=cs%tr_desc(m))
214 
215  ! Set coupled_tracers to be true (hard-coded above) to provide the surface
216  ! values to the coupler (if any). This is meta-code and its arguments will
217  ! currently (deliberately) give fatal errors if it is used.
218  if (cs%coupled_tracers) &
219  cs%ind_tr(m) = aof_set_coupler_flux(trim(name)//'_flux', &
220  flux_type=' ', implementation=' ', caller="register_advection_test_tracer")
221  enddo
222 
223  cs%tr_Reg => tr_reg
224  register_advection_test_tracer = .true.
225 end function register_advection_test_tracer

227 subroutine initialize_advection_test_tracer(restart, day, G, GV, h,diag, OBC, CS, &
228  sponge_csp, diag_to_z_csp)
229  logical, intent(in) :: restart
230  type(time_type), target, intent(in) :: day
231  type(ocean_grid_type), intent(in) :: G
232  type(verticalgrid_type), intent(in) :: GV
233  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
234  type(diag_ctrl), target, intent(in) :: diag
235  type(ocean_obc_type), pointer :: OBC
236  type(advection_test_tracer_cs), pointer :: CS
237  type(sponge_cs), pointer :: sponge_CSp
238  type(diag_to_z_cs), pointer :: diag_to_Z_CSp
239 ! This subroutine initializes the NTR tracer fields in tr(:,:,:,:)
240 ! and it sets up the tracer output.
241 
242 ! Arguments: restart - .true. if the fields have already been read from
243 ! a restart file.
244 ! (in) day - Time of the start of the run.
245 ! (in) G - The ocean's grid structure.
246 ! (in) GV - The ocean's vertical grid structure.
247 ! (in) h - Layer thickness, in m or kg m-2.
248 ! (in) diag - A structure that is used to regulate diagnostic output.
249 ! (in) OBC - This open boundary condition type specifies whether, where,
250 ! and what open boundary conditions are used.
251 ! (in/out) CS - The control structure returned by a previous call to
252 ! register_advection_test_tracer.
253 ! (in/out) sponge_CSp - A pointer to the control structure for the sponges, if
254 ! they are in use. Otherwise this may be unassociated.
255 ! (in/out) diag_to_Z_Csp - A pointer to the control structure for diagnostics
256 ! in depth space.
257  real, allocatable :: temp(:,:,:)
258  real, pointer, dimension(:,:,:) :: &
259  OBC_tr1_u => null(), & ! These arrays should be allocated and set to
260  obc_tr1_v => null() ! specify the values of tracer 1 that should come
261  ! in through u- and v- points through the open
262  ! boundary conditions, in the same units as tr.
263  character(len=16) :: name ! A variable's name in a NetCDF file.
264  character(len=72) :: longname ! The long name of that variable.
265  character(len=48) :: units ! The dimensions of the variable.
266  character(len=48) :: flux_units ! The units for tracer fluxes, usually
267  ! kg(tracer) kg(water)-1 m3 s-1 or kg(tracer) s-1.
268  real, pointer :: tr_ptr(:,:,:) => null()
269  real :: PI ! 3.1415926... calculated as 4*atan(1)
270  real :: tr_y ! Initial zonally uniform tracer concentrations.
271  real :: dist2 ! The distance squared from a line, in m2.
272  real :: h_neglect ! A thickness that is so small it is usually lost
273  ! in roundoff and can be neglected, in m.
274  integer :: i, j, k, is, ie, js, je, isd, ied, jsd, jed, nz, m
275  integer :: IsdB, IedB, JsdB, JedB
276  real :: tmpx, tmpy, locx, locy
277 
278  if (.not.associated(cs)) return
279  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = gv%ke
280  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
281  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
282  h_neglect = gv%H_subroundoff
283 
284  cs%diag => diag
285 
286  if (.not.restart) then
287  do m=1,ntr
288  do k=1,nz ; do j=js,je ; do i=is,ie
289  cs%tr(i,j,k,m) = 0.0
290  enddo ; enddo ; enddo
291  k=1 ! Square wave
292  do j=js,je ; do i=is,ie
293  if (abs(g%geoLonT(i,j)-cs%x_origin)<0.5*cs%x_width .and. &
294  abs(g%geoLatT(i,j)-cs%y_origin)<0.5*cs%y_width) cs%tr(i,j,k,m) = 1.0
295  enddo ; enddo
296  k=2 ! Triangle wave
297  do j=js,je ; do i=is,ie
298  locx=abs(g%geoLonT(i,j)-cs%x_origin)/cs%x_width
299  locy=abs(g%geoLatT(i,j)-cs%y_origin)/cs%y_width
300  cs%tr(i,j,k,m) = max(0.0, 1.0-locx)*max(0.0, 1.0-locy)
301  enddo ; enddo
302  k=3 ! Cosine bell
303  do j=js,je ; do i=is,ie
304  locx=min(1.0, abs(g%geoLonT(i,j)-cs%x_origin)/cs%x_width)*(acos(0.0)*2.)
305  locy=min(1.0, abs(g%geoLatT(i,j)-cs%y_origin)/cs%y_width)*(acos(0.0)*2.)
306  cs%tr(i,j,k,m) = (1.0+cos(locx))*(1.0+cos(locy))*0.25
307  enddo ; enddo
308  k=4 ! Cylinder
309  do j=js,je ; do i=is,ie
310  locx=abs(g%geoLonT(i,j)-cs%x_origin)/cs%x_width
311  locy=abs(g%geoLatT(i,j)-cs%y_origin)/cs%y_width
312  if (locx**2+locy**2<=1.0) cs%tr(i,j,k,m) = 1.0
313  enddo ; enddo
314  k=5 ! Cut cylinder
315  do j=js,je ; do i=is,ie
316  locx=(g%geoLonT(i,j)-cs%x_origin)/cs%x_width
317  locy=(g%geoLatT(i,j)-cs%y_origin)/cs%y_width
318  if (locx**2+locy**2<=1.0) cs%tr(i,j,k,m) = 1.0
319  if (locx>0.0.and.abs(locy)<0.2) cs%tr(i,j,k,m) = 0.0
320  enddo ; enddo
321  enddo
322  endif ! restart
323 
324  ! This needs to be changed if the units of tracer are changed above.
325  if (gv%Boussinesq) then ; flux_units = "kg kg-1 m3 s-1"
326  else ; flux_units = "kg s-1" ; endif
327 
328  do m=1,ntr
329  ! Register the tracer for the restart file.
330  call query_vardesc(cs%tr_desc(m), name, units=units, longname=longname, &
331  caller="initialize_advection_test_tracer")
332  cs%id_tracer(m) = register_diag_field("ocean_model", trim(name), cs%diag%axesTL, &
333  day, trim(longname) , trim(units))
334  cs%id_tr_adx(m) = register_diag_field("ocean_model", trim(name)//"_adx", &
335  cs%diag%axesCuL, day, trim(longname)//" advective zonal flux" , &
336  trim(flux_units))
337  cs%id_tr_ady(m) = register_diag_field("ocean_model", trim(name)//"_ady", &
338  cs%diag%axesCvL, day, trim(longname)//" advective meridional flux" , &
339  trim(flux_units))
340  cs%id_tr_dfx(m) = register_diag_field("ocean_model", trim(name)//"_dfx", &
341  cs%diag%axesCuL, day, trim(longname)//" diffusive zonal flux" , &
342  trim(flux_units))
343  cs%id_tr_dfy(m) = register_diag_field("ocean_model", trim(name)//"_dfy", &
344  cs%diag%axesCvL, day, trim(longname)//" diffusive zonal flux" , &
345  trim(flux_units))
346  if (cs%id_tr_adx(m) > 0) call safe_alloc_ptr(cs%tr_adx(m)%p,isdb,iedb,jsd,jed,nz)
347  if (cs%id_tr_ady(m) > 0) call safe_alloc_ptr(cs%tr_ady(m)%p,isd,ied,jsdb,jedb,nz)
348  if (cs%id_tr_dfx(m) > 0) call safe_alloc_ptr(cs%tr_dfx(m)%p,isdb,iedb,jsd,jed,nz)
349  if (cs%id_tr_dfy(m) > 0) call safe_alloc_ptr(cs%tr_dfy(m)%p,isd,ied,jsdb,jedb,nz)
350 
351 ! Register the tracer for horizontal advection & diffusion.
352  if ((cs%id_tr_adx(m) > 0) .or. (cs%id_tr_ady(m) > 0) .or. &
353  (cs%id_tr_dfx(m) > 0) .or. (cs%id_tr_dfy(m) > 0)) &
354  call add_tracer_diagnostics(name, cs%tr_Reg, cs%tr_adx(m)%p, &
355  cs%tr_ady(m)%p,cs%tr_dfx(m)%p,cs%tr_dfy(m)%p)
356 
357  call register_z_tracer(cs%tr(:,:,:,m), trim(name), longname, units, &
358  day, g, diag_to_z_csp)
359  enddo
360 
361 end subroutine initialize_advection_test_tracer

364 subroutine advection_test_tracer_column_physics(h_old, h_new, ea, eb, fluxes, dt, G, GV, CS)
365  type(ocean_grid_type), intent(in) :: G
366  type(verticalgrid_type), intent(in) :: GV
367  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h_old, h_new, ea, eb
368  type(forcing), intent(in) :: fluxes
369  real, intent(in) :: dt
370  type(advection_test_tracer_cs), pointer :: CS
371 ! This subroutine applies diapycnal diffusion and any other column
372 ! tracer physics or chemistry to the tracers from this file.
373 ! This is a simple example of a set of advected passive tracers.
374 
375 ! Arguments: h_old - Layer thickness before entrainment, in m or kg m-2.
376 ! (in) h_new - Layer thickness after entrainment, in m or kg m-2.
377 ! (in) ea - an array to which the amount of fluid entrained
378 ! from the layer above during this call will be
379 ! added, in m or kg m-2.
380 ! (in) eb - an array to which the amount of fluid entrained
381 ! from the layer below during this call will be
382 ! added, in m or kg m-2.
383 ! (in) fluxes - A structure containing pointers to any possible
384 ! forcing fields. Unused fields have NULL ptrs.
385 ! (in) dt - The amount of time covered by this call, in s.
386 ! (in) G - The ocean's grid structure.
387 ! (in) GV - The ocean's vertical grid structure.
388 ! (in) CS - The control structure returned by a previous call to
389 ! register_advection_test_tracer.
390 !
391 ! The arguments to this subroutine are redundant in that
392 ! h_new[k] = h_old[k] + ea[k] - eb[k-1] + eb[k] - ea[k+1]
393 
394  real :: b1(szi_(g)) ! b1 and c1 are variables used by the
395  real :: c1(szi_(g),szk_(g)) ! tridiagonal solver.
396  integer :: i, j, k, is, ie, js, je, nz, m
397  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = gv%ke
398 
399  if (.not.associated(cs)) return
400 
401 ! do m=1,NTR
402 ! call tracer_vertdiff(h_old, ea, eb, dt, CS%tr(:,:,:,m), G, GV)
403 ! enddo
404 
405  if (cs%mask_tracers) then
406  do m = 1,ntr ; if (cs%id_tracer(m) > 0) then
407  do k=1,nz ; do j=js,je ; do i=is,ie
408  if (h_new(i,j,k) < 1.1*gv%Angstrom) then
409  cs%tr_aux(i,j,k,m) = cs%land_val(m)
410  else
411  cs%tr_aux(i,j,k,m) = cs%tr(i,j,k,m)
412  endif
413  enddo ; enddo ; enddo
414  endif ; enddo
415  endif
416 
417  do m=1,ntr
418  if (cs%mask_tracers) then
419  if (cs%id_tracer(m)>0) &
420  call post_data(cs%id_tracer(m),cs%tr_aux(:,:,:,m),cs%diag)
421  else
422  if (cs%id_tracer(m)>0) &
423  call post_data(cs%id_tracer(m),cs%tr(:,:,:,m),cs%diag)
424  endif
425  if (cs%id_tr_adx(m)>0) &
426  call post_data(cs%id_tr_adx(m),cs%tr_adx(m)%p(:,:,:),cs%diag)
427  if (cs%id_tr_ady(m)>0) &
428  call post_data(cs%id_tr_ady(m),cs%tr_ady(m)%p(:,:,:),cs%diag)
