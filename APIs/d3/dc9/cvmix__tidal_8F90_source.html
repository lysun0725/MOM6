314 
315  if (present(cvmix_tidal_params_user)) then

317  else
318  cvmix_tidal_params => cvmix_tidal_params_saved
319  end if
320 
321  rho = cvmix_params%FreshWaterDensity
322 
323  select case (trim(cvmix_tidal_params%mix_scheme))
324  case ('simmons','Simmons')
325  vert_dep = cvmix_compute_vert_dep(zw(1:nlev+1), zt(1:nlev), nlev, &
326  cvmix_tidal_params)
327  coef = cvmix_tidal_params%local_mixing_frac * &
328  cvmix_tidal_params%efficiency * &
329  energy_flux
330  tdiff_out = cvmix_zero
331  if (oceandepth.ge.cvmix_tidal_params%depth_cutoff) then
332  do k=1, nlev+1
333  if (nsqr(k).gt.cvmix_zero) &
334  tdiff_out(k) = coef*vert_dep(k)/(rho*nsqr(k))
335  if (tdiff_out(k).gt.cvmix_tidal_params%max_coefficient) &
336  tdiff_out(k) = cvmix_tidal_params%max_coefficient
337  end do
338  end if
339 
340  case DEFAULT

342  print*, "ERROR: invalid choice for type of tidal mixing."
343  stop 1

345  end select

356  function cvmix_compute_vert_dep(zw, zt, nlev, CVmix_tidal_params)
357 
358 ! !DESCRIPTION:
359 ! Computes the vertical deposition function needed for Simmons et al tidal
360 ! mixing.
361 !\\
362 !\\
363 !
364 ! !USES:
365 ! only those used by entire module.
366 
367 ! !INPUT PARAMETERS:
368  type(cvmix_tidal_params_type), intent(in) :: CVmix_tidal_params
369  integer, intent(in) :: nlev
370  real(cvmix_r8), dimension(nlev+1), intent(in) :: zw
371  real(cvmix_r8), dimension(nlev), intent(in) :: zt
372 
373 ! !OUTPUT PARAMETERS:
374  real(cvmix_r8), dimension(nlev+1) :: cvmix_compute_vert_dep
375 
376 !EOP
377 !BOC
378 
379  ! Local variables
380  real(cvmix_r8) :: tot_area, num, thick
381  integer :: k
382 
383  ! Compute vertical deposition
384  tot_area = cvmix_zero
385  cvmix_compute_vert_dep(1) = cvmix_zero
386  cvmix_compute_vert_dep(nlev+1) = cvmix_zero
387  do k=2,nlev
388  num = -zw(k)/cvmix_tidal_params%vertical_decay_scale
389  ! Simmons vertical deposition
390  ! Note that it is getting normalized (divide through by tot_area)
391  ! So multiplicative constants that are independent of z are omitted
392  cvmix_compute_vert_dep(k) = exp(num)
393 
394  ! Compute integral of vert_dep via trapezoid rule
395  ! (looks like midpoint rule, but vert_dep = 0 at z=0 and z=-ocn_depth)
396  thick = zt(k-1) - zt(k)
397  tot_area = tot_area + cvmix_compute_vert_dep(k)*thick
398  end do
399  ! Normalize vert_dep (need integral = 1.0D0)
400  cvmix_compute_vert_dep = cvmix_compute_vert_dep/tot_area
401 
402 !EOC
403 
404  end function cvmix_compute_vert_dep
