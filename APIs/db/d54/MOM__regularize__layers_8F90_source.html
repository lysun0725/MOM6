26 !* This file contains the code to do vertical remapping of mass, *
27 !* temperature and salinity in MOM. Other tracers and the horizontal *
28 !* velocity components will be remapped outside of this subroutine *
29 !* using the values that are stored in ea and eb. *
30 !* The code that is here now only applies in very limited cases *
31 !* where the mixed- and buffer-layer structures are problematic, but *
32 !* future additions will include the ability to emulate arbitrary *
33 !* vertical coordinates. *
34 !* *
35 !* Macros written all in capital letters are defined in MOM_memory.h. *
36 !* *
37 !* A small fragment of the grid is shown below: *
38 !* *
39 !* j+1 x ^ x ^ x At x: q *
40 !* j+1 > o > o > At ^: v *
41 !* j x ^ x ^ x At >: u *
42 !* j > o > o > At o: h, T, S, ea, eb, etc. *
43 !* j-1 x ^ x ^ x *
44 !* i-1 i i+1 At x & ^: *
45 !* i i+1 At > & o: *
46 !* *
47 !* The boundaries always run through q grid points (x). *

125 ! This subroutine partially steps the bulk mixed layer model.

135 ! as h - usually m or kg m-2 (i.e., H).
136 ! (in/out) eb - The amount of fluid moved upward into a layer; this should
137 ! be increased due to mixed layer entrainment, in the same units
138 ! as h - usually m or kg m-2 (i.e., H).
139 ! (in) G - The ocean's grid structure.
140 ! (in) GV - The ocean's vertical grid structure.

412  wt * 0.125 * ((e_e + e_w) + (e_n + e_s))
413  e_2d(i,k) = e(i,j,k)
414  endif ; enddo ; enddo
415  do k=1,nz ; do i=is,ie
416  h_2d(i,k) = h(i,j,k)
417  t_2d(i,k) = tv%T(i,j,k) ; s_2d(i,k) = tv%S(i,j,k)
418  enddo ; enddo
419 
420  if (debug) then
421  do k=1,nz ; do i=is,ie ; if (do_i(i)) then
422  h_2d_init(i,k) = h(i,j,k)
423  t_2d_init(i,k) = tv%T(i,j,k) ; s_2d_init(i,k) = tv%S(i,j,k)
424  endif ; enddo ; enddo
425  endif
426 
427  ! First, try to entrain from the interior.
428  ent_any = .false.
429  do i=is,ie
430  more_ent_i(i) = .false. ; ent_i(i) = .false.
431  h_add_tgt(i) = 0.0 ; h_add_tot(i) = 0.0
432  if (do_i(i) .and. (e_2d(i,nkmb+1) > e_filt(i,nkmb+1))) then
433  more_ent_i(i) = .true. ; ent_i(i) = .true. ; ent_any = .true.
434  h_add_tgt(i) = e_2d(i,nkmb+1) - e_filt(i,nkmb+1)
435  endif
436  enddo
437 
438  if (ent_any) then
439  do k=nkmb+1,nz
440  cols_left = .false.
441  do i=is,ie ; if (more_ent_i(i)) then
442  if (h_2d(i,k) - gv%Angstrom > h_neglect) then
443  if (e_2d(i,nkmb+1)-e_filt(i,nkmb+1) > h_2d(i,k) - gv%Angstrom) then
444  h_add = h_2d(i,k) - gv%Angstrom
445  h_2d(i,k) = gv%Angstrom
446  else
447  h_add = e_2d(i,nkmb+1)-e_filt(i,nkmb+1)
448  h_2d(i,k) = h_2d(i,k) - h_add
449  endif
450  d_eb(i,k-1) = d_eb(i,k-1) + h_add
451  h_add_tot(i) = h_add_tot(i) + h_add
452  e_2d(i,nkmb+1) = e_2d(i,nkmb+1) - h_add
453  h_prev = h_2d(i,nkmb)
454  h_2d(i,nkmb) = h_2d(i,nkmb) + h_add
455 
456  t_2d(i,nkmb) = (h_prev*t_2d(i,nkmb) + h_add*t_2d(i,k)) / h_2d(i,nkmb)
457  s_2d(i,nkmb) = (h_prev*s_2d(i,nkmb) + h_add*s_2d(i,k)) / h_2d(i,nkmb)
458 
459  if ((e_2d(i,nkmb+1) <= e_filt(i,nkmb+1)) .or. &
460  (h_add_tot(i) > 0.6*h_add_tgt(i))) then !### 0.6 is adjustable?.
461  more_ent_i(i) = .false.
462  else
463  cols_left = .true.
464  endif
465  else
466  cols_left = .true.
467  endif
468  endif ; enddo
469  if (.not.cols_left) exit
470  enddo
471 
472  ks = min(k-1,nz-1)
473  do k=ks,nkmb,-1 ; do i=is,ie ; if (ent_i(i)) then
474  d_eb(i,k) = d_eb(i,k) + d_eb(i,k+1)
475  endif ; enddo ; enddo
476  endif ! ent_any
477 
478  ! This is where code to detrain to the interior will go.
479  ! The buffer layers can only detrain water into layers when the buffer
480  ! layer potential density is between (c*Rlay(k-1) + (1-c)*Rlay(k)) and
481  ! (c*Rlay(k+1) + (1-c)*Rlay(k)), where 0.5 <= c < 1.0.
482  ! Do not detrain if the 2-layer deficit ratio is not significant.
483  ! Detrainment must be able to come from all mixed and buffer layers.
484  ! All water is moved out of the buffer layers below before moving from
485  ! a shallower layer (characteristics do not cross).
486  det_any = .false.
487  if ((max_def_rat > cs%h_def_tol3) .and. (cs%reg_sfc_detrain)) then
488  do i=is,ie
489  det_i(i) = .false. ; rcv_tol(i) = 0.0
490  if (do_i(i) .and. (e_2d(i,nkmb+1) < e_filt(i,nkmb+1)) .and. &
491  (def_rat_h(i,j) > cs%h_def_tol3)) then
492  det_i(i) = .true. ; det_any = .true.
493  rcv_tol(i) = min((def_rat_h(i,j) - cs%h_def_tol3), 1.0)
494  endif
495  enddo
496  endif
497  if (det_any) then
498  call cpu_clock_begin(id_clock_eos)
499  do k=1,nkmb
500  call calculate_density(t_2d(:,k),s_2d(:,k),p_ref_cv,rcv(:,k), &
501  is,ie-is+1,tv%eqn_of_state)
502  enddo
503  call cpu_clock_end(id_clock_eos)
504 
505  do i=is,ie ; if (det_i(i)) then
506  k1 = nkmb ; k2 = nz
507  h_det_tot = 0.0
508  do ! This loop is terminated by exits.
509  if (k1 <= 1) exit
510  if (k2 <= nkmb) exit
511  ! ### The 0.6 here should be adjustable? It gives 20% overlap for now.
512  rcv_min_det = gv%Rlay(k2) + 0.6*rcv_tol(i)*(gv%Rlay(k2-1)-gv%Rlay(k2))
513  if (k2 < nz) then
514  rcv_max_det = gv%Rlay(k2) + 0.6*rcv_tol(i)*(gv%Rlay(k2+1)-gv%Rlay(k2))
515  else
516  rcv_max_det = gv%Rlay(nz) + 0.6*rcv_tol(i)*(gv%Rlay(nz)-gv%Rlay(nz-1))
517  endif
518  if (rcv(i,k1) > rcv_max_det) &
519  exit ! All shallower interior layers are too light for detrainment.
520 
521  h_deficit = (e_filt(i,k2)-e_filt(i,k2+1)) - h_2d(i,k2)
522  if ((e_filt(i,k2) > e_2d(i,k1+1)) .and. (h_deficit > 0.0) .and. &
523  (rcv(i,k1) < rcv_max_det) .and. (rcv(i,k1) > rcv_min_det)) then
524  ! Detrainment will occur.
525  h_add = min(e_filt(i,k2) - e_2d(i,k2), h_deficit )
526  if (h_add < h_2d(i,k1)) then
527  ! Only part of layer k1 detrains.
528  if (h_add > 0.0) then
529  h_prev = h_2d(i,k2)
530  h_2d(i,k2) = h_2d(i,k2) + h_add
531  e_2d(i,k2) = e_2d(i,k2+1) + h_2d(i,k2)
532  d_ea(i,k2) = d_ea(i,k2) + h_add
533  ! ### THIS IS UPWIND. IT SHOULD BE HIGHER ORDER...
534  t_2d(i,k2) = (h_prev*t_2d(i,k2) + h_add*t_2d(i,k1)) / h_2d(i,k2)
535  s_2d(i,k2) = (h_prev*s_2d(i,k2) + h_add*s_2d(i,k1)) / h_2d(i,k2)
536  h_det_tot = h_det_tot + h_add
537 
538  h_2d(i,k1) = h_2d(i,k1) - h_add
539  do k3=k1,nkmb ; e_2d(i,k3+1) = e_2d(i,k3) - h_2d(i,k3) ; enddo
540  do k3=k1+1,nkmb ; d_ea(i,k3) = d_ea(i,k3) + h_add ; enddo
541  else
542  if (h_add < 0.0) &
543  call mom_error(fatal, "h_add is negative. Some logic is wrong.")
544  h_add = 0.0 ! This usually should not happen...
545  endif
546 
547  ! Move up to the next target layer.
548  k2 = k2-1
549  if (k2>nkmb+1) e_2d(i,k2) = e_2d(i,k2) + h_det_tot
550  else
551  h_add = h_2d(i,k1)
552  h_prev = h_2d(i,k2)
553  h_2d(i,k2) = h_2d(i,k2) + h_add
554  e_2d(i,k2) = e_2d(i,k2+1) + h_2d(i,k2)
555  d_ea(i,k2) = d_ea(i,k2) + h_add
556  t_2d(i,k2) = (h_prev*t_2d(i,k2) + h_add*t_2d(i,k1)) / h_2d(i,k2)
557  s_2d(i,k2) = (h_prev*s_2d(i,k2) + h_add*s_2d(i,k1)) / h_2d(i,k2)
558  h_det_tot = h_det_tot + h_add
559 
560  h_2d(i,k1) = 0.0
561  do k3=k1,nkmb ; e_2d(i,k3+1) = e_2d(i,k3) - h_2d(i,k3) ; enddo
562  do k3=k1+1,nkmb ; d_ea(i,k3) = d_ea(i,k3) + h_add ; enddo
563 
564  ! Move up to the next source layer.
565  k1 = k1-1
566  endif
567 
568  else
569  ! Move up to the next target layer.
570  k2 = k2-1
571  if (k2>nkmb+1) e_2d(i,k2) = e_2d(i,k2) + h_det_tot
572  endif
573 
574  enddo ! exit terminated loop.
575  endif ; enddo
576  ! ### This could be faster if the deepest k with nonzero d_ea were kept.
577  do k=nz-1,nkmb+1,-1 ; do i=is,ie ; if (det_i(i)) then
578  d_ea(i,k) = d_ea(i,k) + d_ea(i,k+1)
579  endif ; enddo ; enddo
580  endif ! Detrainment to the interior.
581  if (debug) then
582  do i=is,ie ; h_tot3(i) = 0.0 ; th_tot3(i) = 0.0 ; sh_tot3(i) = 0.0 ; enddo
583  do k=1,nz ; do i=is,ie ; if (do_i(i)) then
584  h_tot3(i) = h_tot3(i) + h_2d(i,k)
585  th_tot3(i) = th_tot3(i) + h_2d(i,k) * t_2d(i,k)
586  sh_tot3(i) = sh_tot3(i) + h_2d(i,k) * s_2d(i,k)
587  endif ; enddo ; enddo
588  endif
589 
590  do i=is,ie ; if (do_i(i)) then
591  ! Rescale the interface targets so the depth at the bottom of the deepest
592  ! buffer layer matches.
593  scale = e_2d(i,nkmb+1) / e_filt(i,nkmb+1)
594  do k=2,nkmb+1 ; e_filt(i,k) = e_filt(i,k) * scale ; enddo
595 
596  ! Ensure that layer 1 only has water from layers 1 to nkml and rescale
597  ! the remaining layer thicknesses if necessary.
598  if (e_filt(i,2) < e_2d(i,nkml)) then
599  scale = (e_2d(i,nkml) - e_filt(i,nkmb+1)) / &
600  ((e_filt(i,2) - e_filt(i,nkmb+1)) + h_neglect)
601  do k=3,nkmb
602  e_filt(i,k) = e_filt(i,nkmb+1) + scale * (e_filt(i,k) - e_filt(i,nkmb+1))
603  enddo
604  e_filt(i,2) = e_2d(i,nkml)
605  endif
606 
607  ! Map the water back into the layers.
608  k1 = 1 ; k2 = 1
609  int_top = 0.0
610  do k=1,nkmb+1
611  int_flux(k) = 0.0 ; int_rflux(k) = 0.0
612  int_tflux(k) = 0.0 ; int_sflux(k) = 0.0
613  enddo
614  do k=1,2*nkmb
615  int_bot = max(e_2d(i,k1+1),e_filt(i,k2+1))
616  h_add = int_top - int_bot
617 
618  if (k2 > k1) then
619  do k3=k1+1,k2
620  d_ea(i,k3) = d_ea(i,k3) + h_add
621  int_flux(k3) = int_flux(k3) + h_add
622  int_tflux(k3) = int_tflux(k3) + h_add*t_2d(i,k1)
623  int_sflux(k3) = int_sflux(k3) + h_add*s_2d(i,k1)
624  enddo
625  elseif (k1 > k2) then
626  do k3=k2,k1-1
627  d_eb(i,k3) = d_eb(i,k3) + h_add
628  int_flux(k3+1) = int_flux(k3+1) - h_add
629  int_tflux(k3+1) = int_tflux(k3+1) - h_add*t_2d(i,k1)
630  int_sflux(k3+1) = int_sflux(k3+1) - h_add*s_2d(i,k1)
631  enddo
632  endif
633 
634  if (int_bot <= e_filt(i,k2+1)) then
635  ! Increment the target layer.
636  k2 = k2 + 1
637  elseif (int_bot <= e_2d(i,k1+1)) then
638  ! Increment the source layer.
639  k1 = k1 + 1
640  else
641  call mom_error(fatal, &
642  "Regularize_surface: Could not increment target or source.")
643  endif
644  if ((k1 > nkmb) .or. (k2 > nkmb)) exit
645  int_top = int_bot
646  enddo
647  if (k2 < nkmb) &
648  call mom_error(fatal, "Regularize_surface: Did not assign fluid to layer nkmb.")
649 
650  ! Note that movement of water across the base of the bottommost buffer
651  ! layer has already been dealt with separately.
652  do k=1,nkmb ; h_prev_1d(k) = h_2d(i,k) ; enddo
653  h_2d(i,1) = h_2d(i,1) - int_flux(2)
654  do k=2,nkmb-1
655  h_2d(i,k) = h_2d(i,k) + (int_flux(k) - int_flux(k+1))
656  enddo
657  ! Note that movement of water across the base of the bottommost buffer
658  ! layer has already been dealt with separately.
659  h_2d(i,nkmb) = h_2d(i,nkmb) + int_flux(nkmb)
660 
661  t_2d(i,1) = (t_2d(i,1)*h_prev_1d(1) - int_tflux(2)) / h_2d(i,1)
662  s_2d(i,1) = (s_2d(i,1)*h_prev_1d(1) - int_sflux(2)) / h_2d(i,1)
663  do k=2,nkmb-1
664  t_2d(i,k) = (t_2d(i,k)*h_prev_1d(k) + (int_tflux(k) - int_tflux(k+1))) / h_2d(i,k)
665  s_2d(i,k) = (s_2d(i,k)*h_prev_1d(k) + (int_sflux(k) - int_sflux(k+1))) / h_2d(i,k)
666  enddo
667  t_2d(i,nkmb) = (t_2d(i,nkmb)*h_prev_1d(nkmb) + int_tflux(nkmb) ) / h_2d(i,nkmb)
668  s_2d(i,nkmb) = (s_2d(i,nkmb)*h_prev_1d(nkmb) + int_sflux(nkmb) ) / h_2d(i,nkmb)
669 
670  endif ; enddo ! i-loop
671 
672  ! Copy the interior thicknesses and other fields back to the 3-d arrays.
673  do k=1,nz ; do i=is,ie ; if (do_i(i)) then
674  h(i,j,k) = h_2d(i,k)
675  tv%T(i,j,k) = t_2d(i,k) ; tv%S(i,j,k) = s_2d(i,k)
676  ea(i,j,k) = ea(i,j,k) + d_ea(i,k)
677  eb(i,j,k) = eb(i,j,k) + d_eb(i,k)
678  endif ; enddo ; enddo
679 
680  if (debug) then
681  do i=is,ie ; h_tot1(i) = 0.0 ; th_tot1(i) = 0.0 ; sh_tot1(i) = 0.0 ; enddo
682  do i=is,ie ; h_tot2(i) = 0.0 ; th_tot2(i) = 0.0 ; sh_tot2(i) = 0.0 ; enddo
683 
684  do k=1,nz ; do i=is,ie ; if (do_i(i)) then
685  h_tot1(i) = h_tot1(i) + h_2d_init(i,k)
686  h_tot2(i) = h_tot2(i) + h(i,j,k)
687 
688  th_tot1(i) = th_tot1(i) + h_2d_init(i,k) * t_2d_init(i,k)
689  th_tot2(i) = th_tot2(i) + h(i,j,k) * tv%T(i,j,k)
690  sh_tot1(i) = sh_tot1(i) + h_2d_init(i,k) * s_2d_init(i,k)
691  sh_tot2(i) = sh_tot2(i) + h(i,j,k) * tv%S(i,j,k)
692  if (h(i,j,k) < 0.0) &
693  call mom_error(fatal,"regularize_surface: Negative thicknesses.")
694  if (k==1) then ; h_predicted = h_2d_init(i,k) + (d_eb(i,k) - d_ea(i,k+1))
695  elseif (k==nz) then ; h_predicted = h_2d_init(i,k) + (d_ea(i,k) - d_eb(i,k-1))
696  else
697  h_predicted = h_2d_init(i,k) + ((d_ea(i,k) - d_eb(i,k-1)) + &
698  (d_eb(i,k) - d_ea(i,k+1)))
699  endif
700  if (abs(h(i,j,k) - h_predicted) > max(1e-9*abs(h_predicted),gv%Angstrom)) &
701  call mom_error(fatal, "regularize_surface: d_ea mismatch.")
702  endif ; enddo ; enddo
703  do i=is,ie ; if (do_i(i)) then
704  fatal_error = .false.
705  if (abs(h_tot1(i) - h_tot2(i)) > 1e-12*h_tot1(i)) then
706  write(mesg,'(ES11.4," became ",ES11.4," diff ",ES11.4)') &
707  h_tot1(i), h_tot2(i), (h_tot1(i) - h_tot2(i))
708  call mom_error(warning, "regularize_surface: Mass non-conservation."//&
709  trim(mesg), .true.)
710  fatal_error = .true.
711  endif
712  if (abs(th_tot1(i) - th_tot2(i)) > 1e-12*(th_tot1(i)+10.0*h_tot1(i))) then
713  write(mesg,'(ES11.4," became ",ES11.4," diff ",ES11.4," int diff ",ES11.4)') &
714  th_tot1(i), th_tot2(i), (th_tot1(i) - th_tot2(i)), (th_tot1(i) - th_tot3(i))
715  call mom_error(warning, "regularize_surface: Heat non-conservation."//&
716  trim(mesg), .true.)
717  fatal_error = .true.
718  endif
719  if (abs(sh_tot1(i) - sh_tot2(i)) > 1e-12*(sh_tot1(i)+10.0*h_tot1(i))) then
720  write(mesg,'(ES11.4," became ",ES11.4," diff ",ES11.4," int diff ",ES11.4)') &
721  sh_tot1(i), sh_tot2(i), (sh_tot1(i) - sh_tot2(i)), (sh_tot1(i) - sh_tot3(i))
722  call mom_error(warning, "regularize_surface: Salinity non-conservation."//&
723  trim(mesg), .true.)
724  fatal_error = .true.
725  endif
726  if (fatal_error) then
727  write(mesg,'("Error at lat/lon ",2(ES11.4))') g%geoLatT(i,j), g%geoLonT(i,j)
728  call mom_error(fatal, "regularize_surface: Terminating with fatal error. "//&
729  trim(mesg))
730  endif
731  endif ; enddo
732  endif
733 
734  endif ; enddo ! j-loop.
735 
736  if (cs%id_def_rat > 0) call post_data(cs%id_def_rat, def_rat_h, cs%diag)
737 
738 #ifdef DEBUG_CODE
739  if (cs%id_e1 > 0) call post_data(cs%id_e1, e, cs%diag)
740  if (cs%id_def_rat_u > 0) call post_data(cs%id_def_rat_u, def_rat_u, cs%diag)
741  if (cs%id_def_rat_u_1b > 0) call post_data(cs%id_def_rat_u_1b, def_rat_u_1b, cs%diag)
742  if (cs%id_def_rat_v > 0) call post_data(cs%id_def_rat_v, def_rat_v, cs%diag)
743  if (cs%id_def_rat_v_1b > 0) call post_data(cs%id_def_rat_v_1b, def_rat_v_1b, cs%diag)
744 
745  if ((cs%id_def_rat_2 > 0) .or. &
746  (cs%id_def_rat_u_2 > 0) .or. (cs%id_def_rat_u_2b > 0) .or. &
747  (cs%id_def_rat_v_2 > 0) .or. (cs%id_def_rat_v_2b > 0) ) then
748  do j=js-1,je+1 ; do i=is-1,ie+1
749  e(i,j,1) = 0.0
750  enddo ; enddo
751  do k=1,nz ; do j=js-1,je+1 ; do i=is-1,ie+1
752  e(i,j,k+1) = e(i,j,k) - h(i,j,k)
753  enddo ; enddo ; enddo
754 
755  call find_deficit_ratios(e, def_rat_u_2, def_rat_v_2, g, gv, cs, def_rat_u_2b, def_rat_v_2b, h=h)
756 
757  ! Determine which columns are problematic
758  do j=js,je ; do i=is,ie
759  def_rat_h2(i,j) = max(def_rat_u_2(i-1,j), def_rat_u_2(i,j), &
760  def_rat_v_2(i,j-1), def_rat_v_2(i,j))
761  enddo ; enddo
762 
763  if (cs%id_def_rat_2 > 0) call post_data(cs%id_def_rat_2, def_rat_h2, cs%diag)
764  if (cs%id_e2 > 0) call post_data(cs%id_e2, e, cs%diag)
765  if (cs%id_def_rat_u_2 > 0) call post_data(cs%id_def_rat_u_2, def_rat_u_2, cs%diag)
766  if (cs%id_def_rat_u_2b > 0) call post_data(cs%id_def_rat_u_2b, def_rat_u_2b, cs%diag)
767  if (cs%id_def_rat_v_2 > 0) call post_data(cs%id_def_rat_v_2, def_rat_v_2, cs%diag)
768  if (cs%id_def_rat_v_2b > 0) call post_data(cs%id_def_rat_v_2b, def_rat_v_2b, cs%diag)
769  endif
770 #endif
771 
772 end subroutine regularize_surface

774 subroutine find_deficit_ratios(e, def_rat_u, def_rat_v, G, GV, CS, &
775  def_rat_u_2lay, def_rat_v_2lay, halo, h)
776  type(ocean_grid_type), intent(in) :: G
777  type(verticalgrid_type), intent(in) :: GV
778  real, dimension(SZI_(G),SZJ_(G),SZK_(G)+1), intent(in) :: e
779  real, dimension(SZIB_(G),SZJ_(G)), intent(out) :: def_rat_u
