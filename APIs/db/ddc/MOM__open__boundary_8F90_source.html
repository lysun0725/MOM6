25 !* *
26 !* This module implements some aspects of internal open boundary *
27 !* conditions in MOM. *
28 !* *
29 !* A small fragment of the grid is shown below: *
30 !* *
31 !* j+1 x ^ x ^ x At x: q, CoriolisBu *
32 !* j+1 > o > o > At ^: v, tauy *
33 !* j x ^ x ^ x At >: u, taux *
34 !* j > o > o > At o: h, bathyT, buoy, tr, T, S, Rml, ustar *
35 !* j-1 x ^ x ^ x *
36 !* i-1 i i+1 At x & ^: *
37 !* i i+1 At > & o: *
38 !* *
39 !* The boundaries always run through q grid points (x). *
40 !* *

112  rx_avg = (1.0-gamma_u)*obc%rx_old_u(i,j,k) + gamma_u*rx_new

146  if (obc%OBC_kind_v(i,j) == obc_flather_n) then
147  dhdt = v_old(i,j-1,k)-v_new(i,j-1,k) !old-new
148  dhdx = v_new(i,j-1,k)-v_new(i,j-2,k) !in new time backward sasha for J-1

152  obc%ry_old_v(i,j,k) = rx_avg

154 
155  ! dhdt = h_old(i,J,k)-h_new(i,J,k) !old-new
156  ! dhdx = h_new(i,J,k)-h_new(i,J-1,k) !in new time
157  ! rx_new = 0.0
158  ! if (dhdt*dhdx > 0.0) rx_new = min( (dhdt/dhdx), rx_max)

172 
173  ! dhdt = h_old(i,J+1,k)-h_new(i,J+1,k) !old-new
174  ! dhdx = h_new(i,J+1,k)-h_new(i,J+2,k) !in new time
