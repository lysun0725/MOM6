123 subroutine calc_resoln_function(h, tv, G, GV, CS)
124  type(ocean_grid_type), intent(inout) :: G
125  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
126  type(thermo_var_ptrs), intent(in) :: tv
127  type(verticalgrid_type), intent(in) :: GV
128  type(varmix_cs), pointer :: CS
129 ! This subroutine determines a function of the ratio of the grid
130 ! spacing to the deformation radius that is used to scale horizontal
131 ! viscosities or diffusivities.
132 ! In the current implementation, the function is F(R) = 1/(1+R) where
133 ! R = cg^2/(f^2*(dx^2+dy^2))
134 ! Arguments: h - Layer thickness, in m or kg m-2.
135 ! (in) tv - A structure containing the thermodynamic variables.
136 ! (in) G - The ocean's grid structure.
137 ! (in) GV - The ocean's vertical grid structure.
138 ! (in) CS - The control structure returned by a previous call to
139 ! VarMix_init.
140 
141  real :: cg1_q ! The gravity wave speed interpolated to q points, in m s-1.
142  real :: cg1_u ! The gravity wave speed interpolated to u points, in m s-1.
143  real :: cg1_v ! The gravity wave speed interpolated to v points, in m s-1.
144  real :: dx_term
145  integer :: power_2
146  integer :: is, ie, js, je, Isq, Ieq, Jsq, Jeq, nz
147  integer :: i, j, k
148  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
149  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
150 
151  if (.not. ASSOCIATED(cs)) call mom_error(fatal, "calc_resoln_function:"// &
152  "Module must be initialized before it is used.")
153  if (.not. (cs%Resoln_scaled_Kh .or. cs%Resoln_scaled_KhTh .or. &
154  cs%Resoln_scaled_KhTr)) return
155  if (.not. ASSOCIATED(cs%cg1)) call mom_error(fatal, &
156  "calc_resoln_function: %cg1 is not associated with Resoln_scaled_Kh.")
157  if (.not. ASSOCIATED(cs%Res_fn_h)) call mom_error(fatal, &
158  "calc_resoln_function: %Res_fn_h is not associated with Resoln_scaled_Kh.")
159  if (.not. ASSOCIATED(cs%Res_fn_q)) call mom_error(fatal, &
160  "calc_resoln_function: %Res_fn_q is not associated with Resoln_scaled_Kh.")
161  if (.not. ASSOCIATED(cs%Res_fn_u)) call mom_error(fatal, &
162  "calc_resoln_function: %Res_fn_u is not associated with Resoln_scaled_Kh.")
163  if (.not. ASSOCIATED(cs%Res_fn_v)) call mom_error(fatal, &
164  "calc_resoln_function: %Res_fn_v is not associated with Resoln_scaled_Kh.")
165  if (.not. ASSOCIATED(cs%f2_dx2_h)) call mom_error(fatal, &
166  "calc_resoln_function: %f2_dx2_h is not associated with Resoln_scaled_Kh.")
167  if (.not. ASSOCIATED(cs%f2_dx2_q)) call mom_error(fatal, &
168  "calc_resoln_function: %f2_dx2_q is not associated with Resoln_scaled_Kh.")
169  if (.not. ASSOCIATED(cs%f2_dx2_u)) call mom_error(fatal, &
170  "calc_resoln_function: %f2_dx2_u is not associated with Resoln_scaled_Kh.")
171  if (.not. ASSOCIATED(cs%f2_dx2_v)) call mom_error(fatal, &
172  "calc_resoln_function: %f2_dx2_v is not associated with Resoln_scaled_Kh.")
173  if (.not. ASSOCIATED(cs%beta_dx2_h)) call mom_error(fatal, &
174  "calc_resoln_function: %beta_dx2_h is not associated with Resoln_scaled_Kh.")
175  if (.not. ASSOCIATED(cs%beta_dx2_q)) call mom_error(fatal, &
176  "calc_resoln_function: %beta_dx2_q is not associated with Resoln_scaled_Kh.")
177  if (.not. ASSOCIATED(cs%beta_dx2_u)) call mom_error(fatal, &
178  "calc_resoln_function: %beta_dx2_u is not associated with Resoln_scaled_Kh.")
179  if (.not. ASSOCIATED(cs%beta_dx2_v)) call mom_error(fatal, &
180  "calc_resoln_function: %beta_dx2_v is not associated with Resoln_scaled_Kh.")
181 
182  call wave_speed(h, tv, g, gv, cs%cg1, cs%wave_speed_CSp)
183 
184  call create_group_pass(cs%pass_cg1, cs%cg1, g%Domain)
185  call do_group_pass(cs%pass_cg1, g%Domain)
186 
187  ! Do this calculation on the extent used in MOM_hor_visc.F90, and
188  ! MOM_tracer.F90 so that no halo update is needed.
189 
190 !$OMP parallel default(none) shared(is,ie,js,je,Ieq,Jeq,CS) &
191 !$OMP private(dx_term,cg1_q,power_2,cg1_u,cg1_v)
192  if (cs%Res_fn_power_visc >= 100) then
193 !$OMP do
194  do j=js-1,je+1 ; do i=is-1,ie+1
195  dx_term = cs%f2_dx2_h(i,j) + cs%cg1(i,j)*cs%beta_dx2_h(i,j)
196  if ((cs%Res_coef_visc * cs%cg1(i,j))**2 > dx_term) then
197  cs%Res_fn_h(i,j) = 0.0
198  else
199  cs%Res_fn_h(i,j) = 1.0
200  endif
201  enddo ; enddo
202 !$OMP do
203  do j=js-1,jeq ; do i=is-1,ieq
204  cg1_q = 0.25 * ((cs%cg1(i,j) + cs%cg1(i+1,j+1)) + &
205  (cs%cg1(i+1,j) + cs%cg1(i,j+1)))
206  dx_term = cs%f2_dx2_q(i,j) + cg1_q * cs%beta_dx2_q(i,j)
207  if ((cs%Res_coef_visc * cg1_q)**2 > dx_term) then
208  cs%Res_fn_q(i,j) = 0.0
209  else
210  cs%Res_fn_q(i,j) = 1.0
211  endif
212  enddo ; enddo
213  elseif (cs%Res_fn_power_visc == 2) then
214 !$OMP do
215  do j=js-1,je+1 ; do i=is-1,ie+1
216  dx_term = cs%f2_dx2_h(i,j) + cs%cg1(i,j)*cs%beta_dx2_h(i,j)
217  cs%Res_fn_h(i,j) = dx_term / (dx_term + (cs%Res_coef_visc * cs%cg1(i,j))**2)
218  enddo ; enddo
219 !$OMP do
220  do j=js-1,jeq ; do i=is-1,ieq
221  cg1_q = 0.25 * ((cs%cg1(i,j) + cs%cg1(i+1,j+1)) + &
222  (cs%cg1(i+1,j) + cs%cg1(i,j+1)))
223  dx_term = cs%f2_dx2_q(i,j) + cg1_q * cs%beta_dx2_q(i,j)
224  cs%Res_fn_q(i,j) = dx_term / (dx_term + (cs%Res_coef_visc * cg1_q)**2)
225  enddo ; enddo
226  elseif (mod(cs%Res_fn_power_visc, 2) == 0) then
227  power_2 = cs%Res_fn_power_visc / 2
228 !$OMP do
229  do j=js-1,je+1 ; do i=is-1,ie+1
230  dx_term = (cs%f2_dx2_h(i,j) + cs%cg1(i,j)*cs%beta_dx2_h(i,j))**power_2
231  cs%Res_fn_h(i,j) = dx_term / &
232  (dx_term + (cs%Res_coef_visc * cs%cg1(i,j))**cs%Res_fn_power_visc)
233  enddo ; enddo
234 !$OMP do
235  do j=js-1,jeq ; do i=is-1,ieq
236  cg1_q = 0.25 * ((cs%cg1(i,j) + cs%cg1(i+1,j+1)) + &
237  (cs%cg1(i+1,j) + cs%cg1(i,j+1)))
238  dx_term = (cs%f2_dx2_q(i,j) + cg1_q * cs%beta_dx2_q(i,j))**power_2
239  cs%Res_fn_q(i,j) = dx_term / &
240  (dx_term + (cs%Res_coef_visc * cg1_q)**cs%Res_fn_power_visc)
241  enddo ; enddo
242  else
243 !$OMP do
244  do j=js-1,je+1 ; do i=is-1,ie+1
245  dx_term = (sqrt(cs%f2_dx2_h(i,j) + &
246  cs%cg1(i,j)*cs%beta_dx2_h(i,j)))**cs%Res_fn_power_visc
247  cs%Res_fn_h(i,j) = dx_term / &
248  (dx_term + (cs%Res_coef_visc * cs%cg1(i,j))**cs%Res_fn_power_visc)
249  enddo ; enddo
250 !$OMP do
251  do j=js-1,jeq ; do i=is-1,ieq
252  cg1_q = 0.25 * ((cs%cg1(i,j) + cs%cg1(i+1,j+1)) + &
253  (cs%cg1(i+1,j) + cs%cg1(i,j+1)))
254  dx_term = (sqrt(cs%f2_dx2_q(i,j) + &
255  cg1_q * cs%beta_dx2_q(i,j)))**cs%Res_fn_power_visc
256  cs%Res_fn_q(i,j) = dx_term / &
257  (dx_term + (cs%Res_coef_visc * cg1_q)**cs%Res_fn_power_visc)
258  enddo ; enddo
259  endif
260 
261  if (cs%interpolate_Res_fn) then
262  do j=js,je ; do i=is-1,ieq
263  cs%Res_fn_u(i,j) = 0.5*(cs%Res_fn_h(i,j) + cs%Res_fn_h(i+1,j))
264  enddo ; enddo
265  do j=js-1,jeq ; do i=is,ie
266  cs%Res_fn_v(i,j) = 0.5*(cs%Res_fn_h(i,j) + cs%Res_fn_h(i,j+1))
267  enddo ; enddo
268  else ! .not.CS%interpolate_Res_fn
269  if (cs%Res_fn_power_khth >= 100) then
270 !$OMP do
271  do j=js,je ; do i=is-1,ieq
272  cg1_u = 0.5 * (cs%cg1(i,j) + cs%cg1(i+1,j))
273  dx_term = cs%f2_dx2_u(i,j) + cg1_u * cs%beta_dx2_u(i,j)
274  if ((cs%Res_coef_khth * cg1_u)**2 > dx_term) then
275  cs%Res_fn_u(i,j) = 0.0
276  else
277  cs%Res_fn_u(i,j) = 1.0
278  endif
279  enddo ; enddo
280 !$OMP do
281  do j=js-1,jeq ; do i=is,ie
282  cg1_v = 0.5 * (cs%cg1(i,j) + cs%cg1(i,j+1))
283  dx_term = cs%f2_dx2_v(i,j) + cg1_v * cs%beta_dx2_v(i,j)
284  if ((cs%Res_coef_khth * cg1_v)**2 > dx_term) then
285  cs%Res_fn_v(i,j) = 0.0
286  else
287  cs%Res_fn_v(i,j) = 1.0
288  endif
289  enddo ; enddo
290  elseif (cs%Res_fn_power_khth == 2) then
291 !$OMP do
292  do j=js,je ; do i=is-1,ieq
293  cg1_u = 0.5 * (cs%cg1(i,j) + cs%cg1(i+1,j))
294  dx_term = cs%f2_dx2_u(i,j) + cg1_u * cs%beta_dx2_u(i,j)
295  cs%Res_fn_u(i,j) = dx_term / (dx_term + (cs%Res_coef_khth * cg1_u)**2)
296  enddo ; enddo
297 !$OMP do
298  do j=js-1,jeq ; do i=is,ie
299  cg1_v = 0.5 * (cs%cg1(i,j) + cs%cg1(i,j+1))
300  dx_term = cs%f2_dx2_v(i,j) + cg1_v * cs%beta_dx2_v(i,j)
301  cs%Res_fn_v(i,j) = dx_term / (dx_term + (cs%Res_coef_khth * cg1_v)**2)
302  enddo ; enddo
303  elseif (mod(cs%Res_fn_power_khth, 2) == 0) then
304  power_2 = cs%Res_fn_power_khth / 2
305 !$OMP do
306  do j=js,je ; do i=is-1,ieq
307  cg1_u = 0.5 * (cs%cg1(i,j) + cs%cg1(i+1,j))
308  dx_term = (cs%f2_dx2_u(i,j) + cg1_u * cs%beta_dx2_u(i,j))**power_2
309  cs%Res_fn_u(i,j) = dx_term / &
310  (dx_term + (cs%Res_coef_khth * cg1_u)**cs%Res_fn_power_khth)
311  enddo ; enddo
312 !$OMP do
313  do j=js-1,jeq ; do i=is,ie
314  cg1_v = 0.5 * (cs%cg1(i,j) + cs%cg1(i,j+1))
315  dx_term = (cs%f2_dx2_v(i,j) + cg1_v * cs%beta_dx2_v(i,j))**power_2
316  cs%Res_fn_v(i,j) = dx_term / &
317  (dx_term + (cs%Res_coef_khth * cg1_v)**cs%Res_fn_power_khth)
318  enddo ; enddo
319  else
320 !$OMP do
321  do j=js,je ; do i=is-1,ieq
322  cg1_u = 0.5 * (cs%cg1(i,j) + cs%cg1(i+1,j))
323  dx_term = (sqrt(cs%f2_dx2_u(i,j) + &
324  cg1_u * cs%beta_dx2_u(i,j)))**cs%Res_fn_power_khth
325  cs%Res_fn_u(i,j) = dx_term / &
326  (dx_term + (cs%Res_coef_khth * cg1_u)**cs%Res_fn_power_khth)
327  enddo ; enddo
328 !$OMP do
329  do j=js-1,jeq ; do i=is,ie
330  cg1_v = 0.5 * (cs%cg1(i,j) + cs%cg1(i,j+1))
331  dx_term = (sqrt(cs%f2_dx2_v(i,j) + &
332  cg1_v * cs%beta_dx2_v(i,j)))**cs%Res_fn_power_khth
333  cs%Res_fn_v(i,j) = dx_term / &
334  (dx_term + (cs%Res_coef_khth * cg1_v)**cs%Res_fn_power_khth)
335  enddo ; enddo
336  endif
337  endif
338 
339  ! Calculate and store the ratio between deformation radius and grid-spacing
340  ! at h-points (non-dimensional).
341 !$OMP do
342  do j=js-1,je+1 ; do i=is-1,ie+1
343  cs%Rd_dx_h(i,j) = cs%cg1(i,j) / &
344  (sqrt(cs%f2_dx2_h(i,j) + cs%cg1(i,j)*cs%beta_dx2_h(i,j)))
345  enddo ; enddo
346 !$OMP end parallel
347 
348  if (query_averaging_enabled(cs%diag)) then
349  if (cs%id_Res_fn > 0) call post_data(cs%id_Res_fn, cs%Res_fn_h, cs%diag)
350  if (cs%id_Rd_dx > 0) call post_data(cs%id_Rd_dx, cs%Rd_dx_h, cs%diag)
351  endif

355 !> Calculates functions of isopycnal slopes
356 subroutine calc_slope_functions(h, tv, dt, G, GV, CS)
357  type(ocean_grid_type), intent(inout) :: G
358  type(verticalgrid_type), intent(in) :: GV
359  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(inout) :: h
360  type(thermo_var_ptrs), intent(in) :: tv
361  real, intent(in) :: dt
362  type(varmix_cs), pointer :: CS
363 ! This subroutine calls for the calculation of the interface heights, and
364 ! then calls for the calculation of the slope function S*N for the Visbeck et
365 ! al. style scaling for the various horizontal diffusivities.
366  real, dimension(SZI_(G), SZJ_(G), SZK_(G)+1) :: &
367  e ! The interface heights relative to mean sea level, in m.
368  real, dimension(SZIB_(G), SZJ_(G), SZK_(G)+1) :: N2_u ! Square of Brunt-Vaisala freq at u-points
369  real, dimension(SZI_(G), SZJB_(G), SZK_(G)+1) :: N2_v ! Square of Brunt-Vaisala freq at u-points
370 
371  if (.not. ASSOCIATED(cs)) call mom_error(fatal, "MOM_lateral_mixing_coeffs.F90, calc_slope_functions:"//&
372  "Module must be initialized before it is used.")
373 
374  call find_eta(h, tv, g%g_Earth, g, gv, e, halo_size=2)
375  if (cs%use_variable_mixing) then
376  if (cs%use_stored_slopes) then
377  call calc_isoneutral_slopes(g, gv, h, e, tv, dt*cs%kappa_smooth, &
378  cs%slope_x, cs%slope_y, n2_u, n2_v, 1)
379  call calc_visbeck_coeffs(h, e, cs%slope_x, cs%slope_y, n2_u, n2_v, g, gv, cs)
380 ! call calc_slope_functions_using_just_e(h, G, CS, e, .false.)
381  else
382  !call calc_isoneutral_slopes(G, GV, h, e, tv, dt*CS%kappa_smooth, CS%slope_x, CS%slope_y)
383  call calc_slope_functions_using_just_e(h, g, gv, cs, e, .true.)
384  endif

400 !> Calculates diffusivity coefficients similar to Visbeck et al.
401 subroutine calc_visbeck_coeffs(h, e, slope_x, slope_y, N2_u, N2_v, G, GV, CS)
402  type(ocean_grid_type), intent(inout) :: G
403  type(verticalgrid_type), intent(in) :: GV
404  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
405  real, dimension(SZI_(G),SZJ_(G),SZK_(G)+1), intent(in) :: e
406  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)+1), intent(in) :: slope_x
407  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)+1), intent(in) :: N2_u
408  real, dimension(SZI_(G),SZJB_(G),SZK_(G)+1), intent(in) :: slope_y
409  real, dimension(SZI_(G),SZJB_(G),SZK_(G)+1), intent(in) :: N2_v
410  type(varmix_cs), intent(inout) :: CS
411  ! Local variables
412  real :: E_x(szib_(g), szj_(g)) ! X-slope of interface at u points (for diagnostics)
413  real :: E_y(szi_(g), szjb_(g)) ! Y-slope of interface at u points (for diagnostics)
414  real :: Khth_Loc ! Locally calculated thickness mixing coefficient (m2/s)
415  real :: S2 ! Interface slope squared (non-dim)
416  real :: N2 ! Brunt-Vaisala frequency (1/s)
417  real :: Hup, Hdn ! Thickness from above, below (m or kg m-2)
418  real :: H_geom ! The geometric mean of Hup*Hdn, in m or kg m-2.
419  integer :: is, ie, js, je, nz
420  integer :: i, j, k, kb_max
421  real :: S2max, wNE, wSE, wSW, wNW
422  real :: SN_u_local(szib_(g), szj_(g),szk_(g))
423  real :: SN_v_local(szi_(g), szjb_(g),szk_(g))
424  real :: H_u(szib_(g)), H_v(szi_(g))
425  real :: S2_u(szib_(g), szj_(g))
426  real :: S2_v(szi_(g), szjb_(g))
427 
428  if (loc(cs)==0) call mom_error(fatal, "calc_slope_function:"// &
429  "Module must be initialized before it is used.")
430  if (.not. cs%use_variable_mixing) return
431  if (.not. ASSOCIATED(cs%SN_u)) call mom_error(fatal, "calc_slope_function:"// &
432  "%SN_u is not associated with use_variable_mixing.")
433  if (.not. ASSOCIATED(cs%SN_v)) call mom_error(fatal, "calc_slope_function:"// &
434  "%SN_v is not associated with use_variable_mixing.")
435 
436  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
437 
438  s2max = cs%Visbeck_S_max**2
439 
440 !$OMP parallel default(none) shared(is,ie,js,je,CS,nz,e,G,GV,h, &
441 !$OMP H_u,H_v,S2_u,S2_v,slope_x,slope_y, &
442 !$OMP SN_u_local,SN_v_local,N2_u,N2_v, S2max) &
443 !$OMP private(E_x,E_y,S2,Hdn,Hup,H_geom,N2, &
444 !$OMP wNE, wSE, wSW, wNW)
445 !$OMP do
446  do j=js-1,je+1 ; do i=is-1,ie+1
447  cs%SN_u(i,j) = 0.0
448  cs%SN_v(i,j) = 0.0
449  enddo ; enddo
450 
451 
452  ! To set the length scale based on the deformation radius, use wave_speed to
453  ! calculate the first-mode gravity wave speed and then blend the equatorial
454  ! and midlatitude deformation radii, using calc_resoln_function as a template.
455 
456  ! Set the length scale at u-points.
457 !$OMP do
458  do j=js,je ; do i=is-1,ie
459  cs%L2u(i,j) = cs%Visbeck_L_scale**2
460  enddo ; enddo
461  ! Set length scale at v-points
462 !$OMP do
463  do j=js-1,je ; do i=is,ie
464  cs%L2v(i,j) = cs%Visbeck_L_scale**2
465  enddo ; enddo
466 
467 !$OMP do
468  do j = js,je
469  do i=is-1,ie
470  cs%SN_u(i,j) = 0. ; h_u(i) = 0. ; s2_u(i,j) = 0.
471  enddo
472  do k=2,nz ; do i=is-1,ie
473  hdn = sqrt( h(i,j,k) * h(i+1,j,k) )
474  hup = sqrt( h(i,j,k-1) * h(i+1,j,k-1) )
475  h_geom = sqrt( hdn * hup )
476  !H_geom = H_geom * sqrt(N2) ! WKB-ish
477  !H_geom = H_geom * N2 ! WKB-ish
478  wse = h(i+1,j,k)*h(i+1,j-1,k) * h(i+1,j,k)*h(i+1,j-1,k-1)
479  wnw = h(i ,j,k)*h(i ,j+1,k) * h(i ,j,k)*h(i ,j+1,k-1)
480  wne = h(i+1,j,k)*h(i+1,j+1,k) * h(i+1,j,k)*h(i+1,j+1,k-1)
481  wsw = h(i ,j,k)*h(i ,j-1,k) * h(i ,j,k)*h(i ,j-1,k-1)
482  s2 = slope_x(i,j,k)**2 + ( &
483  (wnw*slope_y(i,j,k)**2+wse*slope_y(i+1,j-1,k)**2) &
484  +(wne*slope_y(i+1,j,k)**2+wsw*slope_y(i,j-1,k)**2) ) / &
485  ( ((wse+wnw) + (wne+wsw)) + gv%H_subroundoff**2 ) !### This should be **4 for consistent units.
486  if (s2max>0.) s2 = s2 * s2max / (s2 + s2max) ! Limit S2
487  n2 = max(0., n2_u(i,j,k))
488  cs%SN_u(i,j) = cs%SN_u(i,j) + sqrt( s2*n2 )*h_geom
489  s2_u(i,j) = s2_u(i,j) + s2*h_geom
490  h_u(i) = h_u(i) + h_geom
491  enddo ; enddo
492  do i=is-1,ie
493  if (h_u(i)>0.) then
494  cs%SN_u(i,j) = cs%SN_u(i,j) / h_u(i)
495  s2_u(i,j) = s2_u(i,j) / h_u(i)
496  else
497  cs%SN_u(i,j) = 0.
498  endif
499  enddo
500  enddo
501 
502 !$OMP do
503  do j = js-1,je
504  do i=is,ie
505  cs%SN_v(i,j) = 0. ; h_v(i) = 0. ; s2_v(i,j) = 0.
506  enddo
507  do k=2,nz ; do i=is,ie
508  hdn = sqrt( h(i,j,k) * h(i,j+1,k) )
509  hup = sqrt( h(i,j,k-1) * h(i,j+1,k-1) )
510  h_geom = sqrt( hdn * hup )
511  !H_geom = H_geom * sqrt(N2) ! WKB-ish
512  !H_geom = H_geom * N2 ! WKB-ish
513  wse = h(i,j ,k)*h(i+1,j ,k) * h(i,j ,k)*h(i+1,j ,k-1)
514  wnw = h(i,j+1,k)*h(i-1,j+1,k) * h(i,j+1,k)*h(i-1,j+1,k-1)
515  wne = h(i,j+1,k)*h(i+1,j+1,k) * h(i,j+1,k)*h(i+1,j+1,k-1)
516  wsw = h(i,j ,k)*h(i-1,j ,k) * h(i,j ,k)*h(i-1,j ,k-1)
517  s2 = slope_y(i,j,k)**2 + ( &
518  (wse*slope_x(i,j,k)**2+wnw*slope_x(i-1,j+1,k)**2) &
519  +(wne*slope_x(i,j+1,k)**2+wsw*slope_x(i-1,j,k)**2) ) / &
520  ( ((wse+wnw) + (wne+wsw)) + gv%H_subroundoff**2 ) !### This should be **4 for consistent units.
521  if (s2max>0.) s2 = s2 * s2max / (s2 + s2max) ! Limit S2
522  n2 = max(0., n2_v(i,j,k))
523  cs%SN_v(i,j) = cs%SN_v(i,j) + sqrt( s2*n2 )*h_geom
524  s2_v(i,j) = s2_v(i,j) + s2*h_geom
525  h_v(i) = h_v(i) + h_geom
526  enddo ; enddo
527  do i=is,ie
528  if (h_v(i)>0.) then
529  cs%SN_v(i,j) = cs%SN_v(i,j) / h_v(i)
530  s2_v(i,j) = s2_v(i,j) / h_v(i)
531  else
532  cs%SN_v(i,j) = 0.
533  endif
534  enddo
535  enddo
536 
537 !$OMP end parallel
538 
539 ! Offer diagnostic fields for averaging.
540  if (query_averaging_enabled(cs%diag)) then
541  if (cs%id_S2_u > 0) call post_data(cs%id_S2_u, s2_u, cs%diag)
542  if (cs%id_S2_v > 0) call post_data(cs%id_S2_v, s2_v, cs%diag)
543  endif

545 end subroutine calc_visbeck_coeffs

547 !> The original calc_slope_function() that calculated slopes using
548 !! interface positions only, not accounting for density variations.
549 subroutine calc_slope_functions_using_just_e(h, G, GV, CS, e, calculate_slopes)
550  type(ocean_grid_type), intent(inout) :: G
551  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(inout) :: h
552  type(verticalgrid_type), intent(in) :: GV
553  type(varmix_cs), pointer :: CS
554  real, dimension(SZI_(G),SZJ_(G),SZK_(G)+1), intent(in) :: e
555  logical, intent(in) :: calculate_slopes
556 ! This subroutine calculates the slope function S*N for the Visbeck et
557 ! al. style scaling for the various horizontal diffusivities.
558 
559 ! Arguments: h - Layer thickness, in m or kg m-2.
560 ! (in) G - The ocean's grid structure.
561 ! (in) GV - The ocean's vertical grid structure.
562 ! (inout) CS - The control structure returned by a previous call to

570  ! in roundoff and can be neglected, in H.
571  real :: S2 ! Interface slope squared (non-dim)
572  real :: N2 ! Brunt-Vaisala frequency (1/s)
573  real :: Hup, Hdn ! Thickness from above, below (m or kg m-2)
574  real :: H_geom ! The geometric mean of Hup*Hdn, in m or kg m-2.
575  real :: one_meter ! One meter in thickness units of m or kg m-2.
576  integer :: is, ie, js, je, nz
577  integer :: i, j, k, kb_max
578  real :: SN_u_local(szib_(g), szj_(g),szk_(g))
579  real :: SN_v_local(szi_(g), szjb_(g),szk_(g))
580

582  "Module must be initialized before it is used.")
583  if (.not. cs%use_variable_mixing) return
584  if (.not. ASSOCIATED(cs%SN_u)) call mom_error(fatal, "calc_slope_function:"// &
585  "%SN_u is not associated with use_variable_mixing.")
586  if (.not. ASSOCIATED(cs%SN_v)) call mom_error(fatal, "calc_slope_function:"// &
587  "%SN_v is not associated with use_variable_mixing.")
588 
589  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
590 
591  one_meter = 1.0 * gv%m_to_H
592  h_neglect = gv%H_subroundoff
593  h_cutoff = real(2*nz) * (gv%angstrom + h_neglect)
594 
595 !$OMP parallel default(none) shared(is,ie,js,je,CS,nz,e,G,GV,h,H_cutoff,h_neglect, &
596 !$OMP one_meter,SN_u_local,SN_v_local,calculate_slopes) &
597 !$OMP private(E_x,E_y,S2,Hdn,Hup,H_geom,N2)
598 !$OMP do
599  do j=js-1,je+1 ; do i=is-1,ie+1
600  cs%SN_u(i,j) = 0.0
601  cs%SN_v(i,j) = 0.0
602  enddo ; enddo
603 
604  ! To set the length scale based on the deformation radius, use wave_speed to
605  ! calculate the first-mode gravity wave speed and then blend the equatorial
606  ! and midlatitude deformation radii, using calc_resoln_function as a template.
607 
608  ! Set the length scale at u-points.
609 !$OMP do
610  do j=js,je ; do i=is-1,ie
611  cs%L2u(i,j) = cs%Visbeck_L_scale**2
612  enddo ; enddo
613  ! Set length scale at v-points
614 !$OMP do
615  do j=js-1,je ; do i=is,ie
616  cs%L2v(i,j) = cs%Visbeck_L_scale**2
617  enddo ; enddo
618 !$OMP do
619  do k=nz,cs%VarMix_Ktop,-1
620 
621  if (calculate_slopes) then
622  ! Calculate the interface slopes E_x and E_y and u- and v- points respectively
623  do j=js-1,je+1 ; do i=is-1,ie
624  e_x(i,j) = (e(i+1,j,k)-e(i,j,k))*g%IdxCu(i,j)
625  ! Mask slopes where interface intersects topography
626  if (min(h(i,j,k),h(i+1,j,k)) < h_cutoff) e_x(i,j) = 0.
627  enddo ; enddo
628  do j=js-1,je ; do i=is-1,ie+1
629  e_y(i,j) = (e(i,j+1,k)-e(i,j,k))*g%IdyCv(i,j)
630  ! Mask slopes where interface intersects topography
631  if (min(h(i,j,k),h(i,j+1,k)) < h_cutoff) e_y(i,j) = 0.
632  enddo ; enddo
633  else
634  do j=js-1,je+1 ; do i=is-1,ie
635  e_x(i,j) = cs%slope_x(i,j,k)
636  if (min(h(i,j,k),h(i+1,j,k)) < h_cutoff) e_x(i,j) = 0.
637  enddo ; enddo
638  do j=js-1,je ; do i=is-1,ie+1
639  e_y(i,j) = cs%slope_y(i,j,k)
640  if (min(h(i,j,k),h(i,j+1,k)) < h_cutoff) e_y(i,j) = 0.
641  enddo ; enddo
642  endif
643 
644  ! Calculate N*S*h from this layer and add to the sum
645  do j=js,je ; do i=is-1,ie
646  s2 = ( e_x(i,j)**2 + 0.25*( &
647  (e_y(i,j)**2+e_y(i+1,j-1)**2)+(e_y(i+1,j)**2+e_y(i,j-1)**2) ) )
648  hdn = 2.*h(i,j,k)*h(i,j,k-1) / (h(i,j,k) + h(i,j,k-1) + h_neglect)
649  hup = 2.*h(i+1,j,k)*h(i+1,j,k-1) / (h(i+1,j,k) + h(i+1,j,k-1) + h_neglect)
650  h_geom = sqrt(hdn*hup)
651  n2 = gv%g_prime(k) / (gv%H_to_m * max(hdn,hup,one_meter))
652  if (min(h(i,j,k-1), h(i+1,j,k-1), h(i,j,k), h(i+1,j,k)) < h_cutoff) &
653  s2 = 0.0

658  (e_x(i,j)**2+e_x(i-1,j+1)**2)+(e_x(i,j+1)**2+e_x(i-1,j)**2) ) )
659  hdn = 2.*h(i,j,k)*h(i,j,k-1) / (h(i,j,k) + h(i,j,k-1) + h_neglect)
660  hup = 2.*h(i,j+1,k)*h(i,j+1,k-1) / (h(i,j+1,k) + h(i,j+1,k-1) + h_neglect)
661  h_geom = sqrt(hdn*hup)
662  n2 = gv%g_prime(k) / (gv%H_to_m * max(hdn,hup,one_meter))
663  if (min(h(i,j,k-1), h(i,j+1,k-1), h(i,j,k), h(i,j+1,k)) < h_cutoff) &
664  s2 = 0.0
665  sn_v_local(i,j,k) = (h_geom * gv%H_to_m) * s2 * n2
666  enddo ; enddo
667 
668  enddo ! k
669 !$OMP do
670  do j = js,je;
671  do k=nz,cs%VarMix_Ktop,-1 ; do i=is-1,ie
672  cs%SN_u(i,j) = cs%SN_u(i,j) + sn_u_local(i,j,k)
673  enddo ; enddo
674  ! SN above contains S^2*N^2*H, convert to vertical average of S*N
675  do i=is-1,ie
676  !SN_u(I,j) = sqrt( SN_u(I,j) / ( max(G%bathyT(I,j), G%bathyT(I+1,j)) + GV%Angstrom ) )
677  !The code below behaves better than the line above. Not sure why? AJA
678  if ( min(g%bathyT(i,j), g%bathyT(i+1,j)) > h_cutoff ) then
679  cs%SN_u(i,j) = sqrt( cs%SN_u(i,j) / max(g%bathyT(i,j), g%bathyT(i+1,j)) )
680  else
681  cs%SN_u(i,j) = 0.0
682  endif
683  enddo
684  enddo
685 !$OMP do
686  do j=js-1,je
687  do k=nz,cs%VarMix_Ktop,-1 ; do i=is,ie
688  cs%SN_v(i,j) = cs%SN_v(i,j) + sn_v_local(i,j,k)
689  enddo ; enddo
690  do i=is,ie
691  !SN_v(i,J) = sqrt( SN_v(i,J) / ( max(G%bathyT(i,J), G%bathyT(i,J+1)) + GV%Angstrom ) )
692  !The code below behaves better than the line above. Not sure why? AJA
693  if ( min(g%bathyT(i,j), g%bathyT(i+1,j)) > h_cutoff ) then
694  cs%SN_v(i,j) = sqrt( cs%SN_v(i,j) / max(g%bathyT(i,j), g%bathyT(i,j+1)) )
695  else
696  cs%SN_v(i,j) = 0.0
697  endif
698  enddo
699  enddo
700 !$OMP end parallel
