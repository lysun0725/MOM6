132 ! * Arguments: T - potential temperature relative to the surface in C. *
133 ! * (in) S - salinity in g/kg. *
134 ! * (in) pressure - pressure in Pa. *

136 ! * potential temperature, in m3 kg-1 K-1. *
137 ! * (out) dSV_dS - the partial derivative of specific volume with *

214 subroutine int_density_dz_linear(T, S, z_t, z_b, rho_ref, rho_0_pres, G_e, HII, HIO, &
215  rho_t0_s0, drho_dt, drho_ds, dpa, intz_dpa, intx_dpa, inty_dpa)
216  type(hor_index_type), intent(in) :: HII, HIO
217  real, dimension(SZDI_(HII),SZDJ_(HII)), intent(in) :: T, S, z_t, z_b
218  real, intent(in) :: rho_ref, rho_0_pres, G_e
219  real, intent(in) :: Rho_T0_S0, dRho_dT, dRho_dS
220  real, dimension(SZDI_(HIO),SZDJ_(HIO)), intent(out) :: dpa
221  real, dimension(SZDI_(HIO),SZDJ_(HIO)), optional, intent(out) :: intz_dpa
222  real, dimension(SZDIB_(HIO),SZDJ_(HIO)), optional, intent(out) :: intx_dpa
223  real, dimension(SZDI_(HIO),SZDJB_(HIO)), optional, intent(out) :: inty_dpa
224 ! This subroutine calculates analytical and nearly-analytical integrals of
225 ! pressure anomalies across layers, which are required for calculating the
226 ! finite-volume form pressure accelerations in a Boussinesq model.
227 !
228 ! Arguments: T - potential temperature relative to the surface in C.
229 ! (in) S - salinity in PSU.
230 ! (in) z_t - height at the top of the layer in m.
231 ! (in) z_b - height at the top of the layer in m.
232 ! (in) rho_ref - A mean density, in kg m-3, that is subtracted out to reduce
233 ! the magnitude of each of the integrals.
234 ! (in) rho_0_pres - A density, in kg m-3, that is used to calculate the
235 ! pressure (as p~=-z*rho_0_pres*G_e) used in the equation of
236 ! state. rho_0_pres is not used here.
237 ! (in) G_e - The Earth's gravitational acceleration, in m s-2.
238 ! (in) G - The ocean's grid structure.
239 ! (in) Rho_T0_S0 - The density at T=0, S=0, in kg m-3.
240 ! (in) dRho_dT - The derivative of density with temperature in kg m-3 C-1.
241 ! (in) dRho_dS - The derivative of density with salinity, in kg m-3 psu-1.
242 ! (out) dpa - The change in the pressure anomaly across the layer, in Pa.
243 ! (out,opt) intz_dpa - The integral through the thickness of the layer of the
244 ! pressure anomaly relative to the anomaly at the top of
245 ! the layer, in Pa m.
246 ! (out,opt) intx_dpa - The integral in x of the difference between the
247 ! pressure anomaly at the top and bottom of the layer
248 ! divided by the x grid spacing, in Pa.
249 ! (out,opt) inty_dpa - The integral in y of the difference between the
250 ! pressure anomaly at the top and bottom of the layer
251 ! divided by the y grid spacing, in Pa.
252  real :: rho_anom ! The density anomaly from rho_ref, in kg m-3.
253  real :: raL, raR ! rho_anom to the left and right, in kg m-3.
254  real :: dz, dzL, dzR ! Layer thicknesses in m.
255  real :: C1_6
256  integer :: is, ie, js, je, Isq, Ieq, Jsq, Jeq, i, j, ioff, joff
257 
258  ioff = hio%idg_offset - hii%idg_offset
259  joff = hio%jdg_offset - hii%jdg_offset
260 
261  ! These array bounds work for the indexing convention of the input arrays, but
262  ! on the computational domain defined for the output arrays.
263  isq = hio%IscB + ioff ; ieq = hio%IecB + ioff
264  jsq = hio%JscB + joff ; jeq = hio%JecB + joff
265  is = hio%isc + ioff ; ie = hio%iec + ioff
266  js = hio%jsc + joff ; je = hio%jec + joff
267  c1_6 = 1.0 / 6.0
268 
269  do j=jsq,jeq+1 ; do i=isq,ieq+1
270  dz = z_t(i,j) - z_b(i,j)
271  rho_anom = (rho_t0_s0 - rho_ref) + drho_dt*t(i,j) + drho_ds*s(i,j)
272  dpa(i-ioff,j-joff) = g_e*rho_anom*dz
273  if (present(intz_dpa)) intz_dpa(i-ioff,j-joff) = 0.5*g_e*rho_anom*dz**2
274  enddo ; enddo
275 
276  if (present(intx_dpa)) then ; do j=js,je ; do i=isq,ieq
277  dzl = z_t(i,j) - z_b(i,j) ; dzr = z_t(i+1,j) - z_b(i+1,j)
278  ral = (rho_t0_s0 - rho_ref) + (drho_dt*t(i,j) + drho_ds*s(i,j))
279  rar = (rho_t0_s0 - rho_ref) + (drho_dt*t(i+1,j) + drho_ds*s(i+1,j))
280 
281  intx_dpa(i-ioff,j-joff) = g_e*c1_6 * (dzl*(2.0*ral + rar) + dzr*(2.0*rar + ral))
282  enddo ; enddo ; endif
283 
284  if (present(inty_dpa)) then ; do j=jsq,jeq ; do i=is,ie
285  dzl = z_t(i,j) - z_b(i,j) ; dzr = z_t(i,j+1) - z_b(i,j+1)
286  ral = (rho_t0_s0 - rho_ref) + (drho_dt*t(i,j) + drho_ds*s(i,j))
287  rar = (rho_t0_s0 - rho_ref) + (drho_dt*t(i,j+1) + drho_ds*s(i,j+1))
288 
289  inty_dpa(i-ioff,j-joff) = g_e*c1_6 * (dzl*(2.0*ral + rar) + dzr*(2.0*rar + ral))
290  enddo ; enddo ; endif
291 end subroutine int_density_dz_linear

293 subroutine int_spec_vol_dp_linear(T, S, p_t, p_b, alpha_ref, HI, Rho_T0_S0, &
294  drho_dt, drho_ds, dza, intp_dza, intx_dza, inty_dza, halo_size)
295  type(hor_index_type), intent(in) :: HI
296  real, dimension(SZDI_(HI),SZDJ_(HI)), intent(in) :: T, S, p_t, p_b
297  real, intent(in) :: alpha_ref
298  real, intent(in) :: Rho_T0_S0, dRho_dT, dRho_dS
299  real, dimension(SZDI_(HI),SZDJ_(HI)), intent(out) :: dza
300  real, dimension(SZDI_(HI),SZDJ_(HI)), optional, intent(out) :: intp_dza
301  real, dimension(SZDIB_(HI),SZDJ_(HI)), optional, intent(out) :: intx_dza
302  real, dimension(SZDI_(HI),SZDJB_(HI)), optional, intent(out) :: inty_dza
303  integer, optional, intent(in) :: halo_size
304 ! This subroutine calculates analytical and nearly-analytical integrals in
305 ! pressure across layers of geopotential anomalies, which are required for
306 ! calculating the finite-volume form pressure accelerations in a non-Boussinesq
307 ! model. Specific volume is assumed to vary linearly between adjacent points.
308 !
309 ! Arguments: T - potential temperature relative to the surface in C.
310 ! (in) S - salinity in PSU.
311 ! (in) p_t - pressure at the top of the layer in Pa.
312 ! (in) p_b - pressure at the top of the layer in Pa.
313 ! (in) alpha_ref - A mean specific volume that is subtracted out to reduce
314 ! the magnitude of each of the integrals, m3 kg-1.
315 ! The calculation is mathematically identical with
316 ! different values of alpha_ref, but this reduces the
317 ! effects of roundoff.
318 ! (in) HI - The ocean's horizontal index type.
319 ! (in) Rho_T0_S0 - The density at T=0, S=0, in kg m-3.
320 ! (in) dRho_dT - The derivative of density with temperature in kg m-3 C-1.
321 ! (in) dRho_dS - The derivative of density with salinity, in kg m-3 psu-1.
322 ! (out) dza - The change in the geopotential anomaly across the layer,
323 ! in m2 s-2.
324 ! (out,opt) intp_dza - The integral in pressure through the layer of the
325 ! geopotential anomaly relative to the anomaly at the
326 ! bottom of the layer, in Pa m2 s-2.
327 ! (out,opt) intx_dza - The integral in x of the difference between the
328 ! geopotential anomaly at the top and bottom of the layer
329 ! divided by the x grid spacing, in m2 s-2.
330 ! (out,opt) inty_dza - The integral in y of the difference between the
331 ! geopotential anomaly at the top and bottom of the layer
332 ! divided by the y grid spacing, in m2 s-2.
333  real :: dRho_TS ! The density anomaly due to T and S, in kg m-3.
334  real :: alpha_anom ! The specific volume anomaly from 1/rho_ref, in m3 kg-1.
335  real :: aaL, aaR ! rho_anom to the left and right, in kg m-3.
336  real :: dp, dpL, dpR ! Layer pressure thicknesses in Pa.
337  real :: C1_6
338  integer :: Isq, Ieq, Jsq, Jeq, ish, ieh, jsh, jeh, i, j, halo
339 
340  isq = hi%IscB ; ieq = hi%IecB ; jsq = hi%JscB ; jeq = hi%JecB
341  halo = 0 ; if (present(halo_size)) halo = max(halo_size,0)
342  ish = hi%isc-halo ; ieh = hi%iec+halo ; jsh = hi%jsc-halo ; jeh = hi%jec+halo
343  if (present(intx_dza)) then ; ish = min(isq,ish) ; ieh = max(ieq+1,ieh); endif
344  if (present(inty_dza)) then ; jsh = min(jsq,jsh) ; jeh = max(jeq+1,jeh); endif
345  c1_6 = 1.0 / 6.0
346 
347  do j=jsh,jeh ; do i=ish,ieh
348  dp = p_b(i,j) - p_t(i,j)
349  drho_ts = drho_dt*t(i,j) + drho_ds*s(i,j)
350  ! alpha_anom = 1.0/(Rho_T0_S0 + dRho_TS)) - alpha_ref
351  alpha_anom = ((1.0-rho_t0_s0*alpha_ref) - drho_ts*alpha_ref) / (rho_t0_s0 + drho_ts)
352  dza(i,j) = alpha_anom*dp
353  if (present(intp_dza)) intp_dza(i,j) = 0.5*alpha_anom*dp**2
354  enddo ; enddo
355 
356  if (present(intx_dza)) then ; do j=hi%jsc,hi%jec ; do i=isq,ieq
357  dpl = p_b(i,j) - p_t(i,j) ; dpr = p_b(i+1,j) - p_t(i+1,j)
358  drho_ts = drho_dt*t(i,j) + drho_ds*s(i,j)
359  aal = ((1.0 - rho_t0_s0*alpha_ref) - drho_ts*alpha_ref) / (rho_t0_s0 + drho_ts)
360  drho_ts = drho_dt*t(i+1,j) + drho_ds*s(i+1,j)
361  aar = ((1.0 - rho_t0_s0*alpha_ref) - drho_ts*alpha_ref) / (rho_t0_s0 + drho_ts)
362 
363  intx_dza(i,j) = c1_6 * (2.0*(dpl*aal + dpr*aar) + (dpl*aar + dpr*aal))
364  enddo ; enddo ; endif
365 
366  if (present(inty_dza)) then ; do j=jsq,jeq ; do i=hi%isc,hi%iec
367  dpl = p_b(i,j) - p_t(i,j) ; dpr = p_b(i,j+1) - p_t(i,j+1)
368  drho_ts = drho_dt*t(i,j) + drho_ds*s(i,j)
369  aal = ((1.0 - rho_t0_s0*alpha_ref) - drho_ts*alpha_ref) / (rho_t0_s0 + drho_ts)
370  drho_ts = drho_dt*t(i,j+1) + drho_ds*s(i,j+1)
371  aar = ((1.0 - rho_t0_s0*alpha_ref) - drho_ts*alpha_ref) / (rho_t0_s0 + drho_ts)
372 
373  inty_dza(i,j) = c1_6 * (2.0*(dpl*aal + dpr*aar) + (dpl*aar + dpr*aal))
374  enddo ; enddo ; endif
375 end subroutine int_spec_vol_dp_linear
