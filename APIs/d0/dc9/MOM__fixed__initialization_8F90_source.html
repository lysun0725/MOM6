96  " \t none - All channels have the grid width.\n"//&
97  " \t global_1deg - Sets 16 specific channels appropriate \n"//&

126  endif
127 
128 ! Calculate the value of the Coriolis parameter at the latitude !

130  call mom_initialize_rotation(g%CoriolisBu, g, pf)
131 ! Calculate the components of grad f (beta)
132  call mom_calculate_grad_coriolis(g%dF_dx, g%dF_dy, g)
133  if (debug) then
134  call qchksum(g%CoriolisBu, "MOM_initialize_fixed: f ", g%HI)

137  endif
138 
139 ! Compute global integrals of grid values for later use in scalar diagnostics !
140  call compute_global_grid_integrals(g)
141 
142 ! Write out all of the grid data used by this run.
143  if (write_geom) call write_ocean_geometry_file(g, pf, output_dir)
144 
145  call calltree_leave('MOM_initialize_fixed()')
146

186 !> Calculates the components of grad f (Coriolis parameter)
187 subroutine mom_calculate_grad_coriolis(dF_dx, dF_dy, G)
188  type(ocean_grid_type), intent(inout) :: G !< Grid type
189  real, dimension(SZI_(G),SZJ_(G)), intent(out) :: dF_dx !< x-component of grad f
190  real, dimension(SZI_(G),SZJ_(G)), intent(out) :: dF_dy !< y-component of grad f
191  ! Local variables
192  integer :: i,j
193  real :: f1, f2
194  do j=g%jsc, g%jec ; do i=g%isc, g%iec
195  f1 = 0.5*( g%CoriolisBu(i,j) + g%CoriolisBu(i,j-1) )
196  f2 = 0.5*( g%CoriolisBu(i-1,j) + g%CoriolisBu(i-1,j-1) )
197  df_dx(i,j) = g%IdxT(i,j) * ( f1 - f2 )
198  f1 = 0.5*( g%CoriolisBu(i,j) + g%CoriolisBu(i-1,j) )
199  f2 = 0.5*( g%CoriolisBu(i,j-1) + g%CoriolisBu(i-1,j-1) )
200  df_dy(i,j) = g%IdyT(i,j) * ( f1 - f2 )
201  enddo ; enddo
202  call pass_vector(df_dx, df_dy, g%Domain, stagger=agrid)
203 end subroutine mom_calculate_grad_coriolis

205 subroutine mom_initialize_topography(D, max_depth, G, PF)

215 ! This subroutine makes the appropriate call to set up the bottom depth.
216 ! This is a separate subroutine so that it can be made public and shared with

218 ! Set up the bottom depth, G%bathyT either analytically or from file

221 
222  call get_param(pf, mod, "TOPO_CONFIG", config, &
223  "This specifies how bathymetry is specified: \n"//&

225  " \t\t specified by (TOPO_FILE).\n"//&
226  " \t flat - flat bottom set to MAXIMUM_DEPTH. \n"//&

232  " \t\t profile in the meridional direction. \n"//&
233  " \t benchmark - use the benchmark test case topography. \n"//&
234  " \t DOME - use a slope and channel configuration for the \n"//&
235  " \t\t DOME sill-overflow test case. \n"//&
236  " \t ISOMIP - use a slope and channel configuration for the \n"//&
237  " \t\t ISOMIP test case. \n"//&
238  " \t DOME2D - use a shelf and slope configuration for the \n"//&
239  " \t\t DOME2D gravity current/overflow test case. \n"//&
240  " \t seamount - Gaussian bump for spontaneous motion test case.\n"//&
241  " \t Phillips - ACC-like idealized topography used in the Phillips config.\n"//&
242  " \t USER - call a user modified routine.", &
243  fail_if_missing=.true.)
244  max_depth = -1.e9; call read_param(pf, "MAXIMUM_DEPTH", max_depth)
245  select case ( trim(config) )
246  case ("file"); call initialize_topography_from_file(d, g, pf)

249  case ("bowl"); call initialize_topography_named(d, g, pf, config, max_depth)
250  case ("halfpipe"); call initialize_topography_named(d, g, pf, config, max_depth)
251  case ("DOME"); call dome_initialize_topography(d, g, pf, max_depth)
252  case ("ISOMIP"); call isomip_initialize_topography(d, g, pf, max_depth)
253  case ("benchmark"); call benchmark_initialize_topography(d, g, pf, max_depth)
254  case ("DOME2D"); call dome2d_initialize_topography(d, g, pf, max_depth)
255  case ("sloshing"); call sloshing_initialize_topography(d, g, pf, max_depth)
256  case ("seamount"); call seamount_initialize_topography(d, g, pf, max_depth)
257  case ("Phillips"); call phillips_initialize_topography(d, g, pf)
258  case ("USER"); call user_initialize_topography(d, g, pf)
259  case default ; call mom_error(fatal,"MOM_initialize_topography: "// &
260  "Unrecognized topography setup '"//trim(config)//"'")
261  end select
262  if (max_depth>0.) then
263  call log_param(pf, mod, "MAXIMUM_DEPTH", max_depth, &
264  "The maximum depth of the ocean.", units="m")
265  else
266  max_depth = diagnosemaximumdepth(d,g)
267  call log_param(pf, mod, "!MAXIMUM_DEPTH", max_depth, &
268  "The (diagnosed) maximum depth of the ocean.", units="m")
269  endif
270  if (trim(config) .ne. "DOME") then
271  call limit_topography(d, g, pf, max_depth)
272  endif
273 
274 end subroutine mom_initialize_topography
