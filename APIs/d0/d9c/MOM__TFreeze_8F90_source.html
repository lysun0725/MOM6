47 ! This subroutine computes the freezing point potential temparature
48 ! (in deg C) from salinity (in psu), and pressure (in Pa) using a simple
49 ! linear expression, with coefficients passed in as arguments.
50 !
51 ! Arguments: S - salinity in PSU.
52 ! (in) pres - pressure in Pa.
53 ! (out) T_Fr - Freezing point potential temperature in deg C.
54 ! (in) TFr_S0_P0 - The freezing point at S=0, p=0, in deg C.
55 ! (in) dTFr_dS - The derivatives of freezing point with salinity, in
56 ! deg C PSU-1.
57 ! (in) dTFr_dp - The derivatives of freezing point with pressure, in
58 ! deg C Pa-1.
59 
60  t_fr = (tfr_s0_p0 + dtfr_ds*s) + dtfr_dp*pres
61 
62 end subroutine calculate_tfreeze_linear_scalar

92 subroutine calculate_tfreeze_millero_scalar(S, pres, T_Fr)
93  real, intent(in) :: S, pres
94  real, intent(out) :: T_Fr
95 ! This subroutine computes the freezing point potential temparature
96 ! (in deg C) from salinity (in psu), and pressure (in Pa) using the expression
97 ! from Millero (1978) (and in appendix A of Gill 1982), but with the of the
98 ! pressure dependence changed from 7.53e-8 to 7.75e-8 to make this an
99 ! expression for potential temperature (not in situ temperature), using a
100 ! value that is correct at the freezing point at 35 PSU and 5e6 Pa (500 dbar).
101 !
102 ! Arguments: S - salinity in PSU.
103 ! (in) pres - pressure in Pa.
104 ! (out) T_Fr - Freezing point potential temperature in deg C.
105  real, parameter :: cS1 = -0.0575, cs3_2 = 1.710523e-3, cs2 = -2.154996e-4
106  real, parameter :: dTFr_dp = -7.75e-8
107 
108  t_fr = s*(cs1 + (cs3_2 * sqrt(max(s,0.0)) + cs2 * s)) + dtfr_dp*pres
109 
110 end subroutine calculate_tfreeze_millero_scalar

122 ! value that is correct at the freezing point at 35 PSU and 5e6 Pa (500 dbar).
123 !
124 ! Arguments: S - salinity in PSU.
125 ! (in) pres - pressure in Pa.
126 ! (out) T_Fr - Freezing point potential temperature in deg C.
127 ! (in) start - the starting point in the arrays.
128 ! (in) npts - the number of values to calculate.
129  real, parameter :: cS1 = -0.0575, cs3_2 = 1.710523e-3, cs2 = -2.154996e-4
