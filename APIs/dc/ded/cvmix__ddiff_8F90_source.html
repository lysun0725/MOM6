131 ! !DESCRIPTION:
132 ! Initialization routine for double diffusion mixing. This mixing technique
133 ! looks for two unstable cases in a column - salty water over fresher
134 ! water and colder water over warmer water - and computes different
135 ! diffusivity coefficients in each of these two locations. The parameter
136 ! \begin{eqnarray*}
137 ! R_\rho = \frac{\alpha (\partial \Theta / \partial z)}
138 ! {\beta (\partial S / \partial z)}
139 ! \end{eqnarray*}
140 ! to determine as a stratification parameter. If $(\partial S / \partial z)$
141 ! is positive and $1 < R_\rho < R_\rho^0$ then salt water sits on top
142 ! of fresh water and the diffusivity is given by
143 ! \begin{eqnarray*}
144 ! \kappa = \kappa^0 \left[ 1 - \left(\frac{R_\rho - 1}{R_\rho^0 - 1} \right)^{p_1}\right]^{p_2}
145 ! \end{eqnarray*}
146 ! By default, $R_\rho^0 = 2.55$, but that can be changed by setting
147 ! \verb|strat_param_max| in the code. Similarly, by default $p_1 = 1$
148 ! (\verb|ddiff_exp1|), $p_2 = 3$ (\verb|ddiff_exp2|), and
149 ! \begin{eqnarray*}
150 ! \kappa^0 = \left\{ \begin{array}{r l}
151 ! 7 \cdot 10^{-5}\ \textrm{m}^2\textrm{/s} & \textrm{for temperature}
152 ! \ (0.7 \cdot \verb|kappa_ddiff_s|\ \textrm{in this routine})\\
153 ! 10^{-4}\ \textrm{m}^2\textrm{/s} & \textrm{for salinity and other tracers}
154 ! \ (\verb|kappa_ddiff_s|\ \textrm{in this routine}).
155 ! \end{array} \right.
156 ! \end{eqnarray*}
157 ! On the other hand, if $(\partial \Theta / \partial z)$ is negative and
158 ! $0 < R_\rho < 1$ then cold water sits on warm warm water and the
159 ! diffusivity for temperature is given by
160 ! \begin{eqnarray*}
161 ! \kappa = \nu_\textrm{molecular} \cdot 0.909\exp\left\{ 4.6\exp\left[
162 ! -0.54\left( \frac{1}{R_\rho} - 1 \right) \right] \right\}
163 ! \end{eqnarray*}
164 ! where $\nu_\textrm{molecular}$ Is the molecular viscosity of water. By default it
165 ! is set to $1.5 \cdot 10^{-6}\ \textrm{m}^2\textrm{/s}$, but it can be changed
166 ! through \verb|mol_diff| in the code. Similarly, 0.909, 4.6, and -0.54 are the
167 ! default values of \verb|kappa_ddiff_param1|, \verb|kappa_ddiff_param2|, and
168 ! \verb|kappa_ddiff_param3|, respectively.\\
169 !\\
170 ! For salinity and other tracers, $\kappa$ above is multiplied by the factor
171 ! \begin{eqnarray*}
172 ! \textrm{factor} = \left\{ \begin{array}{c l}
173 ! 0.15R_\rho & R_\rho < 0.5\\
174 ! 1.85R_\rho - 0.85 & 0.5 \le R_\rho < 1\\
175 ! \end{array} \right.
176 ! \end{eqnarray*}
177 ! $\kappa$ is stored in \verb|CVmix_vars%diff_iface(:,1)|, while the modified value
178 ! for non-temperature tracers is stored in \verb|CVmix_vars%diff_iface(:,2)|.
179 ! Note that CVMix assumes units are |'mks'|.\\
180 !\\
181 ! !USES:
182 ! Only those used by entire module.
183 
184 ! !INPUT PARAMETERS:

186  kappa_ddiff_s, &
187  ddiff_exp1, &
188  ddiff_exp2, &

264  else

266  cvmix_ddiff_params_user)
267  end if
268

270  select case (trim(old_vals))
271  case ("overwrite")
272  call cvmix_put_ddiff('handle_old_vals', cvmix_overwrite_old_val, &
273  cvmix_ddiff_params_user)
274  case ("sum")
275  call cvmix_put_ddiff('handle_old_vals', cvmix_sum_old_and_new_vals, &
276  cvmix_ddiff_params_user)

278  call cvmix_put_ddiff('handle_old_vals', cvmix_max_old_and_new_vals, &
279  cvmix_ddiff_params_user)
280  case DEFAULT
281  print*, "ERROR: ", trim(old_vals), " is not a valid option for ", &
282  "handling old values of diff and visc."
283  stop 1
284  end select
285  else
286  call cvmix_put_ddiff('handle_old_vals', cvmix_overwrite_old_val, &
287  cvmix_ddiff_params_user)
288  end if

355  subroutine cvmix_coeffs_ddiff_low(Tdiff_out, Sdiff_out, strat_param_num, &

357  cvmix_ddiff_params_user)
358 
359 ! !DESCRIPTION:
360 ! Computes vertical diffusion coefficients for the double diffusion mixing
361 ! parameterization.
362 !\\
363 !\\
364 !
365 ! !USES:
366 ! only those used by entire module.
367 
368 ! !INPUT PARAMETERS:
369  type(cvmix_ddiff_params_type), optional, target, intent(in) :: &
370  CVmix_ddiff_params_user
371  integer, intent(in) :: nlev, max_nlev
372  real(cvmix_r8), dimension(max_nlev), intent(in) :: strat_param_num, &
373  strat_param_denom
374 
375 ! !INPUT/OUTPUT PARAMETERS:
376  real(cvmix_r8), dimension(max_nlev+1), intent(inout) :: Tdiff_out, &
377  Sdiff_out
378 
379 ! !LOCAL VARIABLES:
380  integer :: k
381  real(cvmix_r8) :: ddiff, Rrho
382 
383 !EOP
384 !BOC
385 
386  type(cvmix_ddiff_params_type), pointer :: CVmix_ddiff_params_in
387 
388  if (present(cvmix_ddiff_params_user)) then
389  cvmix_ddiff_params_in => cvmix_ddiff_params_user
390  else
391  cvmix_ddiff_params_in => cvmix_ddiff_params_saved
392  end if
393 
394  ! Determine coefficients
395  tdiff_out=cvmix_zero
396  sdiff_out=cvmix_zero
397  do k = 1, nlev
398  if ((strat_param_num(k).ge.strat_param_denom(k)).and. &
399  (strat_param_denom(k).gt.cvmix_zero)) then
400  ! Rrho > 1 and dS/dz < 0 => Salt fingering
401  rrho = strat_param_num(k) / strat_param_denom(k)
402  if (rrho.lt.cvmix_ddiff_params_in%strat_param_max) then
403  ddiff = (cvmix_one - ((rrho-cvmix_one)/ &
404  (cvmix_ddiff_params_in%strat_param_max-cvmix_one))** &
405  cvmix_ddiff_params_in%ddiff_exp1)**cvmix_ddiff_params_in%ddiff_exp2
406  sdiff_out(k) = cvmix_ddiff_params_in%kappa_ddiff_s*ddiff
407  end if
408  tdiff_out = sdiff_out*0.7_cvmix_r8
409  end if
410  if ((strat_param_num(k).ge.strat_param_denom(k)).and. &
411  (strat_param_num(k).lt.cvmix_zero)) then
412  ! Rrho < 1 and dT/dz > 0 => Diffusive convection
413  rrho = strat_param_num(k) / strat_param_denom(k)
414  select case (trim(cvmix_ddiff_params_in%diff_conv_type))
415  case ("MC76")
416  ddiff = cvmix_ddiff_params_in%mol_diff * &
417  cvmix_ddiff_params_in%kappa_ddiff_param1 * &
418  exp(cvmix_ddiff_params_in%kappa_ddiff_param2*exp( &
419  cvmix_ddiff_params_in%kappa_ddiff_param3* &
420  (cvmix_one/rrho-cvmix_one)))
421  case ("K88")
422  ddiff = cvmix_ddiff_params_in%mol_diff * 8.7_cvmix_r8 * &
423  (rrho**1.1_cvmix_r8)
424  case DEFAULT
425  print*, "ERROR: ", trim(cvmix_ddiff_params_in%diff_conv_type), &
426  " is not a valid value for diff_conv_type"
427  stop 1
428  end select
429  tdiff_out(k) = ddiff
430  if (rrho.lt.0.5_cvmix_r8) then
431  sdiff_out(k) = 0.15_cvmix_r8*rrho*ddiff
432  else
433  sdiff_out(k) = (1.85_cvmix_r8*rrho-0.85_cvmix_r8)*ddiff
434  end if
435  end if
436  end do
437  tdiff_out(nlev+1) = cvmix_zero
438  sdiff_out(nlev+1) = cvmix_zero
439 
440 !EOC
441 
442  end subroutine cvmix_coeffs_ddiff_low
