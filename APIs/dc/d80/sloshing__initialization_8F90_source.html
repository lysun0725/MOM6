77 !> Initialization of thicknesses
78 !! This routine is called when THICKNESS_CONFIG is set to 'sloshing'
79 !!
80 !! This routine initializes layer positions to set off a sloshing motion in
81 !! the zonal direction in a rectangular basin. All layers have initially the
82 !! same thickness but all interfaces (except bottom and sea surface) are
83 !! displaced according to a half-period cosine, with maximum value on the
84 !! left and minimum value on the right. This sets off a regular sloshing motion.
85 subroutine sloshing_initialize_thickness ( h, G, GV, param_file )
86  type(ocean_grid_type), intent(in) :: G !< The ocean's grid structure.
87  real, intent(out), dimension(SZI_(G),SZJ_(G), SZK_(G)) :: h !< The thicknesses being
88  !! initialized.
89  type(verticalgrid_type), intent(in) :: GV !< The ocean's vertical grid structure.
90  type(param_file_type), intent(in) :: param_file !< A structure indicating the
91  !! open file to parse for model
92  !! parameter values.
93 
94  real :: displ(szk_(g)+1)
95  real :: z_unif(szk_(g)+1)
96  real :: z_inter(szk_(g)+1)
97  real :: x
98  real :: a0
99  real :: deltah
100  real :: total_height
101  real :: weight_z
102  real :: x1, y1, x2, y2
103  real :: t
104  integer :: n
105 
106  integer :: i, j, k, is, ie, js, je, nx, nz
107 
108  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
109 
110  deltah = g%max_depth / nz
111 
112  ! Define thicknesses
113  do j=g%jsc,g%jec ; do i=g%isc,g%iec
114 
115  ! Define uniform interfaces
116  do k = 0,nz
117  z_unif(k+1) = -real(k)/real(nz)
118  end do
119 
120  ! 1. Define stratification
121  n = 3
122  do k = 1,nz+1
123 
124  ! Thin pycnocline in the middle
125  !z_inter(k) = (2.0**(n-1)) * (z_unif(k) + 0.5)**n - 0.5
126 
127  ! Thin pycnocline in the middle (piecewise linear profile)
128  x1 = 0.30; y1 = 0.48; x2 = 0.70; y2 = 0.52
129 
130  x = -z_unif(k)
131 
132  if ( x .le. x1 ) then
133  t = y1*x/x1;
134  else if ( (x .gt. x1 ) .and. ( x .lt. x2 )) then
135  t = y1 + (y2-y1) * (x-x1) / (x2-x1)
136  else
137  t = y2 + (1.0-y2) * (x-x2) / (1.0-x2)
138  end if
139 
140  t = - z_unif(k)
141 
142  z_inter(k) = -t * g%max_depth
143 
144  end do
145 
146  ! 2. Define displacement
147  a0 = 75.0; ! Displacement amplitude (meters)
148  do k = 1,nz+1
149 
150  weight_z = - 4.0 * ( z_unif(k) + 0.5 )**2 + 1
151 
152  x = g%geoLonT(i,j) / g%len_lon
153  displ(k) = a0 * cos(acos(-1.0)*x) + weight_z;
154 
155  if ( k .EQ. 1 ) then
156  displ(k) = 0.0
157  end if
158 
159  if ( k .EQ. nz+1 ) then
160  displ(k) = 0.0
161  end if
162 
163  z_inter(k) = z_inter(k) + displ(k)
164 
165  end do
166 
167  ! 3. The last interface must coincide with the seabed
168  z_inter(nz+1) = -g%bathyT(i,j)
169 
170  ! Modify interface heights to make sure all thicknesses
171  ! are strictly positive
172  do k = nz,1,-1
173 
174  if ( z_inter(k) .LT. (z_inter(k+1) + gv%Angstrom) ) then
175  z_inter(k) = z_inter(k+1) + gv%Angstrom
176  end if
177 
178  end do
179 
180  ! 4. Define layers
181  total_height = 0.0
182  do k = 1,nz
183  h(i,j,k) = z_inter(k) - z_inter(k+1)
184 
185  total_height = total_height + h(i,j,k)
186  end do
187 
188  enddo ; enddo
189 
190 end subroutine sloshing_initialize_thickness
