27 !* *
28 !* The subroutines here calculate the heights of interfaces in a *
29 !* way that is consistent with the calculations in the pressure *
30 !* gradient acceleration calculation. In a Boussinseq model this is *
31 !* pretty simple vertical sum, but in a non-Boussinesq model it uses *
32 !* integrals of the equation of state. *
33 !* *
34 !* Macros written all in capital letters are defined in MOM_memory.h. *
35 !* *
36 !* A small fragment of the grid is shown below: *
37 !* *
38 !* j+1 x ^ x ^ x At x: q, CoriolisBu *
39 !* j+1 > o > o > At ^: v *
40 !* j x ^ x ^ x At >: u *
41 !* j > o > o > At o: h, bathyT *
42 !* j-1 x ^ x ^ x *
43 !* i-1 i i+1 At x & ^: *
44 !* i i+1 At > & o: *
45 !* *
46 !* The boundaries always run through q grid points (x). *
47 !* *
48 !********+*********+*********+*********+*********+*********+*********+**

69 subroutine find_eta_3d(h, tv, G_Earth, G, GV, eta, eta_bt, halo_size)
70  type(ocean_grid_type), intent(in) :: G
71  type(verticalgrid_type), intent(in) :: GV
72  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
73  type(thermo_var_ptrs), intent(in) :: tv
74  real, intent(in) :: G_Earth
75  real, dimension(SZI_(G),SZJ_(G),SZK_(G)+1), intent(out) :: eta
76  real, dimension(SZI_(G),SZJ_(G)), optional, intent(in) :: eta_bt
77  integer, optional, intent(in) :: halo_size
78 
79 ! This subroutine determines the heights of all interfaces between layers,
80 ! using the appropriate form for consistency with the calculation of the
81 ! pressure gradient forces. Additionally, these height may be dilated for
82 ! consistency with the corresponding time-average quantity from the barotropic
83 ! calculation.
84 
85 ! Arguments:
86 ! (in) h - layer thickness H (meter or kg/m2)
87 ! (in) tv - structure pointing to thermodynamic variables
88 ! (in) G_Earth - Earth gravitational acceleration (m/s2)
89 ! (in) G - ocean grid structure
90 ! (in) GV - The ocean's vertical grid structure.
91 ! (out) eta - layer interface heights (meter)
92 ! (in,opt) eta_bt - optional barotropic variable that gives the "correct"
93 ! free surface height (Boussinesq) or total water column
94 ! mass per unit aread (non-Boussinesq). This is used to
95 ! dilate the layer thicknesses when calculating interface
96 ! heights, in H (m or kg m-2).
97 ! (in,opt) halo_size - width of halo points on which to calculate eta
98 
99  real :: p(szi_(g),szj_(g),szk_(g)+1)
100  real :: dz_geo(szi_(g),szj_(g),szk_(g)) ! The change in geopotential height
101  ! across a layer, in m2 s-2.
102  real :: dilate(szi_(g)) ! non-dimensional dilation factor
103  real :: htot(szi_(g)) ! total thickness H
104  real :: I_gEarth
105  integer i, j, k, isv, iev, jsv, jev, nz, halo
106 
107  halo = 0 ; if (present(halo_size)) halo = max(0,halo_size)
108 
109  isv = g%isc-halo ; iev = g%iec+halo ; jsv = g%jsc-halo ; jev = g%jec+halo
110  nz = g%ke
111 
112  if ((isv<g%isd) .or. (iev>g%ied) .or. (jsv<g%jsd) .or. (jev>g%jed)) &
113  call mom_error(fatal,"find_eta called with an overly large halo_size.")
114 
115  i_gearth = 1.0 / g_earth
116 
117 !$OMP parallel default(none) shared(isv,iev,jsv,jev,nz,eta,G,GV,h,eta_bt,tv,p, &
118 !$OMP G_Earth,dz_geo,halo,I_gEarth) &
119 !$OMP private(dilate,htot)
120 !$OMP do
121  do j=jsv,jev ; do i=isv,iev ; eta(i,j,nz+1) = -g%bathyT(i,j) ; enddo ; enddo
122 
123  if (gv%Boussinesq) then
124 !$OMP do
125  do j=jsv,jev ; do k=nz,1,-1; do i=isv,iev
126  eta(i,j,k) = eta(i,j,k+1) + h(i,j,k)*gv%H_to_m
127  enddo ; enddo ; enddo
128  if (present(eta_bt)) then
129  ! Dilate the water column to agree with the free surface height
130  ! that is used for the dynamics.
131 !$OMP do
132  do j=jsv,jev
133  do i=isv,iev
134  dilate(i) = (eta_bt(i,j)*gv%H_to_m + g%bathyT(i,j)) / &
135  (eta(i,j,1) + g%bathyT(i,j))
136  enddo
137  do k=1,nz ; do i=isv,iev
138  eta(i,j,k) = dilate(i) * (eta(i,j,k) + g%bathyT(i,j)) - g%bathyT(i,j)
139  enddo ; enddo
140  enddo
141  endif
142  else
143  if (associated(tv%eqn_of_state)) then
144  ! ### THIS SHOULD BE P_SURF, IF AVAILABLE.
145 !$OMP do
146  do j=jsv,jev
147  do i=isv,iev ; p(i,j,1) = 0.0 ; enddo
148  do k=1,nz ; do i=isv,iev
149  p(i,j,k+1) = p(i,j,k) + g_earth*gv%H_to_kg_m2*h(i,j,k)
150  enddo ; enddo
151  enddo
152 !$OMP do
153  do k=1,nz
154  call int_specific_vol_dp(tv%T(:,:,k), tv%S(:,:,k), p(:,:,k), p(:,:,k+1), &
155  0.0, g%HI, tv%eqn_of_state, dz_geo(:,:,k), halo_size=halo)
156  enddo
157 !$OMP do
158  do j=jsv,jev
159  do k=nz,1,-1 ; do i=isv,iev
160  eta(i,j,k) = eta(i,j,k+1) + i_gearth * dz_geo(i,j,k)
161  enddo ; enddo
162  enddo
163  else
164 !$OMP do
165  do j=jsv,jev ; do k=nz,1,-1; do i=isv,iev
166  eta(i,j,k) = eta(i,j,k+1) + gv%H_to_kg_m2*h(i,j,k)/gv%Rlay(k)
167  enddo ; enddo ; enddo
168  endif
169  if (present(eta_bt)) then
170  ! Dilate the water column to agree with the free surface height
171  ! from the time-averaged barotropic solution.
172 !$OMP do
173  do j=jsv,jev
174  do i=isv,iev ; htot(i) = gv%H_subroundoff ; enddo
175  do k=1,nz ; do i=isv,iev ; htot(i) = htot(i) + h(i,j,k) ; enddo ; enddo
176  do i=isv,iev ; dilate(i) = eta_bt(i,j) / htot(i) ; enddo
177  do k=1,nz ; do i=isv,iev
178  eta(i,j,k) = dilate(i) * (eta(i,j,k) + g%bathyT(i,j)) - g%bathyT(i,j)
179  enddo ; enddo
180  enddo
181  endif
182  endif
183 !$OMP end parallel
184 
185 end subroutine find_eta_3d

188 subroutine find_eta_2d(h, tv, G_Earth, G, GV, eta, eta_bt, halo_size)
189  type(ocean_grid_type), intent(in) :: G
190  type(verticalgrid_type), intent(in) :: GV
191  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
192  type(thermo_var_ptrs), intent(in) :: tv
193  real, intent(in) :: G_Earth
194  real, dimension(SZI_(G),SZJ_(G)), intent(out) :: eta
195  real, dimension(SZI_(G),SZJ_(G)), optional, intent(in) :: eta_bt
196  integer, optional, intent(in) :: halo_size
197 
198 ! This subroutine determines the free surface height, using the appropriate
199 ! form for consistency with the calculation of the pressure gradient forces.
200 ! Additionally, the sea surface height may be adjusted for consistency with the
201 ! corresponding time-average quantity from the barotropic calculation.
202 
203 ! Arguments:
204 ! (in) h - layer thickness (meter or kg/m2)
205 ! (in) tv - structure pointing to various thermodynamic variables
206 ! (in) G_Earth - Earth gravitational acceleration (m/s2)
207 ! (in) G - ocean grid structure
208 ! (in) GV - The ocean's vertical grid structure.
209 ! (out) eta - free surface height relative to mean sea level (z=0) (m)
210 ! (in,opt) eta_bt - optional barotropic variable that gives the "correct"
211 ! free surface height (Boussinesq) or total water column
212 ! mass per unit aread (non-Boussinesq), in H (m or kg m-2)
213 ! (in,opt) halo_size - width of halo points on which to calculate eta
214 
215  real, dimension(SZI_(G),SZJ_(G),SZK_(G)+1) :: &
216  p ! The pressure in Pa.
217  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: &
218  dz_geo ! The change in geopotential height across a layer, in m2 s-2.
219  real :: htot(szi_(g)) ! The sum of all layers' thicknesses, in kg m-2 or m.
220  real :: I_gEarth
221  integer i, j, k, is, ie, js, je, nz, halo
222 
223  halo = 0 ; if (present(halo_size)) halo = max(0,halo_size)
224  is = g%isc-halo ; ie = g%iec+halo ; js = g%jsc-halo ; je = g%jec+halo
225  nz = g%ke
226 
227  i_gearth = 1.0 / g_earth
228 
229 !$OMP parallel default(none) shared(is,ie,js,je,nz,eta,G,GV,eta_bt,h,tv,p, &
230 !$OMP G_Earth,dz_geo,halo,I_gEarth) &
231 !$OMP private(htot)
232 !$OMP do
233  do j=js,je ; do i=is,ie ; eta(i,j) = -g%bathyT(i,j) ; enddo ; enddo
234 
235  if (gv%Boussinesq) then
236  if (present(eta_bt)) then
237 !$OMP do
238  do j=js,je ; do i=is,ie
239  eta(i,j) = eta_bt(i,j)
240  enddo ; enddo
241  else
242 !$OMP do
243  do j=js,je ; do k=1,nz ; do i=is,ie
244  eta(i,j) = eta(i,j) + h(i,j,k)*gv%H_to_m
245  enddo ; enddo ; enddo
246  endif
247  else
248  if (associated(tv%eqn_of_state)) then
249 !$OMP do
250  do j=js,je
251  do i=is,ie ; p(i,j,1) = 0.0 ; enddo
252 
253  do k=1,nz ; do i=is,ie
254  p(i,j,k+1) = p(i,j,k) + g_earth*gv%H_to_kg_m2*h(i,j,k)
255  enddo ; enddo
256  enddo
257 !$OMP do
258  do k = 1, nz
259  call int_specific_vol_dp(tv%T(:,:,k), tv%S(:,:,k), p(:,:,k), p(:,:,k+1), 0.0, &
260  g%HI, tv%eqn_of_state, dz_geo(:,:,k), halo_size=halo)
261  enddo
262 !$OMP do
263  do j=js,je ; do k=1,nz ; do i=is,ie
264  eta(i,j) = eta(i,j) + i_gearth * dz_geo(i,j,k)
265  enddo ; enddo ; enddo
266  else
267 !$OMP do
268  do j=js,je ; do k=1,nz ; do i=is,ie
269  eta(i,j) = eta(i,j) + gv%H_to_kg_m2*h(i,j,k)/gv%Rlay(k)
270  enddo ; enddo ; enddo
271  endif
272  if (present(eta_bt)) then
273  ! Dilate the water column to agree with the the time-averaged column
274  ! mass from the barotropic solution.
275 !$OMP do
276  do j=js,je
277  do i=is,ie ; htot(i) = gv%H_subroundoff ; enddo
278  do k=1,nz ; do i=is,ie ; htot(i) = htot(i) + h(i,j,k) ; enddo ; enddo
279  do i=is,ie
280  eta(i,j) = (eta_bt(i,j) / htot(i)) * (eta(i,j) + g%bathyT(i,j)) - &
281  g%bathyT(i,j)
282  enddo
283  enddo
284  endif
285  endif
286 !$OMP end parallel
287 
288 end subroutine find_eta_2d
