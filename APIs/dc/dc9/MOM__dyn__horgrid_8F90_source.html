247 !> set_derived_dyn_horgrid calculates metric terms that are derived from other metrics.
248 subroutine set_derived_dyn_horgrid(G)
249  type(dyn_horgrid_type), intent(inout) :: G !< The dynamic horizontal grid type
250 ! Various inverse grid spacings and derived areas are calculated within this
251 ! subroutine.
252  integer :: i, j, isd, ied, jsd, jed
253  integer :: IsdB, IedB, JsdB, JedB
254 
255  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
256  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
257 
258  do j=jsd,jed ; do i=isd,ied
259  if (g%dxT(i,j) < 0.0) g%dxT(i,j) = 0.0
260  if (g%dyT(i,j) < 0.0) g%dyT(i,j) = 0.0
261  g%IdxT(i,j) = adcroft_reciprocal(g%dxT(i,j))
262  g%IdyT(i,j) = adcroft_reciprocal(g%dyT(i,j))
263  g%IareaT(i,j) = adcroft_reciprocal(g%areaT(i,j))
264  enddo ; enddo
265 
266  do j=jsd,jed ; do i=isdb,iedb
267  if (g%dxCu(i,j) < 0.0) g%dxCu(i,j) = 0.0
268  if (g%dyCu(i,j) < 0.0) g%dyCu(i,j) = 0.0
269  g%IdxCu(i,j) = adcroft_reciprocal(g%dxCu(i,j))
270  g%IdyCu(i,j) = adcroft_reciprocal(g%dyCu(i,j))
271  enddo ; enddo
272 
273  do j=jsdb,jedb ; do i=isd,ied
274  if (g%dxCv(i,j) < 0.0) g%dxCv(i,j) = 0.0
275  if (g%dyCv(i,j) < 0.0) g%dyCv(i,j) = 0.0
276  g%IdxCv(i,j) = adcroft_reciprocal(g%dxCv(i,j))
277  g%IdyCv(i,j) = adcroft_reciprocal(g%dyCv(i,j))
278  enddo ; enddo
279 
280  do j=jsdb,jedb ; do i=isdb,iedb
281  if (g%dxBu(i,j) < 0.0) g%dxBu(i,j) = 0.0
282  if (g%dyBu(i,j) < 0.0) g%dyBu(i,j) = 0.0
283 
284  g%IdxBu(i,j) = adcroft_reciprocal(g%dxBu(i,j))
285  g%IdyBu(i,j) = adcroft_reciprocal(g%dyBu(i,j))
286  ! areaBu has usually been set to a positive area elsewhere.
287  if (g%areaBu(i,j) <= 0.0) g%areaBu(i,j) = g%dxBu(i,j) * g%dyBu(i,j)
288  g%IareaBu(i,j) = adcroft_reciprocal(g%areaBu(i,j))
