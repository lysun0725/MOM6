23 !* *
24 !* By Robert Hallberg, April 1994 - June 2000 *
25 !* *
26 !* This file contains the subroutine that calculates the hori- *
27 !* zontal accelerations due to pressure gradients. A second-order *
28 !* accurate, centered scheme is used. If a split time stepping *
29 !* scheme is used, the vertical decomposition into barotropic and *
30 !* baroclinic contributions described by Hallberg (J Comp Phys 1997) *
31 !* is used. With a nonlinear equation of state, compressibility is *
32 !* added along the lines proposed by Sun et al. (JPO 1999), but with *
33 !* compressibility coefficients based on a fit to a user-provided *
34 !* reference profile. *
35 !* *
36 !* PressureForce takes 8 arguments, which are described below. If *
37 !* a non-split time stepping scheme is used, the last three arguments *
38 !* are ignored. *
39 !* *
40 !* Macros written all in capital letters are defined in MOM_memory.h. *
41 !* *
42 !* A small fragment of the grid is shown below: *
43 !* *
44 !* j+1 x ^ x ^ x At x: q, CoriolisBu *
45 !* j+1 > o > o > At ^: v, PFv *
46 !* j x ^ x ^ x At >: u, PFu *
47 !* j > o > o > At o: h, bathyT, M, e, p, pbce, T, S, rho_star *
48 !* j-1 x ^ x ^ x *
49 !* i-1 i i+1 *
50 !* i i+1 *
51 !* *
52 !* The boundaries always run through q grid points (x). *
53 !* *
54 !********+*********+*********+*********+*********+*********+*********+**

93 subroutine pressureforce_mont_nonbouss(h, tv, PFu, PFv, G, GV, CS, p_atm, pbce, eta)
94  type(ocean_grid_type), intent(in) :: G
95  type(verticalgrid_type), intent(in) :: GV
96  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
97  type(thermo_var_ptrs), intent(in) :: tv
98  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out) :: PFu
99  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(out) :: PFv
100  type(pressureforce_mont_cs), pointer :: CS
101  real, dimension(:,:), optional, pointer :: p_atm
102  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), optional, intent(out) :: pbce
103  real, dimension(SZI_(G),SZJ_(G)), optional, intent(out) :: eta
104 
105 ! This subroutine determines the acceleration due to pressure forces in a
106 ! non-Boussinesq fluid using the compressibility compensated (if appropriate)
107 ! Montgomery-potential form described in Hallberg (Ocean Mod., 2005).
108 ! To work, the following fields must be set outside of the usual
109 ! ie to ie, je to je range before this subroutine is called:
110 ! h[ie+1] and h[je+1] and and (if tv%form_of_EOS is set) T[ie+1], S[ie+1],
111 ! T[je+1], and S[je+1].
112 ! Arguments: h - Layer thickness, in m.
113 ! (in) tv - A structure containing pointers to any available
114 ! thermodynamic fields, including potential temperature and
115 ! salinity or mixed layer density. Absent fields have NULL ptrs.
116 ! (out) PFu - Zonal acceleration due to pressure gradients
117 ! (equal to -dM/dx) in m s-2.
118 ! (out) PFv - Meridional acceleration due to pressure
119 ! gradients (equal to -dM/dy) in m s-2.
120 ! (in) G - The ocean's grid structure.
121 ! (in) GV - The ocean's vertical grid structure.
122 ! (in) CS - The control structure returned by a previous call to
123 ! PressureForce_init.
124 ! (in) p_atm - The pressure at the ice-ocean or atmosphere-ocean
125 ! interface in Pa.
126 ! (out) pbce - The baroclinic pressure anomaly in each layer
127 ! due to free surface height anomalies, in m s-2.
128 ! pbce points to a space with nz layers or NULL.
129 ! (out) eta - The free surface height used to calculate PFu and PFv, in m,
130 ! with any tidal contributions or compressibility compensation.
131  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: &
132  M, & ! The Montgomery potential, M = (p/rho + gz) , in m2 s-2.
133  alpha_star, & ! Compression adjusted specific volume, in m3 kg-1.
134  dz_geo ! The change in geopotential across a layer, in m2 s-2.
135  real, dimension(SZI_(G),SZJ_(G),SZK_(G)+1) :: p ! Interface pressure in Pa.
136  ! p may be adjusted (with a nonlinear equation of state) so that
137  ! its derivative compensates for the adiabatic compressibility
138  ! in seawater, but p will still be close to the pressure.
139  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), target :: &
140  T_tmp, & ! Temporary array of temperatures where layers that are lighter
141  ! than the mixed layer have the mixed layer's properties, in C.
142  s_tmp ! Temporary array of salinities where layers that are lighter
143  ! than the mixed layer have the mixed layer's properties, in psu.
144 
145  real, dimension(SZI_(G)) :: Rho_cv_BL ! The coordinate potential density in the
146  ! deepest variable density near-surface layer, in kg m-3.
147 
148  real, dimension(SZI_(G),SZJ_(G)) :: &
149  dM, & ! A barotropic correction to the Montgomery potentials to
150  ! enable the use of a reduced gravity form of the equations,
151  ! in m2 s-2.
152  dp_star, & ! Layer thickness after compensation for compressibility, in Pa.
153  e_tidal, & ! Bottom geopotential anomaly due to tidal forces from
154  ! astronomical sources and self-attraction and loading, in m.
155  geopot_bot, & ! Bottom geopotential relative to time-mean sea level,
156  ! including any tidal contributions, in units of m2 s-2.
157  ssh ! Sea surface height anomalies, in m.
158  real :: p_ref(szi_(g)) ! The pressure used to calculate the coordinate
159  ! density, in Pa (usually 2e7 Pa = 2000 dbar).
160  real :: rho_in_situ(szi_(g)) !In-situ density of a layer, in kg m-3.
161  real :: PFu_bc, PFv_bc ! The pressure gradient force due to along-layer
162  ! compensated density gradients, in m s-2.
163  real :: dp_neglect ! A thickness that is so small it is usually lost
164  ! in roundoff and can be neglected, in Pa.
165  logical :: use_p_atm ! If true, use the atmospheric pressure.
166  logical :: use_EOS ! If true, density is calculated from T & S using
167  ! an equation of state.
168  logical :: is_split ! A flag indicating whether the pressure
169  ! gradient terms are to be split into
170  ! barotropic and baroclinic pieces.
171  type(thermo_var_ptrs) :: tv_tmp! A structure of temporary T & S.
172 
173  real :: I_gEarth
174  real :: dalpha
175  real :: Pa_to_H ! A factor to convert from Pa to the thicknesss units (H).
176  real :: alpha_Lay(szk_(g)) ! The specific volume of each layer, in kg m-3.
177  real :: dalpha_int(szk_(g)+1) ! The change in specific volume across each
178  ! interface, in kg m-3.
179  integer :: is, ie, js, je, Isq, Ieq, Jsq, Jeq, nz, nkmb
180  integer :: i, j, k
181  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
182  nkmb=gv%nk_rho_varies
183  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
184 
185  use_p_atm = .false.
186  if (present(p_atm)) then ; if (associated(p_atm)) use_p_atm = .true. ; endif
187  is_split = .false. ; if (present(pbce)) is_split = .true.
188  use_eos = associated(tv%eqn_of_state)
189 
190  if (.not.associated(cs)) call mom_error(fatal, &
191  "MOM_PressureForce_Mont: Module must be initialized before it is used.")
192  if (use_eos) then
193  if (query_compressible(tv%eqn_of_state)) call mom_error(fatal, &
194  "PressureForce_Mont_nonBouss: The Montgomery form of the pressure force "//&
195  "can no longer be used with a compressible EOS. Use #define ANALYTIC_FV_PGF.")
196  endif
197 
198  i_gearth = 1.0 / g%g_Earth
199  dp_neglect = gv%H_to_Pa * gv%H_subroundoff
200 !$OMP parallel default(none) shared(nz,alpha_Lay,GV,dalpha_int)
201 !$OMP do
202  do k=1,nz ; alpha_lay(k) = 1.0 / gv%Rlay(k) ; enddo
203 !$OMP do
204  do k=2,nz ; dalpha_int(k) = alpha_lay(k-1) - alpha_lay(k) ; enddo
205 !$OMP end parallel
206 
207 !$OMP parallel default(none) shared(Isq,Ieq,Jsq,Jeq,nz,p,p_atm,GV,h,use_p_atm)
208  if (use_p_atm) then
209 !$OMP do
210  do j=jsq,jeq+1 ; do i=isq,ieq+1 ; p(i,j,1) = p_atm(i,j) ; enddo ; enddo
211  else
212 !$OMP do
213  do j=jsq,jeq+1 ; do i=isq,ieq+1 ; p(i,j,1) = 0.0 ; enddo ; enddo
214  endif
215 !$OMP do
216  do j=jsq,jeq+1 ; do k=1,nz ; do i=isq,ieq+1
217  p(i,j,k+1) = p(i,j,k) + gv%H_to_Pa * h(i,j,k)
218  enddo ; enddo ; enddo
219 !$OMP end parallel
220 
221  if (present(eta)) then
222  pa_to_h = 1.0 / gv%H_to_Pa
223  if (use_p_atm) then
224 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,eta,p,p_atm,Pa_to_H)
225  do j=jsq,jeq+1 ; do i=isq,ieq+1
226  eta(i,j) = (p(i,j,nz+1) - p_atm(i,j))*pa_to_h ! eta has the same units as h.
227  enddo ; enddo
228  else
229 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,eta,p,Pa_to_H)
230  do j=jsq,jeq+1 ; do i=isq,ieq+1
231  eta(i,j) = p(i,j,nz+1)*pa_to_h ! eta has the same units as h.
232  enddo ; enddo
233  endif
234  endif
235 
236  if (cs%tides) then
237  ! Determine the sea surface height anomalies, to enable the calculation
238  ! of self-attraction and loading.
239 !$OMP parallel default(none) shared(Isq,Ieq,Jsq,Jeq,nz,SSH,G,GV,use_EOS,tv,p,dz_geo, &
240 !$OMP I_gEarth,h,alpha_Lay)
241 !$OMP do
242  do j=jsq,jeq+1 ; do i=isq,ieq+1
243  ssh(i,j) = -g%bathyT(i,j)
244  enddo ; enddo
245  if (use_eos) then
246 !$OMP do
247  do k=1,nz
248  call int_specific_vol_dp(tv%T(:,:,k), tv%S(:,:,k), p(:,:,k), p(:,:,k+1), &
249  0.0, g%HI, tv%eqn_of_state, dz_geo(:,:,k), halo_size=1)
250  enddo
251 !$OMP do
252  do j=jsq,jeq+1 ; do k=1,nz; do i=isq,ieq+1
253  ssh(i,j) = ssh(i,j) + i_gearth * dz_geo(i,j,k)
254  enddo ; enddo ; enddo
255  else
256 !$OMP do
257  do j=jsq,jeq+1 ; do k=1,nz ; do i=isq,ieq+1
258  ssh(i,j) = ssh(i,j) + gv%H_to_kg_m2*h(i,j,k)*alpha_lay(k)
259  enddo ; enddo ; enddo
260  endif
261 !$OMP end parallel
262 
263  call calc_tidal_forcing(cs%Time, ssh, e_tidal, g, cs%tides_CSp)
264 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,geopot_bot,G,e_tidal)
265  do j=jsq,jeq+1 ; do i=isq,ieq+1
266  geopot_bot(i,j) = -g%g_Earth*(e_tidal(i,j) + g%bathyT(i,j))
267  enddo ; enddo
268  else
269 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,geopot_bot,G)
270  do j=jsq,jeq+1 ; do i=isq,ieq+1
271  geopot_bot(i,j) = -g%g_Earth*g%bathyT(i,j)
272  enddo ; enddo
273  endif
274 
275  if (use_eos) then
276  ! Calculate in-situ specific volumes (alpha_star).
277 
278  ! With a bulk mixed layer, replace the T & S of any layers that are
279  ! lighter than the the buffer layer with the properties of the buffer
280  ! layer. These layers will be massless anyway, and it avoids any
281  ! formal calculations with hydrostatically unstable profiles.
282  if (nkmb>0) then
283  tv_tmp%T => t_tmp ; tv_tmp%S => s_tmp
284  tv_tmp%eqn_of_state => tv%eqn_of_state
285  do i=isq,ieq+1 ; p_ref(i) = tv%P_Ref ; enddo
286 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,nkmb,tv_tmp,tv,p_ref,GV) &
287 !$OMP private(Rho_cv_BL)
288  do j=jsq,jeq+1
289  do k=1,nkmb ; do i=isq,ieq+1
290  tv_tmp%T(i,j,k) = tv%T(i,j,k) ; tv_tmp%S(i,j,k) = tv%S(i,j,k)
291  enddo ; enddo
292  call calculate_density(tv%T(:,j,nkmb), tv%S(:,j,nkmb), p_ref, &
293  rho_cv_bl(:), isq, ieq-isq+2, tv%eqn_of_state)
294  do k=nkmb+1,nz ; do i=isq,ieq+1
295  if (gv%Rlay(k) < rho_cv_bl(i)) then
296  tv_tmp%T(i,j,k) = tv%T(i,j,nkmb) ; tv_tmp%S(i,j,k) = tv%S(i,j,nkmb)
297  else
298  tv_tmp%T(i,j,k) = tv%T(i,j,k) ; tv_tmp%S(i,j,k) = tv%S(i,j,k)
299  endif
300  enddo ; enddo
301  enddo
302  else
303  tv_tmp%T => tv%T ; tv_tmp%S => tv%S
304  tv_tmp%eqn_of_state => tv%eqn_of_state
305  do i=isq,ieq+1 ; p_ref(i) = 0 ; enddo
306  endif
307 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,tv_tmp,p_ref,tv,alpha_star) &
308 !$OMP private(rho_in_situ)
309  do k=1,nz ; do j=jsq,jeq+1
310  call calculate_density(tv_tmp%T(:,j,k),tv_tmp%S(:,j,k),p_ref, &
311  rho_in_situ,isq,ieq-isq+2,tv%eqn_of_state)
312  do i=isq,ieq+1 ; alpha_star(i,j,k) = 1.0 / rho_in_situ(i) ; enddo
313  enddo ; enddo
314  endif ! use_EOS
315 
316  if (use_eos) then
317 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,M,geopot_bot,p,alpha_star)
318  do j=jsq,jeq+1
319  do i=isq,ieq+1
320  m(i,j,nz) = geopot_bot(i,j) + p(i,j,nz+1) * alpha_star(i,j,nz)
321  enddo
322  do k=nz-1,1,-1 ; do i=isq,ieq+1
323  m(i,j,k) = m(i,j,k+1) + p(i,j,k+1) * (alpha_star(i,j,k) - alpha_star(i,j,k+1))
324  enddo ; enddo
325  enddo
326  else ! not use_EOS
327 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,M,geopot_bot,p,&
328 !$OMP alpha_Lay,dalpha_int)
329  do j=jsq,jeq+1
330  do i=isq,ieq+1
331  m(i,j,nz) = geopot_bot(i,j) + p(i,j,nz+1) * alpha_lay(nz)
332  enddo
333  do k=nz-1,1,-1 ; do i=isq,ieq+1
334  m(i,j,k) = m(i,j,k+1) + p(i,j,k+1) * dalpha_int(k+1)
335  enddo ; enddo
336  enddo
337  endif ! use_EOS
338 
339  if (cs%GFS_scale < 1.0) then
340  ! Adjust the Montgomery potential to make this a reduced gravity model.
341 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,dM,CS,M)
342  do j=jsq,jeq+1 ; do i=isq,ieq+1
343  dm(i,j) = (cs%GFS_scale - 1.0) * m(i,j,1)
344  enddo ; enddo
345 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,dM,M)
346  do k=1,nz ; do j=jsq,jeq+1 ; do i=isq,ieq+1
347  m(i,j,k) = m(i,j,k) + dm(i,j)
348  enddo ; enddo ; enddo
349 
350  ! Could instead do the following, to avoid taking small differences
351  ! of large numbers...
352 ! do j=Jsq,Jeq+1 ; do i=Isq,Ieq+1
353 ! M(i,j,1) = CS%GFS_scale * M(i,j,1)
354 ! enddo ; enddo
355 ! if (use_EOS) then
356 ! do k=2,nz ; do j=Jsq,Jeq+1 ; do i=Isq,Ieq+1
357 ! M(i,j,k) = M(i,j,k-1) - p(i,j,K) * (alpha_star(i,j,k-1) - alpha_star(i,j,k))
358 ! enddo ; enddo ; enddo
359 ! else ! not use_EOS
360 ! do k=2,nz ; do j=Jsq,Jeq+1 ; do i=Isq,Ieq+1
361 ! M(i,j,k) = M(i,j,k-1) - p(i,j,K) * dalpha_int(K)
362 ! enddo ; enddo ; enddo
363 ! endif ! use_EOS
364 
365  endif
366 
367  ! Note that ddM/dPb = alpha_star(i,j,1)
368  if (present(pbce)) then
369  call set_pbce_nonbouss(p, tv_tmp, g, gv, g%g_Earth, cs%GFS_scale, pbce, &
370  alpha_star)
371  endif
372 
373 ! Calculate the pressure force. On a Cartesian grid,
374 ! PFu = - dM/dx and PFv = - dM/dy.
375  if (use_eos) then
376 !$OMP parallel do default(none) shared(is,ie,js,je,Isq,Ieq,Jsq,Jeq,nz,p,dp_neglect, &
377 !$OMP alpha_star,G,PFu,PFv,M,CS) &
378 !$OMP private(dp_star,PFu_bc,PFv_bc)
379  do k=1,nz
380  do j=jsq,jeq+1 ; do i=isq,ieq+1
381  dp_star(i,j) = (p(i,j,k+1) - p(i,j,k)) + dp_neglect
382  enddo ; enddo
383  do j=js,je ; do i=isq,ieq
384  ! PFu_bc = p* grad alpha*
385  pfu_bc = (alpha_star(i+1,j,k) - alpha_star(i,j,k)) * (g%IdxCu(i,j) * &
386  ((dp_star(i,j) * dp_star(i+1,j) + (p(i,j,k) * dp_star(i+1,j) + &
387  p(i+1,j,k) * dp_star(i,j))) / (dp_star(i,j) + dp_star(i+1,j))))
388  pfu(i,j,k) = -(m(i+1,j,k) - m(i,j,k)) * g%IdxCu(i,j) + pfu_bc
389  if (ASSOCIATED(cs%PFu_bc)) cs%PFu_bc(i,j,k) = pfu_bc
390  enddo ; enddo
391  do j=jsq,jeq ; do i=is,ie
392  pfv_bc = (alpha_star(i,j+1,k) - alpha_star(i,j,k)) * (g%IdyCv(i,j) * &
393  ((dp_star(i,j) * dp_star(i,j+1) + (p(i,j,k) * dp_star(i,j+1) + &
394  p(i,j+1,k) * dp_star(i,j))) / (dp_star(i,j) + dp_star(i,j+1))))
395  pfv(i,j,k) = -(m(i,j+1,k) - m(i,j,k)) * g%IdyCv(i,j) + pfv_bc
396  if (ASSOCIATED(cs%PFv_bc)) cs%PFv_bc(i,j,k) = pfv_bc
397  enddo ; enddo
398  enddo ! k-loop
399  else ! .not. use_EOS
400 !$OMP parallel do default(none) shared(is,ie,js,je,Isq,Ieq,Jsq,Jeq,nz,PFu,PFv,M,G)
401  do k=1,nz
402  do j=js,je ; do i=isq,ieq
403  pfu(i,j,k) = -(m(i+1,j,k) - m(i,j,k)) * g%IdxCu(i,j)
404  enddo ; enddo
405  do j=jsq,jeq ; do i=is,ie
406  pfv(i,j,k) = -(m(i,j+1,k) - m(i,j,k)) * g%IdyCv(i,j)
407  enddo ; enddo
408  enddo
409  endif ! use_EOS

411  if (cs%id_PFu_bc>0) call post_data(cs%id_PFu_bc, cs%PFu_bc, cs%diag)
412  if (cs%id_PFv_bc>0) call post_data(cs%id_PFv_bc, cs%PFv_bc, cs%diag)
413  if (cs%id_e_tidal>0) call post_data(cs%id_e_tidal, e_tidal, cs%diag)
414 
415 end subroutine pressureforce_mont_nonbouss

417 subroutine pressureforce_mont_bouss(h, tv, PFu, PFv, G, GV, CS, p_atm, pbce, eta)
418  type(ocean_grid_type), intent(in) :: G
419  type(verticalgrid_type), intent(in) :: GV
420  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
421  type(thermo_var_ptrs), intent(in) :: tv
422  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out) :: PFu
423  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(out) :: PFv
424  type(pressureforce_mont_cs), pointer :: CS
425  real, dimension(:,:), optional, pointer :: p_atm
426  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), optional, intent(out) :: pbce
427  real, dimension(SZI_(G),SZJ_(G)), optional, intent(out) :: eta
428 
429 ! This subroutine determines the acceleration due to pressure
430 ! forces.
431 ! To work, the following fields must be set outside of the usual
432 ! ie to ie, je to je range before this subroutine is called:
433 ! h[ie+1] and h[je+1] and (if tv%form_of_EOS is set) T[ie+1], S[ie+1],
434 ! T[je+1], and S[je+1].
435 ! Arguments: h - Layer thickness, in m.
436 ! (in) tv - A structure containing pointers to any available
437 ! thermodynamic fields, including potential temperature and
438 ! salinity or mixed layer density. Absent fields have NULL ptrs.
439 ! (out) PFu - Zonal acceleration due to pressure gradients
440 ! (equal to -dM/dx) in m s-2.
441 ! (out) PFv - Meridional acceleration due to pressure
442 ! gradients (equal to -dM/dy) in m s-2.
443 ! (in) G - The ocean's grid structure.
444 ! (in) GV - The ocean's vertical grid structure.
445 ! (in) CS - The control structure returned by a previous call to
446 ! PressureForce_init.
447 ! (in) p_atm - the pressure at the ice-ocean or atmosphere-ocean
448 ! interface in Pa.
449 ! (out) pbce - the baroclinic pressure anomaly in each layer
450 ! due to free surface height anomalies, in m s-2.
451 ! (out) eta - the free surface height used to calculate PFu and PFv, in m,
452 ! with any tidal contributions or compressibility compensation.
453 
454  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: &
455  M, & ! The Montgomery potential, M = (p/rho + gz) , in m2 s-2.
456  rho_star ! In-situ density divided by the derivative with depth of the
457  ! corrected e times (G_Earth/Rho0). In units of m s-2.
458  real, dimension(SZI_(G),SZJ_(G),SZK_(G)+1) :: e ! Interface height in m.
459  ! e may be adjusted (with a nonlinearequation of state) so that
460  ! its derivative compensates for the adiabatic compressibility
461  ! in seawater, but e will still be close to the interface depth.
462  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), target :: &
463  T_tmp, & ! Temporary array of temperatures where layers that are lighter
464  ! than the mixed layer have the mixed layer's properties, in C.
465  s_tmp ! Temporary array of salinities where layers that are lighter
466  ! than the mixed layer have the mixed layer's properties, in psu.
467 
468  real :: Rho_cv_BL(szi_(g)) ! The coordinate potential density in
469  ! the deepest variable density near-surface layer, in kg m-3.
470  real :: h_star(szi_(g),szj_(g)) ! Layer thickness after compensation
471  ! for compressibility, in m.
472  real :: e_tidal(szi_(g),szj_(g)) ! Bottom geopotential anomaly due to tidal
473  ! forces from astronomical sources and self-
474  ! attraction and loading, in m.
475  real :: p_ref(szi_(g)) ! The pressure used to calculate the coordinate
476  ! density, in Pa (usually 2e7 Pa = 2000 dbar).
477  real :: I_Rho0 ! 1/Rho0.
478  real :: G_Rho0 ! G_Earth / Rho0 in m4 s-2 kg-1.
479  real :: PFu_bc, PFv_bc ! The pressure gradient force due to along-layer
480  ! compensated density gradients, in m s-2.
481  real :: dr ! Temporary variables.
482  real :: h_neglect ! A thickness that is so small it is usually lost
483  ! in roundoff and can be neglected, in m.
484  logical :: use_p_atm ! If true, use the atmospheric pressure.
485  logical :: use_EOS ! If true, density is calculated from T & S using
486  ! an equation of state.
487  logical :: is_split ! A flag indicating whether the pressure
488  ! gradient terms are to be split into
489  ! barotropic and baroclinic pieces.
490  type(thermo_var_ptrs) :: tv_tmp! A structure of temporary T & S.
491  integer :: is, ie, js, je, Isq, Ieq, Jsq, Jeq, nz, nkmb
492  integer :: i, j, k
493 
494  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
495  nkmb=gv%nk_rho_varies
496  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
497 
498  use_p_atm = .false.
499  if (present(p_atm)) then ; if (associated(p_atm)) use_p_atm = .true. ; endif
500  is_split = .false. ; if (present(pbce)) is_split = .true.
501  use_eos = associated(tv%eqn_of_state)
502 
503  if (.not.associated(cs)) call mom_error(fatal, &
504  "MOM_PressureForce_Mont: Module must be initialized before it is used.")
505  if (use_eos) then
506  if (query_compressible(tv%eqn_of_state)) call mom_error(fatal, &
507  "PressureForce_Mont_Bouss: The Montgomery form of the pressure force "//&
508  "can no longer be used with a compressible EOS. Use #define ANALYTIC_FV_PGF.")
509  endif
510 
511  h_neglect = gv%H_subroundoff * gv%H_to_m
512  i_rho0 = 1.0/cs%Rho0
513  g_rho0 = g%g_Earth/gv%Rho0
514 
515  if (cs%tides) then
516  ! Determine the surface height anomaly for calculating self attraction
517  ! and loading. This should really be based on bottom pressure anomalies,
518  ! but that is not yet implemented, and the current form is correct for
519  ! barotropic tides.
520 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,e,h,G,GV)
521  do j=jsq,jeq+1
522  do i=isq,ieq+1 ; e(i,j,1) = -1.0*g%bathyT(i,j) ; enddo
523  do k=1,nz ; do i=isq,ieq+1
524  e(i,j,1) = e(i,j,1) + h(i,j,k)*gv%H_to_m
525  enddo ; enddo
526  enddo
527  call calc_tidal_forcing(cs%Time, e(:,:,1), e_tidal, g, cs%tides_CSp)
528  endif
529 
530 ! Here layer interface heights, e, are calculated.
531 !$OMP parallel default(none) shared(Isq,Ieq,Jsq,Jeq,nz,e,h,G,GV,e_tidal,CS)
532  if (cs%tides) then
533 !$OMP do
534  do j=jsq,jeq+1 ; do i=isq,ieq+1
535  e(i,j,nz+1) = -1.0*g%bathyT(i,j) - e_tidal(i,j)
536  enddo ; enddo
537  else
538 !$OMP do
539  do j=jsq,jeq+1 ; do i=isq,ieq+1
540  e(i,j,nz+1) = -1.0*g%bathyT(i,j)
541  enddo ; enddo
542  endif
543 !$OMP do
544  do j=jsq,jeq+1 ; do k=nz,1,-1 ; do i=isq,ieq+1
545  e(i,j,k) = e(i,j,k+1) + h(i,j,k)*gv%H_to_m
546  enddo ; enddo ; enddo
547 !$OMP end parallel
548  if (use_eos) then
549 
550 ! Calculate in-situ densities (rho_star).
551 
552 ! With a bulk mixed layer, replace the T & S of any layers that are
553 ! lighter than the the buffer layer with the properties of the buffer
554 ! layer. These layers will be massless anyway, and it avoids any
555 ! formal calculations with hydrostatically unstable profiles.
556 
557  if (nkmb>0) then
558  tv_tmp%T => t_tmp ; tv_tmp%S => s_tmp
559  tv_tmp%eqn_of_state => tv%eqn_of_state
560 
561  do i=isq,ieq+1 ; p_ref(i) = tv%P_Ref ; enddo
562 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,nkmb,tv_tmp,tv,p_ref,GV) &
563 !$OMP private(Rho_cv_BL)
564  do j=jsq,jeq+1
565  do k=1,nkmb ; do i=isq,ieq+1
566  tv_tmp%T(i,j,k) = tv%T(i,j,k) ; tv_tmp%S(i,j,k) = tv%S(i,j,k)
567  enddo ; enddo
568  call calculate_density(tv%T(:,j,nkmb), tv%S(:,j,nkmb), p_ref, &
569  rho_cv_bl(:), isq, ieq-isq+2, tv%eqn_of_state)
570 
571  do k=nkmb+1,nz ; do i=isq,ieq+1
572  if (gv%Rlay(k) < rho_cv_bl(i)) then
573  tv_tmp%T(i,j,k) = tv%T(i,j,nkmb) ; tv_tmp%S(i,j,k) = tv%S(i,j,nkmb)
574  else
575  tv_tmp%T(i,j,k) = tv%T(i,j,k) ; tv_tmp%S(i,j,k) = tv%S(i,j,k)
576  endif
577  enddo ; enddo
578  enddo
579  else
580  tv_tmp%T => tv%T ; tv_tmp%S => tv%S
581  tv_tmp%eqn_of_state => tv%eqn_of_state
582  do i=isq,ieq+1 ; p_ref(i) = 0.0 ; enddo
583  endif
584 
585  ! This no longer includes any pressure dependency, since this routine
586  ! will come down with a fatal error if there is any compressibility.
587 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,tv_tmp,p_ref,rho_star,tv,G_Rho0)
588  do k=1,nz+1 ; do j=jsq,jeq+1
589  call calculate_density(tv_tmp%T(:,j,k), tv_tmp%S(:,j,k), p_ref, rho_star(:,j,k), &
590  isq,ieq-isq+2,tv%eqn_of_state)
591  do i=isq,ieq+1 ; rho_star(i,j,k) = g_rho0*rho_star(i,j,k) ; enddo
592  enddo ; enddo
593  endif ! use_EOS
594 
595 ! Here the layer Montgomery potentials, M, are calculated.
596  if (use_eos) then
597 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,M,CS,rho_star,e,use_p_atm, &
598 !$OMP p_atm,I_Rho0)
599  do j=jsq,jeq+1
600  do i=isq,ieq+1
601  m(i,j,1) = cs%GFS_scale * (rho_star(i,j,1) * e(i,j,1))
602  if (use_p_atm) m(i,j,1) = m(i,j,1) + p_atm(i,j) * i_rho0
603  enddo
604  do k=2,nz ; do i=isq,ieq+1
605  m(i,j,k) = m(i,j,k-1) + (rho_star(i,j,k) - rho_star(i,j,k-1)) * e(i,j,k)
606  enddo ; enddo
607  enddo
608  else ! not use_EOS
609 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,M,GV,e,use_p_atm,p_atm,I_Rho0)
610  do j=jsq,jeq+1
611  do i=isq,ieq+1
612  m(i,j,1) = gv%g_prime(1) * e(i,j,1)
613  if (use_p_atm) m(i,j,1) = m(i,j,1) + p_atm(i,j) * i_rho0
614  enddo
615  do k=2,nz ; do i=isq,ieq+1
616  m(i,j,k) = m(i,j,k-1) + gv%g_prime(k) * e(i,j,k)
617  enddo ; enddo
618  enddo
619  endif ! use_EOS
620 
621  if (present(pbce)) then
622  call set_pbce_bouss(e, tv_tmp, g, gv, g%g_Earth, cs%Rho0, cs%GFS_scale, pbce, &
623  rho_star)
624  endif
625 
626 ! Calculate the pressure force. On a Cartesian grid,
627 ! PFu = - dM/dx and PFv = - dM/dy.
628  if (use_eos) then
629 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,js,je,is,ie,nz,e,h_neglect, &
630 !$OMP rho_star,G,PFu,CS,PFv,M) &
631 !$OMP private(h_star,PFu_bc,PFv_bc)
632  do k=1,nz
633  do j=jsq,jeq+1 ; do i=isq,ieq+1
634  h_star(i,j) = (e(i,j,k) - e(i,j,k+1)) + h_neglect
635  enddo ; enddo
636  do j=js,je ; do i=isq,ieq
637  pfu_bc = -1.0*(rho_star(i+1,j,k) - rho_star(i,j,k)) * (g%IdxCu(i,j) * &
638  ((h_star(i,j) * h_star(i+1,j) - (e(i,j,k) * h_star(i+1,j) + &
639  e(i+1,j,k) * h_star(i,j))) / (h_star(i,j) + h_star(i+1,j))))
640  pfu(i,j,k) = -(m(i+1,j,k) - m(i,j,k)) * g%IdxCu(i,j) + pfu_bc
641  if (ASSOCIATED(cs%PFu_bc)) cs%PFu_bc(i,j,k) = pfu_bc
642  enddo ; enddo
643  do j=jsq,jeq ; do i=is,ie
644  pfv_bc = -1.0*(rho_star(i,j+1,k) - rho_star(i,j,k)) * (g%IdyCv(i,j) * &
645  ((h_star(i,j) * h_star(i,j+1) - (e(i,j,k) * h_star(i,j+1) + &
646  e(i,j+1,k) * h_star(i,j))) / (h_star(i,j) + h_star(i,j+1))))
647  pfv(i,j,k) = -(m(i,j+1,k) - m(i,j,k)) * g%IdyCv(i,j) + pfv_bc
648  if (ASSOCIATED(cs%PFv_bc)) cs%PFv_bc(i,j,k) = pfv_bc
649  enddo ; enddo
650  enddo ! k-loop
651  else ! .not. use_EOS
652 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,is,ie,js,je,nz,PFu,PFv,M,G)
653  do k=1,nz
654  do j=js,je ; do i=isq,ieq
655  pfu(i,j,k) = -(m(i+1,j,k) - m(i,j,k)) * g%IdxCu(i,j)
656  enddo ; enddo
657  do j=jsq,jeq ; do i=is,ie
658  pfv(i,j,k) = -(m(i,j+1,k) - m(i,j,k)) * g%IdyCv(i,j)
659  enddo ; enddo
660  enddo
661  endif ! use_EOS
662 
663  if (present(eta)) then
664  if (cs%tides) then
665  ! eta is the sea surface height relative to a time-invariant geoid, for
666  ! comparison with what is used for eta in btstep. See how e was calculated
667  ! about 200 lines above.
668 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,eta,e,e_tidal,GV)
669  do j=jsq,jeq+1 ; do i=isq,ieq+1
670  eta(i,j) = e(i,j,1)*gv%m_to_H + e_tidal(i,j)*gv%m_to_H
671  enddo ; enddo
672  else
673 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,eta,e,GV)
674  do j=jsq,jeq+1 ; do i=isq,ieq+1
675  eta(i,j) = e(i,j,1)*gv%m_to_H
676  enddo ; enddo
677  endif
678  endif
679 
680  if (cs%id_PFu_bc>0) call post_data(cs%id_PFu_bc, cs%PFu_bc, cs%diag)
681  if (cs%id_PFv_bc>0) call post_data(cs%id_PFv_bc, cs%PFv_bc, cs%diag)
682  if (cs%id_e_tidal>0) call post_data(cs%id_e_tidal, e_tidal, cs%diag)
683 
684 end subroutine pressureforce_mont_bouss

686 subroutine set_pbce_bouss(e, tv, G, GV, g_Earth, Rho0, GFS_scale, pbce, rho_star)
687  type(ocean_grid_type), intent(in) :: G
688  type(verticalgrid_type), intent(in) :: GV
689  real, dimension(SZI_(G),SZJ_(G),SZK_(G)+1), intent(in) :: e
690  type(thermo_var_ptrs), intent(in) :: tv
691  real, intent(in) :: g_Earth
692  real, intent(in) :: Rho0
693  real, intent(in) :: GFS_scale
694 ! type(PressureForce_Mont_CS), pointer :: CS
695  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(out) :: pbce
696  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), optional, intent(in) :: rho_star
697 ! This subroutine determines the partial derivative of the acceleration due
698 ! to pressure forces with the free surface height.
699 ! Arguments: e - Interface height, in H (m).
700 ! (in) tv - A structure containing pointers to any available
701 ! thermodynamic fields, including potential temperature and
702 ! salinity or mixed layer density. Absent fields have NULL ptrs.
703 ! (in) G - The ocean's grid structure.
704 ! (in) GV - The ocean's vertical grid structure.
705 ! (in) g_Earth - The gravitational acceleration, in m s-2.
706 ! (in) Rho0 - The "Boussinesq" ocean density, in kg m-3.
707 ! (in) Rho_atmos - The atmospheric density, in kg m-3. Typically this
708 ! should be 0, but it could be set to a significant
709 ! fraction of the ocean density to use the model in a
710 ! reduced gravity mode.
711 ! (in) CS - The control structure returned by a previous call to
712 ! PressureForce_init.
713 ! (out) pbce - the baroclinic pressure anomaly in each layer
714 ! due to free surface height anomalies, in m2 H-1 s-2.
715 ! (in,opt) rho_star - The layer densities (maybe compressibility compensated),
716 ! times g/rho_0, in m s-2.
717 
718  real :: Ihtot(szi_(g)) ! The inverse of the sum of the layer
719  ! thicknesses, in m-1.
720  real :: press(szi_(g)) ! Interface pressure, in Pa.
721  real :: T_int(szi_(g)) ! Interface temperature in C.
722  real :: S_int(szi_(g)) ! Interface salinity in PSU.
723  real :: dR_dT(szi_(g)) ! Partial derivatives of density with temperature
724  real :: dR_dS(szi_(g)) ! and salinity in kg m-3 K-1 and kg m-3 PSU-1.
725  real :: rho_in_situ(szi_(g)) !In-situ density at the top of a layer.
726  real :: G_Rho0 ! g_Earth / Rho0 in m4 s-2 kg-1.
727  real :: Rho0xG ! g_Earth * Rho0 in kg s-2 m-2.
728  logical :: use_EOS ! If true, density is calculated from T & S using
729  ! an equation of state.
730  real :: h_neglect ! A thickness that is so small it is usually lost
731  ! in roundoff and can be neglected, in m.
732  integer :: Isq, Ieq, Jsq, Jeq, nz, i, j, k
733 
734  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB ; nz = g%ke
735 
736  rho0xg = rho0*g_earth
737  g_rho0 = g_earth/rho0
738  use_eos = associated(tv%eqn_of_state)
739  h_neglect = gv%H_subroundoff*gv%H_to_m
740 
741  if (use_eos) then
742  if (present(rho_star)) then
743 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,e,h_neglect,pbce,rho_star,&
744 !$OMP GFS_scale,GV) &
745 !$OMP private(Ihtot)
746  do j=jsq,jeq+1
747  do i=isq,ieq+1
748  ihtot(i) = 1.0 / (((e(i,j,1)-e(i,j,nz+1)) + h_neglect) * gv%m_to_H)
749  pbce(i,j,1) = gfs_scale * rho_star(i,j,1) * gv%H_to_m
750  enddo
751  do k=2,nz ; do i=isq,ieq+1
752  pbce(i,j,k) = pbce(i,j,k-1) + (rho_star(i,j,k)-rho_star(i,j,k-1)) * &
753  ((e(i,j,k) - e(i,j,nz+1)) * ihtot(i))
754  enddo ; enddo
755  enddo ! end of j loop
756  else
757 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,e,tv,h_neglect,G_Rho0,Rho0xG,&
758 !$OMP pbce,GFS_scale,GV) &
759 !$OMP private(Ihtot,press,rho_in_situ,T_int,S_int,dR_dT,dR_dS)
760  do j=jsq,jeq+1
761  do i=isq,ieq+1
762  ihtot(i) = 1.0 / (((e(i,j,1)-e(i,j,nz+1)) + h_neglect) * gv%m_to_H)
763  press(i) = -rho0xg*e(i,j,1)
764  enddo
765  call calculate_density(tv%T(:,j,1), tv%S(:,j,1), press, rho_in_situ, &
766  isq, ieq-isq+2, tv%eqn_of_state)
767  do i=isq,ieq+1
768  pbce(i,j,1) = g_rho0*(gfs_scale * rho_in_situ(i)) * gv%H_to_m
769  enddo
770  do k=2,nz
771  do i=isq,ieq+1
772  press(i) = -rho0xg*e(i,j,k)
773  t_int(i) = 0.5*(tv%T(i,j,k-1)+tv%T(i,j,k))
774  s_int(i) = 0.5*(tv%S(i,j,k-1)+tv%S(i,j,k))
775  enddo
776  call calculate_density_derivs(t_int, s_int, press, dr_dt, dr_ds, &
777  isq, ieq-isq+2, tv%eqn_of_state)
778  do i=isq,ieq+1
779  pbce(i,j,k) = pbce(i,j,k-1) + g_rho0 * &
780  ((e(i,j,k) - e(i,j,nz+1)) * ihtot(i)) * &
781  (dr_dt(i)*(tv%T(i,j,k)-tv%T(i,j,k-1)) + &
782  dr_ds(i)*(tv%S(i,j,k)-tv%S(i,j,k-1)))
783  enddo
784  enddo
785  enddo ! end of j loop
786  endif
787  else ! not use_EOS
788 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,e,GV,h_neglect,pbce) private(Ihtot)
789  do j=jsq,jeq+1
790  do i=isq,ieq+1
791  ihtot(i) = 1.0 / (((e(i,j,1)-e(i,j,nz+1)) + h_neglect) * gv%m_to_H)
792  pbce(i,j,1) = gv%g_prime(1) * gv%H_to_m
793  enddo
794  do k=2,nz ; do i=isq,ieq+1
795  pbce(i,j,k) = pbce(i,j,k-1) + &

805  type(verticalgrid_type), intent(in) :: GV

812  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), optional, intent(in) :: alpha_star
813 ! This subroutine determines the partial derivative of the acceleration due
814 ! to pressure forces with the column mass.
815 ! Arguments: p - Interface pressures, in Pa.
816 ! (in) tv - A structure containing pointers to any available
817 ! thermodynamic fields, including potential temperature and

823 ! should be 0, but it could be set to a significant
824 ! fraction of the ocean density to use the model in a
825 ! reduced gravity mode.
826 ! (in) CS - The control structure returned by a previous call to
827 ! PressureForce_init.
828 ! (out) pbce - the baroclinic pressure anomaly in each layer
829 ! due to total thickness anomalies, in m4 s-2 kg-1.
830 ! (in,opt) alpha_star - The layer specific volumes (maybe compressibility
831 ! compensated), in m3 kg-1.
832  real, dimension(SZI_(G),SZJ_(G)) :: &
833  dpbce, & ! A barotropic correction to the pbce to enable the use of
834  ! a reduced gravity form of the equations, in m4 s-2 kg-1.
835  c_htot ! dP_dH divided by the total ocean pressure, m2 kg-1.

839  real :: dR_dS(szi_(g)) ! and salinity in kg m-3 K-1 and kg m-3 PSU-1.

841  real :: alpha_Lay(szk_(g)) ! The specific volume of each layer, in kg m-3.
842  real :: dalpha_int(szk_(g)+1) ! The change in specific volume across each
843  ! interface, in kg m-3.

845  ! usually in Pa m2 kg-1.
846  real :: dp_neglect ! A thickness that is so small it is usually lost
847  ! in roundoff and can be neglected, in Pa.
848  logical :: use_EOS ! If true, density is calculated from T & S using
849  ! an equation of state.
850  integer :: Isq, Ieq, Jsq, Jeq, nz, i, j, k
851 
852  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB ; nz = g%ke
853

857  dp_neglect = dp_dh * gv%H_subroundoff

860  do k=2,nz ; dalpha_int(k) = alpha_lay(k-1) - alpha_lay(k) ; enddo

874  enddo ; enddo
875  enddo
876  else

878 !$OMP dP_dH,dp_neglect,pbce) &
879 !$OMP private(T_int,S_int,dR_dT,dR_dS,rho_in_situ)
880  do j=jsq,jeq+1
881  call calculate_density(tv%T(:,j,nz), tv%S(:,j,nz), p(:,j,nz+1), &
882  rho_in_situ, isq, ieq-isq+2, tv%eqn_of_state)
883  do i=isq,ieq+1
884  c_htot(i,j) = dp_dh / ((p(i,j,nz+1)-p(i,j,1)) + dp_neglect)
885  pbce(i,j,nz) = dp_dh / rho_in_situ(i)
886  enddo
887  do k=nz-1,1,-1
888  do i=isq,ieq+1
889  t_int(i) = 0.5*(tv%T(i,j,k)+tv%T(i,j,k+1))
890  s_int(i) = 0.5*(tv%S(i,j,k)+tv%S(i,j,k+1))
891  enddo
892  call calculate_density(t_int, s_int, p(:,j,k+1), rho_in_situ, &
893  isq, ieq-isq+2, tv%eqn_of_state)
894  call calculate_density_derivs(t_int, s_int, p(:,j,k+1), dr_dt, dr_ds, &
895  isq, ieq-isq+2, tv%eqn_of_state)
896  do i=isq,ieq+1
897  pbce(i,j,k) = pbce(i,j,k+1) + ((p(i,j,k+1)-p(i,j,1))*c_htot(i,j)) * &
898  ((dr_dt(i)*(tv%T(i,j,k+1)-tv%T(i,j,k)) + &
899  dr_ds(i)*(tv%S(i,j,k+1)-tv%S(i,j,k))) / rho_in_situ(i)**2)
900  enddo
901  enddo
902  enddo
903  endif
904  else ! not use_EOS
905 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,C_htot,dP_dH,p,dp_neglect, &
906 !$OMP pbce,alpha_Lay,dalpha_int)
907  do j=jsq,jeq+1
908  do i=isq,ieq+1
909  c_htot(i,j) = dp_dh / ((p(i,j,nz+1)-p(i,j,1)) + dp_neglect)
910  pbce(i,j,nz) = dp_dh * alpha_lay(nz)
911  enddo
912  do k=nz-1,1,-1 ; do i=isq,ieq+1
913  pbce(i,j,k) = pbce(i,j,k+1) + ((p(i,j,k+1)-p(i,j,1))*c_htot(i,j)) * &
914  dalpha_int(k+1)
915  enddo ; enddo
916  enddo
917  endif ! use_EOS
918 
919  if (gfs_scale < 1.0) then
920  ! Adjust the Montgomery potential to make this a reduced gravity model.
921 !$OMP parallel default(none) shared(Isq,Ieq,Jsq,Jeq,dpbce,GFS_scale,pbce,nz)
922 !$OMP do
923  do j=jsq,jeq+1 ; do i=isq,ieq+1
924  dpbce(i,j) = (gfs_scale - 1.0) * pbce(i,j,1)
925  enddo ; enddo
926 !$OMP do
927  do k=1,nz ; do j=jsq,jeq+1 ; do i=isq,ieq+1
928  pbce(i,j,k) = pbce(i,j,k) + dpbce(i,j)
929  enddo ; enddo ; enddo
930 !$OMP end parallel
931  endif
