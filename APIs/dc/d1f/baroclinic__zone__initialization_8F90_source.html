51 !> Initialization of temperature and salinity with the baroclinic zone initial conditions
52 subroutine baroclinic_zone_init_temperature_salinity(T, S, h, G, param_file)

58 
59  integer :: i, j, k, is, ie, js, je, nz
60  real :: T_ref, dTdz, dTdx, delta_T ! Parameters describing temperature distribution
61  real :: S_ref, dSdz, dSdx, delta_S ! Parameters describing salinity distribution
62  real :: L_zone ! Width of baroclinic zone
63  real :: zc, zi, x, xd, xs, y, yd, fn
64  real :: PI ! 3.1415926... calculated as 4*atan(1)
65 
66  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
67  call bcz_params(g, param_file, s_ref, dsdz, delta_s, dsdx, t_ref, dtdz, delta_t, dtdx, l_zone)
68 
69  t(:,:,:) = 0.
70  s(:,:,:) = 0.
71  pi = 4.*atan(1.)
72 
73  do j = g%jsc,g%jec ; do i = g%isc,g%iec
74  zi = -g%bathyT(i,j)
75  x = g%geoLonT(i,j) - (g%west_lon + 0.5*g%len_lon) ! Relative to center of domain
76  xd = x / g%len_lon ! -1/2 < xd 1/2
77  y = g%geoLatT(i,j) - (g%south_lat + 0.5*g%len_lat) ! Relative to center of domain
78  yd = y / g%len_lat ! -1/2 < yd 1/2
79  if (l_zone/=0.) then
80  xs = min(1., max(-1., x/l_zone)) ! -1 < ys < 1
81  fn = sin((0.5*pi)*xs)
82  else
83  xs = sign(1., x) ! +/- 1
84  fn = xs
85  endif
86  do k = nz, 1, -1
87  zc = zi + 0.5*h(i,j,k) ! Position of middle of cell
88  zi = zi + h(i,j,k) ! Top interface position
89  t(i,j,k) = t_ref + dtdz * zc & ! Linear temperature stratification
90  + dtdx * x & ! Linear gradient
91  + delta_t * fn ! Smooth fn of width L_zone
92  s(i,j,k) = s_ref + dsdz * zc & ! Linear temperature stratification
93  + dsdx * x & ! Linear gradient
94  + delta_s * fn ! Smooth fn of width L_zone
95  enddo
96  end do ; end do
