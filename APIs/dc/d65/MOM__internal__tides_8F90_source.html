687 subroutine itidal_lowmode_loss(G, CS, Nb, Ub, En, TKE_loss_fixed, TKE_loss, dt, full_halos)

691  real, dimension(G%isd:G%ied,G%jsd:G%jed,CS%nFreq,CS%nMode), intent(inout) :: Ub

693  real, dimension(G%isd:G%ied,G%jsd:G%jed,CS%NAngle,CS%nFreq,CS%nMode), intent(inout) :: En
694  real, dimension(G%isd:G%ied,G%jsd:G%jed,CS%NAngle,CS%nFreq,CS%nMode), intent(out) :: TKE_loss
695  real, intent(in) :: dt
696  logical,optional, intent(in) :: full_halos
697 
698  ! This subroutine calculates the energy lost from the propagating internal tide due to
699  ! scattering over small-scale roughness along the lines of Jayne & St. Laurent (2001).
700  !
701  ! Arguments:
702  ! (in) Nb - near-bottom stratification, in s-1.
703  ! (in) Ub - rms (over one period) near-bottom horizontal mode velocity , in m s-1.
704  ! (inout) En - energy density of the internal waves, in J m-2.
705  ! (in) TKE_loss_fixed - fixed part of energy loss, in kg m-2 (rho*kappa*h^2)
706  ! (out) TKE_loss - energy loss rate, in W m-2 (q*rho*kappa*h^2*N*U^2)
707  ! (in) dt - time increment, in s

709  ! computational domain.

721  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec

2100 subroutine ppm_limit_pos(h_in, h_L, h_R, h_min, G, iis, iie, jis, jie)
2101  type(ocean_grid_type), intent(in) :: G
2102  real, dimension(SZI_(G),SZJ_(G)), intent(in) :: h_in
2103  real, dimension(SZI_(G),SZJ_(G)), intent(inout) :: h_L, h_R
2104  real, intent(in) :: h_min
2105  integer, intent(in) :: iis, iie, jis, jie
2106  ! This subroutine limits the left/right edge values of the PPM reconstruction
2107  ! to give a reconstruction that is positive-definite. Here this is
2108  ! reinterpreted as giving a constant thickness if the mean thickness is less
2109  ! than h_min, with a minimum of h_min otherwise.
2110  ! Arguments: h_in - thickness of layer (2D)
2111  ! (inout) h_L - left edge value (2D)
2112  ! (inout) h_R - right edge value (2D)
2113  ! (in) h_min - The minimum thickness that can be obtained by a
2114  ! concave parabolic fit.
2115  ! (in) iis, iie, jis, jie - Index range for computation.
2116  ! (in) G - The ocean's grid structure.
2117 
2118  ! Local variables
2119  real :: curv, dh, scale
2120  character(len=256) :: mesg
2121  integer :: i,j
2122 
2123  do j=jis,jie ; do i=iis,iie
2124  ! This limiter prevents undershooting minima within the domain with
2125  ! values less than h_min.
2126  curv = 3.0*(h_l(i,j) + h_r(i,j) - 2.0*h_in(i,j))
2127  if (curv > 0.0) then ! Only minima are limited.
2128  dh = h_r(i,j) - h_l(i,j)
2129  if (abs(dh) < curv) then ! The parabola's minimum is within the cell.
2130  if (h_in(i,j) <= h_min) then
2131  h_l(i,j) = h_in(i,j) ; h_r(i,j) = h_in(i,j)
2132  elseif (12.0*curv*(h_in(i,j) - h_min) < (curv**2 + 3.0*dh**2)) then
2133  ! The minimum value is h_in - (curv^2 + 3*dh^2)/(12*curv), and must
2134  ! be limited in this case. 0 < scale < 1.
2135  scale = 12.0*curv*(h_in(i,j) - h_min) / (curv**2 + 3.0*dh**2)
2136  h_l(i,j) = h_in(i,j) + scale*(h_l(i,j) - h_in(i,j))
2137  h_r(i,j) = h_in(i,j) + scale*(h_r(i,j) - h_in(i,j))
2138  endif
2139  endif
2140  endif
2141  enddo ; enddo
2142 end subroutine ppm_limit_pos
