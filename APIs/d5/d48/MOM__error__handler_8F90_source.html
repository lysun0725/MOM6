72  integer, intent(in) :: level
73  character(len=*), intent(in) :: message
74  logical, optional, intent(in) :: all_print
75  ! This provides a convenient interface for writing an mpp_error message
76  ! with run-time filter based on a verbosity.
77  logical :: write_msg
78 
79  write_msg = is_root_pe()
80  if (present(all_print)) write_msg = write_msg .or. all_print
81 
82  select case (level)
83  case (note)
84  if (write_msg.and.verbosity>=2) call mpp_error(note, message)
85  case (warning)

87  case (fatal)
88  if (verbosity>=0) call mpp_error(fatal, message)
89  case default
