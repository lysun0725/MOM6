24 !* By Robert Hallberg, 2002 *
25 !* *
26 !* This file contains an example of the code that is needed to set *
27 !* up and use a set (in this case two) of dynamically passive tracers *
28 !* for diagnostic purposes. The tracers here are an ideal age tracer *
29 !* that ages at a rate of 1/year once it is isolated from the surface,*
30 !* and a vintage tracer, whose surface concentration grows exponen- *
31 !* with time with a 30-year timescale (similar to CFCs). *
32 !* *
33 !* A single subroutine is called from within each file to register *
34 !* each of the tracers for reinitialization and advection and to *
35 !* register the subroutine that initializes the tracers and set up *
36 !* their output and the subroutine that does any tracer physics or *
37 !* chemistry along with diapycnal mixing (included here because some *
38 !* tracers may float or swim vertically or dye diapycnal processes). *
39 !* *
40 !* *
41 !* Macros written all in capital letters are defined in MOM_memory.h. *
42 !* *
43 !* A small fragment of the grid is shown below: *
44 !* *
45 !* j+1 x ^ x ^ x At x: q *
46 !* j+1 > o > o > At ^: v *
47 !* j x ^ x ^ x At >: u *
48 !* j > o > o > At o: h, tr *
49 !* j-1 x ^ x ^ x *
50 !* i-1 i i+1 At x & ^: *
51 !* i i+1 At > & o: *
52 !* *
53 !* The boundaries always run through q grid points (x). *
54 !* *
55 !********+*********+*********+*********+*********+*********+*********+**

145 function register_oil_tracer(HI, GV, param_file, CS, tr_Reg, restart_CS)
146  type(hor_index_type), intent(in) :: HI
147  type(verticalgrid_type), intent(in) :: GV
148  type(param_file_type), intent(in) :: param_file
149  type(oil_tracer_cs), pointer :: CS
150  type(tracer_registry_type), pointer :: tr_Reg
151  type(mom_restart_cs), pointer :: restart_CS
152 ! This subroutine is used to register tracer fields and subroutines
153 ! to be used with MOM.
154 ! Arguments: HI - A horizontal index type structure.
155 ! (in) GV - The ocean's vertical grid structure.
156 ! (in) param_file - A structure indicating the open file to parse for
157 ! model parameter values.
158 ! (in/out) CS - A pointer that is set to point to the control structure
159 ! for this module
160 ! (in/out) tr_Reg - A pointer that is set to point to the control structure
161 ! for the tracer advection and diffusion module.
162 ! (in) restart_CS - A pointer to the restart control structure.
163 
164 ! This include declares and sets the variable "version".
165 #include "version_variable.h"
166  character(len=40) :: mod = "oil_tracer" ! This module's name.
167  character(len=200) :: inputdir ! The directory where the input files are.
168  character(len=48) :: var_name ! The variable's name.
169  character(len=3) :: name_tag ! String for creating identifying oils
170  real, pointer :: tr_ptr(:,:,:) => null()
171  logical :: register_oil_tracer
172  integer :: isd, ied, jsd, jed, nz, m, i, j
173  isd = hi%isd ; ied = hi%ied ; jsd = hi%jsd ; jed = hi%jed ; nz = gv%ke
174 
175  if (associated(cs)) then
176  call mom_error(warning, "register_oil_tracer called with an "// &
177  "associated control structure.")
178  return
179  endif
180  allocate(cs)
181 
182  ! Read all relevant parameters and write them to the model log.
183  call log_version(param_file, mod, version, "")
184  call get_param(param_file, mod, "OIL_IC_FILE", cs%IC_file, &
185  "The file in which the oil tracer initial values can be \n"//&
186  "found, or an empty string for internal initialization.", &
187  default=" ")
188  if ((len_trim(cs%IC_file) > 0) .and. (scan(cs%IC_file,'/') == 0)) then
189  ! Add the directory if CS%IC_file is not already a complete path.
190  call get_param(param_file, mod, "INPUTDIR", inputdir, default=".")
191  cs%IC_file = trim(slasher(inputdir))//trim(cs%IC_file)
192  call log_param(param_file, mod, "INPUTDIR/CFC_IC_FILE", cs%IC_file)
193  endif
194  call get_param(param_file, mod, "OIL_IC_FILE_IS_Z", cs%Z_IC_file, &
195  "If true, OIL_IC_FILE is in depth space, not layer space", &
196  default=.false.)
197 
198  call get_param(param_file, mod, "MASK_MASSLESS_TRACERS", cs%mask_tracers, &
199  "If true, the tracers are masked out in massless layer. \n"//&
200  "This can be a problem with time-averages.", default=.false.)
201  call get_param(param_file, mod, "OIL_MAY_REINIT", cs%oil_may_reinit, &
202  "If true, oil tracers may go through the initialization \n"//&
203  "code if they are not found in the restart files. \n"//&
204  "Otherwise it is a fatal error if the oil tracers are not \n"//&
205  "found in the restart files of a restarted run.", &
206  default=.false.)
207  call get_param(param_file, mod, "OIL_SOURCE_LONGITUDE", cs%oil_source_longitude, &
208  "The geographic longitude of the oil source.", units="degrees E", &
209  fail_if_missing=.true.)
210  call get_param(param_file, mod, "OIL_SOURCE_LATITUDE", cs%oil_source_latitude, &
211  "The geographic latitude of the oil source.", units="degrees N", &
212  fail_if_missing=.true.)
213  call get_param(param_file, mod, "OIL_SOURCE_LAYER", cs%oil_source_k, &
214  "The layer into which the oil is introduced, or a \n"//&
215  "negative number for a vertically uniform source, \n"//&
216  "or 0 not to use this tracer.", units="Layer", default=0)
217  call get_param(param_file, mod, "OIL_SOURCE_RATE", cs%oil_source_rate, &
218  "The rate of oil injection.", units="kg s-1", default=1.0)
219  call get_param(param_file, mod, "OIL_DECAY_DAYS", cs%oil_decay_days, &
220  "The decay timescale in days (if positive), or no decay \n"//&
221  "if 0, or use the temperature dependent decay rate of \n"//&
222  "Adcroft et al. (GRL, 2010) if negative.", units="days", &
223  default=0.0)
224  call get_param(param_file, mod, "OIL_DATED_START_YEAR", cs%oil_start_year, &
225  "The time at which the oil source starts", units="years", &
226  default=0.0)
227  call get_param(param_file, mod, "OIL_DATED_END_YEAR", cs%oil_end_year, &
228  "The time at which the oil source ends", units="years", &
229  default=1.0e99)
230 
231  cs%ntr = 0
232  cs%oil_decay_rate(:) = 0.
233  do m=1,ntr_max
234  if (cs%oil_source_k(m)/=0) then
235  write(name_tag(1:3),'("_",I2.2)') m
236  cs%ntr = cs%ntr + 1
237  cs%tr_desc(m) = var_desc("oil"//trim(name_tag), "kg/m3", "Oil Tracer", caller=mod)
238  cs%IC_val(m) = 0.0
239  if (cs%oil_decay_days(m)>0.) then
240  cs%oil_decay_rate(m)=1./(86400.0*cs%oil_decay_days(m))
241  elseif (cs%oil_decay_days(m)<0.) then
242  cs%oil_decay_rate(m)=-1.
243  endif
244  endif
245  enddo
246  call log_param(param_file, mod, "OIL_DECAY_RATE", cs%oil_decay_rate(1:cs%ntr))
247 
248  allocate(cs%tr(isd:ied,jsd:jed,nz,cs%ntr)) ; cs%tr(:,:,:,:) = 0.0
249 
250  do m=1,cs%ntr
251  ! This is needed to force the compiler not to do a copy in the registration
252  ! calls. Curses on the designers and implementers of Fortran90.
253  tr_ptr => cs%tr(:,:,:,m)
254  call query_vardesc(cs%tr_desc(m), name=var_name, caller="register_oil_tracer")
255  ! Register the tracer for the restart file.
256  call register_restart_field(tr_ptr, cs%tr_desc(m), &
257  .not.cs%oil_may_reinit, restart_cs)
258  ! Register the tracer for horizontal advection & diffusion.
259  call register_tracer(tr_ptr, cs%tr_desc(m), param_file, hi, gv, tr_reg, &
260  tr_desc_ptr=cs%tr_desc(m))
261 
262  ! Set coupled_tracers to be true (hard-coded above) to provide the surface
263  ! values to the coupler (if any). This is meta-code and its arguments will
264  ! currently (deliberately) give fatal errors if it is used.
265  if (cs%coupled_tracers) &
266  cs%ind_tr(m) = aof_set_coupler_flux(trim(var_name)//'_flux', &
267  flux_type=' ', implementation=' ', caller="register_oil_tracer")
268  enddo
269 
270  cs%tr_Reg => tr_reg
271  cs%restart_CSp => restart_cs
272  register_oil_tracer = .true.
273 
274 end function register_oil_tracer

276 subroutine initialize_oil_tracer(restart, day, G, GV, h, diag, OBC, CS, &
277  sponge_csp, diag_to_z_csp)
278  logical, intent(in) :: restart
279  type(time_type), target, intent(in) :: day
280  type(ocean_grid_type), intent(in) :: G
281  type(verticalgrid_type), intent(in) :: GV
282  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
283  type(diag_ctrl), target, intent(in) :: diag
284  type(ocean_obc_type), pointer :: OBC
285  type(oil_tracer_cs), pointer :: CS
286  type(sponge_cs), pointer :: sponge_CSp
287  type(diag_to_z_cs), pointer :: diag_to_Z_CSp
288 ! This subroutine initializes the CS%ntr tracer fields in tr(:,:,:,:)
289 ! and it sets up the tracer output.
290 
291 ! Arguments: restart - .true. if the fields have already been read from
292 ! a restart file.
293 ! (in) day - Time of the start of the run.
294 ! (in) G - The ocean's grid structure.
295 ! (in) GV - The ocean's vertical grid structure.
296 ! (in) h - Layer thickness, in m or kg m-2.
297 ! (in) diag - A structure that is used to regulate diagnostic output.
298 ! (in) OBC - This open boundary condition type specifies whether, where,
299 ! and what open boundary conditions are used.
300 ! (in/out) CS - The control structure returned by a previous call to
301 ! register_oil_tracer.
302 ! (in/out) sponge_CSp - A pointer to the control structure for the sponges, if
303 ! they are in use. Otherwise this may be unassociated.
304 ! (in/out) diag_to_Z_Csp - A pointer to the control structure for diagnostics
305 ! in depth space.
306  character(len=16) :: name ! A variable's name in a NetCDF file.
307  character(len=72) :: longname ! The long name of that variable.
308  character(len=48) :: units ! The dimensions of the variable.
309  character(len=48) :: flux_units ! The units for age tracer fluxes, either
310  ! years m3 s-1 or years kg s-1.
311  logical :: OK
312  integer :: i, j, k, is, ie, js, je, isd, ied, jsd, jed, nz, m
313  integer :: IsdB, IedB, JsdB, JedB
314 
315  if (.not.associated(cs)) return
316  if (cs%ntr < 1) return
317  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = gv%ke
318  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
319  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
320 
321  ! Establish location of source
322  do j=g%jsdB+1,g%jed ; do i=g%isdB+1,g%ied
323  ! This test for i,j index is specific to a lat/lon (non-rotated grid).
324  ! and needs to be generalized to work properly on the tri-polar grid.
325  if (cs%oil_source_longitude<g%geoLonBu(i,j) .and. &
326  cs%oil_source_longitude>=g%geoLonBu(i-1,j) .and. &
327  cs%oil_source_latitude<g%geoLatBu(i,j) .and. &
328  cs%oil_source_latitude>=g%geoLatBu(i,j-1) ) then
329  cs%oil_source_i=i
330  cs%oil_source_j=j
331  endif
332  enddo; enddo
333 
334  cs%Time => day
335  cs%diag => diag
336 
337  do m=1,cs%ntr
338  call query_vardesc(cs%tr_desc(m), name=name, caller="initialize_oil_tracer")
339  if ((.not.restart) .or. (cs%oil_may_reinit .and. .not. &
340  query_initialized(cs%tr(:,:,:,m), name, cs%restart_CSp))) then
341 
342  if (len_trim(cs%IC_file) > 0) then
343  ! Read the tracer concentrations from a netcdf file.
344  if (.not.file_exists(cs%IC_file, g%Domain)) &
345  call mom_error(fatal, "initialize_oil_tracer: "// &
346  "Unable to open "//cs%IC_file)
347 
348  if (cs%Z_IC_file) then
349  ok = tracer_z_init(cs%tr(:,:,:,m), h, cs%IC_file, name, &
350  g, -1e34, 0.0) ! CS%land_val(m))
351  if (.not.ok) then
352  ok = tracer_z_init(cs%tr(:,:,:,m), h, cs%IC_file, &
353  trim(name), g, -1e34, 0.0) ! CS%land_val(m))
354  if (.not.ok) call mom_error(fatal,"initialize_oil_tracer: "//&
355  "Unable to read "//trim(name)//" from "//&
356  trim(cs%IC_file)//".")
357  endif
358  else
359  call read_data(cs%IC_file, trim(name), cs%tr(:,:,:,m), &
360  domain=g%Domain%mpp_domain)
361  endif
362  else
363  do k=1,nz ; do j=js,je ; do i=is,ie
364  if (g%mask2dT(i,j) < 0.5) then
365  cs%tr(i,j,k,m) = cs%land_val(m)
366  else
367  cs%tr(i,j,k,m) = cs%IC_val(m)
368  endif
369  enddo ; enddo ; enddo
370  endif
371 
372  endif ! restart
373  enddo ! Tracer loop
374 
375  if (associated(obc)) then
376  ! All tracers but the first have 0 concentration in their inflows. As this
377  ! is the default value, the following calls are unnecessary.
378  ! do m=1,CS%ntr
379  ! call add_tracer_OBC_values(trim(CS%tr_desc(m)%name), CS%tr_Reg, 0.0)
380  ! enddo
381  endif
382 
383  ! This needs to be changed if the units of tracer are changed above.
384  if (gv%Boussinesq) then ; flux_units = "years m3 s-1"
385  else ; flux_units = "years kg s-1" ; endif
386 
387  do m=1,cs%ntr
388  ! Register the tracer for the restart file.
389  call query_vardesc(cs%tr_desc(m), name, units=units, longname=longname, &
390  caller="initialize_oil_tracer")
391  cs%id_tracer(m) = register_diag_field("ocean_model", trim(name), cs%diag%axesTL, &
392  day, trim(longname) , trim(units))
393  cs%id_tr_adx(m) = register_diag_field("ocean_model", trim(name)//"_adx", &
394  cs%diag%axesCuL, day, trim(longname)//" advective zonal flux" , &
395  trim(flux_units))
396  cs%id_tr_ady(m) = register_diag_field("ocean_model", trim(name)//"_ady", &
397  cs%diag%axesCvL, day, trim(longname)//" advective meridional flux" , &
398  trim(flux_units))
399  cs%id_tr_dfx(m) = register_diag_field("ocean_model", trim(name)//"_dfx", &
400  cs%diag%axesCuL, day, trim(longname)//" diffusive zonal flux" , &
401  trim(flux_units))
402  cs%id_tr_dfy(m) = register_diag_field("ocean_model", trim(name)//"_dfy", &
403  cs%diag%axesCvL, day, trim(longname)//" diffusive zonal flux" , &
404  trim(flux_units))
405  if (cs%id_tr_adx(m) > 0) call safe_alloc_ptr(cs%tr_adx(m)%p,isdb,iedb,jsd,jed,nz)
406  if (cs%id_tr_ady(m) > 0) call safe_alloc_ptr(cs%tr_ady(m)%p,isd,ied,jsdb,jedb,nz)
407  if (cs%id_tr_dfx(m) > 0) call safe_alloc_ptr(cs%tr_dfx(m)%p,isdb,iedb,jsd,jed,nz)
408  if (cs%id_tr_dfy(m) > 0) call safe_alloc_ptr(cs%tr_dfy(m)%p,isd,ied,jsdb,jedb,nz)
409 
410 ! Register the tracer for horizontal advection & diffusion.
411  if ((cs%id_tr_adx(m) > 0) .or. (cs%id_tr_ady(m) > 0) .or. &
412  (cs%id_tr_dfx(m) > 0) .or. (cs%id_tr_dfy(m) > 0)) &
413  call add_tracer_diagnostics(name, cs%tr_Reg, cs%tr_adx(m)%p, &
414  cs%tr_ady(m)%p,cs%tr_dfx(m)%p,cs%tr_dfy(m)%p)
415 
416  call register_z_tracer(cs%tr(:,:,:,m), trim(name), longname, units, &
417  day, g, diag_to_z_csp)
418  enddo
419 
420 end subroutine initialize_oil_tracer

422 subroutine oil_tracer_column_physics(h_old, h_new, ea, eb, fluxes, dt, G, GV, CS, tv)
423  type(ocean_grid_type), intent(in) :: G
424  type(verticalgrid_type), intent(in) :: GV
425  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h_old, h_new, ea, eb
426  type(forcing), intent(in) :: fluxes
427  real, intent(in) :: dt
428  type(oil_tracer_cs), pointer :: CS
429  type(thermo_var_ptrs), intent(in) :: tv
430 ! This subroutine applies diapycnal diffusion and any other column
431 ! tracer physics or chemistry to the tracers from this file.
432 ! This is a simple example of a set of advected passive tracers.
433 
434 ! Arguments: h_old - Layer thickness before entrainment, in m or kg m-2.
435 ! (in) h_new - Layer thickness after entrainment, in m or kg m-2.
436 ! (in) ea - an array to which the amount of fluid entrained
437 ! from the layer above during this call will be
438 ! added, in m or kg m-2.
