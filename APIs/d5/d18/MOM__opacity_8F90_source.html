23 !* *
24 !* This module contains the routines used to calculate the opacity *
25 !* of the ocean. *
26 !* *
27 !* CHL_from_file: *
28 !* In this routine, the Morel (modified) and Manizza (modified) *
29 !* schemes use the "blue" band in the paramterizations to determine *
30 !* the e-folding depth of the incoming shortwave attenuation. The red *
31 !* portion is lumped into the net heating at the surface. *
32 !* *
33 !* Morel, A., 1988: Optical modeling of the upper ocean in relation *
34 !* to itsbiogenous matter content (case-i waters)., J. Geo. Res., *
35 !* 93, 10,749-10,768. *
36 !* *
37 !* Manizza, M., C. LeQuere, A. J. Watson, and E. T. Buitenhuis, 2005: *
38 !* Bio-optical feedbacks amoung phytoplankton, upper ocean physics *
39 !* and sea-ice in a global model, Geophys. Res. Let., 32, L05603, *
40 !* doi:10.1029/2004GL020778. *
41 !* *
42 !* A small fragment of the grid is shown below: *
43 !* *
44 !* j+1 x ^ x ^ x At x: q *
45 !* j+1 > o > o > At ^: v *
46 !* j x ^ x ^ x At >: u *
47 !* j > o > o > At o: h, buoy, Rml, eaml, ebml, etc. *
48 !* j-1 x ^ x ^ x *
49 !* i-1 i i+1 At x & ^: *
50 !* i i+1 At > & o: *
51 !* *
52 !* The boundaries always run through q grid points (x). *
53 !* *
54 !********+*********+*********+*********+*********+*********+*********+**

75 type, public :: opacity_cs ; private

78  ! depth of incoming short wave radiation.

81  ! used to translate water properties into the
82  ! opacity (i.e., the e-folding depth) and (perhaps)
83  ! the number of bands of penetrating shortwave
84  ! radiation to use.
85  real :: pen_sw_scale ! The vertical absorption e-folding depth of the
86  ! penetrating shortwave radiation, in m.
87  real :: pen_sw_frac ! The fraction of shortwave radiation that is
88  ! penetrating with a constant e-folding approach.
89  real :: blue_frac ! The fraction of the penetrating shortwave

98  type(time_type), pointer :: time ! A pointer to the ocean model's clock.
99  type(diag_ctrl), pointer :: diag ! A structure that is used to regulate the
100  ! timing of diagnostic output.

102  ! A pointer to the control structure of the tracer modules.
103 
104  integer :: id_sw_pen = -1, id_sw_vis_pen = -1, id_chl = -1
105  integer, pointer :: id_opacity(:) => null()
106 end type opacity_cs

115 subroutine set_opacity(optics, fluxes, G, GV, CS)
116  type(optics_type), intent(inout) :: optics
117  type(forcing), intent(in) :: fluxes
118  type(ocean_grid_type), intent(in) :: G
119  type(verticalgrid_type), intent(in) :: GV
120  type(opacity_cs), pointer :: CS
121 ! Arguments: (inout) opacity - The inverse of the vertical absorption decay
122 ! scale for penetrating shortwave radiation, in m-1.
123 ! (inout) fluxes - A structure containing pointers to any possible
124 ! forcing fields. Unused fields have NULL ptrs.
125 ! (in) G - The ocean's grid structure.
126 ! (in) GV - The ocean's vertical grid structure.
127 ! (in) CS - The control structure earlier set up by opacity_init.
128 
129 ! local variables
130  integer :: i, j, k, n, is, ie, js, je, nz
131  real :: inv_sw_pen_scale ! The inverse of the e-folding scale, in m-1.
132  real :: Inv_nbands ! The inverse of the number of bands of penetrating
133  ! shortwave radiation.
134  logical :: call_for_surface ! if horizontal slice is the surface layer
135  real :: tmp(szi_(g),szj_(g),szk_(g)) ! A 3-d temporary array.
136  real :: chl(szi_(g),szj_(g),szk_(g)) ! The concentration of chlorophyll-A
137  ! in mg m-3.
138  real :: Pen_SW_tot(szi_(g),szj_(g)) ! The penetrating shortwave radiation
139  ! summed across all bands, in W m-2.
140  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
141 
142  if (.not. associated(cs)) call mom_error(fatal, "set_opacity: "// &
143  "Module must be initialized via opacity_init before it is used.")
144 
145  if (cs%var_pen_sw) then
146  if (cs%chl_from_file) then
147  call opacity_from_chl(optics, fluxes, g, cs)
148  else
149  call get_chl_from_model(chl, g, cs%tracer_flow_CSp)
150  call opacity_from_chl(optics, fluxes, g, cs, chl)
151  endif
152  else ! Use sw e-folding scale set by MOM_input
153  if (optics%nbands <= 1) then ; inv_nbands = 1.0
154  else ; inv_nbands = 1.0 / real(optics%nbands) ; endif
155 
156  ! Make sure there is no division by 0.
157  inv_sw_pen_scale = 1.0 / max(cs%pen_sw_scale, 0.1*gv%Angstrom_z, &
158  gv%H_to_m*gv%H_subroundoff)
159 !$OMP parallel default(none) shared(is,ie,js,je,nz,optics,inv_sw_pen_scale,fluxes,CS,Inv_nbands)
160 !$OMP do
161  do k=1,nz ; do j=js,je ; do i=is,ie ; do n=1,optics%nbands
162  optics%opacity_band(n,i,j,k) = inv_sw_pen_scale
163  enddo ; enddo ; enddo ; enddo
164  if (.not.associated(fluxes%sw) .or. (cs%pen_SW_scale <= 0.0)) then
165 !$OMP do
166  do j=js,je ; do i=is,ie ; do n=1,optics%nbands
167  optics%sw_pen_band(n,i,j) = 0.0
168  enddo ; enddo ; enddo
169  else
170 !$OMP do
171  do j=js,je ; do i=is,ie ; do n=1,optics%nbands
172  optics%sw_pen_band(n,i,j) = cs%pen_SW_frac * inv_nbands * fluxes%sw(i,j)
173  enddo ; enddo ; enddo
174  endif
175 !$OMP end parallel
176  endif
177 
178  if (query_averaging_enabled(cs%diag)) then
179  if (cs%id_sw_pen > 0) then
180 !$OMP parallel do default(none) shared(is,ie,js,je,Pen_SW_tot,optics)
181  do j=js,je ; do i=is,ie
182  pen_sw_tot(i,j) = 0.0
183  do n=1,optics%nbands
184  pen_sw_tot(i,j) = pen_sw_tot(i,j) + optics%sw_pen_band(n,i,j)
185  enddo
186  enddo ; enddo
187  call post_data(cs%id_sw_pen, pen_sw_tot, cs%diag)
188  endif
189  if (cs%id_sw_vis_pen > 0) then
190  if (cs%opacity_scheme == manizza_05) then
191 !$OMP parallel do default(none) shared(is,ie,js,je,Pen_SW_tot,optics)
192  do j=js,je ; do i=is,ie
193  pen_sw_tot(i,j) = 0.0
194  do n=1,min(optics%nbands,2)
195  pen_sw_tot(i,j) = pen_sw_tot(i,j) + optics%sw_pen_band(n,i,j)
196  enddo
197  enddo ; enddo
198  else
199 !$OMP parallel do default(none) shared(is,ie,js,je,Pen_SW_tot,optics)
200  do j=js,je ; do i=is,ie
201  pen_sw_tot(i,j) = 0.0
202  do n=1,optics%nbands
203  pen_sw_tot(i,j) = pen_sw_tot(i,j) + optics%sw_pen_band(n,i,j)
204  enddo
205  enddo ; enddo
206  endif
207  call post_data(cs%id_sw_vis_pen, pen_sw_tot, cs%diag)
208  endif
209  do n=1,optics%nbands ; if (cs%id_opacity(n) > 0) then
210 !$OMP parallel do default(none) shared(nz,is,ie,js,je,tmp,optics,n)
211  do k=1,nz ; do j=js,je ; do i=is,ie
212  tmp(i,j,k) = optics%opacity_band(n,i,j,k)
213  enddo ; enddo ; enddo
214  call post_data(cs%id_opacity(n), tmp, cs%diag)
215  endif ; enddo
216  endif
217 
218 end subroutine set_opacity

221 subroutine opacity_from_chl(optics, fluxes, G, CS, chl_in)
222  type(optics_type), intent(inout) :: optics
223  type(forcing), intent(in) :: fluxes
224  type(ocean_grid_type), intent(in) :: G
225  type(opacity_cs), pointer :: CS
226  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in), optional :: chl_in
227 ! Arguments: fluxes - A structure containing pointers to any possible
228 ! forcing fields. Unused fields have NULL ptrs.
229 ! (out) opacity - The inverse of the vertical absorption decay
230 ! scale for penetrating shortwave radiation, in m-1.
231 ! (in) G - The ocean's grid structure.
232 ! (in) chl_in - A 3-d field of chlorophyll A, in mg m-3.
233 
234  real :: chl_data(szi_(g),szj_(g)) ! The chlorophyll A concentrations in
235  ! a layer, in mg/m^3.
236  real :: Inv_nbands ! The inverse of the number of bands of penetrating
237  ! shortwave radiation.
238  real :: Inv_nbands_nir ! The inverse of the number of bands of penetrating
239  ! near-infrafed radiation.
240  real :: SW_pen_tot ! The sum across the bands of the penetrating
241  ! shortwave radiation, in W m-2.
242  real :: SW_vis_tot ! The sum across the visible bands of shortwave
243  ! radiation, in W m-2.
244  real :: SW_nir_tot ! The sum across the near infrared bands of shortwave
245  ! radiation, in W m-2.
246  type(time_type) :: day
247  character(len=128) :: mesg
248  integer :: days, seconds
249  integer :: i, j, k, n, is, ie, js, je, nz, nbands
250  logical :: multiband_vis_input, multiband_nir_input
251 
252  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
253 
254 ! In this model, the Morel (modified) and Manizza (modified) schemes
255 ! use the "blue" band in the paramterizations to determine the e-folding
256 ! depth of the incoming shortwave attenuation. The red portion is lumped
257 ! into the net heating at the surface.
258 !
259 ! Morel, A., Optical modeling of the upper ocean in relation to its biogenous
260 ! matter content (case-i waters).,J. Geo. Res., {93}, 10,749--10,768, 1988.
261 !
262 
263 ! Manizza, M., C.~L. Quere, A.~Watson, and E.~T. Buitenhuis, Bio-optical
264 ! feedbacks amoung phytoplankton, upper ocean physics and sea-ice in a
265 ! global model, Geophys. Res. Let., , L05,603, 2005.
266 
267  nbands = optics%nbands
268 
269  if (nbands <= 1) then ; inv_nbands = 1.0
270  else ; inv_nbands = 1.0 / real(nbands) ; endif
271 
272  if (nbands <= 2) then ; inv_nbands_nir = 0.0
273  else ; inv_nbands_nir = 1.0 / real(nbands - 2.0) ; endif
274 
275  multiband_vis_input = (associated(fluxes%sw_vis_dir) .and. &
276  associated(fluxes%sw_vis_dif))
277  multiband_nir_input = (associated(fluxes%sw_nir_dir) .and. &
278  associated(fluxes%sw_nir_dif))
279 
280  chl_data(:,:) = 0.0
281  if(present(chl_in)) then
282  do j=js,je ; do i=is,ie ; chl_data(i,j) = chl_in(i,j,1) ; enddo ; enddo
283  do k=1,nz; do j=js,je ; do i=is,ie
284  if ((g%mask2dT(i,j) > 0.5) .and. (chl_in(i,j,k) < 0.0)) then
285  write(mesg,'(" Negative chl_in of ",(1pe12.4)," found at i,j,k = ", &
286  & 3(1x,i3), " lon/lat = ",(1pe12.4)," E ", (1pe12.4), " N.")') &
287  chl_in(i,j,k), i, j, k, g%geoLonT(i,j), g%geoLatT(i,j)
288  call mom_error(fatal,"MOM_opacity opacity_from_chl: "//trim(mesg))
289  endif
290  enddo; enddo; enddo
291  else
292  ! Only the 2-d surface chlorophyll can be read in from a file. The
293  ! same value is assumed for all layers.
294  call get_time(cs%Time,seconds,days)
295  call time_interp_external(cs%sbc_chl, cs%Time, chl_data)
296  do j=js,je ; do i=is,ie
297  if ((g%mask2dT(i,j) > 0.5) .and. (chl_data(i,j) < 0.0)) then
298  write(mesg,'(" Time_interp negative chl of ",(1pe12.4)," at i,j = ",&
299  & 2(i3), "lon/lat = ",(1pe12.4)," E ", (1pe12.4), " N.")') &
300  chl_data(i,j), i, j, g%geoLonT(i,j), g%geoLatT(i,j)
301  call mom_error(fatal,"MOM_opacity opacity_from_chl: "//trim(mesg))
302  endif
303  enddo ; enddo
304  endif
305 
306  if (cs%id_chl > 0) then
307  if(present(chl_in)) then
308  call post_data(cs%id_chl, chl_in(:,:,1), cs%diag)
309  else
310  call post_data(cs%id_chl, chl_data, cs%diag)
311  endif
312  endif
313 
314  select case (cs%opacity_scheme)
315  case (manizza_05)
316 !$OMP parallel do default(none) shared(is,ie,js,je,fluxes,optics,CS,G,multiband_nir_input, &
317 !$OMP nbands,Inv_nbands_nir,multiband_vis_input ) &
318 !$OMP private(SW_vis_tot,SW_nir_tot)
319  do j=js,je ; do i=is,ie
320  sw_vis_tot = 0.0 ; sw_nir_tot = 0.0
321  if (g%mask2dT(i,j) > 0.5) then
322  if (multiband_vis_input) then
323  sw_vis_tot = fluxes%sw_vis_dir(i,j) + fluxes%sw_vis_dif(i,j)
324  else ! Follow Manizza 05 in assuming that 42% of SW is visible.
325  sw_vis_tot = 0.42 * fluxes%sw(i,j)
326  endif
327  if (multiband_nir_input) then
328  sw_nir_tot = fluxes%sw_nir_dir(i,j) + fluxes%sw_nir_dif(i,j)
329  else
330  sw_nir_tot = fluxes%sw(i,j) - sw_vis_tot
331  endif
332  endif
333 
334  ! Band 1 is Manizza blue.
335  optics%sw_pen_band(1,i,j) = cs%blue_frac*sw_vis_tot
336  ! Band 2 (if used) is Manizza red.
337  if (nbands > 1) &
338  optics%sw_pen_band(2,i,j) = (1.0-cs%blue_frac)*sw_vis_tot
339  ! All remaining bands are NIR, for lack of something better to do.
340  do n=3,nbands
341  optics%sw_pen_band(n,i,j) = inv_nbands_nir * sw_nir_tot
342  enddo
343  enddo ; enddo
344  case (morel_88)
345 !$OMP parallel do default(none) shared(is,ie,js,je,G,multiband_vis_input,chl_data, &
346 !$OMP fluxes,nbands,optics,Inv_nbands) &
347 !$OMP private(SW_pen_tot)
348  do j=js,je ; do i=is,ie
349  sw_pen_tot = 0.0
350  if (g%mask2dT(i,j) > 0.5) then ; if (multiband_vis_input) then
351  sw_pen_tot = sw_pen_frac_morel(chl_data(i,j)) * &
352  (fluxes%sw_vis_dir(i,j) + fluxes%sw_vis_dif(i,j))
353  else
354  sw_pen_tot = sw_pen_frac_morel(chl_data(i,j)) * &
355  0.5*fluxes%sw(i,j)
356  endif ; endif
357 
358  do n=1,nbands
359  optics%sw_pen_band(n,i,j) = inv_nbands*sw_pen_tot
360  enddo
361  enddo ; enddo
362  case default
363  call mom_error(fatal, "opacity_from_chl: CS%opacity_scheme is not valid.")
364  end select
365 
366 !$OMP parallel do default(none) shared(nz,is,ie,js,je,CS,G,chl_in,optics,nbands) &
367 !$OMP firstprivate(chl_data)
368  do k=1,nz
369  if (present(chl_in)) then
370  do j=js,je ; do i=is,ie ; chl_data(i,j) = chl_in(i,j,k) ; enddo ; enddo
371  endif
372 
373  select case (cs%opacity_scheme)
374  case (manizza_05)
375  do j=js,je ; do i=is,ie
376  if (g%mask2dT(i,j) <= 0.5) then
377  do n=1,optics%nbands
378  optics%opacity_band(n,i,j,k) = cs%opacity_land_value
379  enddo
380  else
381  ! Band 1 is Manizza blue.
382  optics%opacity_band(1,i,j,k) = 0.0232 + 0.074*chl_data(i,j)**0.674
383  if (nbands >= 2) & ! Band 2 is Manizza red.
384  optics%opacity_band(2,i,j,k) = 0.225 + 0.037*chl_data(i,j)**0.629
385  ! All remaining bands are NIR, for lack of something better to do.
386  do n=3,nbands ; optics%opacity_band(n,i,j,k) = 2.86 ; enddo
387  endif
388  enddo ; enddo
389  case (morel_88)
390  do j=js,je ; do i=is,ie
391  optics%opacity_band(1,i,j,k) = cs%opacity_land_value
392  if (g%mask2dT(i,j) > 0.5) &
393  optics%opacity_band(1,i,j,k) = opacity_morel(chl_data(i,j))
394 
395  do n=2,optics%nbands
396  optics%opacity_band(n,i,j,k) = optics%opacity_band(1,i,j,k)
397  enddo
398  enddo ; enddo
399 
400  case default
401  call mom_error(fatal, "opacity_from_chl: CS%opacity_scheme is not valid.")
402  end select
403  enddo
404 
405 
406 end subroutine opacity_from_chl

408 function opacity_morel(chl_data)
409  real, intent(in) :: chl_data
410  real :: opacity_morel
411 ! Argument : chl_data - The chlorophyll-A concentration in mg m-3.
412 ! The following are coefficients for the optical model taken from Morel and
413 ! Antoine (1994). These coeficients represent a non uniform distribution of
414 ! chlorophyll-a through the water column. Other approaches may be more
415 ! appropriate when using an interactive ecosystem model that predicts
416 ! three-dimensional chl-a values.
417  real, dimension(6), parameter :: &
418  Z2_coef=(/7.925, -6.644, 3.662, -1.815, -0.218, 0.502/)
419  real :: Chl, Chl2 ! The log10 of chl_data (in mg m-3), and Chl^2.
420 
421  chl = log10(min(max(chl_data,0.02),60.0)) ; chl2 = chl*chl
422  opacity_morel = 1.0 / ( (z2_coef(1) + z2_coef(2)*chl) + chl2 * &
423  ((z2_coef(3) + chl*z2_coef(4)) + chl2*(z2_coef(5) + chl*z2_coef(6))) )
424 end function
425 
426 function sw_pen_frac_morel(chl_data)
427  real, intent(in) :: chl_data
428  real :: SW_pen_frac_morel
429 ! Argument : chl_data - The chlorophyll-A concentration in mg m-3.
430 ! The following are coefficients for the optical model taken from Morel and
431 ! Antoine (1994). These coeficients represent a non uniform distribution of
432 ! chlorophyll-a through the water column. Other approaches may be more
433 ! appropriate when using an interactive ecosystem model that predicts
434 ! three-dimensional chl-a values.
435  real :: Chl, Chl2 ! The log10 of chl_data in mg m-3, and Chl^2.
436  real, dimension(6), parameter :: &
437  V1_coef=(/0.321, 0.008, 0.132, 0.038, -0.017, -0.007/)
438 
439  chl = log10(min(max(chl_data,0.02),60.0)) ; chl2 = chl*chl
440  sw_pen_frac_morel = 1.0 - ( (v1_coef(1) + v1_coef(2)*chl) + chl2 * &
441  ((v1_coef(3) + chl*v1_coef(4)) + chl2*(v1_coef(5) + chl*v1_coef(6))) )
442 end function sw_pen_frac_morel

444 function opacity_manizza(chl_data)
445  real, intent(in) :: chl_data
446  real :: opacity_manizza
447 ! Argument : chl_data - The chlorophyll-A concentration in mg m-3.
448 ! This sets the blue-wavelength opacity according to the scheme proposed by
449 ! Manizza, M. et al, 2005.
450 
451  opacity_manizza = 0.0232 + 0.074*chl_data**0.674
452 end function

454 subroutine opacity_init(Time, G, param_file, diag, tracer_flow, CS, optics)
455  type(time_type), target, intent(in) :: Time
456  type(ocean_grid_type), intent(in) :: G
