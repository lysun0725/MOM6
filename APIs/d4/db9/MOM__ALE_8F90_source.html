364 !> Takes care of (1) building a new grid and (2) remapping all variables between
365 !! the old grid and the new grid. The creation of the new grid can be based
366 !! on z coordinates, target interface densities, sigma coordinates or any
367 !! arbitrary coordinate system.
368 subroutine ale_main( G, GV, h, u, v, tv, Reg, CS, dt)
369  type(ocean_grid_type), intent(in) :: G !< Ocean grid informations
370  type(verticalgrid_type), intent(in) :: GV !< Ocean vertical grid structure
371  real, dimension(SZI_(G),SZJ_(G),SZK_(GV)), intent(inout) :: h !< Current 3D grid obtained after last time step (m or Pa)
372  real, dimension(SZIB_(G),SZJ_(G),SZK_(GV)), intent(inout) :: u !< Zonal velocity field (m/s)
373  real, dimension(SZI_(G),SZJB_(G),SZK_(GV)), intent(inout) :: v !< Meridional velocity field (m/s)
374  type(thermo_var_ptrs), intent(inout) :: tv !< Thermodynamic variable structure
375  type(tracer_registry_type), pointer :: Reg !< Tracer registry structure
376  type(ale_cs), pointer :: CS !< Regridding parameters and options
377  real, optional, intent(in) :: dt !< Time step between calls to ALE_main()
378 
379  ! Local variables
380  real, dimension(SZI_(G), SZJ_(G), SZK_(GV)+1) :: dzRegrid ! The change in grid interface positions
381  real, dimension(SZI_(G),SZJ_(G),SZK_(GV)) :: h_new ! New 3D grid obtained after last time step (m or Pa)
382  integer :: nk, i, j, k, isc, iec, jsc, jec
383 
384  nk = gv%ke; isc = g%isc; iec = g%iec; jsc = g%jsc; jec = g%jec
385 
386  if (cs%show_call_tree) call calltree_enter("ALE_main(), MOM_ALE.F90")
387 
388  if (present(dt)) then
389  call ale_update_regrid_weights( dt, cs )
390  endif
391  dzregrid(:,:,:) = 0.0
392 
393  ! Build new grid. The new grid is stored in h_new. The old grid is h.
394  ! Both are needed for the subsequent remapping of variables.
395  call regridding_main( cs%remapCS, cs%regridCS, g, gv, h, tv, h_new, dzregrid )
396 
397  call check_grid( g, gv, h, 0. )
398 
399  if (cs%show_call_tree) call calltree_waypoint("new grid generated (ALE_main)")
400 
401  ! Remap all variables from old grid h onto new grid h_new
402  call remap_all_state_vars( cs%remapCS, cs, g, gv, h, h_new, -dzregrid, reg, &
403  u, v, cs%show_call_tree, dt )
404 
405  if (cs%show_call_tree) call calltree_waypoint("state remapped (ALE_main)")
406 
407  ! Override old grid with new one. The new grid 'h_new' is built in
408  ! one of the 'build_...' routines above.
409 !$OMP parallel do default(none) shared(isc,iec,jsc,jec,nk,h,h_new,CS)
410  do k = 1,nk
411  do j = jsc-1,jec+1 ; do i = isc-1,iec+1
412  h(i,j,k) = h_new(i,j,k)
413  enddo ; enddo
414  enddo
415 
416  if (cs%show_call_tree) call calltree_leave("ALE_main()")
417 end subroutine ale_main

499  integer :: nz, ntr

610  enddo
611  enddo
612  enddo

616  endif
617  endif

640  endif
641  call remapping_core_h( nz, h1, u(i,j,:), nz, h2, u_column, cs_remapping )
642  u(i,j,:) = u_column(:)
643  endif
644  enddo
645  enddo
646  endif

669  enddo
670  enddo
671  endif
672 !$OMP end parallel
673 
674  if (show_call_tree) call calltree_waypoint("v remapped (remap_all_state_vars)")

1261 !> Parses a string and generates a dz(:) profile
1262 subroutine dz_function1( string, dz )
1263  character(len=*), intent(in) :: string !< String with list of parameters
1264  real, dimension(:), intent(inout) :: dz !< Profile of nominal thicknesses
1265 
1266  ! Local variables
1267  integer :: nk, k
1268  real :: dz_min, power, prec, H_total
1269 
1270  nk = size(dz) ! Number of cells
1271  prec = -1024.
1272  read( string, *) dz_min, h_total, power, prec
1273  if (prec == -1024.) call mom_error(fatal,"dz_function1: "// &
1274  "Problem reading FNC1: string ="//trim(string))
1275  ! Create profile of ( dz - dz_min )
1276  do k = 1, nk
1277  dz(k) = (real(k-1)/real(nk-1))**power
1278  enddo
1279  dz(:) = ( h_total - real(nk) * dz_min ) * ( dz(:) / sum(dz) ) ! Rescale to so total is H_total
1280  dz(:) = anint( dz(:) / prec ) * prec ! Rounds to precision prec
1281  dz(:) = ( h_total - real(nk) * dz_min ) * ( dz(:) / sum(dz) ) ! Rescale to so total is H_total
1282  dz(:) = anint( dz(:) / prec ) * prec ! Rounds to precision prec
1283  dz(nk) = dz(nk) + ( h_total - sum( dz(:) + dz_min ) ) ! Adjust bottommost layer
1284  dz(:) = anint( dz(:) / prec ) * prec ! Rounds to precision prec
1285  dz(:) = dz(:) + dz_min ! Finally add in the constant dz_min
1286 
1287 end subroutine dz_function1
