213 subroutine convert_iob_to_fluxes(IOB, fluxes, index_bounds, Time, G, CS, state, restore_salt)
214  type(ice_ocean_boundary_type), intent(in), target :: IOB
215  type(forcing), intent(inout) :: fluxes
216  integer, dimension(4), intent(in) :: index_bounds
217  type(time_type), intent(in) :: Time
218  type(ocean_grid_type), intent(inout) :: G
219  type(surface_forcing_cs), pointer :: CS
220  type(surface), intent(in) :: state
221  logical, optional, intent(in) :: restore_salt
222 
223 ! This subroutine translates the Ice_ocean_boundary_type into a
224 ! MOM forcing type, including changes of units, sign conventions,
225 ! and puting the fields into arrays with MOM-standard halos.
226 
227 ! Arguments:
228 ! IOB ice-ocean boundary type w/ fluxes to drive ocean in a coupled model
229 ! (out) fluxes - A structure containing pointers to any possible
230 ! forcing fields. Unused fields have NULL ptrs.
231 ! (in) index_bounds - the i- and j- size of the arrays in IOB.
232 ! (in) Time - The time of the fluxes, used for interpolating the salinity
233 ! to the right time, when it is being restored.
234 ! (in) G - The ocean's grid structure.
235 ! (in) CS - A pointer to the control structure returned by a previous
236 ! call to surface_forcing_init.
237 ! (in) state - A structure containing fields that describe the
238 ! surface state of the ocean.
239 ! (in) restore_salt - if true, salinity is restored to a target value.
240 
241  real, dimension(SZIB_(G),SZJB_(G)) :: &
242  taux_at_q, & ! Zonal wind stresses at q points (Pa)
243  tauy_at_q ! Meridional wind stresses at q points (Pa)
244 
245  real, dimension(SZI_(G),SZJ_(G)) :: &
246  taux_at_h, & ! Zonal wind stresses at h points (Pa)
247  tauy_at_h, & ! Meridional wind stresses at h points (Pa)
248  data_srestore, & ! The surface salinity toward which to restore (g/kg)
249  SST_anom, & ! Instantaneous sea surface temperature anomalies from a target value (deg C)
250  SSS_anom, & ! Instantaneous sea surface salinity anomalies from a target value (g/kg)
251  SSS_mean, & ! A (mean?) salinity about which to normalize local salinity
252  ! anomalies when calculating restorative precipitation anomalies (g/kg)
253  pme_adj, & ! The adjustment to PminusE that will cause the salinity
254  ! to be restored toward its target value (kg/(m^2 * s))
255  net_fw, & ! The area integrated net freshwater flux into the ocean (kg/s)
256  net_fw2, & ! The area integrated net freshwater flux into the ocean (kg/s)
257  work_sum, & ! A 2-d array that is used as the work space for a global
258  ! sum, used with units of m2 or (kg/s)
259  open_ocn_mask ! a binary field indicating where ice is present based on frazil criteria
260 
261  real :: gustiness ! unresolved gustiness that contributes to ustar (Pa)
262  real :: Irho0 ! inverse of the mean density in (m^3/kg)
263  real :: taux2, tauy2 ! squared wind stresses (Pa^2)
264  real :: tau_mag ! magnitude of the wind stress (Pa)
265  real :: I_GEarth ! 1.0 / G%G_Earth (s^2/m)
266  real :: Kv_rho_ice ! (CS%kv_sea_ice / CS%density_sea_ice) ( m^5/(s*kg) )
267  real :: mass_ice ! mass of sea ice at a face (kg/m^2)
268  real :: mass_eff ! effective mass of sea ice for rigidity (kg/m^2)
269 
270  integer :: wind_stagger ! AGRID, BGRID_NE, or CGRID_NE (integers from MOM_domains)
271  integer :: i, j, is, ie, js, je, Isq, Ieq, Jsq, Jeq, i0, j0
272  integer :: isd, ied, jsd, jed, IsdB, IedB, JsdB, JedB, isr, ier, jsr, jer
273  integer :: isc_bnd, iec_bnd, jsc_bnd, jec_bnd
274 
275  logical :: restore_salinity ! local copy of the argument restore_salt, if it
276  ! is present, or false (no restoring) otherwise.
277  real :: delta_sss ! temporary storage for sss diff from restoring value
278 
279  real :: C_p ! heat capacity of seawater ( J/(K kg) )
280 
281  call cpu_clock_begin(id_clock_forcing)
282 
283  isc_bnd = index_bounds(1) ; iec_bnd = index_bounds(2)
284  jsc_bnd = index_bounds(3) ; jec_bnd = index_bounds(4)
285  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
286  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
287  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
288  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
289  isr = is-isd+1 ; ier = ie-isd+1 ; jsr = js-jsd+1 ; jer = je-jsd+1
290 
291  c_p = fluxes%C_p
292  irho0 = 1.0/cs%Rho0
293  open_ocn_mask(:,:) = 1.0
294  pme_adj(:,:) = 0.0
295  fluxes%vPrecGlobalAdj = 0.0
296  fluxes%vPrecGlobalScl = 0.0
297  fluxes%saltFluxGlobalAdj = 0.0
298  fluxes%saltFluxGlobalScl = 0.0
299  fluxes%netFWGlobalAdj = 0.0
300  fluxes%netFWGlobalScl = 0.0
301 
302  restore_salinity = .false.
303  if (present(restore_salt)) restore_salinity = restore_salt
304 
305  ! allocation and initialization if this is the first time that this
306  ! flux type has been used.
307  if (fluxes%dt_buoy_accum < 0) then
308  call allocate_forcing_type(g, fluxes, stress=.true., ustar=.true., &
309  water=.true., heat=.true.)
310 
311  call safe_alloc_ptr(fluxes%sw_vis_dir,isd,ied,jsd,jed)
312  call safe_alloc_ptr(fluxes%sw_vis_dif,isd,ied,jsd,jed)
313  call safe_alloc_ptr(fluxes%sw_nir_dir,isd,ied,jsd,jed)
314  call safe_alloc_ptr(fluxes%sw_nir_dif,isd,ied,jsd,jed)
315 
316  call safe_alloc_ptr(fluxes%p_surf,isd,ied,jsd,jed)
317  call safe_alloc_ptr(fluxes%p_surf_full,isd,ied,jsd,jed)
318 
319  call safe_alloc_ptr(fluxes%salt_flux,isd,ied,jsd,jed)
320  call safe_alloc_ptr(fluxes%salt_flux_in,isd,ied,jsd,jed)
321  call safe_alloc_ptr(fluxes%salt_flux_restore,isd,ied,jsd,jed)
322 
323  call safe_alloc_ptr(fluxes%TKE_tidal,isd,ied,jsd,jed)
324  call safe_alloc_ptr(fluxes%ustar_tidal,isd,ied,jsd,jed)
325 
326  do j=js-2,je+2 ; do i=is-2,ie+2
327  fluxes%TKE_tidal(i,j) = cs%TKE_tidal(i,j)
328  fluxes%ustar_tidal(i,j) = cs%ustar_tidal(i,j)
329  enddo; enddo
330 
331  if (cs%rigid_sea_ice) then
332  call safe_alloc_ptr(fluxes%rigidity_ice_u,isdb,iedb,jsd,jed)
333  call safe_alloc_ptr(fluxes%rigidity_ice_v,isd,ied,jsdb,jedb)
334  endif
335 
336  fluxes%dt_buoy_accum = 0.0
337  endif ! endif for allocation and initialization
338 
339  ! allocation and initialization on first call to this routine
340  if (cs%area_surf < 0.0) then
341  do j=js,je ; do i=is,ie
342  work_sum(i,j) = g%areaT(i,j) * g%mask2dT(i,j)
343  enddo ; enddo
344  cs%area_surf = reproducing_sum(work_sum, isr, ier, jsr, jer)
345  endif ! endif for allocation and initialization
346 
347  do j=js,je ; do i=is,ie
348  fluxes%salt_flux(i,j) = 0.0
349  fluxes%vprec(i,j) = 0.0
350  enddo ; enddo
351 
352  ! Salinity restoring logic
353  if (restore_salinity) then
354  call time_interp_external(cs%id_srestore,time,data_srestore)
355  ! open_ocn_mask indicates where to restore salinity (1 means restore, 0 does not)
356  open_ocn_mask(:,:) = 1.0
357  if (cs%mask_srestore_under_ice) then ! Do not restore under sea-ice
358  do j=js,je ; do i=is,ie
359  if (state%SST(i,j) .le. -0.0539*state%SSS(i,j)) open_ocn_mask(i,j)=0.0
360  enddo; enddo
361  endif
362  if (cs%salt_restore_as_sflux) then
363  do j=js,je ; do i=is,ie
364  delta_sss = data_srestore(i,j)- state%SSS(i,j)
365  delta_sss = sign(1.0,delta_sss)*min(abs(delta_sss),cs%max_delta_srestore)
366  fluxes%salt_flux(i,j) = 1.e-3*g%mask2dT(i,j) * (cs%Rho0*cs%Flux_const)* &
367  (cs%basin_mask(i,j)*open_ocn_mask(i,j)) *delta_sss ! kg Salt m-2 s-1
368  enddo; enddo
369  if (cs%adjust_net_srestore_to_zero) then
370  if (cs%adjust_net_srestore_by_scaling) then
371  call adjust_area_mean_to_zero(fluxes%salt_flux, g, fluxes%saltFluxGlobalScl)
372  fluxes%saltFluxGlobalAdj = 0.
373  else
374  work_sum(is:ie,js:je) = g%areaT(is:ie,js:je)*fluxes%salt_flux(is:ie,js:je)
375  fluxes%saltFluxGlobalAdj = reproducing_sum(work_sum(:,:), isr,ier, jsr,jer)/cs%area_surf
376  fluxes%salt_flux(is:ie,js:je) = fluxes%salt_flux(is:ie,js:je) - fluxes%saltFluxGlobalAdj
377  endif
378  endif
379  fluxes%salt_flux_restore(is:ie,js:je) = fluxes%salt_flux(is:ie,js:je) ! Diagnostic
380  else
381  do j=js,je ; do i=is,ie
382  if (g%mask2dT(i,j) > 0.5) then
383  delta_sss = state%SSS(i,j) - data_srestore(i,j)
384  delta_sss = sign(1.0,delta_sss)*min(abs(delta_sss),cs%max_delta_srestore)
385  fluxes%vprec(i,j) = (cs%basin_mask(i,j)*open_ocn_mask(i,j))* &
386  (cs%Rho0*cs%Flux_const) * &
387  delta_sss / (0.5*(state%SSS(i,j) + data_srestore(i,j)))
388  endif
389  enddo; enddo
390  if (cs%adjust_net_srestore_to_zero) then
391  if (cs%adjust_net_srestore_by_scaling) then
392  call adjust_area_mean_to_zero(fluxes%vprec, g, fluxes%vPrecGlobalScl)
393  fluxes%vPrecGlobalAdj = 0.
394  else
395  work_sum(is:ie,js:je) = g%areaT(is:ie,js:je)*fluxes%vprec(is:ie,js:je)
396  fluxes%vPrecGlobalAdj = reproducing_sum(work_sum(:,:), isr, ier, jsr, jer) / cs%area_surf
397  do j=js,je ; do i=is,ie
398  fluxes%vprec(i,j) = ( fluxes%vprec(i,j) - fluxes%vPrecGlobalAdj ) * g%mask2dT(i,j)
399  enddo; enddo
400  endif
401  endif
402  endif
403  endif
404 
405  wind_stagger = cs%wind_stagger
406  if ((iob%wind_stagger == agrid) .or. (iob%wind_stagger == bgrid_ne) .or. &
407  (iob%wind_stagger == cgrid_ne)) wind_stagger = iob%wind_stagger
408  if (wind_stagger == bgrid_ne) then
409  ! This is necessary to fill in the halo points.
410  taux_at_q(:,:) = 0.0 ; tauy_at_q(:,:) = 0.0
411  endif
412  if (wind_stagger == agrid) then
413  ! This is necessary to fill in the halo points.
414  taux_at_h(:,:) = 0.0 ; tauy_at_h(:,:) = 0.0
415  endif
416 
417 
418  ! obtain fluxes from IOB; note the staggering of indices
419  i0 = is - isc_bnd ; j0 = js - jsc_bnd
420  do j=js,je ; do i=is,ie
421 
422  if (wind_stagger == bgrid_ne) then
423  if (ASSOCIATED(iob%u_flux)) taux_at_q(i,j) = iob%u_flux(i-i0,j-j0) * cs%wind_stress_multiplier
424  if (ASSOCIATED(iob%v_flux)) tauy_at_q(i,j) = iob%v_flux(i-i0,j-j0) * cs%wind_stress_multiplier
425  elseif (wind_stagger == agrid) then
426  if (ASSOCIATED(iob%u_flux)) taux_at_h(i,j) = iob%u_flux(i-i0,j-j0) * cs%wind_stress_multiplier

697 !> Adds flux adjustments obtained via data_override
698 !! Component name is 'OCN'
699 !! Available adjustments are:
700 !! - taux_adj (Zonal wind stress delta, positive to the east, in Pa)
701 !! - tauy_adj (Meridional wind stress delta, positive to the north, in Pa)
702 subroutine apply_flux_adjustments(G, CS, Time, fluxes)
703  type(ocean_grid_type), intent(inout) :: G !< Ocean grid structure
704  type(surface_forcing_cs), pointer :: CS !< Surface forcing control structure
705  type(time_type), intent(in) :: Time !< Model time structure
706  type(forcing), optional, intent(inout) :: fluxes !< Surface fluxes structure
707  ! Local variables
708  real, dimension(SZI_(G),SZJ_(G)) :: tempx_at_h ! Delta to zonal wind stress at h points (Pa)
709  real, dimension(SZI_(G),SZJ_(G)) :: tempy_at_h ! Delta to meridional wind stress at h points (Pa)
710  integer :: is_in, ie_in, js_in, je_in, i, j
711  real :: dLonDx, dLonDy, rDlon, cosA, sinA, zonal_tau, merid_tau
712  logical :: overrode_x, overrode_y
713 
714  is_in = g%isc - g%isd + 1
715  ie_in = g%iec - g%isd + 1
716  js_in = g%jsc - g%jsd + 1
717  je_in = g%jec - g%jsd + 1
718 
719  tempx_at_h(:,:) = 0.0 ; tempy_at_h(:,:) = 0.0
720  ! Either reads data or leaves contents unchanged
721  overrode_x = .false. ; overrode_y = .false.
722  call data_override('OCN', 'taux_adj', tempx_at_h, time, override=overrode_x, &
723  is_in=is_in, ie_in=ie_in, js_in=js_in, je_in=je_in)
724  call data_override('OCN', 'tauy_adj', tempy_at_h, time, override=overrode_y, &
725  is_in=is_in, ie_in=ie_in, js_in=js_in, je_in=je_in)
726 
727  if (overrode_x .or. overrode_y) then
728  if (.not. (overrode_x .and. overrode_y)) call mom_error(fatal,"apply_flux_adjustments: "//&
729  "Both taux_adj and tauy_adj must be specified, or neither, in data_table")
730 
731  ! Rotate winds
732  call pass_vector(tempx_at_h, tempy_at_h, g%Domain, to_all, agrid)
733  do j=g%jsc-1,g%jec+1 ; do i=g%isc-1,g%iec+1
734  dlondx = g%geoLonCu(i,j) - g%geoLonCu(i-1,j)
735  dlondy = g%geoLonCv(i,j) - g%geoLonCv(i,j-1)
736  rdlon = sqrt( dlondx * dlondx + dlondy * dlondy )
737  if (rdlon > 0.) rdlon = 1. / rdlon
738  cosa = dlondx * rdlon
739  sina = dlondy * rdlon
740  zonal_tau = tempx_at_h(i,j)
741  merid_tau = tempy_at_h(i,j)
742  tempx_at_h(i,j) = cosa * zonal_tau - sina * merid_tau
743  tempy_at_h(i,j) = sina * zonal_tau + cosa * merid_tau
744  enddo ; enddo
745 
746  ! Average to C-grid locations
747  do j=g%jsc,g%jec ; do i=g%isc-1,g%iec
748  fluxes%taux(i,j) = fluxes%taux(i,j) + 0.5 * ( tempx_at_h(i,j) + tempx_at_h(i+1,j) )
749  enddo ; enddo
750 
751  do j=g%jsc-1,g%jec ; do i=g%isc,g%iec
752  fluxes%tauy(i,j) = fluxes%tauy(i,j) + 0.5 * ( tempy_at_h(i,j) + tempy_at_h(i,j+1) )
753  enddo ; enddo
754  endif ! overrode_x .or. overrode_y
755 
756 end subroutine apply_flux_adjustments
