27 !* This program contains the subroutines that calculate the *
28 !* surface wind stresses and fluxes of buoyancy or temperature and *
29 !* fresh water. These subroutines will be called every time step, *
30 !* even if the wind stresses or buoyancy fluxes are constant in time *
31 !* - in that case these routines return quickly without doing *
32 !* anything. In addition, any I/O of forcing fields is controlled *
33 !* by surface_forcing_init, located in this file. *
34 !* *
35 !* set_forcing is a small entry subroutine for the subroutines in *
36 !* this file. It provides the external access to these subroutines. *
37 !* *
38 !* wind_forcing determines the wind stresses and places them into *
39 !* taux[][] and tauy[][]. Often wind_forcing must be tailored for *
40 !* a particular application - either by specifying file and variable *
41 !* names or by providing appropriate internal expressions for the *
42 !* stresses. *
43 !* *
44 !* buoyancy_forcing determines the surface fluxes of buoyancy, *
45 !* temperature, and fresh water, as is appropriate. A restoring *
46 !* boundary condition is implemented, but the code for any other *
47 !* boundary condition will usually be modified - either to specify *
48 !* file and variable names and which time level to read, or to set *
49 !* an internal expression for the variables. *
50 !* *
51 !* Macros written all in capital letters are defined in MOM_memory.h. *
52 !* *
53 !* A small fragment of the grid is shown below: *
54 !* *
55 !* j+1 x ^ x ^ x At x: q *
56 !* j+1 > o > o > At ^: v, tauy *
57 !* j x ^ x ^ x At >: u, taux *
58 !* j > o > o > At o: h, fluxes. *
59 !* j-1 x ^ x ^ x *
60 !* i-1 i i+1 At x & ^: *
61 !* i i+1 At > & o: *
62 !* *
63 !* The boundaries always run through q grid points (x). *
64 !* *

178 subroutine set_forcing(state, fluxes, day_start, day_interval, G, CS)
179  type(surface), intent(inout) :: state
180  type(forcing), intent(inout) :: fluxes
181  type(time_type), intent(in) :: day_start
182  type(time_type), intent(in) :: day_interval
183  type(ocean_grid_type), intent(inout) :: G
184  type(surface_forcing_cs), pointer :: CS
185 
186 ! This subroutine calls other subroutines in this file to get surface forcing fields.
187 ! It also allocates and initializes the fields in the flux type.
188 
189 ! Arguments:
190 ! (inout) state = structure describing ocean surface state
191 ! (inout) fluxes = structure with pointers to forcing fields; unused have NULL ptrs
192 ! (in) day_start = Start time of the fluxes
193 ! (in) day_interval = Length of time over which these fluxes applied
194 ! (in) G = ocean grid structure
195 ! (in) CS = pointer to control struct returned by previous surface_forcing_init call
196 
197  real :: dt ! length of time in seconds over which fluxes applied
198  type(time_type) :: day_center ! central time of the fluxes.
199  integer :: intdt
200 
201  call cpu_clock_begin(id_clock_forcing)
202 
203  day_center = day_start + day_interval/2
204  call get_time(day_interval, intdt)
205  dt = real(intdt)
206 
207  if (cs%variable_winds .or. cs%first_call_set_forcing) then
208  if (trim(cs%wind_config) == "file") then
209  call wind_forcing_from_file(state, fluxes, day_center, g, cs)
210  elseif (trim(cs%wind_config) == "2gyre") then
211  call wind_forcing_2gyre(state, fluxes, day_center, g, cs)
212  elseif (trim(cs%wind_config) == "1gyre") then
213  call wind_forcing_1gyre(state, fluxes, day_center, g, cs)
214  elseif (trim(cs%wind_config) == "gyres") then
215  call wind_forcing_gyres(state, fluxes, day_center, g, cs)
216  elseif (trim(cs%wind_config) == "zero") then
217  call wind_forcing_zero(state, fluxes, day_center, g, cs)
218  elseif (trim(cs%wind_config) == "MESO") then
219  call mom_error(fatal, "MESO forcing is not available with the ice-shelf"//&
220  "version of MOM_surface_forcing.")
221 ! call MESO_wind_forcing(state, fluxes, day_center, G, CS%MESO_forcing_CSp)
222  elseif (trim(cs%wind_config) == "USER") then
223  call user_wind_forcing(state, fluxes, day_center, g, cs%user_forcing_CSp)
224  elseif (cs%variable_winds .and. .not.cs%first_call_set_forcing) then
225  call mom_error(fatal, &
226  "MOM_surface_forcing: Variable winds defined with no wind config")
227  else
228  call mom_error(fatal, &
229  "MOM_surface_forcing:Unrecognized wind config "//trim(cs%wind_config))
230  endif
231  endif
232  if ((cs%variable_buoyforce .or. cs%first_call_set_forcing) .and. &
233  (.not.cs%adiabatic)) then
234  if (trim(cs%buoy_config) == "file") then
235  call buoyancy_forcing_from_files(state, fluxes, day_center, dt, g, cs)
236  elseif (trim(cs%buoy_config) == "zero") then
237  call buoyancy_forcing_zero(state, fluxes, day_center, dt, g, cs)
238  elseif (trim(cs%buoy_config) == "linear") then
239  call buoyancy_forcing_linear(state, fluxes, day_center, dt, g, cs)
240  elseif (trim(cs%buoy_config) == "MESO") then
241  call mom_error(fatal, "MESO forcing is not available with the ice-shelf"//&
242  "version of MOM_surface_forcing.")
243 ! call MESO_buoyancy_forcing(state, fluxes, day_center, dt, G, CS%MESO_forcing_CSp)
244  elseif (trim(cs%buoy_config) == "USER") then
245  call user_buoyancy_forcing(state, fluxes, day_center, dt, g, cs%user_forcing_CSp)
246  elseif (trim(cs%buoy_config) == "NONE") then
247  call mom_mesg("MOM_surface_forcing: buoyancy forcing has been set to omitted.")
248  elseif (cs%variable_buoyforce .and. .not.cs%first_call_set_forcing) then
249  call mom_error(fatal, &
250  "MOM_surface_forcing: Variable buoy defined with no buoy config.")
251  else
252  call mom_error(fatal, &
253  "MOM_surface_forcing: Unrecognized buoy config "//trim(cs%buoy_config))
254  endif
255  endif
256  if (associated(cs%tracer_flow_CSp)) then
257  if (.not.associated(fluxes%tr_fluxes)) allocate(fluxes%tr_fluxes)
258  call call_tracer_set_forcing(state, fluxes, day_start, day_interval, g, cs%tracer_flow_CSp)
259  endif
260 
261  ! Allow for user-written code to alter the fluxes after all the above
262  call user_alter_forcing(state, fluxes, day_center, g, cs%urf_CS)
263 
264  cs%first_call_set_forcing = .false.
265 
266  call cpu_clock_end(id_clock_forcing)
267 end subroutine set_forcing

414 subroutine wind_forcing_zero(state, fluxes, day, G, CS)
415  type(surface), intent(inout) :: state
416  type(forcing), intent(inout) :: fluxes
417  type(time_type), intent(in) :: day
418  type(ocean_grid_type), intent(in) :: G
419  type(surface_forcing_cs), pointer :: CS
420 
421 ! subroutine sets the surface wind stresses to zero
422 
423 ! Arguments:
424 ! state = structure describing ocean surface state
425 ! (out) fluxes = structure with pointers to forcing fields; unused have NULL ptrs
426 ! (in) day = time of the fluxes
427 ! (in) G = ocean grid structure
428 ! (in) CS = pointer to control struct returned by previous surface_forcing_init call
429 
430  real :: PI
431  integer :: i, j, is, ie, js, je, Isq, Ieq, Jsq, Jeq
432  integer :: isd, ied, jsd, jed, IsdB, IedB, JsdB, JedB
433 
434  call calltree_enter("wind_forcing_zero, MOM_surface_forcing.F90")
435  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
436  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
437  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
438  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
439 
440  call allocate_forcing_type(g, fluxes, stress=.true., ustar=.true.)
441 
442  !set steady surface wind stresses, in units of Pa.
443  pi = 4.0*atan(1.0)
444 
445  do j=js,je ; do i=isq,ieq
446  fluxes%taux(i,j) = 0.0
447  enddo ; enddo
448 
449  do j=jsq,jeq ; do i=is,ie
450  fluxes%tauy(i,j) = 0.0
451  enddo ; enddo
452 
453  if (cs%read_gust_2d) then
454  if (associated(fluxes%ustar)) then ; do j=js,je ; do i=is,ie
455  fluxes%ustar(i,j) = sqrt(cs%gust(i,j)/cs%Rho0)
456  enddo ; enddo ; endif
457  else
458  if (associated(fluxes%ustar)) then ; do j=js,je ; do i=is,ie
459  fluxes%ustar(i,j) = sqrt(cs%gust_const/cs%Rho0)
460  enddo ; enddo ; endif
461  endif
462 
463  call calltree_leave("wind_forcing_zero")
464 end subroutine wind_forcing_zero

615

629  real :: temp_y(szi_(g),szj_(g)) ! wind stresses at h-points, in Pa.

644  filename = trim(cs%inputdir) // trim(cs%wind_file)
645 ! if (is_root_pe()) &
646 ! write(*,'("Wind_forcing Reading time level ",I," last was ",I,".")')&
647 ! time_lev-1,CS%wind_last_lev_read-1
648  select case ( uppercase(cs%wind_stagger(1:1)) )
649  case ("A")
650  temp_x(:,:) = 0.0 ; temp_y(:,:) = 0.0
651  call read_data(filename,cs%stress_x_var,temp_x(:,:), &
652  domain=g%Domain%mpp_domain,timelevel=time_lev)
653  call read_data(filename,cs%stress_y_var,temp_y(:,:), &
654  domain=g%Domain%mpp_domain,timelevel=time_lev)

659  enddo ; enddo

667  temp_y(i,j)*temp_y(i,j)) + cs%gust(i,j)) / cs%Rho0)
668  enddo ; enddo
669  else
670  do j=js,je ; do i=is,ie
671  fluxes%ustar(i,j) = sqrt(sqrt(temp_x(i,j)*temp_x(i,j) + &
672  temp_y(i,j)*temp_y(i,j))/cs%Rho0 + (cs%gust_const/cs%Rho0))
673  enddo ; enddo
674  endif
675  case ("C")
676  call read_data(filename,cs%stress_x_var,fluxes%taux(:,:), &
677  domain=g%Domain%mpp_domain,timelevel=time_lev)
678  call read_data(filename,cs%stress_y_var,fluxes%tauy(:,:), &
679  domain=g%Domain%mpp_domain,timelevel=time_lev)
680  if (cs%wind_scale /= 1.0) then
681  do j=js,je ; do i=isq,ieq
682  fluxes%taux(i,j) = cs%wind_scale * fluxes%taux(i,j)
683  enddo ; enddo

693  fluxes%tauy(i,j)**2) + (fluxes%taux(i-1,j)**2 + &
694  fluxes%taux(i,j)**2))) + cs%gust(i,j)) / cs%Rho0 )
695  enddo ; enddo
696  else
697  do j=js, je ; do i=is, ie
698  fluxes%ustar(i,j) = sqrt(sqrt(0.5*((fluxes%tauy(i,j-1)**2 + &
699  fluxes%tauy(i,j)**2) + (fluxes%taux(i-1,j)**2 + &
700  fluxes%taux(i,j)**2)))/cs%Rho0 + (cs%gust_const/cs%Rho0))

702  endif

704  call mom_error(fatal, "wind_forcing_from_file: Unrecognized stagger "//&
705  trim(cs%wind_stagger)//" is not 'A' or 'C'.")
706  end select
707  cs%wind_last_lev_read = time_lev
708  endif ! time_lev /= CS%wind_last_lev_read

714 subroutine buoyancy_forcing_from_files(state, fluxes, day, dt, G, CS)
715  type(surface), intent(inout) :: state
716  type(forcing), intent(inout) :: fluxes
717  type(time_type), intent(in) :: day
718  real, intent(in) :: dt
719  type(ocean_grid_type), intent(inout) :: G
720  type(surface_forcing_cs), pointer :: CS
721 
722 ! This subroutine specifies the current surface fluxes of buoyancy
723 ! temperature and fresh water. It may also be modified to add
724 ! surface fluxes of user provided tracers.
725 !
726 ! Arguments:
727 ! state = structure describing ocean surface state
728 ! (out) fluxes = structure with pointers to forcing fields; unused have NULL ptrs
729 ! (in) day = time of the fluxes
730 ! (in) dt = amount of time over which the fluxes apply
731 ! (in) G = ocean grid structure
732 ! (in) CS = pointer to control struct returned by previous surface_forcing_init call
733 
734  real :: rhoXcp ! mean density times the heat capacity, in J m-3 K-1.
735  real :: Irho0 ! inverse Boussinesq reference density, in m3 kg-1.
736  integer :: i, j, is, ie, js, je, isd, ied, jsd, jed
737 
738  integer :: time_lev ! With fields from a file, this must
739  ! be reset, depending on the time.
740  integer :: time_lev_monthly ! With fields from a file, this must
741  ! be reset, depending on the time.
742  integer :: days, seconds
743  real, dimension(SZI_(G),SZJ_(G)) :: &
744  temp, & ! A 2-d temporary work array with various units.
745  SST_anom, & ! Instantaneous sea surface temperature anomalies from a
746  ! target (observed) value, in deg C.
747  sss_anom, & ! Instantaneous sea surface salinity anomalies from a target
748  ! (observed) value, in g kg-1.
749  sss_mean ! A (mean?) salinity about which to normalize local salinity
750  ! anomalies when calculating restorative precipitation
751  ! anomalies, in g kg-1.
752 
753  call calltree_enter("buoyancy_forcing_from_files, MOM_surface_forcing.F90")
754 
755  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
756  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
757 
758  ! allocate and initialize arrays
759  call buoyancy_forcing_allocate(fluxes, g, cs)
760 
761  if (cs%use_temperature) rhoxcp = cs%Rho0 * fluxes%C_p
762  irho0 = 1.0/cs%Rho0
763 
764  ! Read the file containing the buoyancy forcing.
765  call get_time(day,seconds,days)
766 
767  time_lev = days - 365*floor(real(days) / 365.0)
768 
769  if (time_lev < 31) then ; time_lev_monthly = 0
770  else if (time_lev < 59) then ; time_lev_monthly = 1
771  else if (time_lev < 90) then ; time_lev_monthly = 2
772  else if (time_lev < 120) then ; time_lev_monthly = 3
773  else if (time_lev < 151) then ; time_lev_monthly = 4
774  else if (time_lev < 181) then ; time_lev_monthly = 5
775  else if (time_lev < 212) then ; time_lev_monthly = 6
776  else if (time_lev < 243) then ; time_lev_monthly = 7
777  else if (time_lev < 273) then ; time_lev_monthly = 8
778  else if (time_lev < 304) then ; time_lev_monthly = 9
779  else if (time_lev < 334) then ; time_lev_monthly = 10
780  else ; time_lev_monthly = 11
781  endif
782 
783  time_lev = time_lev+1
784  time_lev_monthly = time_lev_monthly+1
785 
786  if (time_lev /= cs%buoy_last_lev_read) then
787 
788 ! if (is_root_pe()) &
789 ! write(*,'("buoyancy_forcing : Reading time level ",I3,", last was ",I3,".")')&
790 ! time_lev,CS%buoy_last_lev_read
791 
792 
793  call read_data(trim(cs%inputdir)//trim(cs%longwavedown_file), "lwdn_sfc", &
794  fluxes%LW(:,:), domain=g%Domain%mpp_domain, timelevel=time_lev)
795  call read_data(trim(cs%inputdir)//trim(cs%longwaveup_file), "lwup_sfc", &
796  temp(:,:), domain=g%Domain%mpp_domain, timelevel=time_lev)
797  do j=js,je ; do i=is,ie ; fluxes%LW(i,j) = fluxes%LW(i,j) - temp(i,j) ; enddo ; enddo
798 
799  call read_data(trim(cs%inputdir)//trim(cs%evaporation_file), "evap", &
800  temp(:,:), domain=g%Domain%mpp_domain, timelevel=time_lev)
801  do j=js,je ; do i=is,ie
802  fluxes%latent(i,j) = -hlv*temp(i,j)
803  fluxes%evap(i,j) = -temp(i,j)
804  fluxes%latent_evap_diag(i,j) = fluxes%latent(i,j)
805 
806  enddo ; enddo
807 
808  call read_data(trim(cs%inputdir)//trim(cs%sensibleheat_file), "shflx", &
809  temp(:,:), domain=g%Domain%mpp_domain, timelevel=time_lev)
810  do j=js,je ; do i=is,ie ; fluxes%sens(i,j) = -temp(i,j) ; enddo ; enddo
811 
812  call read_data(trim(cs%inputdir)//trim(cs%shortwavedown_file), "swdn_sfc", &
813  fluxes%sw(:,:), domain=g%Domain%mpp_domain, timelevel=time_lev)
814  call read_data(trim(cs%inputdir)//trim(cs%shortwaveup_file), "swup_sfc", &
815  temp(:,:), domain=g%Domain%mpp_domain, timelevel=time_lev)
816  do j=js,je ; do i=is,ie
817  fluxes%sw(i,j) = fluxes%sw(i,j) - temp(i,j)
818  enddo ; enddo
819 
820  call read_data(trim(cs%inputdir)//trim(cs%snow_file), "snow", &
821  fluxes%fprec(:,:), domain=g%Domain%mpp_domain, timelevel=time_lev)
822  call read_data(trim(cs%inputdir)//trim(cs%precip_file), "precip", &
823  fluxes%lprec(:,:), domain=g%Domain%mpp_domain, timelevel=time_lev)
824  do j=js,je ; do i=is,ie
825  fluxes%lprec(i,j) = fluxes%lprec(i,j) - fluxes%fprec(i,j)
826  enddo ; enddo
827 
828  call read_data(trim(cs%inputdir)//trim(cs%freshdischarge_file), "disch_w", &
829  temp(:,:), domain=g%Domain%mpp_domain, timelevel=time_lev_monthly)
830  do j=js,je ; do i=is,ie
831  fluxes%lrunoff(i,j) = temp(i,j)*g%IareaT(i,j)
832  enddo ; enddo
833  call read_data(trim(cs%inputdir)//trim(cs%freshdischarge_file), "disch_s", &
834  temp(:,:), domain=g%Domain%mpp_domain, timelevel=time_lev_monthly)
835  do j=js,je ; do i=is,ie
836  fluxes%frunoff(i,j) = temp(i,j)*g%IareaT(i,j)
837  enddo ; enddo
838 
839 ! Read the SST and SSS fields for damping.
840  if (cs%restorebuoy) then !### .or. associated(CS%ctrl_forcing_CSp)) then
841  call read_data(trim(cs%inputdir)//trim(cs%SSTrestore_file), "TEMP", &
842  cs%T_Restore(:,:), domain=g%Domain%mpp_domain, timelevel=time_lev_monthly)
843  call read_data(trim(cs%inputdir)//trim(cs%salinityrestore_file), "SALT", &
844  cs%S_Restore(:,:), domain=g%Domain%mpp_domain, timelevel=time_lev_monthly)
845  endif
846  cs%buoy_last_lev_read = time_lev
847 
848  ! mask out land points and compute heat content of water fluxes
849  ! assume liquid precip enters ocean at SST
850  ! assume frozen precip enters ocean at 0degC
851  ! assume liquid runoff enters ocean at SST
852  ! assume solid runoff (calving) enters ocean at 0degC
853  do j=js,je ; do i=is,ie
854  fluxes%evap(i,j) = fluxes%evap(i,j) * g%mask2dT(i,j)
855  fluxes%lprec(i,j) = fluxes%lprec(i,j) * g%mask2dT(i,j)
856  fluxes%fprec(i,j) = fluxes%fprec(i,j) * g%mask2dT(i,j)
857  fluxes%lrunoff(i,j) = fluxes%lrunoff(i,j) * g%mask2dT(i,j)
858  fluxes%frunoff(i,j) = fluxes%frunoff(i,j) * g%mask2dT(i,j)
859  fluxes%LW(i,j) = fluxes%LW(i,j) * g%mask2dT(i,j)
860  fluxes%sens(i,j) = fluxes%sens(i,j) * g%mask2dT(i,j)
861  fluxes%sw(i,j) = fluxes%sw(i,j) * g%mask2dT(i,j)
862  fluxes%latent(i,j) = fluxes%latent(i,j) * g%mask2dT(i,j)
863 
864  fluxes%heat_content_lrunoff(i,j) = fluxes%C_p*fluxes%lrunoff(i,j)*state%SST(i,j)
865  fluxes%latent_evap_diag(i,j) = fluxes%latent_evap_diag(i,j) * g%mask2dT(i,j)
866  fluxes%latent_fprec_diag(i,j) = -fluxes%fprec(i,j)*hlf
867  fluxes%latent_frunoff_diag(i,j) = -fluxes%frunoff(i,j)*hlf
868  enddo ; enddo
869 
870  endif ! time_lev /= CS%buoy_last_lev_read
871 
872  if (cs%restorebuoy) then
873  if (cs%use_temperature) then
874  do j=js,je ; do i=is,ie
875  if (g%mask2dT(i,j) > 0) then
876  fluxes%heat_restore(i,j) = g%mask2dT(i,j) * &
877  ((cs%T_Restore(i,j) - state%SST(i,j)) * rhoxcp * cs%Flux_const)
878  fluxes%vprec(i,j) = - (cs%Rho0*cs%Flux_const) * &
879  (cs%S_Restore(i,j) - state%SSS(i,j)) / &
880  (0.5*(state%SSS(i,j) + cs%S_Restore(i,j)))
881  else
882  fluxes%heat_restore(i,j) = 0.0
883  fluxes%vprec(i,j) = 0.0
884  endif
885  enddo ; enddo
886  else
887  do j=js,je ; do i=is,ie
888  if (g%mask2dT(i,j) > 0) then
889  fluxes%buoy(i,j) = (cs%Dens_Restore(i,j) - state%sfc_density(i,j)) * &
890  (cs%G_Earth*cs%Flux_const/cs%Rho0)
891  else
892  fluxes%buoy(i,j) = 0.0
893  endif
894  enddo ; enddo
895  endif
896  else ! not RESTOREBUOY
897  if (.not.cs%use_temperature) then
898  call mom_error(fatal, "buoyancy_forcing in MOM_surface_forcing: "// &
899  "The fluxes need to be defined without RESTOREBUOY.")
900  endif
901  endif ! end RESTOREBUOY
902 
903 !### if (associated(CS%ctrl_forcing_CSp)) then
904 !### do j=js,je ; do i=is,ie
905 !### SST_anom(i,j) = state%SST(i,j) - CS%T_Restore(i,j)
906 !### SSS_anom(i,j) = state%SSS(i,j) - CS%S_Restore(i,j)
907 !### SSS_mean(i,j) = 0.5*(state%SSS(i,j) + CS%S_Restore(i,j))
908 !### enddo ; enddo
909 !### call apply_ctrl_forcing(SST_anom, SSS_anom, SSS_mean, fluxes%heat_restore, &
910 !### fluxes%vprec, day, dt, G, CS%ctrl_forcing_CSp)
911 !### endif
912 
913  call calltree_leave("buoyancy_forcing_from_files")
914 end subroutine buoyancy_forcing_from_files
