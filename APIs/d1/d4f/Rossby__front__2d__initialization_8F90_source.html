65 
66  case (regridding_layer, regridding_rho)
67  do j = g%jsc,g%jec ; do i = g%isc,g%iec

69  eta = -( -drho_dt / gv%Rho0 ) * tz * 0.5 * ( dml * dml )
70  stretch = ( ( g%max_depth + eta ) / g%max_depth )
71  h0 = ( g%max_depth / real(nz) ) * stretch
72  do k = 1, nz
73  h(i,j,k) = h0
74  enddo
75  end do ; end do
76 
77  case (regridding_zstar, regridding_sigma)
78  do j = g%jsc,g%jec ; do i = g%isc,g%iec

80  eta = -( -drho_dt / gv%Rho0 ) * tz * 0.5 * ( dml * dml )

167  f = 0.5*( g%CoriolisBu(i,j) + g%CoriolisBu(i,j-1) )
168  dudt = 0.0 ; if (abs(f) > 0.0) &
169  dudt = ( g%g_Earth * drho_dt ) / ( f * gv%Rho0 )
170  dml = hml( g, g%geoLatT(i,j) )
171  ty = dtdy( g, t_range, g%geoLatT(i,j) )
172  zi = 0.
173  do k = 1, nz
174  hatu = 0.5*(h(i,j,k)+h(i+1,j,k))
175  zi = zi - hatu ! Bottom interface position
176  zc = zi - 0.5*hatu ! Position of middle of cell
177  zm = max( zc + dml, 0. ) ! Height above bottom of mixed layer
178  u(i,j,k) = dudt * ty * zm ! Thermal wind starting at base of ML
179  enddo

229 !> \namespace Rossby_front_2d_initialization
230 !!
231 !! \section section_Rossby_front_2d Description of the 2d Rossby front initial conditions
232 !!
233 !! Consistent with a linear equation of state, the system has a constant stratification
234 !! below the mixed layer, stratified in temperature only. Isotherms are flat below the
235 !! mixed layer and vertical within. Salinity is constant. The mixed layer has a half sine
236 !! form so that there are no mixed layer or temperature gradients at the side walls.
237 !!
238 !! Below the mixed layer the potential temperature, \f$\theta(z)\f$, is given by
239 !! \f[ \theta(z) = \theta_0 - \Delta \theta \left( z + h_{ML} \right) \f]
240 !! where \f$ \theta_0 \f$ and \f$ \Delta \theta \f$ are external model parameters.
241 !!
242 !! The depth of the mixed layer, \f$H_{ML}\f$ is
243 !! \f[ h_{ML}(y) = h_{min} + \left( h_{max} - h_{min} \right) \cos{\pi y/L} \f].
244 !! The temperature in mixed layer is given by the reference temperature at \f$z=h_{ML}\f$
245 !! so that
246 !! \f[ \theta(y,z) =
247 !! \theta_0 - \Delta \theta \left( z + h_{ML} \right) & \forall & z < h_{ML}(y) T.B.D.
248 !! \f]
