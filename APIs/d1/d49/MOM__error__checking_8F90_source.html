81 subroutine check_redundant_v2d(mesg, u_comp, v_comp, G, is, ie, js, je, &
82  direction, stagger)
83  type(ocean_grid_type), intent(inout) :: G
84  character(len=*), intent(in) :: mesg
85  real, dimension(SZIB_(G),SZJ_(G)), intent(in) :: u_comp
86  real, dimension(SZI_(G),SZJB_(G)), intent(in) :: v_comp
87  integer, optional, intent(in) :: is, ie, js, je
88  integer, optional, intent(in) :: direction
89  integer, optional, intent(in) :: stagger
90 ! Arguments: u_comp - The u-component of the vector being checked.
91 ! (in) v_comp - The v-component of the vector being checked.
92 ! (in) mesg - A message indicating what is being checked.
93 ! (in) G - The ocean's grid structure.
94 ! (in/opt) is, ie, js, je - the i- and j- range of indices to check.
95 ! (in/opt) direction - the direction flag to be passed to pass_vector.
96 ! (in/opt) stagger - the stagger flag to be passed to pass_vector.
97 
98  real :: u_nonsym(szi_(g),szj_(g))
99  real :: v_nonsym(szi_(g),szj_(g))
100  real :: u_resym(szib_(g),szj_(g))
101  real :: v_resym(szi_(g),szjb_(g))
102  character(len=128) :: mesg2
103  type(group_pass_type), save :: pass_nonsym_uv, pass_resym_uv !For group halo pass
104 
105  integer :: i, j, is_ch, ie_ch, js_ch, je_ch
106  integer :: Isq, Ieq, Jsq, Jeq, isd, ied, jsd, jed, IsdB, IedB, JsdB, JedB
107  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
108  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
109  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
110 
111  if ((isd == isdb) .and. (jsd == jsdb)) return
112 
113  do i=isd,ied ; do j=jsd,jed
114  u_nonsym(i,j) = u_comp(i,j) ; v_nonsym(i,j) = v_comp(i,j)
115  enddo ; enddo
116 
117  if (.not.associated(g%Domain_aux)) call mom_error(fatal," check_redundant"//&
118  " called with a non-associated auxiliary domain the grid type.")
119 
120  call create_group_pass(pass_nonsym_uv, u_nonsym, v_nonsym, g%Domain_aux, &
121  direction, stagger)
122  call create_group_pass(pass_resym_uv, u_resym, v_resym, g%Domain, &
123  direction, stagger)
124  call do_group_pass(pass_nonsym_uv, g%Domain_aux)
125 
126  do i=isdb,iedb ; do j=jsd,jed ; u_resym(i,j) = u_comp(i,j) ; enddo ; enddo
127  do i=isd,ied ; do j=jsdb,jedb ; v_resym(i,j) = v_comp(i,j) ; enddo ; enddo
128  do i=isd,ied ; do j=jsd,jed
129  u_resym(i,j) = u_nonsym(i,j) ; v_resym(i,j) = v_nonsym(i,j)
130  enddo ; enddo
131  call do_group_pass(pass_resym_uv, g%Domain)
132 
133  is_ch = isq ; ie_ch = ieq ; js_ch = jsq ; je_ch = jeq
134  if (present(is)) is_ch = is ; if (present(ie)) ie_ch = ie
135  if (present(js)) js_ch = js ; if (present(js)) je_ch = je
136 
137  do i=is_ch,ie_ch ; do j=js_ch+1,je_ch
138  if (u_resym(i,j) /= u_comp(i,j)) then
139  write(mesg2,'(" redundant u-components",2(1pe12.4)," differ by ", &
140  & 1pe12.4," at i,j = ",2i4," on pe ",i4)') &
141  u_comp(i,j), u_resym(i,j),u_comp(i,j)-u_resym(i,j),i,j,pe_here()
142  write(*,'(A130)') trim(mesg)//trim(mesg2)
143  endif
144  enddo ; enddo
145  do i=is_ch+1,ie_ch ; do j=js_ch,je_ch
146  if (v_resym(i,j) /= v_comp(i,j)) then
147  write(mesg2,'(" redundant v-comps",2(1pe12.4)," differ by ", &
148  & 1pe12.4," at i,j = ",2i4," x,y = ",2(1pe12.4)" on pe ",i4)') &
149  v_comp(i,j), v_resym(i,j),v_comp(i,j)-v_resym(i,j),i,j, &
150  g%geoLonBu(i,j), g%geoLatBu(i,j), pe_here()
151  write(*,'(A155)') trim(mesg)//trim(mesg2)
152  endif
153  enddo ; enddo
154 
155 end subroutine check_redundant_v2d
