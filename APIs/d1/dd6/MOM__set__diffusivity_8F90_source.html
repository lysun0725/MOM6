944 subroutine find_tke_to_kd(h, tv, dRho_int, N2_lay, j, dt, G, GV, CS, &
945  tke_to_kd, maxtke, kb)
946  type(ocean_grid_type), intent(in) :: G
947  type(verticalgrid_type), intent(in) :: GV
948  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
949  type(thermo_var_ptrs), intent(in) :: tv
950  real, dimension(SZI_(G),SZK_(G)+1), intent(in) :: dRho_int
951  real, dimension(SZI_(G),SZK_(G)), intent(in) :: N2_lay
952  integer, intent(in) :: j
953  real, intent(in) :: dt
954  type(set_diffusivity_cs), pointer :: CS
955  real, dimension(SZI_(G),SZK_(G)), intent(out) :: TKE_to_Kd, maxTKE
956  integer, dimension(SZI_(G)), intent(out) :: kb
957 
958  real, dimension(SZI_(G),SZK_(G)) :: &
959  ds_dsp1, & ! coordinate variable (sigma-2) difference across an
960  ! interface divided by the difference across the interface
961  ! below it (nondimensional)
962  dsp1_ds, & ! inverse coordinate variable (sigma-2) difference
963  ! across an interface times the difference across the
964  ! interface above it (nondimensional)
965  rho_0, & ! Layer potential densities relative to surface pressure (kg/m3)
966  maxent ! maxEnt is the maximum value of entrainment from below (with
967  ! compensating entrainment from above to keep the layer
968  ! density from changing) that will not deplete all of the
969  ! layers above or below a layer within a timestep (meter)
970  real, dimension(SZI_(G)) :: &
971  htot, & ! total thickness above or below a layer, or the
972  ! integrated thickness in the BBL (meter)
973  mfkb, & ! total thickness in the mixed and buffer layers
974  ! times ds_dsp1 (meter)
975  p_ref, & ! array of tv%P_Ref pressures
976  rcv_kmb, & ! coordinate density in the lowest buffer layer
977  p_0 ! An array of 0 pressures
978 
979  real :: dh_max ! maximum amount of entrainment a layer could
980  ! undergo before entraining all fluid in the layers
981  ! above or below (meter)
982  real :: dRho_lay ! density change across a layer (kg/m3)
983  real :: Omega2 ! rotation rate squared (1/s2)
984  real :: G_Rho0 ! gravitation accel divided by Bouss ref density (m4 s-2 kg-1)
985  real :: I_Rho0 ! inverse of Boussinesq reference density (m3/kg)
986  real :: I_dt ! 1/dt (1/sec)
987  real :: H_neglect ! negligibly small thickness (units as h)
988  real :: hN2pO2 ! h * (N^2 + Omega^2), in m s-2.
989  logical :: do_i(szi_(g))
990 
991  integer :: i, k, is, ie, nz, i_rem, kmb, kb_min
992  is = g%isc ; ie = g%iec ; nz = g%ke
993 
994  i_dt = 1.0/dt
995  omega2 = cs%Omega**2
996  g_rho0 = g%g_Earth / gv%Rho0
997  h_neglect = gv%H_subroundoff
998  i_rho0 = 1.0/gv%Rho0
999 
1000  ! Simple but coordinate-independent estimate of Kd/TKE
1001  if (cs%simple_TKE_to_Kd) then
1002  do k=1,nz ; do i=is,ie
1003  hn2po2 = ( gv%H_to_m * h(i,j,k) ) * ( n2_lay(i,k) + omega2 ) ! Units of m s-2.
1004  if (hn2po2>0.) then
1005  tke_to_kd(i,k) = 1./ hn2po2 ! Units of s2 m-1.
1006  else; tke_to_kd(i,k) = 0.; endif
1007  ! The maximum TKE conversion we allow is really a statement
1008  ! about the upper diffusivity we allow. Kd_max must be set.
1009  maxtke(i,k) = hn2po2 * cs%Kd_max ! Units of m3 s-3.
1010  enddo ; enddo
1011  kb(is:ie) = -1 ! kb should not be used by any code in non-layered mode -AJA
1012  return
1013  endif
1014 
1015  ! Determine kb - the index of the shallowest active interior layer.
1016  if (cs%bulkmixedlayer) then
1017  kmb = gv%nk_rho_varies
1018  do i=is,ie ; p_0(i) = 0.0 ; p_ref(i) = tv%P_Ref ; enddo
1019  do k=1,nz
1020  call calculate_density(tv%T(:,j,k),tv%S(:,j,k),p_0,rho_0(:,k),&
1021  is,ie-is+1,tv%eqn_of_state)
1022  enddo
1023  call calculate_density(tv%T(:,j,kmb),tv%S(:,j,kmb),p_ref,rcv_kmb,&
1024  is,ie-is+1,tv%eqn_of_state)
1025 
1026  kb_min = kmb+1
1027  do i=is,ie
1028  ! Determine the next denser layer than the buffer layer in the
1029  ! coordinate density (sigma-2).
1030  do k=kmb+1,nz-1 ; if (rcv_kmb(i) <= gv%Rlay(k)) exit ; enddo
1031  kb(i) = k
1032 
1033  ! Backtrack, in case there are massive layers above that are stable
1034  ! in sigma-0.
1035  do k=kb(i)-1,kmb+1,-1
1036  if (rho_0(i,kmb) > rho_0(i,k)) exit
1037  if (h(i,j,k)>2.0*gv%Angstrom) kb(i) = k
1038  enddo
1039  enddo
1040 
1041  call set_density_ratios(h, tv, kb, g, gv, cs, j, ds_dsp1, rho_0)
1042  else ! not bulkmixedlayer
1043  kb_min = 2 ; kmb = 0
1044  do i=is,ie ; kb(i) = 1 ; enddo
1045  call set_density_ratios(h, tv, kb, g, gv, cs, j, ds_dsp1)
1046  endif
1047 
1048  ! Determine maxEnt - the maximum permitted entrainment from below by each
1049  ! interior layer.
1050  do k=2,nz-1 ; do i=is,ie
1051  dsp1_ds(i,k) = 1.0 / ds_dsp1(i,k)
1052  enddo ; enddo
1053  do i=is,ie ; dsp1_ds(i,nz) = 0.0 ; enddo
1054 
1055  if (cs%bulkmixedlayer) then
1056  kmb = gv%nk_rho_varies
1057  do i=is,ie
1058  htot(i) = gv%H_to_m*h(i,j,kmb)
1059  mfkb(i) = 0.0
1060  if (kb(i) < nz) &
1061  mfkb(i) = ds_dsp1(i,kb(i)) * (gv%H_to_m*(h(i,j,kmb) - gv%Angstrom))
1062  enddo
1063  do k=1,kmb-1 ; do i=is,ie
1064  htot(i) = htot(i) + gv%H_to_m*h(i,j,k)
1065  mfkb(i) = mfkb(i) + ds_dsp1(i,k+1)*(gv%H_to_m*(h(i,j,k) - gv%Angstrom))
1066  enddo ; enddo
1067  else
1068  do i=is,i
1069  maxent(i,1) = 0.0 ; htot(i) = gv%H_to_m*(h(i,j,1) - gv%Angstrom)
1070  enddo
1071  endif
1072  do k=kb_min,nz-1 ; do i=is,ie
1073  if (k == kb(i)) then
1074  maxent(i,kb(i))= mfkb(i)
1075  elseif (k > kb(i)) then
1076  maxent(i,k) = (1.0/dsp1_ds(i,k))*(maxent(i,k-1) + htot(i))
1077 ! maxEnt(i,k) = ds_dsp1(i,k)*(maxEnt(i,k-1) + htot(i)) ! BITWISE CHG
1078  htot(i) = htot(i) + gv%H_to_m*(h(i,j,k) - gv%Angstrom)
1079  endif
1080  enddo ; enddo
1081 
1082  do i=is,ie
1083  htot(i) = gv%H_to_m*(h(i,j,nz) - gv%Angstrom) ; maxent(i,nz) = 0.0
1084  do_i(i) = (g%mask2dT(i,j) > 0.5)
1085  enddo
1086  do k=nz-1,kb_min,-1
1087  i_rem = 0
1088  do i=is,ie ; if (do_i(i)) then
1089  if (k<kb(i)) then ; do_i(i) = .false. ; cycle ; endif
1090  i_rem = i_rem + 1 ! Count the i-rows that are still being worked on.
1091  maxent(i,k) = min(maxent(i,k),dsp1_ds(i,k+1)*maxent(i,k+1) + htot(i))
1092  htot(i) = htot(i) + gv%H_to_m*(h(i,j,k) - gv%Angstrom)
1093  endif ; enddo
1094  if (i_rem == 0) exit
1095  enddo ! k-loop
1096 
1097  ! Now set maxTKE and TKE_to_Kd.
1098  do i=is,ie
1099  maxtke(i,1) = 0.0 ; tke_to_kd(i,1) = 0.0
1100  maxtke(i,nz) = 0.0 ; tke_to_kd(i,nz) = 0.0
1101  enddo
1102  do k=2,kmb ; do i=is,ie
1103  maxtke(i,k) = 0.0
1104  tke_to_kd(i,k) = 1.0 / ((n2_lay(i,k) + omega2) * &
1105  (gv%H_to_m*(h(i,j,k) + h_neglect)))
1106  enddo ; enddo
1107  do k=kmb+1,kb_min-1 ; do i=is,ie
1108  ! These are the properties in the deeper mixed and buffer layers, and
1109  ! should perhaps be revisited.
1110  maxtke(i,k) = 0.0 ; tke_to_kd(i,k) = 0.0
1111  enddo ; enddo
1112  do k=kb_min,nz-1 ; do i=is,ie
1113  if (k<kb(i)) then
1114  maxtke(i,k) = 0.0
1115  tke_to_kd(i,k) = 0.0
1116  else
1117  ! maxTKE is found by determining the kappa that gives maxEnt.
1118  ! ### This should be 1 / G_Earth * (delta rho_InSitu)
1119  ! kappa_max = I_dt * dRho_int(i,K+1) * maxEnt(i,k) * &
1120  ! (GV%H_to_m*h(i,j,k) + dh_max) / dRho_lay
1121  ! maxTKE(i,k) = G%g_Earth * dRho_lay * kappa_max
1122  ! dRho_int should already be non-negative, so the max is redundant?
1123  dh_max = maxent(i,k) * (1.0 + dsp1_ds(i,k))
1124  drho_lay = 0.5 * max(drho_int(i,k) + drho_int(i,k+1), 0.0)
1125  maxtke(i,k) = i_dt * ((g%g_Earth * i_rho0) * &
1126  (0.5*max(drho_int(i,k+1) + dsp1_ds(i,k)*drho_int(i,k),0.0))) * &
1127  ((gv%H_to_m*h(i,j,k) + dh_max) * maxent(i,k))
1128  tke_to_kd(i,k) = 1.0 / (g_rho0 * drho_lay + &
1129  cs%Omega**2 * gv%H_to_m*(h(i,j,k) + h_neglect))
1130  endif
1131  enddo ; enddo
1132 
1133 end subroutine find_tke_to_kd

1135 subroutine find_n2(h, tv, T_f, S_f, fluxes, j, G, GV, CS, dRho_int, &

1150  dRho_int_unfilt, & ! unfiltered density differences across interfaces
1151  dRho_dT, & ! partial derivative of density wrt temp (kg m-3 degC-1)
1152  dRho_dS ! partial derivative of density wrt saln (kg m-3 PPT-1)

1156  Temp_int, & ! temperature at each interface (degC)
1157  Salin_int, & ! salinity at each interface (PPT)
1158  drho_bot, &
1159  h_amp, &
1160  hb, &
1161  z_from_bot
1162

1164  real :: dz_int ! thickness associated with an interface (meter)
1165  real :: G_Rho0 ! gravitation acceleration divided by Bouss reference density (m4 s-2 kg-1)
1166  real :: H_neglect ! negligibly small thickness, in the same units as h.
1167 
1168  logical :: do_i(szi_(g)), do_any
1169  integer :: i, k, is, ie, nz
1170 
1171  is = g%isc ; ie = g%iec ; nz = g%ke
1172  g_rho0 = g%g_Earth / gv%Rho0
1173  h_neglect = gv%H_subroundoff
1174 
1175  ! Find the (limited) density jump across each interface.
1176  do i=is,ie
1177  drho_int(i,1) = 0.0 ; drho_int(i,nz+1) = 0.0
1178  drho_int_unfilt(i,1) = 0.0 ; drho_int_unfilt(i,nz+1) = 0.0
1179  enddo

1181  if (associated(fluxes%p_surf)) then
1182  do i=is,ie ; pres(i) = fluxes%p_surf(i,j) ; enddo
1183  else
1184  do i=is,ie ; pres(i) = 0.0 ; enddo
1185  endif

1187  do i=is,ie
1188  pres(i) = pres(i) + gv%H_to_Pa*h(i,j,k-1)
1189  temp_int(i) = 0.5 * (t_f(i,j,k) + t_f(i,j,k-1))
1190  salin_int(i) = 0.5 * (s_f(i,j,k) + s_f(i,j,k-1))
1191  enddo
1192  call calculate_density_derivs(temp_int, salin_int, pres, &
1193  drho_dt(:,k), drho_ds(:,k), is, ie-is+1, tv%eqn_of_state)
1194  do i=is,ie
1195  drho_int(i,k) = max(drho_dt(i,k)*(t_f(i,j,k) - t_f(i,j,k-1)) + &
1196  drho_ds(i,k)*(s_f(i,j,k) - s_f(i,j,k-1)), 0.0)

1198  drho_ds(i,k)*(tv%S(i,j,k) - tv%S(i,j,k-1)), 0.0)
1199  enddo
1200  enddo
1201  else
1202  do k=2,nz ; do i=is,ie
1203  drho_int(i,k) = gv%Rlay(k) - gv%Rlay(k-1)
1204  enddo ; enddo
1205  endif
1206

1208  do k=1,nz ; do i=is,ie
1209  n2_lay(i,k) = g_rho0 * 0.5*(drho_int(i,k) + drho_int(i,k+1)) / &
1210  (gv%H_to_m*(h(i,j,k) + h_neglect))
1211  enddo ; enddo
1212  do i=is,ie ; n2_int(i,1) = 0.0 ; n2_int(i,nz+1) = 0.0 ; enddo
1213  do k=2,nz ; do i=is,ie
1214  n2_int(i,k) = g_rho0 * drho_int(i,k) / &
1215  (0.5*gv%H_to_m*(h(i,j,k-1) + h(i,j,k) + h_neglect))
1216  enddo ; enddo

1735  endif

1737  endif

1746  if (do_diag_kd_bbl) kd_bbl(i,j,k) = kd_wall

1748  enddo ! i

2420  do i=is,ie

2432  kmb = gv%nk_rho_varies
2433  eps = 0.1

2437  is,ie-is+1,tv%eqn_of_state)
2438  enddo
2439  do i=is,ie
2440  if (kb(i) <= nz-1) then
2441 ! Set up appropriately limited ratios of the reduced gravities of the
2442 ! interfaces above and below the buffer layer and the next denser layer.
2443  k = kb(i)
2444

2446  ! The indexing convention for a is appropriate for the interfaces.
2447  do k3=1,kmb
2448  a(k3+1) = (gv%Rlay(k) - rcv(i,k3)) * i_drho
2449  enddo
2450  if ((present(rho_0)) .and. (a(kmb+1) < 2.0*eps*ds_dsp1(i,k))) then
2451 ! If the buffer layer nearly matches the density of the layer below in the
2452 ! coordinate variable (sigma-2), use the sigma-0-based density ratio if it is
2453 ! greater (and stable).
2454  if ((rho_0(i,k) > rho_0(i,kmb)) .and. &
2455  (rho_0(i,k+1) > rho_0(i,k))) then
2456  i_drho = 1.0 / (rho_0(i,k+1)-rho_0(i,k))
2457  a_0(kmb+1) = min((rho_0(i,k)-rho_0(i,kmb)) * i_drho, ds_dsp1(i,k))
2458  if (a_0(kmb+1) > a(kmb+1)) then
2459  do k3=2,kmb
2460  a_0(k3) = a_0(kmb+1) + (rho_0(i,kmb)-rho_0(i,k3-1)) * i_drho
2461  enddo
2462  if (a(kmb+1) <= eps*ds_dsp1(i,k)) then
2463  do k3=2,kmb+1 ; a(k3) = a_0(k3) ; enddo
2464  else
2465 ! Alternative... tmp = 0.5*(1.0 - cos(PI*(a(K2+1)/(eps*ds_dsp1(i,k)) - 1.0)) )
2466  tmp = a(kmb+1)/(eps*ds_dsp1(i,k)) - 1.0
2467  do k3=2,kmb+1 ; a(k3) = tmp*a(k3) + (1.0-tmp)*a_0(k3) ; enddo
2468  endif
2469  endif
2470  endif
2471  endif
2472 
2473  ds_dsp1(i,k) = max(a(kmb+1),1e-5)
2474 
2475  do k3=2,kmb
2476 ! ds_dsp1(i,k3) = MAX(a(k3),1e-5)
2477  ! Deliberately treat convective instabilies of the upper mixed
2478  ! and buffer layers with respect to the deepest buffer layer as
2479  ! though they don't exist. They will be eliminated by the upcoming
2480  ! call to the mixedlayer code anyway.
2481  ! The indexing convention is appropriate for the interfaces.
2482  ds_dsp1(i,k3) = max(a(k3),ds_dsp1(i,k))
2483  enddo
2484  endif ! (kb(i) <= nz-1)
2485  enddo ! I-loop.
2486  endif ! bulkmixedlayer
