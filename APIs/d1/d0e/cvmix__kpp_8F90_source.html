347  else

349  call cvmix_put_kpp('lscalar_Cv', .false., cvmix_kpp_params_user)

363  case DEFAULT

366  stop 1
367  end select
368  else
369  call cvmix_put_kpp('interp_type', cvmix_math_interp_quad, &
370  cvmix_kpp_params_user)
371  end if
372 
373  if (present(interp_type2)) then

375  case ('line', 'linear')
376  call cvmix_put_kpp('interp_type2', cvmix_math_interp_linear, &
377  cvmix_kpp_params_user)
378  case ('quad', 'quadratic')
379  call cvmix_put_kpp('interp_type2', cvmix_math_interp_quad, &

557 
558  subroutine cvmix_coeffs_kpp_low(Mdiff_out, Tdiff_out, Sdiff_out, zw, zt, &
559  old_mdiff, old_tdiff, old_sdiff, obl_depth, &
560  kobl_depth, tnonlocal, snonlocal, surf_fric,&
561  surf_buoy, nlev, max_nlev, &
562  cvmix_kpp_params_user)
563 
564 ! !DESCRIPTION:
565 ! Computes vertical diffusion coefficients for the KPP boundary layer mixing
566 ! parameterization.
567 !\\
568 !\\
569 !
570 ! !USES:
571 ! only those used by entire module.
572 
573 ! !INPUT PARAMETERS:
574  type(cvmix_kpp_params_type), intent(in), optional, target :: &
575  CVmix_kpp_params_user
576 
577  integer, intent(in) :: nlev, max_nlev
578  real(cvmix_r8), dimension(max_nlev+1), intent(in) :: old_Mdiff, &
579  old_Tdiff, &
580  old_Sdiff, &
581  zw
582  real(cvmix_r8), dimension(max_nlev), intent(in) :: zt
583  real(cvmix_r8), intent(in) :: OBL_depth, &
584  surf_fric, &
585  surf_buoy, &
586  kOBL_depth
587 
588 
589 ! !INPUT/OUTPUT PARAMETERS:
590  real(cvmix_r8), dimension(max_nlev+1), intent(inout) :: Mdiff_out, &
591  Tdiff_out, &
592  Sdiff_out, &
593  Tnonlocal, &
594  Snonlocal
595 
596 !EOP
597 !BOC
598 
599  ! Local variables
600  type(cvmix_kpp_params_type), pointer :: CVmix_kpp_params_in
601 
602  ! OBL_[MTS]diff are the diffusivities in the whole OBL
603  real(cvmix_r8), dimension(nint(kOBL_depth)) :: OBL_Mdiff, OBL_Tdiff, &
604  OBL_Sdiff
605 
606  ! [MTS]diff_ktup are the enhanced diffusivity and viscosity values at the
607  ! deepest cell center above OBL_depth. Other _ktup vars are intermediary
608  ! variables needed to compute [MTS]diff_ktup
609  real(cvmix_r8) :: Mdiff_ktup, Tdiff_ktup, Sdiff_ktup
610  real(cvmix_r8) :: sigma_ktup, wm_ktup, ws_ktup
611 
612  real(cvmix_r8) :: delta
613 
614  real(cvmix_r8), dimension(nlev+1) :: sigma, w_m, w_s
615 
616  ! [MTS]shape are the coefficients of the shape function in the gradient
617  ! term; [TS]shape2 are the coefficients for the nonlocal term
618  real(cvmix_r8), dimension(4) :: Mshape, Tshape, Sshape, Tshape2, Sshape2
619 
620  ! [MTS]shapeAt1 is value of shape function at sigma = 1
621  ! d[MTS]shapeAt1 is value of derivative of shape function at sigma = 1
622  ! (Used for matching the shape function at OBL depth)
623  real(cvmix_r8) :: MshapeAt1, TshapeAt1, SshapeAt1
624  real(cvmix_r8) :: dMshapeAt1, dTshapeAt1, dSshapeAt1
625 
626  ! [MTS]shapeAtS is value of shape function at sigma = S
627  real(cvmix_r8) :: MshapeAtS, TshapeAtS, SshapeAtS, GAtS
628 
629  ! [MTS]diff_OBL is value of diffusivity at OBL depth
630  ! d[MTS]diff_OBL is value of derivative of diffusivity at OBL depth
631  ! w[ms]_OBL is value of wm or ws at OBL depth
632  real(cvmix_r8) :: Mdiff_OBL, Tdiff_OBL, Sdiff_OBL
633  real(cvmix_r8) :: dMdiff_OBL, dTdiff_OBL, dSdiff_OBL
634  real(cvmix_r8) :: wm_OBL, ws_OBL, second_term
635 
636  ! coefficients used for interpolation if interp_type2 is not 'LMD94'
637  real(kind=cvmix_r8), dimension(4) :: coeffs
638 
639  ! Width of column kw_up and kw_up+1
640  real(cvmix_r8), dimension(2) :: col_widths, col_centers
641  real(cvmix_r8), dimension(2) :: Mdiff_vals, Tdiff_vals, Sdiff_vals
642 
643  ! Constant from params
644  integer :: interp_type2, MatchTechnique
645 
646  integer :: kw
647  logical :: lstable
648  integer :: ktup, & ! kt index of cell center above OBL_depth
649  kwup ! kw index of iface above OBL_depth (= kt index of
650  ! cell containing OBL_depth)
651 
652  cvmix_kpp_params_in => cvmix_kpp_params_saved
653  if (present(cvmix_kpp_params_user)) then
654  cvmix_kpp_params_in => cvmix_kpp_params_user
655  end if
656  interp_type2 = cvmix_kpp_params_in%interp_type2
657  matchtechnique = cvmix_kpp_params_in%MatchTechnique
658 
659  ! Output values should be set to input values
660  mdiff_out = old_mdiff
661  tdiff_out = old_tdiff
662  sdiff_out = old_sdiff
663 
664  ! (1) Column-specific parameters
665  !
666  ! Stability => positive surface buoyancy flux
667  lstable = (surf_buoy.gt.cvmix_zero)
668 
669  kwup = floor(kobl_depth)
670  ktup = nint(kobl_depth)-1
671 
672  if (ktup.eq.nlev) then
673  ! OBL_depth between bottom cell center and ocean bottom, assume
674  ! zt(ktup+1) = ocn_bottom (which is zw(nlev+1)
675  delta = (obl_depth+zt(ktup))/(zt(ktup)-zw(ktup+1))
676  else
677  delta = (obl_depth+zt(ktup))/(zt(ktup)-zt(ktup+1))
678  end if
679 
680  ! (2) Compute coefficients of shape function
681  select case (matchtechnique)
682  case (cvmix_kpp_simple_shapes)
683  ! Simple shape function is sigma*(1-sigma)^2
684  mshape(1) = cvmix_zero
685  mshape(2) = cvmix_one
686  mshape(3) = -real(2,cvmix_r8)
687  mshape(4) = cvmix_one
688  tshape = mshape
689  sshape = mshape
690  tshape2 = tshape
691  sshape2 = sshape
692  case (cvmix_kpp_parabolic_nonlocal)
693  ! Shape function is sigma*(1-sigma)^2 for gradient term
694  ! and (1-sigma)^2 for non-local term
695  mshape(1) = cvmix_zero
696  mshape(2) = cvmix_one
697  mshape(3) = -real(2,cvmix_r8)
698  mshape(4) = cvmix_one
699  tshape = mshape
700  sshape = mshape
701  tshape2(1) = cvmix_one
702  tshape2(2) = -real(2,cvmix_r8)
703  tshape2(3) = cvmix_one
704  tshape2(4) = cvmix_zero
705  sshape2 = tshape2
706  case DEFAULT
707  ! (2a) Compute turbulent scales at OBL depth
708  call cvmix_kpp_compute_turbulent_scales(cvmix_one, obl_depth, &
709  surf_buoy, surf_fric, wm_obl, &
710  ws_obl, cvmix_kpp_params_user)
711 
712  ! (2b) Compute diffusivities at OBL depth
713  if (interp_type2.ne.cvmix_kpp_interp_lmd94) then
714  if (kwup.eq.1) then
715  call cvmix_math_poly_interp(coeffs, interp_type2, zw(kwup:kwup+1),&
716  old_mdiff(kwup:kwup+1))
717  mdiff_obl = cvmix_math_evaluate_cubic(coeffs, -obl_depth, &
718  dmdiff_obl)
719 
720  call cvmix_math_poly_interp(coeffs, interp_type2, zw(kwup:kwup+1),&
721  old_tdiff(kwup:kwup+1))
722  tdiff_obl = cvmix_math_evaluate_cubic(coeffs, -obl_depth, &
723  dtdiff_obl)
724 
725  call cvmix_math_poly_interp(coeffs, interp_type2, zw(kwup:kwup+1),&
726  old_sdiff(kwup:kwup+1))
727  sdiff_obl = cvmix_math_evaluate_cubic(coeffs, -obl_depth, &
728  dsdiff_obl)
729  else ! interp_type2 != 'LMD94' and kwup > 1
730  call cvmix_math_poly_interp(coeffs, interp_type2, zw(kwup:kwup+1),&
731  old_mdiff(kwup:kwup+1), zw(kwup-1), &
732  old_mdiff(kwup-1))
733  mdiff_obl = cvmix_math_evaluate_cubic(coeffs, -obl_depth, &
734  dmdiff_obl)
735 
736  call cvmix_math_poly_interp(coeffs, interp_type2, zw(kwup:kwup+1),&
737  old_tdiff(kwup:kwup+1), zw(kwup-1), &
738  old_tdiff(kwup-1))
739  tdiff_obl = cvmix_math_evaluate_cubic(coeffs, -obl_depth, &
740  dtdiff_obl)
741 
742  call cvmix_math_poly_interp(coeffs, interp_type2, zw(kwup:kwup+1),&
743  old_sdiff(kwup:kwup+1), zw(kwup-1), &
744  old_sdiff(kwup-1))
745  sdiff_obl = cvmix_math_evaluate_cubic(coeffs, -obl_depth, &
746  dsdiff_obl)
747  end if
748  else ! interp_type2 == 'LMD94'
749  col_centers(1) = zt(kwup)
750  col_widths(1) = zw(kwup) - zw(kwup+1)
751  mdiff_vals(1) = old_mdiff(kwup+1)
752  tdiff_vals(1) = old_tdiff(kwup+1)
753  sdiff_vals(1) = old_sdiff(kwup+1)
754  if (kwup.eq.nlev) then
755  col_centers(2) = zw(kwup+1)
756  col_widths(2) = 1.0_cvmix_r8 ! Value doesn't matter, will divide
757  ! into zero
758  mdiff_vals(2) = old_mdiff(kwup+1)
759  tdiff_vals(2) = old_tdiff(kwup+1)
760  sdiff_vals(2) = old_sdiff(kwup+1)
761  else
762  col_centers(2) = zt(kwup+1)
763  col_widths(2) = zw(kwup+1) - zw(kwup+2)
764  mdiff_vals(2) = old_mdiff(kwup+2)
765  tdiff_vals(2) = old_tdiff(kwup+2)
766  sdiff_vals(2) = old_sdiff(kwup+2)
767  end if
768 
769  if (kwup.eq.1) then
770  mdiff_obl = cvmix_kpp_compute_nu_at_obl_depth_lmd94(col_centers, &
771  col_widths, &
772  mdiff_vals, obl_depth, &
773  dnu_dz=dmdiff_obl)
774  tdiff_obl = cvmix_kpp_compute_nu_at_obl_depth_lmd94(col_centers, &
775  col_widths, &
776  tdiff_vals, obl_depth, &
777  dnu_dz=dtdiff_obl)
778  sdiff_obl = cvmix_kpp_compute_nu_at_obl_depth_lmd94(col_centers, &
779  col_widths, &
780  sdiff_vals, obl_depth, &
781  dnu_dz=dsdiff_obl)
782  else ! interp_type == 'LMD94' and kwup > 1
783  mdiff_obl = cvmix_kpp_compute_nu_at_obl_depth_lmd94(col_centers, &
784  col_widths, &
785  mdiff_vals, obl_depth, &
786  old_mdiff(kwup), &
787  dnu_dz=dmdiff_obl)
788  tdiff_obl = cvmix_kpp_compute_nu_at_obl_depth_lmd94(col_centers, &
789  col_widths, &
790  tdiff_vals, obl_depth, &
791  old_tdiff(kwup), &
792  dnu_dz=dtdiff_obl)
793  sdiff_obl = cvmix_kpp_compute_nu_at_obl_depth_lmd94(col_centers, &
794  col_widths, &
795  sdiff_vals, obl_depth, &
796  old_sdiff(kwup), &
797  dnu_dz=dsdiff_obl)
798  end if
799  end if ! interp_type != "LMD94"
800 
801  ! (2c) Compute G(1) [shape function when sigma = 1] and G'(1) for three
802  ! cases:
803  ! i) momentum diffusivity (viscosity)
804  ! ii) temperature diffusivity
805  ! iii) other tracers diffusivity
806  ! Notes:
807  ! * We are computing G(1) and G'(1) so we can represent G(sigma) as a
808  ! cubic polynomial and then compute Kx = OBL_depth*wx*G. If either
809  ! OBL_depth or wx are 0, it doesn't matter what G is because Kx
810  ! will be zero everywhere... in these cases, we set G(1)=G'(1)=0.
811  ! * If OBL_depth = 0, the above note applies to all three situations
812  ! listed as (i), (ii), and (iii). If ws = 0, it applies only to (i)
813  ! and (ii). If wm = 0, it applies only to (iii).
814  if (obl_depth.eq.cvmix_zero) then
815  ! Values don't matter, K = 0
816  mshapeat1 = cvmix_zero
817  tshapeat1 = cvmix_zero
818  sshapeat1 = cvmix_zero
819  dmshapeat1 = cvmix_zero
820  dtshapeat1 = cvmix_zero
821  dsshapeat1 = cvmix_zero
822  else ! OBL_depth != 0
823  if (wm_obl.ne.cvmix_zero) then
824  mshapeat1 = mdiff_obl/(wm_obl*obl_depth)
825  else
826  mshapeat1 = cvmix_zero ! value doesn't really matter, Km = 0
827  end if
828  if (ws_obl.ne.cvmix_zero) then
829  tshapeat1 = tdiff_obl/(ws_obl*obl_depth)
830  sshapeat1 = sdiff_obl/(ws_obl*obl_depth)
831  else
832  tshapeat1 = cvmix_zero ! value doesn't really matter, Ks = 0
833  sshapeat1 = cvmix_zero ! value doesn't really matter, Ks = 0
834  end if
835  if (cvmix_kpp_params_in%lnoDGat1) then
836  ! Force G'(1) = 0
837  dmshapeat1 = cvmix_zero
838  dtshapeat1 = cvmix_zero
839  dsshapeat1 = cvmix_zero
840  else
841  second_term = real(5,cvmix_r8)*surf_buoy/(surf_fric**4)
842  if (wm_obl.ne.cvmix_zero) then
843  dmshapeat1 = -dmdiff_obl/wm_obl
844  if (lstable) &
845  dmshapeat1 = dmshapeat1 + second_term*mdiff_obl
846  else
847  dmshapeat1 = cvmix_zero ! value doesn't really matter, Km = 0
848  end if
849  if (ws_obl.ne.cvmix_zero) then
850  dtshapeat1 = -dtdiff_obl/ws_obl
851  dsshapeat1 = -dsdiff_obl/ws_obl
852  if (lstable) then
853  dtshapeat1 = dtshapeat1 + second_term*tdiff_obl
854  dsshapeat1 = dsshapeat1 + second_term*sdiff_obl
855  end if
856  else
857  dtshapeat1 = cvmix_zero ! value doesn't really matter, Ks = 0
858  dsshapeat1 = cvmix_zero ! value doesn't really matter, Ks = 0
859  end if
860  dmshapeat1 = min(dmshapeat1, cvmix_zero) ! non-positive value!
861  dtshapeat1 = min(dtshapeat1, cvmix_zero) ! non-positive value!
862  dsshapeat1 = min(dsshapeat1, cvmix_zero) ! non-positive value!
863  end if ! lnoDGat1
864  end if ! OBL_depth == 0
865 
866  ! (2d) Compute coefficients of shape function
867  call cvmix_kpp_compute_shape_function_coeffs(mshapeat1, dmshapeat1, &
868  mshape)
869  call cvmix_kpp_compute_shape_function_coeffs(tshapeat1, dtshapeat1, &
870  tshape)
871  call cvmix_kpp_compute_shape_function_coeffs(sshapeat1, dsshapeat1, &
872  sshape)
873  if (matchtechnique.eq.cvmix_kpp_match_gradient) then
874  ! Only match for gradient term, use simple shape for nonlocal
875  tshape2(1) = cvmix_zero
876  tshape2(2) = cvmix_one
877  tshape2(3) = -real(2,cvmix_r8)
878  tshape2(4) = cvmix_one
879  sshape2 = tshape2
880  else
881  ! Shape function is the same for gradient and nonlocal
882  tshape2 = tshape
883  sshape2 = sshape
884  end if
885  end select
886 
887  ! (3) Use shape function to compute diffusivities throughout OBL
888  tnonlocal = cvmix_zero
889  snonlocal = cvmix_zero
890  obl_mdiff = cvmix_zero
891  obl_tdiff = cvmix_zero
892  obl_sdiff = cvmix_zero
893  sigma = -zw(1:nlev+1)/obl_depth
894  ! (3a) Compute turbulent scales throghout column
895  call cvmix_kpp_compute_turbulent_scales(sigma, obl_depth, surf_buoy, &
896  surf_fric, w_m, w_s, &
897  cvmix_kpp_params_user)
898  do kw=2,ktup+1
899  ! (3b) Evaluate G(sigma) at each cell interface
900  mshapeats = cvmix_math_evaluate_cubic(mshape, sigma(kw))
901  tshapeats = cvmix_math_evaluate_cubic(tshape, sigma(kw))
902  sshapeats = cvmix_math_evaluate_cubic(sshape, sigma(kw))
903 
904  ! (3c) Compute nonlocal term at each cell interface
905  if ((.not.lstable).and.(kw.le.kwup)) then
906  gats = cvmix_math_evaluate_cubic(tshape2, sigma(kw))
907  tnonlocal(kw) = cvmix_kpp_params_in%nonlocal_coeff*gats
908  gats = cvmix_math_evaluate_cubic(sshape2, sigma(kw))
909  snonlocal(kw) = cvmix_kpp_params_in%nonlocal_coeff*gats
910  end if
911 
912  ! (3d) Diffusivity = OBL_depth * (turbulent scale) * G(sigma)
913  obl_mdiff(kw) = obl_depth * w_m(kw) * mshapeats
914  obl_tdiff(kw) = obl_depth * w_s(kw) * tshapeats
915  obl_sdiff(kw) = obl_depth * w_s(kw) * sshapeats
916  end do
917 
918  ! (4) Compute the enhanced diffusivity
919  ! (4a) Compute shape function at last cell center in OBL
920  sigma_ktup = -zt(ktup)/obl_depth
921  mshapeats = cvmix_math_evaluate_cubic(mshape, sigma_ktup)
922  tshapeats = cvmix_math_evaluate_cubic(tshape, sigma_ktup)
923  sshapeats = cvmix_math_evaluate_cubic(sshape, sigma_ktup)
924  ! (4b) Compute turbulent scales at last cell center in OBL
925  call cvmix_kpp_compute_turbulent_scales(sigma_ktup, obl_depth, surf_buoy, &
926  surf_fric, wm_ktup, ws_ktup, &
927  cvmix_kpp_params_user)
928  ! (4c) Diffusivity = OBL_depth * (turbulent scale) * G(sigma)
929  mdiff_ktup = obl_depth * wm_ktup * mshapeats
930  tdiff_ktup = obl_depth * ws_ktup * tshapeats
931  sdiff_ktup = obl_depth * ws_ktup * sshapeats
932 
933  if (cvmix_kpp_params_in%lenhanced_diff) then
934  if ((ktup.eq.kwup).or.(ktup.eq.kwup-1)) then
935  call cvmix_kpp_compute_enhanced_diff(mdiff_ktup, &
936  tdiff_ktup, &
937  sdiff_ktup, &
938  mdiff_out(ktup+1), &
939  tdiff_out(ktup+1), &
940  sdiff_out(ktup+1), &
941  obl_mdiff(ktup+1), &
942  obl_tdiff(ktup+1), &
943  obl_sdiff(ktup+1), &
944  tnonlocal(ktup+1), &
945  snonlocal(ktup+1), &
946  delta, lkteqkw=(ktup.eq.kwup))
947  else
948  print*, "ERROR: ktup should be either kwup or kwup-1!"
949  print*, "ktup = ", ktup, " and kwup = ", kwup
950  stop 1
951  end if
952  end if
953 
954  ! (5) Combine interior and boundary coefficients
955  mdiff_out(2:ktup+1) = obl_mdiff(2:ktup+1)
956  tdiff_out(2:ktup+1) = obl_tdiff(2:ktup+1)
957  sdiff_out(2:ktup+1) = obl_sdiff(2:ktup+1)
958 
959 !EOC
960  end subroutine cvmix_coeffs_kpp_low

1227  zt_cntr

1232  CVmix_kpp_params_user

1235  real(cvmix_r8), intent(out) :: OBL_depth, kOBL_depth

1238 !BOC
1239

1241  real(kind=cvmix_r8), dimension(:), pointer :: depth
1242  real(kind=cvmix_r8), dimension(4) :: coeffs

1324  exit

1348  ! results in Ri_bulk = Ri_crit at a depth above the center of the

1350  if (k.eq.1) then
1351  obl_depth = max(obl_depth, -zt_cntr(1))
1352  end if

1355  ! Note: maybe there are times when we don't need to do the interpolation
1356  ! because we know OBL_depth will equal OBL_limit?
1357  obl_depth = min(obl_depth, obl_limit)
1358  end if
1359

1361  if (cvmix_kpp_params_in%maxOBLdepth.gt.cvmix_zero) &
1362  obl_depth = min(obl_depth, cvmix_kpp_params_in%maxOBLdepth)

1430  subroutine cvmix_kpp_compute_enhanced_diff(Mdiff_ktup, Tdiff_ktup, &
1431  sdiff_ktup, mdiff, tdiff, sdiff, &
1432  obl_mdiff, obl_tdiff, obl_sdiff, &
1433  tnonlocal, snonlocal, &
1434  delta, lkteqkw)
1435 
1436 ! !DESCRIPTION:
1437 ! The enhanced mixing described in Appendix D of LMD94 changes the diffusivity
1438 ! values at the interface between the cell center above OBL\_depth and the one
1439 ! below it, based on a weighted average of how close to each center OBL\_depth
1440 ! is. Note that we need to know whether OBL\_depth is above this interface or
1441 ! below it - we do this by comparing the indexes of the cell center above
1442 ! OBL\_depth (ktup) and the cell interface above OBL\_depth(kwup).
1443 !\\
1444 !\\
1445 
1446 ! !INPUT PARAMETERS:
1447 
1448  ! Diffusivity and viscosity at cell center above OBL_depth
1449  real(cvmix_r8), intent(in) :: Mdiff_ktup, Tdiff_ktup, Sdiff_ktup
1450 
1451  ! Weight to use in averaging (distance between OBL_depth and cell center
1452  ! above OBL_depth divided by distance between cell centers bracketing
1453  ! OBL_depth).
1454  real(cvmix_r8), intent(in) :: delta
1455 
1456  logical, intent(in) :: lkteqkw ! .true. => interface ktup+1 is outside OBL
1457  ! (update diff and visc)
1458  ! .false. => interface ktup+1 is inside OBL
1459  ! (update OBL_diff and OBL_visc)
1460 
1461 ! !OUTPUT PARAMETERS:
1462  ! Will change either diff & visc or OBL_diff & OBL_visc, depending on value
1463  ! of lkteqkw
1464  real(cvmix_r8), intent(inout) :: Mdiff, Tdiff, Sdiff, &
1465  OBL_Mdiff, OBL_Tdiff, OBL_Sdiff, &
1466  Tnonlocal, Snonlocal
1467 
1468 !EOP
1469 !BOC
1470 
1471  ! Local variables
1472 
1473  ! enh_diff and enh_visc are the enhanced diffusivity and viscosity values
1474  ! at the interface nearest OBL_depth
1475  real(cvmix_r8) :: enh_Mdiff, enh_Tdiff, enh_Sdiff
1476  ! Need to store original OBL_Tdiff and OBL_Sdiff for updating nonlocal
1477  real(cvmix_r8) :: old_Tdiff, old_Sdiff
1478 
1479  real(cvmix_r8) :: omd ! one minus delta
1480 
1481  omd = cvmix_one - delta
1482  old_tdiff = obl_tdiff
1483  old_sdiff = obl_sdiff
1484 
1485  if (lkteqkw) then
1486  ! => ktup = kwup
1487  ! Interface kw = ktup+1 is outside the OBL
1488 
1489  ! (a) compute enhanced diffs: get diffusivity values at kw = ktup+1
1490  ! from diff and visc rather than OBL_diff and OBL_visc
1491  enh_mdiff = (omd**2)*mdiff_ktup + (delta**2)*mdiff
1492  enh_tdiff = (omd**2)*tdiff_ktup + (delta**2)*tdiff
1493  enh_sdiff = (omd**2)*sdiff_ktup + (delta**2)*sdiff
1494 
1495  ! (b) modify diffusivity values at kw = ktup+1 (again in diff and visc)
1496  mdiff = omd*mdiff + delta*enh_mdiff
1497  tdiff = omd*tdiff + delta*enh_tdiff
1498  sdiff = omd*sdiff + delta*enh_sdiff
1499 
1500  ! (c) Update OBL_[MTS]diff
1501  obl_mdiff = mdiff
1502  obl_tdiff = tdiff
1503  obl_sdiff = sdiff
1504  else
1505  ! => ktup = kwup - 1
1506  ! Interface kw = ktup+1 is in the OBL
1507 
1508  ! (a) compute enhanced diffs: get diffusivity values at kw = ktup+1
1509  ! from OBL_diff and OBL_visc rather than diff and visc
1510  enh_mdiff = (omd**2)*mdiff_ktup + (delta**2)*obl_mdiff
1511  enh_tdiff = (omd**2)*tdiff_ktup + (delta**2)*obl_tdiff
1512  enh_sdiff = (omd**2)*sdiff_ktup + (delta**2)*obl_sdiff
1513 
1514  ! (b) modify diffusivity values at kw = ktup+1 (again in OBL_diff and
1515  ! OBL_visc)
1516  obl_mdiff = omd*mdiff + delta*enh_mdiff
1517  obl_tdiff = omd*tdiff + delta*enh_tdiff
1518  obl_sdiff = omd*sdiff + delta*enh_sdiff
1519 
1520  ! (c) update nonlocal term
1521  if (old_tdiff.ne.cvmix_zero) then
1522  tnonlocal = tnonlocal*obl_tdiff/old_tdiff
1523  else
1524  tnonlocal = cvmix_zero
1525  end if
1526  if (old_sdiff.ne.cvmix_zero) then
1527  snonlocal = snonlocal*obl_sdiff/old_sdiff
1528  else
1529  snonlocal = cvmix_zero
1530  end if
1531  end if
1532 
1533 ! EOC
1534 
1535  end subroutine cvmix_kpp_compute_enhanced_diff
