23 !* *
24 !* By Robert Hallberg, 2007 *
25 !* *
26 !* This file contains an example of the code that is needed to set *
27 !* up and use CFC-11 and CFC-12 in a fully coupled or ice-ocean model *
28 !* context. There are 5 subroutines in this file. *
29 !* *
30 !* register_OCMIP2_CFC determines if the module is going to work, *
31 !* then makes several calls registering tracers to be advected and *
32 !* read from a restart file. it also sets various run-time parameters *
33 !* for this module and sets up a "control structure" (CS) to store *
34 !* all information for this module. *
35 !* *
36 !* initialize_OCMIP2_CFC initializes this modules arrays if they *
37 !* have not been found in a restart file. It also determines which *
38 !* diagnostics will need to be calculated. *
39 !* *
40 !* OCMIP2_CFC_column_physics updates the CFC concentrations, *
41 !* applying everthing but horizontal advection and diffusion. *
42 !* Surface fluxes are applied inside an implicit vertical advection *
43 !* and diffusion tridiagonal solver, and any interior sources and *
44 !* sinks (not applicable for CFCs) would also be applied here. This *
45 !* subroutine also sends out any requested interior diagnostics. *
46 !* *
47 !* OCMIP2_CFC_surface_state calculates the information required *
48 !* from the ocean for the FMS coupler to calculate CFC fluxes. *
49 !* *
50 !* OCMIP2_CFC_end deallocates the persistent run-time memory used *
51 !* by this module. *
52 !* *
53 !* A small fragment of the horizontal grid is shown below: *
54 !* *
55 !* j+1 x ^ x ^ x At x: q *
56 !* j+1 > o > o > At ^: v, tr_ady, tr_dfy *
57 !* j x ^ x ^ x At >: u, tr_adx, tr_dfx *
58 !* j > o > o > At o: h, tr, CFC11, CFC12 *
59 !* j-1 x ^ x ^ x *
60 !* i-1 i i+1 At x & ^: *
61 !* i i+1 At > & o: *
62 !* *
63 !* The boundaries always run through q grid points (x). *
64 !* *
65 !********+*********+*********+*********+*********+*********+*********+**

159 function register_ocmip2_cfc(HI, GV, param_file, CS, tr_Reg, restart_CS)
160  type(hor_index_type), intent(in) :: HI
161  type(verticalgrid_type), intent(in) :: GV
162  type(param_file_type), intent(in) :: param_file
163  type(ocmip2_cfc_cs), pointer :: CS
164  type(tracer_registry_type), pointer :: tr_Reg
165  type(mom_restart_cs), pointer :: restart_CS
166 ! This subroutine is used to register tracer fields and subroutines
167 ! to be used with MOM.
168 ! Arguments: HI - A horizontal index type structure.
169 ! (in) param_file - A structure indicating the open file to parse for
170 ! model parameter values.
171 ! (in/out) CS - A pointer that is set to point to the control structure
172 ! for this module
173 ! (in/out) tr_Reg - A pointer to the tracer registry.
174 ! (in) restart_CS - A pointer to the restart control structure.
175 
176 ! This include declares and sets the variable "version".
177 #include "version_variable.h"
178  character(len=40) :: mod = "MOM_OCMIP2_CFC" ! This module's name.
179  character(len=200) :: inputdir ! The directory where NetCDF input files are.
180  ! These can be overridden later in via the field manager?
181  character(len=128) :: default_ice_restart_file = 'ice_ocmip2_cfc.res.nc'
182  character(len=128) :: default_ocean_restart_file = 'ocmip2_cfc.res.nc'
183  real, dimension(:,:,:), pointer :: tr_ptr
184  real :: a11_dflt(4), a12_dflt(4) ! Default values of the various coefficients
185  real :: d11_dflt(4), d12_dflt(4) ! In the expressions for the solubility and
186  real :: e11_dflt(3), e12_dflt(3) ! Schmidt numbers.
187  logical :: register_OCMIP2_CFC
188  integer :: isd, ied, jsd, jed, nz, m
189 
190  isd = hi%isd ; ied = hi%ied ; jsd = hi%jsd ; jed = hi%jed ; nz = gv%ke
191 
192  if (associated(cs)) then
193  call mom_error(warning, "register_OCMIP2_CFC called with an "// &
194  "associated control structure.")
195  return
196  endif
197  allocate(cs)
198 
199  ! These calls obtain the indices for the CFC11 and CFC12 flux coupling.
200  cs%ind_cfc_11_flux = aof_set_coupler_flux('cfc_11_flux', &
201  flux_type = 'air_sea_gas_flux', implementation = 'ocmip2', &
202  param = (/ 9.36e-07, 9.7561e-06 /), &
203  ice_restart_file = default_ice_restart_file, &
204  ocean_restart_file = default_ocean_restart_file, &
205  caller = "register_OCMIP2_CFC")
206  cs%ind_cfc_12_flux = aof_set_coupler_flux('cfc_12_flux', &
207  flux_type = 'air_sea_gas_flux', implementation = 'ocmip2', &
208  param = (/ 9.36e-07, 9.7561e-06 /), &
209  ice_restart_file = default_ice_restart_file, &
210  ocean_restart_file = default_ocean_restart_file, &
211  caller = "register_OCMIP2_CFC")
212  if ((cs%ind_cfc_11_flux < 0) .or. (cs%ind_cfc_11_flux < 0)) then
213  ! This is most likely to happen with the dummy version of aof_set_coupler_flux
214  ! used in ocean-only runs.
215  call mom_error(warning, "CFCs are currently only set up to be run in " // &
216  " coupled model configurations, and will be disabled.")
217  deallocate(cs)
218  register_ocmip2_cfc = .false.
219  return
220  endif
221 
222  ! Read all relevant parameters and write them to the model log.
223  call log_version(param_file, mod, version, "")
224  call get_param(param_file, mod, "CFC_IC_FILE", cs%IC_file, &
225  "The file in which the CFC initial values can be \n"//&
226  "found, or an empty string for internal initialization.", &
227  default=" ")
228  if ((len_trim(cs%IC_file) > 0) .and. (scan(cs%IC_file,'/') == 0)) then
229  ! Add the directory if CS%IC_file is not already a complete path.
230  call get_param(param_file, mod, "INPUTDIR", inputdir, default=".")
231  cs%IC_file = trim(slasher(inputdir))//trim(cs%IC_file)
232  call log_param(param_file, mod, "INPUTDIR/CFC_IC_FILE", cs%IC_file)
233  endif
234  call get_param(param_file, mod, "CFC_IC_FILE_IS_Z", cs%Z_IC_file, &
235  "If true, CFC_IC_FILE is in depth space, not layer space", &
236  default=.false.)
237  call get_param(param_file, mod, "MASK_MASSLESS_TRACERS", cs%mask_tracers, &
238  "If true, the tracers are masked out in massless layer. \n"//&
239  "This can be a problem with time-averages.", default=.false.)
240  call get_param(param_file, mod, "TRACERS_MAY_REINIT", cs%tracers_may_reinit, &
241  "If true, tracers may go through the initialization code \n"//&
242  "if they are not found in the restart files. Otherwise \n"//&
243  "it is a fatal error if tracers are not found in the \n"//&
244  "restart files of a restarted run.", default=.false.)
245 
246  ! The following vardesc types contain a package of metadata about each tracer,
247  ! including, the name; units; longname; and grid information.
248  cs%CFC11_name = "CFC11" ; cs%CFC12_name = "CFC12"
249  cs%CFC11_desc = var_desc(cs%CFC11_name,"mol m-3","CFC-11 Concentration", caller=mod)
250  cs%CFC12_desc = var_desc(cs%CFC12_name,"mol m-3","CFC-12 Concentration", caller=mod)
251 
252  allocate(cs%CFC11(isd:ied,jsd:jed,nz)) ; cs%CFC11(:,:,:) = 0.0
253  allocate(cs%CFC12(isd:ied,jsd:jed,nz)) ; cs%CFC12(:,:,:) = 0.0
254  if (cs%mask_tracers) then
255  allocate(cs%CFC11_aux(isd:ied,jsd:jed,nz)) ; cs%CFC11_aux(:,:,:) = 0.0
256  allocate(cs%CFC12_aux(isd:ied,jsd:jed,nz)) ; cs%CFC12_aux(:,:,:) = 0.0
257  endif
258 
259  ! This pointer assignment is needed to force the compiler not to do a copy in
260  ! the registration calls. Curses on the designers and implementers of F90.
261  tr_ptr => cs%CFC11
262  ! Register CFC11 for the restart file.
263  call register_restart_field(tr_ptr, cs%CFC11_desc, &
264  .not.cs%tracers_may_reinit, restart_cs)
265  ! Register CFC11 for horizontal advection & diffusion.
266  call register_tracer(tr_ptr, cs%CFC11_desc, param_file, hi, gv, tr_reg, &
267  tr_desc_ptr=cs%CFC11_desc)
268  ! Do the same for CFC12
269  tr_ptr => cs%CFC12
270  call register_restart_field(tr_ptr, cs%CFC12_desc, &
271  .not.cs%tracers_may_reinit, restart_cs)
272  call register_tracer(tr_ptr, cs%CFC12_desc, param_file, hi, gv, tr_reg, &
273  tr_desc_ptr=cs%CFC12_desc)
274 
275  ! Set and read the various empirical coefficients.
276

279 ! by Zheng et al (1998), JGR vol 103, C1.
280 !-----------------------------------------------------------------------
281  a11_dflt(:) = (/ 3501.8, -210.31, 6.1851, -0.07513 /)
282  a12_dflt(:) = (/ 3845.4, -228.95, 6.1908, -0.06743 /)

286  call get_param(param_file, mod, "CFC11_A2", cs%a2_11, &
287  "A coefficient in the Schmidt number of CFC11.", &
288  units="degC-1", default=a11_dflt(2))

290  "A coefficient in the Schmidt number of CFC11.", &
291  units="degC-2", default=a11_dflt(3))
292  call get_param(param_file, mod, "CFC11_A4", cs%a4_11, &
293  "A coefficient in the Schmidt number of CFC11.", &
294  units="degC-3", default=a11_dflt(4))
295 
296  call get_param(param_file, mod, "CFC12_A1", cs%a1_12, &
297  "A coefficient in the Schmidt number of CFC12.", &
298  units="nondim", default=a12_dflt(1))
299  call get_param(param_file, mod, "CFC12_A2", cs%a2_12, &
300  "A coefficient in the Schmidt number of CFC12.", &
301  units="degC-1", default=a12_dflt(2))
302  call get_param(param_file, mod, "CFC12_A3", cs%a3_12, &

304  units="degC-2", default=a12_dflt(3))

307  units="degC-3", default=a12_dflt(4))

326  units="none", default=d11_dflt(3))

329  units="hK-2", default=d11_dflt(4))

368 subroutine initialize_ocmip2_cfc(restart, day, G, GV, h, diag, OBC, CS, &
369  sponge_csp, diag_to_z_csp)
370  logical, intent(in) :: restart
371  type(time_type), target, intent(in) :: day
372  type(ocean_grid_type), intent(in) :: G
373  type(verticalgrid_type), intent(in) :: GV
374  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
375  type(diag_ctrl), target, intent(in) :: diag
376  type(ocean_obc_type), pointer :: OBC
377  type(ocmip2_cfc_cs), pointer :: CS
378  type(sponge_cs), pointer :: sponge_CSp
379  type(diag_to_z_cs), pointer :: diag_to_Z_CSp
380 ! This subroutine initializes the NTR tracer fields in tr(:,:,:,:)
381 ! and it sets up the tracer output.
382 
383 ! Arguments: restart - .true. if the fields have already been read from
384 ! a restart file.
385 ! (in) day - Time of the start of the run.
386 ! (in) G - The ocean's grid structure.
387 ! (in) GV - The ocean's vertical grid structure.
388 ! (in) h - Layer thickness, in m or kg m-2.
389 ! (in) diag - A structure that is used to regulate diagnostic output.
390 ! (in) OBC - This open boundary condition type specifies whether, where,
391 ! and what open boundary conditions are used.
392 ! (in/out) CS - The control structure returned by a previous call to
393 ! register_OCMIP2_CFC.
394 ! (in/out) sponge_CSp - A pointer to the control structure for the sponges, if
395 ! they are in use. Otherwise this may be unassociated.
396 ! (in/out) diag_to_Z_Csp - A pointer to the control structure for diagnostics
397 ! in depth space.
398  logical :: from_file = .false.
399  character(len=16) :: name ! A variable's name in a NetCDF file.
400  character(len=72) :: longname ! The long name of that variable.
401  character(len=48) :: units ! The dimensions of the variable.
402  character(len=48) :: flux_units ! The units for tracer fluxes.
403  integer :: i, j, k, is, ie, js, je, isd, ied, jsd, jed, nz, m
404  integer :: IsdB, IedB, JsdB, JedB
405 
406  if (.not.associated(cs)) return
407  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = gv%ke
408  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
409  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
410 
411  cs%Time => day
412  cs%diag => diag
413

527 subroutine ocmip2_cfc_column_physics(h_old, h_new, ea, eb, fluxes, dt, G, GV, CS)
528  type(ocean_grid_type), intent(in) :: G
529  type(verticalgrid_type), intent(in) :: GV
530  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h_old, h_new, ea, eb
531  type(forcing), intent(in) :: fluxes
532  real, intent(in) :: dt
533  type(ocmip2_cfc_cs), pointer :: CS
534 ! This subroutine applies diapycnal diffusion and any other column
535 ! tracer physics or chemistry to the tracers from this file.
536 ! CFCs are relatively simple, as they are passive tracers. with only a surface
537 ! flux as a source.
538

540 ! (in) h_new - Layer thickness after entrainment, in m or kg m-2.
541 ! (in) ea - an array to which the amount of fluid entrained
542 ! from the layer above during this call will be
543 ! added, in m or kg m-2.
544 ! (in) eb - an array to which the amount of fluid entrained
545 ! from the layer below during this call will be
546 ! added, in m or kg m-2.
547 ! (in) fluxes - A structure containing pointers to any possible
548 ! forcing fields. Unused fields have NULL ptrs.
549 ! (in) dt - The amount of time covered by this call, in s.
550 ! (in) G - The ocean's grid structure.
551 ! (in) GV - The ocean's vertical grid structure.
552 ! (in) CS - The control structure returned by a previous call to
553 ! register_OCMIP2_CFC.
554 !
555 ! The arguments to this subroutine are redundant in that
556 ! h_new[k] = h_old[k] + ea[k] - eb[k-1] + eb[k] - ea[k+1]
557 
558  real :: b1(szi_(g)) ! b1 and c1 are variables used by the
559  real :: c1(szi_(g),szk_(g)) ! tridiagonal solver.
560  real, dimension(SZI_(G),SZJ_(G)) :: &
561  CFC11_flux, & ! The fluxes of CFC11 and CFC12 into the ocean, in the
562  CFC12_flux ! units of CFC concentrations times meters per second.

615 function ocmip2_cfc_stock(h, stocks, G, GV, CS, names, units, stock_index)
616  type(ocean_grid_type), intent(in) :: G
617  type(verticalgrid_type), intent(in) :: GV
618  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
619  real, dimension(:), intent(out) :: stocks
620  type(ocmip2_cfc_cs), pointer :: CS
621  character(len=*), dimension(:), intent(out) :: names
622  character(len=*), dimension(:), intent(out) :: units
623  integer, optional, intent(in) :: stock_index
624  integer :: OCMIP2_CFC_stock
625 ! This function calculates the mass-weighted integral of all tracer stocks,
626 ! returning the number of stocks it has calculated. If the stock_index
627 ! is present, only the stock corresponding to that coded index is returned.
628

630 ! (out) stocks - the mass-weighted integrated amount of each tracer,
631 ! in kg times concentration units.

633 ! (in) GV - The ocean's vertical grid structure.
634 ! (in) CS - The control structure returned by a previous call to
635 ! register_OCMIP2_CFC.
636 ! (out) names - the names of the stocks calculated.
637 ! (out) units - the units of the stocks calculated.
638 ! (in,opt) stock_index - the coded index of a specific stock being sought.
639 ! Return value: the number of stocks calculated here.
640 
641  real :: mass
642  integer :: i, j, k, is, ie, js, je, nz
643  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = gv%ke
644 
645  ocmip2_cfc_stock = 0
646  if (.not.associated(cs)) return
647 
648  if (present(stock_index)) then ; if (stock_index > 0) then
649  ! Check whether this stock is available from this routine.
650 
651  ! No stocks from this routine are being checked yet. Return 0.
652  return
653  endif ; endif
654 
655  call query_vardesc(cs%CFC11_desc, name=names(1), units=units(1), caller="OCMIP2_CFC_stock")
656  call query_vardesc(cs%CFC12_desc, name=names(2), units=units(2), caller="OCMIP2_CFC_stock")

662  stocks(1) = stocks(1) + cs%CFC11(i,j,k) * mass
663  stocks(2) = stocks(2) + cs%CFC12(i,j,k) * mass
664  enddo ; enddo ; enddo
665  stocks(1) = gv%H_to_kg_m2 * stocks(1)
666  stocks(2) = gv%H_to_kg_m2 * stocks(2)

672 subroutine ocmip2_cfc_surface_state(state, h, G, CS)
673  type(ocean_grid_type), intent(in) :: G
674  type(surface), intent(inout) :: state
675  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
676  type(ocmip2_cfc_cs), pointer :: CS
677 ! This subroutine sets up the fields that the coupler needs to calculate the
678 ! CFC fluxes between the ocean and atmosphere.
679 ! Arguments: state - A structure containing fields that describe the
680 ! surface state of the ocean.
681 ! (in) h - Layer thickness, in m or kg m-2.
682 ! (in) G - The ocean's grid structure.
683 ! (in) CS - The control structure returned by a previous call to
684 ! register_OCMIP2_CFC.
685 
686  real, dimension(SZI_(G),SZJ_(G)) :: &
687  CFC11_Csurf, & ! The CFC-11 and CFC-12 surface concentrations times the
688  CFC12_Csurf, & ! Schmidt number term, both in mol m-3.
689  CFC11_alpha, & ! The CFC-11 solubility in mol m-3 pptv-1.
690  CFC12_alpha ! The CFC-12 solubility in mol m-3 pptv-1.
691  real :: ta ! Absolute sea surface temperature in units of dekaKelvin!?!
692  real :: sal ! Surface salinity in PSU.
693  real :: SST ! Sea surface temperature in degrees Celsius.
694  real :: alpha_11 ! The solubility of CFC 11 in mol m-3 pptv-1.
695  real :: alpha_12 ! The solubility of CFC 12 in mol m-3 pptv-1.
696  real :: sc_11, sc_12 ! The Schmidt numbers of CFC 11 and CFC 12.
697  real :: sc_no_term ! A term related to the Schmidt number.
698  integer :: i, j, k, is, ie, js, je, m
699 
700  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
701 
702  if (.not.associated(cs)) return
703 
704  do j=js,je ; do i=is,ie
705  ta = max(0.01, (state%SST(i,j) + 273.15) * 0.01) ! Why is this in hectoKelvin?
706  sal = state%SSS(i,j) ; sst = state%SST(i,j)
707  ! Calculate solubilities using Warner and Weiss (1985) DSR, vol 32.
708  ! The final result is in mol/cm3/pptv (1 part per trillion 1e-12)
709  ! Use Bullister and Wisegavger for CCl4.
710  ! The factor 1.e-09 converts from mol/(l * atm) to mol/(m3 * pptv).
711  alpha_11 = exp(cs%d1_11 + cs%d2_11/ta + cs%d3_11*log(ta) + cs%d4_11*ta**2 +&
712  sal * ((cs%e3_11 * ta + cs%e2_11) * ta + cs%e1_11)) * &
713  1.0e-09 * g%mask2dT(i,j)
714  alpha_12 = exp(cs%d1_12 + cs%d2_12/ta + cs%d3_12*log(ta) + cs%d4_12*ta**2 +&
715  sal * ((cs%e3_12 * ta + cs%e2_12) * ta + cs%e1_12)) * &
716  1.0e-09 * g%mask2dT(i,j)
717  ! Calculate Schmidt numbers using coefficients given by
718  ! Zheng et al (1998), JGR vol 103, C1.
719  sc_11 = cs%a1_11 + sst * (cs%a2_11 + sst * (cs%a3_11 + sst * cs%a4_11)) * &
720  g%mask2dT(i,j)
721  sc_12 = cs%a1_12 + sst * (cs%a2_12 + sst * (cs%a3_12 + sst * cs%a4_12)) * &
722  g%mask2dT(i,j)
723  ! The abs here is to avoid NaNs. The model should be failing at this point.
724  sc_no_term = sqrt(660.0 / (abs(sc_11) + 1.0e-30))
725  cfc11_alpha(i,j) = alpha_11 * sc_no_term
726  cfc11_csurf(i,j) = cs%CFC11(i,j,1) * sc_no_term
727 
728  sc_no_term = sqrt(660.0 / (abs(sc_12) + 1.0e-30))
729  cfc12_alpha(i,j) = alpha_12 * sc_no_term
730  cfc12_csurf(i,j) = cs%CFC12(i,j,1) * sc_no_term
731  enddo ; enddo
732 
733  ! These calls load these values into the appropriate arrays in the
734  ! coupler-type structure.
735  call set_coupler_values(cfc11_alpha, state%tr_fields, cs%ind_cfc_11_flux, &
736  ind_alpha, is, ie, js, je)
737  call set_coupler_values(cfc11_csurf, state%tr_fields, cs%ind_cfc_11_flux, &
738  ind_csurf, is, ie, js, je)
739  call set_coupler_values(cfc12_alpha, state%tr_fields, cs%ind_cfc_12_flux, &
740  ind_alpha, is, ie, js, je)
741  call set_coupler_values(cfc12_csurf, state%tr_fields, cs%ind_cfc_12_flux, &
742  ind_csurf, is, ie, js, je)
743 
744 end subroutine ocmip2_cfc_surface_state
