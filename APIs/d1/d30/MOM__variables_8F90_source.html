382 subroutine mom_thermovar_chksum(mesg, tv, G)
383  character(len=*), intent(in) :: mesg
384  type(thermo_var_ptrs), intent(in) :: tv
385  type(ocean_grid_type), intent(in) :: G
386 ! This subroutine writes out chksums for the model's basic state variables.
387 ! Arguments: mesg - A message that appears on the chksum lines.
388 ! (in) u - Zonal velocity, in m s-1.
389 ! (in) v - Meridional velocity, in m s-1.
390 ! (in) h - Layer thickness, in m.
391 ! (in) uh - Volume flux through zonal faces = u*h*dy, m3 s-1.
392 ! (in) vh - Volume flux through meridional faces = v*h*dx, in m3 s-1.
393 ! (in) G - The ocean's grid structure.
394  integer :: is, ie, js, je, nz
395  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
396 
397  ! Note that for the chksum calls to be useful for reproducing across PE
398  ! counts, there must be no redundant points, so all variables use is..ie
399  ! and js...je as their extent.
400  if (associated(tv%T)) &
401  call hchksum(tv%T, mesg//" tv%T",g)
402  if (associated(tv%S)) &
403  call hchksum(tv%S, mesg//" tv%S",g)
404  if (associated(tv%frazil)) &
405  call hchksum(tv%frazil, mesg//" tv%frazil",g)
406  if (associated(tv%salt_deficit)) &
407  call hchksum(tv%salt_deficit, mesg//" tv%salt_deficit",g)
408  if (associated(tv%TempxPmE)) &
409  call hchksum(tv%TempxPmE, mesg//" tv%TempxPmE",g)
410 end subroutine mom_thermovar_chksum
