26 !* This file contains the subroutine (energetic_PBL) that uses an *
27 !* integrated boundary layer energy budget (like a bulk- or refined- *
28 !* bulk mixed layer scheme), but instead of homogenizing this model *
29 !* calculates a finite diffusivity and viscosity, which in this *
30 !* regard is conceptually similar to what is done with KPP or various *
31 !* two-equation closures. However, the scheme that is implemented *
32 !* here has the big advantage that is entirely implicit, but is *
33 !* simple enough that it requires only a single vertical pass to *
34 !* determine the diffusivity. The development of bulk mixed layer *
35 !* models stems from the work of various people, as described in the *
36 !* review paper by Niiler and Kraus (1979). The work here draws in *
37 !* with particular on the form for TKE decay proposed by Oberhuber *
38 !* (JPO, 1993, 808-829), with an extension to a refined bulk mixed *
39 !* layer as described in Hallberg (Aha Huliko'a, 2003). The physical *
40 !* processes portrayed in this subroutine include convectively driven *
41 !* mixing and mechanically driven mixing. Unlike boundary-layer *
42 !* mixing, stratified shear mixing is not a one-directional turbulent *
43 !* process, and it is dealt with elsewhere in the MOM6 code within *
44 !* the module MOM_kappa_shear.F90. It is assumed that the heat, *
45 !* mass, and salt fluxes have been applied elsewhere, but that their *
46 !* implications for the integrated TKE budget have been captured in *
47 !* an array that is provided as an argument to this subroutine. This *
48 !* is a full 3-d array due to the effects of penetrating shortwave *
49 !* radiation. *
50 !* *
51 !* Macros written all in capital letters are defined in MOM_memory.h. *
52 !* *
53 !* A small fragment of the grid is shown below: *
54 !* *
55 !* j+1 x ^ x ^ x At x: q *
56 !* j+1 > o > o > At ^: v *
57 !* j x ^ x ^ x At >: u *
58 !* j > o > o > At o: h, T, S, Kd, etc. *
59 !* j-1 x ^ x ^ x *
60 !* i-1 i i+1 At x & ^: *
61 !* i i+1 At > & o: *
62 !* *
63 !* The boundaries always run through q grid points (x). *
64 !* *

160 ! This subroutine determines the diffusivities from the integrated energetics
161 ! mixed layer model. It assumes that heating, cooling and freshwater fluxes
162 ! have already been applied. All calculations are done implicitly, and there
163 ! is no stability limit on the time step.
164 !
165 ! For each interior interface, first discard the TKE to account for mixing
166 ! of shortwave radiation through the next denser cell. Next drive mixing based
167 ! on the local? values of ustar + wstar, subject to available energy. This
168 ! step sets the value of Kd(K). Any remaining energy is then subject to decay
169 ! before being handed off to the next interface. mech_TKE and conv_PErel are treated
170 ! separately for the purposes of decay, but are used proportionately to drive
171 ! mixing.
172 !
173 ! The key parameters for the mixed layer are found in the control structure.
174 ! These include mstar, nstar, TKE_decay, and conv_decay. For the Oberhuber (1993) mixed layer,
175 ! the values of these are:
176 ! mstar = 1.25, nstar = 1, TKE_decay = 2.5, conv_decay = 0.5
177 ! TKE_decay is 1/kappa in eq. 28 of Oberhuber (1993), while
178 ! conv_decay is 1/mu.
179 ! For a traditional Kraus-Turner mixed layer, the values are:
180 ! mstar = 1.25, nstar = 0.4, TKE_decay = 0.0, conv_decay = 0.0

414 !!OMP dMKE_src_dK,TKE_left,use_Newt, &
415 !!OMP dKddt_h_Newt,Kddt_h_Newt,Kddt_h_next, &
416 !!OMP dKddt_h,Te,Se,Hsfc_used,dS_to_dPE_a, &
417 !!OMP dMKE_max,sfc_connected,TKE_here)
418  do j=js,je
419  ! Copy the thicknesses and other fields to 2-d arrays.
420  do k=1,nz ; do i=is,ie
421  h(i,k) = h_3d(i,j,k) + h_neglect ; u(i,k) = u_3d(i,j,k) ; v(i,k) = v_3d(i,j,k)
422  t(i,k) = tv%T(i,j,k) ; s(i,k) = tv%S(i,j,k)
423  kd(i,k) = 0.0
424  enddo ; enddo
425  do i=is,ie ; cs%ML_depth(i,j) = h(i,1)*gv%H_to_m ; sfc_connected(i) = .true. ; enddo
426 
427  if (debug) then
428  mech_tke_k(:,:) = 0.0 ; conv_perel_k(:,:) = 0.0
429  endif
430 
431  ! Determine the initial mech_TKE and conv_PErel, including the energy required
432  ! to mix surface heating through the topmost cell, the energy released by mixing
433  ! surface cooling & brine rejection down through the topmost cell, and
434  ! homogenizing the shortwave heating within that cell. This sets the energy
435  ! and ustar and wstar available to drive mixing at the first interior
436  ! interface.
437  do i=is,ie ; if (g%mask2dT(i,j) > 0.5) then
438  u_star = fluxes%ustar(i,j)
439  if (associated(fluxes%ustar_shelf) .and. associated(fluxes%frac_shelf_h)) then
440  if (fluxes%frac_shelf_h(i,j) > 0.0) &
441  u_star = (1.0 - fluxes%frac_shelf_h(i,j)) * u_star + &
442  fluxes%frac_shelf_h(i,j) * fluxes%ustar_shelf(i,j)
443  endif
444  if (u_star < cs%ustar_min) u_star = cs%ustar_min
445 
446  if (cs%use_omega) then ; absf(i) = 2.0*cs%omega
447  else
448  absf(i) = 0.25*((abs(g%CoriolisBu(i,j)) + abs(g%CoriolisBu(i-1,j-1))) + &
449  (abs(g%CoriolisBu(i,j-1)) + abs(g%CoriolisBu(i-1,j))))
450  endif
451 
452  mech_tke(i) = (dt*cs%mstar*gv%Rho0)*((u_star**3))
453  conv_perel(i) = 0.0
454 
455  if (cs%TKE_diagnostics) then
456  cs%diag_TKE_wind(i,j) = cs%diag_TKE_wind(i,j) + mech_tke(i) * idtdr0
457  if (tke_forced(i,j,1) <= 0.0) then
458  cs%diag_TKE_forcing(i,j) = cs%diag_TKE_forcing(i,j) + &
459  max(-mech_tke(i), tke_forced(i,j,1)) * idtdr0
460  else
461  cs%diag_TKE_forcing(i,j) = cs%diag_TKE_forcing(i,j) + cs%nstar*tke_forced(i,j,1) * idtdr0
462  endif
463  endif
464 
465  if (tke_forced(i,j,1) <= 0.0) then
466  mech_tke(i) = mech_tke(i) + tke_forced(i,j,1)
467  if (mech_tke(i) < 0.0) mech_tke(i) = 0.0
468  else
469  conv_perel(i) = conv_perel(i) + tke_forced(i,j,1)
470  endif
471 
472 ! endif ; enddo
473 
474 ! do i=is,ie ; if (G%mask2dT(i,j) > 0.5) then
475  if (debug) then
476  mech_tke_k(i,1) = mech_tke(i) ; conv_perel_k(i,1) = conv_perel(i)
477  nstar_k(:) = 0.0 ; nstar_k(1) = cs%nstar
478  endif
479 
480  h_sum(i) = h_neglect
481  do k=1,nz ; h_sum(i) = h_sum(i) + h(i,k) ; enddo
482  i_hs = 0.0 ; if (h_sum(i) > 0.0) i_hs = 1.0 / h_sum(i)
483 
484  h_bot(i) = 0.0 ; hb_hs(i,nz+1) = 0.0
485  do k=nz,1,-1
486  h_bot(i) = h_bot(i) + h(i,k)
487  hb_hs(i,k) = gv%H_to_m * (h_bot(i)*i_hs)
488  enddo
489 ! endif ; enddo
490 
491  ! Note the outer i-loop and inner k-loop loop order!!!
492 ! do i=is,ie ; if (G%mask2dT(i,j) > 0.5) then
493  do k=1,nz ; t0(k) = t(i,k) ; s0(k) = s(i,k) ; enddo
494 
495  num_itts(:) = -1
496 
497  pres(i,1) = 0.0
498  do k=1,nz
499  dmass = gv%H_to_kg_m2 * h(i,k)
500  dpres = g%G_Earth * dmass
501  dt_to_dpe(i,k) = (dmass * (pres(i,k) + 0.5*dpres)) * dsv_dt(i,j,k)
502  ds_to_dpe(i,k) = (dmass * (pres(i,k) + 0.5*dpres)) * dsv_ds(i,j,k)
503  dt_to_dcolht(i,k) = dmass * dsv_dt(i,j,k)
504  ds_to_dcolht(i,k) = dmass * dsv_ds(i,j,k)
505 
506  pres(i,k+1) = pres(i,k) + dpres
507  enddo
508 
509  kd(i,1) = 0.0 ; kddt_h(1) = 0.0
510  b_den_1(i) = h(i,1)
511  dt_to_dpe_a(i,1) = dt_to_dpe(i,1)
512  ds_to_dpe_a(i,1) = ds_to_dpe(i,1)
513  dt_to_dcolht_a(i,1) = dt_to_dcolht(i,1)
514  ds_to_dcolht_a(i,1) = ds_to_dcolht(i,1)
515 
516  htot(i) = h(i,1)
517  uhtot(i) = u(i,1)*h(i,1)
518  vhtot(i) = v(i,1)*h(i,1)
519 
520  do k=2,nz
521  ! Apply dissipation to the TKE, here applied as an exponential decay
522  ! due to 3-d turbulent energy being lost to inefficient rotational modes.
523 
524  ! There should be several different "flavors" of TKE that decay at
525  ! different rates. The following form is often used for mechanical
526  ! stirring from the surface, perhaps due to breaking surface gravity
527  ! waves and wind-driven turbulence.
528  idecay_len_tke(i) = (cs%TKE_decay * absf(i) / u_star) * gv%H_to_m
529  exp_kh = 1.0
530  if (idecay_len_tke(i) > 0.0) exp_kh = exp(-h(i,k-1)*idecay_len_tke(i))
531  if (cs%TKE_diagnostics) cs%diag_TKE_mech_decay(i,j) = &
532  cs%diag_TKE_mech_decay(i,j) + (exp_kh-1.0) * mech_tke(i) * idtdr0
533  mech_tke(i) = mech_tke(i) * exp_kh
534 
535 
536  ! Accumulate any convectively released potential energy to contribute
537  ! to wstar and to drive penetrating convection.
538  if (tke_forced(i,j,k) > 0.0) then
539  conv_perel(i) = conv_perel(i) + tke_forced(i,j,k)
540  if (cs%TKE_diagnostics) then
541  cs%diag_TKE_forcing(i,j) = cs%diag_TKE_forcing(i,j) + cs%nstar*tke_forced(i,j,k) * idtdr0
542  endif
543  endif
544 
545  if (debug) then
546  mech_tke_k(i,k) = mech_tke(i) ; conv_perel_k(i,k) = conv_perel(i)
547  endif
548 
549  ! Determine the total energy
550  nstar_fc = cs%nstar
551  if (cs%nstar * conv_perel(i) > 0.0) then
552  ! Here nstar is a function of the natural Rossby number 0.2/(1+0.2/Ro), based
553  ! on a curve fit from the data of Wang (GRL, 2003).
554  ! Note: Ro = 1.0 / sqrt(0.5 * dt * Rho0 * (absf*htot(i))**3 / conv_PErel(i))
555  nstar_fc = cs%nstar * conv_perel(i) / (conv_perel(i) + 0.2 * &
556  sqrt(0.5 * dt * gv%Rho0 * (absf(i)*(htot(i)*gv%H_to_m))**3 * conv_perel(i)))
557  endif
558  if (debug) nstar_k(k) = nstar_fc
559 
560  tot_tke = mech_tke(i) + nstar_fc * conv_perel(i)
561 
562  ! For each interior interface, first discard the TKE to account for
563  ! mixing of shortwave radiation through the next denser cell.
564  if (tke_forced(i,j,k) < 0.0) then
565  if (tke_forced(i,j,k) + tot_tke < 0.0) then
566  ! The shortwave requirements deplete all the energy in this layer.
567  if (cs%TKE_diagnostics) then
568  cs%diag_TKE_mixing(i,j) = cs%diag_TKE_mixing(i,j) - tot_tke * idtdr0
569  cs%diag_TKE_conv_decay(i,j) = cs%diag_TKE_conv_decay(i,j) + &
570  (1.0-tke_reduc)*(cs%nstar-nstar_fc) * conv_perel(i) * idtdr0
571  endif
572  tot_tke = 0.0 ; mech_tke(i) = 0.0 ; conv_perel(i) = 0.0
573  else
574  ! Reduce the mechanical and convective TKE proportionately.
575  tke_reduc = (tot_tke + tke_forced(i,j,k)) / tot_tke
576  if (cs%TKE_diagnostics) then
577  cs%diag_TKE_mixing(i,j) = cs%diag_TKE_mixing(i,j) - tke_forced(i,j,k) * idtdr0
578  cs%diag_TKE_forcing(i,j) = cs%diag_TKE_forcing(i,j) + tke_forced(i,j,k) * idtdr0
579  cs%diag_TKE_conv_decay(i,j) = cs%diag_TKE_conv_decay(i,j) + &
580  (1.0-tke_reduc)*(cs%nstar-nstar_fc) * conv_perel(i) * idtdr0
581  endif
582  tot_tke = tke_reduc*tot_tke
583  mech_tke(i) = tke_reduc*mech_tke(i)
584  conv_perel(i) = tke_reduc*conv_perel(i)
585  endif
586  endif
587 
588  ! Precalculate some temporary expressions that are independent of Kddt_h(K).
589  if (k==2) then
590  dte_t2 = 0.0 ; dse_t2 = 0.0
591  else
592  dte_t2 = kddt_h(k-1) * ((t0(k-2) - t0(k-1)) + dte(k-2))
593  dse_t2 = kddt_h(k-1) * ((s0(k-2) - s0(k-1)) + dse(k-2))
594  endif
595  dt_h = (gv%m_to_H**2*dt) / max(0.5*(h(i,k-1)+h(i,k)), 1e-15*h_sum(i))
596 
597  ! This tests whether the layers above and below this interface are in
598  ! a convetively stable configuration, without considering any effects of
599  ! mixing at higher interfaces. It is an approximation to the more
600  ! complete test dPEc_dKd_Kd0 >= 0.0, that would include the effects of
601  ! mixing across interface K-1. The dT_to_dColHt here are effectively
602  ! mass-weigted estimates of dSV_dT.
603  convectively_stable = ( 0.0 <= &
604  ( (dt_to_dcolht(i,k) + dt_to_dcolht(i,k-1) ) * (t0(k-1)-t0(k)) + &
605  (ds_to_dcolht(i,k) + ds_to_dcolht(i,k-1) ) * (s0(k-1)-s0(k)) ) )
606 
607  if ((mech_tke(i) + conv_perel(i)) <= 0.0 .and. convectively_stable) then
608  ! Energy is already exhausted, so set Kd = 0 and cycle or exit?
609  tot_tke = 0.0 ; mech_tke(i) = 0.0 ; conv_perel(i) = 0.0
610  kd(i,k) = 0.0 ; kddt_h(k) = 0.0
611  sfc_disconnect = .true.
612  ! if (.not.debug) exit
613 
614  ! The estimated properties for layer k-1 can be calculated, using
615  ! greatly simplified expressions when Kddt_h = 0. This enables the
616  ! tridiagonal solver for the whole column to be completed for debugging
617  ! purposes, and also allows for somethign akin to convective adjustment
618  ! in unstable interior regions?
619  b1 = 1.0 / (b_den_1(i))
620  c1(k) = 0.0
621  dte(k-1) = b1 * ( dte_t2 )
622  dse(k-1) = b1 * ( dse_t2 )
623 
624  b_den_1(i) = h(i,k)
625  dt_to_dpe_a(i,k) = dt_to_dpe(i,k)
626  ds_to_dpe_a(i,k) = ds_to_dpe(i,k)
627  dt_to_dcolht_a(i,k) = dt_to_dcolht(i,k)
628  ds_to_dcolht_a(i,k) = ds_to_dcolht(i,k)
629 
630  else ! tot_TKE > 0.0 or this is a potentially convectively unstable profile.
631  sfc_disconnect = .false.
632 
633  ! Precalculate some more temporary expressions that are independent of
634  ! Kddt_h(K).
635  if (k==2) then
636  dt_km1_t2 = (t0(k)-t0(k-1))
637  ds_km1_t2 = (s0(k)-s0(k-1))
638  else
639  dt_km1_t2 = (t0(k)-t0(k-1)) - &
640  (kddt_h(k-1) / b_den_1(i)) * ((t0(k-2) - t0(k-1)) + dte(k-2))
641  ds_km1_t2 = (s0(k)-s0(k-1)) - &
642  (kddt_h(k-1) / b_den_1(i)) * ((s0(k-2) - s0(k-1)) + dse(k-2))
643  endif
644  dte_term = dte_t2 + b_den_1(i) * (t0(k-1)-t0(k))
645  dse_term = dse_t2 + b_den_1(i) * (s0(k-1)-s0(k))
646 
647  ! Using Pr=1 and the diffusivity at the bottom interface (once it is
648  ! known), determine how much resolved mean kinetic energy (MKE) will be
649  ! extracted within a timestep and add a fraction CS%MKE_to_TKE_effic of
650  ! this to the mTKE budget available for mixing in the next layer.
651 
652  if ((cs%MKE_to_TKE_effic > 0.0) .and. (htot(i)*h(i,k) > 0.0)) then
653  ! This is the energy that would be available from homogenizing the
654  ! velocities between layer k and the layers above.
655  dmke_max = (gv%H_to_kg_m2 * cs%MKE_to_TKE_effic) * 0.5 * &
656  (h(i,k) / ((htot(i) + h(i,k))*htot(i))) * &
657  ((uhtot(i)-u(i,k)*htot(i))**2 + (vhtot(i)-v(i,k)*htot(i))**2)
658  ! A fraction (1-exp(Kddt_h*MKE2_Hharm)) of this energy would be
659  ! extracted by mixing with a finite viscosity.
660  mke2_hharm = (htot(i) + h(i,k) + 2.0*h_neglect) / &
661  ((htot(i)+h_neglect) * (h(i,k)+h_neglect))
662  else
663  dmke_max = 0.0 ; mke2_hharm = 0.0
664  endif
665 
666  ! At this point, Kddt_h(K) will be unknown because its value may depend
667  ! on how much energy is available. mech_TKE might be negative due to
668  ! contributions from TKE_forced.
669  h_tt = htot(i) + h_tt_min
670  tke_here = mech_tke(i) + cs%wstar_ustar_coef*conv_perel(i)
671  if (tke_here > 0.0) then
672  vstar = cs%vstar_scale_coef * (i_dtmrho*tke_here)**c1_3
673  kd_guess0 = vstar * vonkar * ((h_tt*hb_hs(i,k))*vstar) / &
674  ((cs%Ekman_scale_coef * absf(i)) * (h_tt*hb_hs(i,k)) + vstar)
675  else
676  vstar = 0.0 ; kd_guess0 = 0.0
677  endif
678  kddt_h_g0 = kd_guess0*dt_h
679 
680  call find_pe_chg(kddt_h_g0, h(i,k), b_den_1(i), dte_term, dse_term, &
681  dt_km1_t2, ds_km1_t2, dt_to_dpe(i,k), ds_to_dpe(i,k), &
682  dt_to_dpe_a(i,k-1), ds_to_dpe_a(i,k-1), &
683  pres(i,k), dt_to_dcolht(i,k), ds_to_dcolht(i,k), &
684  dt_to_dcolht_a(i,k-1), ds_to_dcolht_a(i,k-1), &
685  pe_chg=pe_chg_g0, dpec_dkd=dpea_dkd_g0, dpe_max=pe_chg_max, &
686  dpec_dkd_0=dpec_dkd_kd0 )
687 
688  mke_src = dmke_max*(1.0 - exp(-kddt_h_g0 * mke2_hharm))
689 
690  if ((pe_chg_g0 < 0.0) .or. ((vstar == 0.0) .and. (dpec_dkd_kd0 < 0.0))) then
691  ! This column is convectively unstable.
692  if (pe_chg_max <= 0.0) then
693  ! Does MKE_src need to be included in the calculation of vstar here?
694  tke_here = mech_tke(i) + cs%wstar_ustar_coef*(conv_perel(i)-pe_chg_max)
695  if (tke_here > 0.0) then
696  vstar = cs%vstar_scale_coef * (i_dtmrho*tke_here)**c1_3
697  kd(i,k) = vstar * vonkar * ((h_tt*hb_hs(i,k))*vstar) / &
698  ((cs%Ekman_scale_coef * absf(i)) * (h_tt*hb_hs(i,k)) + vstar)
699  else
700  vstar = 0.0 ; kd(i,k) = 0.0
701  endif
702 
703  call find_pe_chg(kd(i,k)*dt_h, h(i,k), b_den_1(i), dte_term, dse_term, &
704  dt_km1_t2, ds_km1_t2, dt_to_dpe(i,k), ds_to_dpe(i,k), &
705  dt_to_dpe_a(i,k-1), ds_to_dpe_a(i,k-1), &
706  pres(i,k), dt_to_dcolht(i,k), ds_to_dcolht(i,k), &
707  dt_to_dcolht_a(i,k-1), ds_to_dcolht_a(i,k-1), &
708  pe_chg=dpe_conv)
709  ! Should this be iterated to convergence for Kd?
710  if (dpe_conv > 0.0) then
711  kd(i,k) = kd_guess0 ; dpe_conv = pe_chg_g0
712  else
713  mke_src = dmke_max*(1.0 - exp(-(kd(i,k)*dt_h) * mke2_hharm))
714  endif
715  else
716  ! The energy change does not vary monotonically with Kddt_h. Find the maximum?
717  kd(i,k) = kd_guess0 ; dpe_conv = pe_chg_g0
718  endif
719  conv_perel(i) = conv_perel(i) - dpe_conv
720  mech_tke(i) = mech_tke(i) + mke_src
721  if (cs%TKE_diagnostics) then
722  cs%diag_TKE_conv(i,j) = cs%diag_TKE_conv(i,j) - dpe_conv * idtdr0
723  cs%diag_TKE_MKE(i,j) = cs%diag_TKE_MKE(i,j) + mke_src * idtdr0
724  endif
725  if (sfc_connected(i)) cs%ML_depth(i,j) = cs%ML_depth(i,j) + &
726  gv%H_to_m * h(i,k)
727 
728  kddt_h(k) = kd(i,k)*dt_h
729  elseif (tot_tke + (mke_src - pe_chg_g0) >= 0.0) then
730  ! There is energy to support the suggested mixing. Keep that estimate.
731  kd(i,k) = kd_guess0
732  kddt_h(k) = kddt_h_g0
733 
734  ! Reduce the mechanical and convective TKE proportionately.
735  tot_tke = tot_tke + mke_src
736  tke_reduc = 0.0 ! tot_TKE could be 0 if Convectively_stable is false.
737  if (tot_tke > 0.0) tke_reduc = (tot_tke - pe_chg_g0) / tot_tke
738 
739  if (cs%TKE_diagnostics) then
740  cs%diag_TKE_mixing(i,j) = cs%diag_TKE_mixing(i,j) - pe_chg_g0 * idtdr0
741  cs%diag_TKE_MKE(i,j) = cs%diag_TKE_MKE(i,j) + mke_src * idtdr0
742  cs%diag_TKE_conv_decay(i,j) = cs%diag_TKE_conv_decay(i,j) + &
743  (1.0-tke_reduc)*(cs%nstar-nstar_fc) * conv_perel(i) * idtdr0
744  endif
745  tot_tke = tke_reduc*tot_tke
746  mech_tke(i) = tke_reduc*(mech_tke(i) + mke_src)
747  conv_perel(i) = tke_reduc*conv_perel(i)
748  if (sfc_connected(i)) cs%ML_depth(i,j) = cs%ML_depth(i,j) + &
749  gv%H_to_m * h(i,k)
750  elseif (tot_tke == 0.0) then
751  ! This can arise if nstar_FC = 0.
752  kd(i,k) = 0.0 ; kddt_h(k) = 0.0
753  tot_tke = 0.0 ; conv_perel(i) = 0.0 ; mech_tke(i) = 0.0
754  sfc_disconnect = .true.
755  else
756  ! There is not enough energy to support the mixing, so reduce the
757  ! diffusivity to what can be supported.
758  kddt_h_max = kddt_h_g0 ; kddt_h_min = 0.0
759  tke_left_max = tot_tke + (mke_src - pe_chg_g0) ; tke_left_min = tot_tke
760 
761  ! As a starting guess, take the minimum of a false position estimate
762  ! and a Newton's method estimate starting from Kddt_h = 0.0.
763  kddt_h_guess = tot_tke * kddt_h_max / max( pe_chg_g0 - mke_src, &
764  kddt_h_max * (dpec_dkd_kd0 - dmke_max * mke2_hharm) )
765  ! The above expression is mathematically the same as the following
766  ! except it is not susceptible to division by zero when
767  ! dPEc_dKd_Kd0 = dMKE_max = 0 .
768  ! Kddt_h_guess = tot_TKE * min( Kddt_h_max / (PE_chg_g0 - MKE_src), &
769  ! 1.0 / (dPEc_dKd_Kd0 - dMKE_max * MKE2_Hharm) )
770  if (debug) then
771  tke_left_itt(:) = 0.0 ; dpea_dkd_itt(:) = 0.0 ; pe_chg_itt(:) = 0.0
772  mke_src_itt(:) = 0.0 ; kddt_h_itt(:) = 0.0
773  endif
774  do itt=1,max_itt
775  call find_pe_chg(kddt_h_guess, h(i,k), b_den_1(i), dte_term, dse_term, &
776  dt_km1_t2, ds_km1_t2, dt_to_dpe(i,k), ds_to_dpe(i,k), &
777  dt_to_dpe_a(i,k-1), ds_to_dpe_a(i,k-1), &
778  pres(i,k), dt_to_dcolht(i,k), ds_to_dcolht(i,k), &
779  dt_to_dcolht_a(i,k-1), ds_to_dcolht_a(i,k-1), &
780  pe_chg=pe_chg, dpec_dkd=dpec_dkd )
781  mke_src = dmke_max * (1.0 - exp(-mke2_hharm * kddt_h_guess))
782  dmke_src_dk = dmke_max * mke2_hharm * exp(-mke2_hharm * kddt_h_guess)
783 
784  tke_left = tot_tke + (mke_src - pe_chg)
785  if (debug) then
786  kddt_h_itt(itt) = kddt_h_guess ; mke_src_itt(itt) = mke_src
787  pe_chg_itt(itt) = pe_chg ; tke_left_itt(itt) = tke_left
788  dpea_dkd_itt(itt) = dpec_dkd
789  endif
790  ! Store the new bounding values, bearing in mind that min and max
791  ! here refer to Kddt_h and dTKE_left/dKddt_h < 0:
792  if (tke_left >= 0.0) then
793  kddt_h_min = kddt_h_guess ; tke_left_min = tke_left
794  else
795  kddt_h_max = kddt_h_guess ; tke_left_max = tke_left
796  endif
797 
798  ! Try to use Newton's method, but if it would go outside the bracketed
799  ! values use the false-position method instead.
800  use_newt = .true.
801  if (dpec_dkd - dmke_src_dk <= 0.0) then
802  use_newt = .false.
803  else
804  dkddt_h_newt = tke_left / (dpec_dkd - dmke_src_dk)
805  kddt_h_newt = kddt_h_guess + dkddt_h_newt
806  if ((kddt_h_newt > kddt_h_max) .or. (kddt_h_newt < kddt_h_min)) &
807  use_newt = .false.
808  endif
809 
810  if (use_newt) then
811  kddt_h_next = kddt_h_guess + dkddt_h_newt
812  dkddt_h = dkddt_h_newt
813  else
814  kddt_h_next = (tke_left_max * kddt_h_min - kddt_h_max * tke_left_min) / &
815  (tke_left_max - tke_left_min)
816  dkddt_h = kddt_h_next - kddt_h_guess
817  endif
818 
819  if ((abs(dkddt_h) < 1e-9*kddt_h_guess) .or. (itt==max_itt)) then
820  ! Use the old value so that the energy calculation does not need to be repeated.
821  if (debug) num_itts(k) = itt
822  exit
823  else
824  kddt_h_guess = kddt_h_next
825  endif
826  enddo
827  kd(i,k) = kddt_h_guess / dt_h ; kddt_h(k) = kd(i,k)*dt_h
828 
829  ! All TKE should have been consumed.
830  if (cs%TKE_diagnostics) then
831  cs%diag_TKE_mixing(i,j) = cs%diag_TKE_mixing(i,j) - (tot_tke + mke_src) * idtdr0
832  cs%diag_TKE_MKE(i,j) = cs%diag_TKE_MKE(i,j) + mke_src * idtdr0
833  cs%diag_TKE_conv_decay(i,j) = cs%diag_TKE_conv_decay(i,j) + &
834  (cs%nstar-nstar_fc) * conv_perel(i) * idtdr0
835  endif
836 
837  if (sfc_connected(i)) cs%ML_depth(i,j) = cs%ML_depth(i,j) + &
838  (pe_chg / pe_chg_g0) * gv%H_to_m * h(i,k)
839  tot_tke = 0.0 ; mech_tke(i) = 0.0 ; conv_perel(i) = 0.0
840  sfc_disconnect = .true.
841  endif
842 
843  kddt_h(k) = kd(i,k)*dt_h
844  ! At this point, the final value of Kddt_h(K) is known, so the
845  ! estimated properties for layer k-1 can be calculated.
846  b1 = 1.0 / (b_den_1(i) + kddt_h(k))
847  c1(k) = kddt_h(k) * b1
848  dte(k-1) = b1 * ( kddt_h(k)*(t0(k)-t0(k-1)) + dte_t2 )
849  dse(k-1) = b1 * ( kddt_h(k)*(s0(k)-s0(k-1)) + dse_t2 )
850 
851  b_den_1(i) = h(i,k) + (b_den_1(i) * b1) * kddt_h(k)
852  dt_to_dpe_a(i,k) = dt_to_dpe(i,k) + c1(k)*dt_to_dpe_a(i,k-1)
853  ds_to_dpe_a(i,k) = ds_to_dpe(i,k) + c1(k)*ds_to_dpe_a(i,k-1)
854  dt_to_dcolht_a(i,k) = dt_to_dcolht(i,k) + c1(k)*dt_to_dcolht_a(i,k-1)
855  ds_to_dcolht_a(i,k) = ds_to_dcolht(i,k) + c1(k)*ds_to_dcolht_a(i,k-1)
856 
857  endif ! tot_TKT > 0.0 branch. Kddt_h(K) has been set.
858 
859  ! Store integrated velocities and thicknesses for MKE conversion calculations.
860  if (sfc_disconnect) then
861  ! There is no turbulence at this interface, so zero out the running sums.
862  uhtot(i) = u(i,k)*h(i,k)
863  vhtot(i) = v(i,k)*h(i,k)
864  htot(i) = h(i,k)
865  sfc_connected(i) = .false.
866  else
867  uhtot(i) = uhtot(i) + u(i,k)*h(i,k)
868  vhtot(i) = vhtot(i) + v(i,k)*h(i,k)
869  htot(i) = htot(i) + h(i,k)
870  endif
871 
872  if (debug) then
873  if (k==2) then
874  te(1) = b1*(h(i,1)*t0(1))
875  se(1) = b1*(h(i,1)*s0(1))
876  else
877  te(k-1) = b1 * (h(i,k-1) * t0(k-1) + kddt_h(k-1) * te(k-2))
878  se(k-1) = b1 * (h(i,k-1) * s0(k-1) + kddt_h(k-1) * se(k-2))
879  endif
880  endif
881  enddo
882  kd(i,nz+1) = 0.0
883 
884  if (debug) then
885  ! Complete the tridiagonal solve for Te.
886  b1 = 1.0 / (b_den_1(i))
887  te(nz) = b1 * (h(i,nz) * t0(nz) + kddt_h(nz) * te(nz-1))
888  se(nz) = b1 * (h(i,nz) * s0(nz) + kddt_h(nz) * se(nz-1))
889  do k=nz-1,1,-1
890  te(k) = te(k) + c1(k+1)*te(k+1)
891  se(k) = se(k) + c1(k+1)*se(k+1)
892  enddo
893  endif
894  k = nz ! This is here to allow a breakpoint to be set.
895  else
896  ! For masked points, Kd_int must still be set (to 0) because it has intent(out).
897  do k=1,nz+1
898  kd(i,k) = 0.
899  enddo
900  endif ; enddo ; ! Close of i-loop - Note unusual loop order!
901 
902  if (cs%id_Hsfc_used > 0) then
903  do i=is,ie ; hsfc_used(i,j) = h(i,1)*gv%H_to_m ; enddo
904  do k=2,nz ; do i=is,ie
905  if (kd(i,k) > 0.0) hsfc_used(i,j) = hsfc_used(i,j) + h(i,k)*gv%H_to_m
906  enddo ; enddo
907  endif
908 
909  do k=1,nz+1 ; do i=is,ie
910  kd_int(i,j,k) = kd(i,k)
911  enddo ; enddo
912 
913  enddo ! j-loop
914 
915  if (write_diags) then
916  if (cs%id_ML_depth > 0) &
917  call post_data(cs%id_ML_depth, cs%ML_depth, cs%diag)
918  if (cs%id_TKE_wind > 0) &
919  call post_data(cs%id_TKE_wind, cs%diag_TKE_wind, cs%diag)
920  if (cs%id_TKE_MKE > 0) &
921  call post_data(cs%id_TKE_MKE, cs%diag_TKE_MKE, cs%diag)
922  if (cs%id_TKE_conv > 0) &
923  call post_data(cs%id_TKE_conv, cs%diag_TKE_conv, cs%diag)
924  if (cs%id_TKE_forcing > 0) &
925  call post_data(cs%id_TKE_forcing, cs%diag_TKE_forcing, cs%diag)
926  if (cs%id_TKE_mixing > 0) &
927  call post_data(cs%id_TKE_mixing, cs%diag_TKE_mixing, cs%diag)
928  if (cs%id_TKE_mech_decay > 0) &
929  call post_data(cs%id_TKE_mech_decay, cs%diag_TKE_mech_decay, cs%diag)
930  if (cs%id_TKE_conv_decay > 0) &
931  call post_data(cs%id_TKE_conv_decay, cs%diag_TKE_conv_decay, cs%diag)
932  if (cs%id_Hsfc_used > 0) &
933  call post_data(cs%id_Hsfc_used, hsfc_used, cs%diag)
934  endif
935 
936 end subroutine energetic_pbl

938 subroutine find_pe_chg(Kddt_h, h_k, b_den_1, dTe_term, dSe_term, &
939  dt_km1_t2, ds_km1_t2, dt_to_dpe_k, ds_to_dpe_k, &
940  dt_to_dpea, ds_to_dpea, pres, dt_to_dcolht_k, &
941  ds_to_dcolht_k, dt_to_dcolhta, ds_to_dcolhta, &
942  pe_chg, dpec_dkd, dpe_max, dpec_dkd_0)
943  real, intent(in) :: Kddt_h, h_k, b_den_1, dTe_term, dSe_term
944  real, intent(in) :: dT_km1_t2, dS_km1_t2, pres
945  real, intent(in) :: dT_to_dPE_k, dS_to_dPE_k, dT_to_dPEa, dS_to_dPEa
946  real, intent(in) :: dT_to_dColHt_k, dS_to_dColHt_k, dT_to_dColHta, dS_to_dColHta
947  real, optional, intent(out) :: PE_chg, dPEc_dKd, dPE_max, dPEc_dKd_0
948 
949 ! This subroutine determines the total potential energy change due to mixing
950 ! at an interface, including all of the implicit effects of the prescribed
951 ! mixing at interfaces above. Everything here is derived by careful manipulation
952 ! of the robust tridiagonal solvers used for tracers by MOM6. The results are
953 ! positive for mixing in a stably stratified environment.
954 ! The comments describing these arguments are for a downward mixing pass, but
955 ! this routine can also be used for an upward pass with the sense of direction
956 ! reversed.
957 !
958 ! Arguments: Kddt_h - The diffusivity at an interface times the time step and
959 ! divided by the average of the thicknesses around the
960 ! interface, in units of H (m or kg-2).
961 ! (in) h_k - The thickness of the layer below the interface, in H.
962 ! (in) b_den_1 - The first first term in the denominator of the pivot
963 ! for the tridiagonal solver, given by h_k plus a term that
964 ! is a fraction (determined from the tridiagonal solver) of
965 ! Kddt_h for the interface above, in H.
966 ! (in) dTe_term - A diffusivity-independent term related to the
967 ! temperature change in the layer below the interface, in K H.
968 ! (in) dSe_term - A diffusivity-independent term related to the
969 ! salinity change in the layer below the interface, in ppt H.
970 ! (in) dT_km1_t2 - A diffusivity-independent term related to the
971 ! temperature change in the layer above the interface, in K.
972 ! (in) dS_km1_t2 - A diffusivity-independent term related to the
973 ! salinity change in the layer above the interface, in ppt.
974 ! (in) dT_to_dPE_k - A factor (pres_lay*mass_lay*dSpec_vol/dT) relating
975 ! a layer's temperature change to the change in column
976 ! potential energy, in J m-2 K-1.
977 ! (in) dS_to_dPE_k - A factor (pres_lay*mass_lay*dSpec_vol/dS) relating
978 ! a layer's salinity change to the change in column
979 ! potential energy, in J m-2 ppt-1.
980 ! (in) dT_to_dPEa - A factor (pres_lay*mass_lay*dSpec_vol/dT) relating
981 ! a layer's temperature change to the change in column
982 ! potential energy, including all implicit diffusive changes
983 ! in the temperatures of all the layers above, in J m-2 K-1.
984 ! (in) dS_to_dPEa - A factor (pres_lay*mass_lay*dSpec_vol/dS) relating
985 ! a layer's salinity change to the change in column
986 ! potential energy, including all implicit diffusive changes
987 ! in the salinities of all the layers above, in J m-2 ppt-1.
988 ! (in) pres - The hydrostatic interface pressure, which is used to relate
989 ! the changes in column thickness to the energy that is radiated
990 ! as gravity waves and unavailable to drive mixing, in Pa.
991 ! (in) dT_to_dColHt_k - A factor (mass_lay*dSColHtc_vol/dT) relating
992 ! a layer's temperature change to the change in column
993 ! height, in m K-1.
994 ! (in) dS_to_dColHt_k - A factor (mass_lay*dSColHtc_vol/dS) relating
995 ! a layer's salinity change to the change in column
996 ! height, in m ppt-1.
997 ! (in) dT_to_dColHta - A factor (mass_lay*dSColHtc_vol/dT) relating
998 ! a layer's temperature change to the change in column
999 ! height, including all implicit diffusive changes
1000 ! in the temperatures of all the layers above, in m K-1.
1001 ! (in) dS_to_dColHta - A factor (mass_lay*dSColHtc_vol/dS) relating
1002 ! a layer's salinity change to the change in column
1003 ! height, including all implicit diffusive changes
1004 ! in the salinities of all the layers above, in m ppt-1.
1005 ! (out,opt) PE_chg - The change in column potential energy from applying
1006 ! Kddt_h at the present interface, in J m-2.
1007 ! (out,opt) dPE_chg_dKd - The partial derivative of PE_chg with Kddt_h,
1008 ! in units of J m-2 H-1.
1009 ! (out,opt) PE_max - The maximum change in column potential energy that could
1010 ! be realizedd by applying a huge value of Kddt_h at the
1011 ! present interface, in J m-2.
1012 ! (out,opt) dPEc_dKc0 - The partial derivative of PE_chg with Kddt_h in the
1013 ! limit where Kddt_h = 0, in J m-2 H-1.
1014  ! b_den_1 - The first term in the denominator of b1, in m or kg m-2.
1015  real :: b1 ! b1 is used by the tridiagonal solver, in H-1.
1016  real :: b1Kd ! Temporary array (nondim.)
1017  real :: ColHt_chg ! The change in column thickness in m.
1018  real :: dColHt_max ! The change in column thickess for infinite diffusivity, in m.
1019  real :: dColHt_dKd ! The partial derivative of column thickess with diffusivity, in s m-1.
1020  real :: dT_k, dT_km1 ! Temporary arrays in K.
1021  real :: dS_k, dS_km1 ! Temporary arrays in ppt.
1022  real :: I_Kr_denom, dKr_dKd ! Temporary arrays in H-2 and nondim.
1023  real :: ddT_k_dKd, ddT_km1_dKd ! Temporary arrays in K H-1.
1024  real :: ddS_k_dKd, ddS_km1_dKd ! Temporary arrays in ppt H-1.
1025 
1026  b1 = 1.0 / (b_den_1 + kddt_h)
1027  b1kd = kddt_h*b1
1028 
1029  ! Start with the temperature change in layer k-1 due to the diffusivity at
1030  ! interface K without considering the effects of changes in layer k.
1031 
1032  ! Calculate the change in PE due to the diffusion at interface K
1033  ! if Kddt_h(K+1) = 0.
1034  i_kr_denom = 1.0 / (h_k*b_den_1 + (b_den_1 + h_k)*kddt_h)
1035 
1036  dt_k = (kddt_h*i_kr_denom) * dte_term
1037  ds_k = (kddt_h*i_kr_denom) * dse_term
1038 
1039  if (present(pe_chg)) then
1040  ! Find the change in energy due to diffusion with strength Kddt_h at this interface.
1041  ! Increment the temperature changes in layer k-1 due the changes in layer k.
1042  dt_km1 = b1kd * ( dt_k + dt_km1_t2 )
1043  ds_km1 = b1kd * ( ds_k + ds_km1_t2 )
1044  pe_chg = (dt_to_dpe_k * dt_k + dt_to_dpea * dt_km1) + &
1045  (ds_to_dpe_k * ds_k + ds_to_dpea * ds_km1)
1046  colht_chg = (dt_to_dcolht_k * dt_k + dt_to_dcolhta * dt_km1) + &
1047  (ds_to_dcolht_k * ds_k + ds_to_dcolhta * ds_km1)
1048  if (colht_chg < 0.0) pe_chg = pe_chg - pres * colht_chg
1049  endif
1050 
1051  if (present(dpec_dkd)) then
1052  ! Find the derivatives of the temperature and salinity changes with Kddt_h.
1053  dkr_dkd = (h_k*b_den_1) * i_kr_denom**2
1054 
1055  ddt_k_dkd = dkr_dkd * dte_term
1056  dds_k_dkd = dkr_dkd * dse_term
1057  ddt_km1_dkd = (b1**2 * b_den_1) * ( dt_k + dt_km1_t2 ) + b1kd * ddt_k_dkd
1058  dds_km1_dkd = (b1**2 * b_den_1) * ( ds_k + ds_km1_t2 ) + b1kd * dds_k_dkd
1059 
1060  ! Calculate the partial derivative of Pe_chg with Kddt_h.
1061  dpec_dkd = (dt_to_dpe_k * ddt_k_dkd + dt_to_dpea * ddt_km1_dkd) + &
1062  (ds_to_dpe_k * dds_k_dkd + ds_to_dpea * dds_km1_dkd)
1063  dcolht_dkd = (dt_to_dcolht_k * ddt_k_dkd + dt_to_dcolhta * ddt_km1_dkd) + &
1064  (ds_to_dcolht_k * dds_k_dkd + ds_to_dcolhta * dds_km1_dkd)
1065  if (dcolht_dkd < 0.0) dpec_dkd = dpec_dkd - pres * dcolht_dkd
1066  endif
1067 
1068  if (present(dpe_max)) then
1069  ! This expression is the limit of PE_chg for infinite Kddt_h.
1070  dpe_max = (dt_to_dpea * dt_km1_t2 + ds_to_dpea * ds_km1_t2) + &
1071  ((dt_to_dpe_k + dt_to_dpea) * dte_term + &
1072  (ds_to_dpe_k + ds_to_dpea) * dse_term) / (b_den_1 + h_k)
1073  dcolht_max = (dt_to_dcolhta * dt_km1_t2 + ds_to_dcolhta * ds_km1_t2) + &
1074  ((dt_to_dcolht_k + dt_to_dcolhta) * dte_term + &
1075  (ds_to_dcolht_k + ds_to_dcolhta) * dse_term) / (b_den_1 + h_k)
1076  if (dcolht_max < 0.0) dpe_max = dpe_max - pres*dcolht_max
1077  endif
1078 
1079  if (present(dpec_dkd_0)) then
1080  ! This expression is the limit of dPEc_dKd for Kddt_h = 0.
1081  dpec_dkd_0 = (dt_to_dpea * dt_km1_t2 + ds_to_dpea * ds_km1_t2) / (b_den_1) + &
1082  (dt_to_dpe_k * dte_term + ds_to_dpe_k * dse_term) / (h_k*b_den_1)
1083  dcolht_dkd = (dt_to_dcolhta * dt_km1_t2 + ds_to_dcolhta * ds_km1_t2) / (b_den_1) + &
1084  (dt_to_dcolht_k * dte_term + ds_to_dcolht_k * dse_term) / (h_k*b_den_1)
1085  if (dcolht_dkd < 0.0) dpec_dkd_0 = dpec_dkd_0 - pres*dcolht_dkd
1086  endif
1087 
1088 end subroutine find_pe_chg
