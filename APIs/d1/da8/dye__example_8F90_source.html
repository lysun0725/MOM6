23 !* *
24 !* By Robert Hallberg, 2002 *
25 !* *
26 !* This file contains an example of the code that is needed to set *
27 !* up and use a set (in this case two) of dynamically passive tracers *
28 !* for diagnostic purposes. The tracers here are dye tracers which *
29 !* are set to 1 within the geographical region specified. The depth *
30 !* which a tracer is set is determined by calculating the depth from *
31 !* the seafloor upwards through the column. *
32 !* *
33 !* A single subroutine is called from within each file to register *
34 !* each of the tracers for reinitialization and advection and to *
35 !* register the subroutine that initializes the tracers and set up *
36 !* their output and the subroutine that does any tracer physics or *
37 !* chemistry along with diapycnal mixing (included here because some *
38 !* tracers may float or swim vertically or dye diapycnal processes). *
39 !* *
40 !* *
41 !* Macros written all in capital letters are defined in MOM_memory.h. *
42 !* *
43 !* A small fragment of the grid is shown below: *
44 !* *
45 !* j+1 x ^ x ^ x At x: q *
46 !* j+1 > o > o > At ^: v *
47 !* j x ^ x ^ x At >: u *
48 !* j > o > o > At o: h, tr *
49 !* j-1 x ^ x ^ x *
50 !* i-1 i i+1 At x & ^: *
51 !* i i+1 At > & o: *
52 !* *
53 !* The boundaries always run through q grid points (x). *
54 !* *
55 !********+*********+*********+*********+*********+*********+*********+**

126 function register_dye_tracer(HI, GV, param_file, CS, tr_Reg, restart_CS)
127  type(hor_index_type), intent(in) :: HI
128  type(verticalgrid_type), intent(in) :: GV
129  type(param_file_type), intent(in) :: param_file
130  type(dye_tracer_cs), pointer :: CS
131  type(tracer_registry_type), pointer :: tr_Reg
132  type(mom_restart_cs), pointer :: restart_CS
133 ! This subroutine is used to register tracer fields and subroutines
134 ! to be used with MOM.
135 ! Arguments: HI - A horizontal index type structure.
136 ! (in) param_file - A structure indicating the open file to parse for
137 ! model parameter values.
138 ! (in/out) CS - A pointer that is set to point to the control structure
139 ! for this module
140 ! (in) diag - A structure that is used to regulate diagnostic output.
141 ! (in/out) tr_Reg - A pointer that is set to point to the control structure
142 ! for the tracer advection and diffusion module.
143 ! (in) restart_CS - A pointer to the restart control structure.
144 
145 ! This include declares and sets the variable "version".
146 #include "version_variable.h"
147  character(len=40) :: mod = "regional_dyes" ! This module's name.
148  character(len=200) :: inputdir ! The directory where the input files are.
149  character(len=48) :: var_name ! The variable's name.
150  character(len=48) :: desc_name ! The variable's descriptor.
151  real, pointer :: tr_ptr(:,:,:) => null()
152  logical :: register_dye_tracer
153  integer :: isd, ied, jsd, jed, nz, m
154  isd = hi%isd ; ied = hi%ied ; jsd = hi%jsd ; jed = hi%jed ; nz = gv%ke
155 
156  if (associated(cs)) then
157  call mom_error(warning, "register_dye_tracer called with an "// &
158  "associated control structure.")
159  return
160  endif
161  allocate(cs)
162 
163  ! Read all relevant parameters and write them to the model log.
164  call log_version(param_file, mod, version, "")
165  call get_param(param_file, mod, "NUM_DYE_TRACERS", cs%ntr, &
166  "The number of dye tracers in this run. Each tracer \n"//&
167  "should have a separate region.", default=0)
168  allocate(cs%dye_source_minlon(cs%ntr), &
169  cs%dye_source_maxlon(cs%ntr), &
170  cs%dye_source_minlat(cs%ntr), &
171  cs%dye_source_maxlat(cs%ntr), &
172  cs%dye_source_mindepth(cs%ntr), &
173  cs%dye_source_maxdepth(cs%ntr))
174  allocate(cs%tr_adx(cs%ntr), &
175  cs%tr_ady(cs%ntr), &
176  cs%tr_dfx(cs%ntr), &
177  cs%tr_dfy(cs%ntr))
178  allocate(cs%ind_tr(cs%ntr), &
179  cs%id_tracer(cs%ntr), &
180  cs%id_tr_adx(cs%ntr), &
181  cs%id_tr_ady(cs%ntr), &
182  cs%id_tr_dfx(cs%ntr), &
183  cs%id_tr_dfy(cs%ntr))
184  allocate(cs%tr_desc(cs%ntr))
185 
186  cs%id_tracer(:) = -1
187  cs%id_tr_adx(:) = -1
188  cs%id_tr_ady(:) = -1
189  cs%id_tr_dfx(:) = -1
190  cs%id_tr_dfy(:) = -1
191 
192  cs%dye_source_minlon(:) = -1.e30
193  call get_param(param_file, mod, "DYE_SOURCE_MINLON", cs%dye_source_minlon, &
194  "This is the starting longitude at which we start injecting dyes.", &
195  fail_if_missing=.true.)
196  if (minval(cs%dye_source_minlon(:)) < -1.e29) &
197  call mom_error(fatal, "register_dye_tracer: Not enough values provided for DYE_SOURCE_MINLON ")
198 
199  cs%dye_source_maxlon(:) = -1.e30
200  call get_param(param_file, mod, "DYE_SOURCE_MAXLON", cs%dye_source_maxlon, &
201  "This is the ending longitude at which we finish injecting dyes.", &
202  fail_if_missing=.true.)
203  if (minval(cs%dye_source_maxlon(:)) < -1.e29) &
204  call mom_error(fatal, "register_dye_tracer: Not enough values provided for DYE_SOURCE_MAXLON ")
205 
206  cs%dye_source_minlat(:) = -1.e30
207  call get_param(param_file, mod, "DYE_SOURCE_MINLAT", cs%dye_source_minlat, &
208  "This is the starting latitude at which we start injecting dyes.", &
209  fail_if_missing=.true.)
210  if (minval(cs%dye_source_minlat(:)) < -1.e29) &
211  call mom_error(fatal, "register_dye_tracer: Not enough values provided for DYE_SOURCE_MINLAT ")
212 
213  cs%dye_source_maxlat(:) = -1.e30
214  call get_param(param_file, mod, "DYE_SOURCE_MAXLAT", cs%dye_source_maxlat, &
215  "This is the ending latitude at which we finish injecting dyes.", &
216  fail_if_missing=.true.)
217  if (minval(cs%dye_source_maxlat(:)) < -1.e29) &
218  call mom_error(fatal, "register_dye_tracer: Not enough values provided for DYE_SOURCE_MAXLAT ")
219 
220  cs%dye_source_mindepth(:) = -1.e30
221  call get_param(param_file, mod, "DYE_SOURCE_MINDEPTH", cs%dye_source_mindepth, &
222  "This is the minumum depth at which we inject dyes.", &
223  fail_if_missing=.true.)
224  if (minval(cs%dye_source_mindepth(:)) < -1.e29) &
225  call mom_error(fatal, "register_dye_tracer: Not enough values provided for DYE_SOURCE_MINDEPTH")
226 
227  cs%dye_source_maxdepth(:) = -1.e30
228  call get_param(param_file, mod, "DYE_SOURCE_MAXDEPTH", cs%dye_source_maxdepth, &
229  "This is the maximum depth at which we inject dyes.", &
230  fail_if_missing=.true.)
231  if (minval(cs%dye_source_maxdepth(:)) < -1.e29) &
232  call mom_error(fatal, "register_dye_tracer: Not enough values provided for DYE_SOURCE_MAXDEPTH ")
233 
234  do m = 1, cs%ntr
235  write(var_name(:),'(A,I3.3)') "dye",m
236  write(desc_name(:),'(A,I3.3)') "Dye Tracer ",m
237  cs%tr_desc(m) = var_desc(trim(var_name), "conc", trim(desc_name), caller=mod)
238  enddo
239 
240 
241 
242  allocate(cs%tr(isd:ied,jsd:jed,nz,cs%ntr)) ; cs%tr(:,:,:,:) = 0.0
243 
244  do m=1,cs%ntr
245  ! This is needed to force the compiler not to do a copy in the registration
246  ! calls. Curses on the designers and implementers of Fortran90.
247  tr_ptr => cs%tr(:,:,:,m)
248  call query_vardesc(cs%tr_desc(m), name=var_name, &
249  caller="register_dye_tracer")
250 ! ! Register the tracer for the restart file.
251 ! call register_restart_field(tr_ptr, CS%tr_desc(m), &
252 ! .not.CS%tracers_may_reinit, restart_CS)
253  ! Register the tracer for horizontal advection & diffusion.
254  call register_tracer(tr_ptr, cs%tr_desc(m), param_file, hi, gv, tr_reg, &
255  tr_desc_ptr=cs%tr_desc(m))
256 
257  ! Set coupled_tracers to be true (hard-coded above) to provide the surface
258  ! values to the coupler (if any). This is meta-code and its arguments will
259  ! currently (deliberately) give fatal errors if it is used.
260  if (cs%coupled_tracers) &
261  cs%ind_tr(m) = aof_set_coupler_flux(trim(var_name)//'_flux', &
262  flux_type=' ', implementation=' ', caller="register_dye_tracer")
263  enddo
264 
265  cs%tr_Reg => tr_reg
266  cs%restart_CSp => restart_cs
267  register_dye_tracer = .true.
268 end function register_dye_tracer

270 subroutine initialize_dye_tracer(restart, day, G, GV, h, diag, OBC, CS, sponge_CSp, &
271  diag_to_z_csp)
272  logical, intent(in) :: restart
273  type(time_type), target, intent(in) :: day
274  type(ocean_grid_type), intent(in) :: G
275  type(verticalgrid_type), intent(in) :: GV
276  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
277  type(diag_ctrl), target, intent(in) :: diag
278  type(ocean_obc_type), pointer :: OBC
279  type(dye_tracer_cs), pointer :: CS
280  type(sponge_cs), pointer :: sponge_CSp
281  type(diag_to_z_cs), pointer :: diag_to_Z_CSp
282 ! This subroutine initializes the CS%ntr tracer fields in tr(:,:,:,:)
283 ! and it sets up the tracer output.
284 
285 ! Arguments: restart - .true. if the fields have already been read from
286 ! a restart file.
287 ! (in) day - Time of the start of the run.
288 ! (in) G - The ocean's grid structure.
289 ! (in) GV - The ocean's vertical grid structure.
290 ! (in) h - Layer thickness, in m or kg m-2.
291 ! (in) OBC - This open boundary condition type specifies whether, where,
292 ! and what open boundary conditions are used.
293 ! (in/out) CS - The control structure returned by a previous call to
294 ! register_dye_tracer.
295 ! (in/out) sponge_CSp - A pointer to the control structure for the sponges, if
296 ! they are in use. Otherwise this may be unassociated.
297 ! (in/out) diag_to_Z_Csp - A pointer to the control structure for diagnostics
298 ! in depth space.
299  character(len=24) :: name ! A variable's name in a NetCDF file.
300  character(len=72) :: longname ! The long name of that variable.
301  character(len=48) :: units ! The dimensions of the variable.
302  character(len=48) :: flux_units ! The units for age tracer fluxes, either
303  ! years m3 s-1 or years kg s-1.
304  logical :: OK
305  integer :: i, j, k, m
306  real :: z_bot, z_center
307 
308  if (.not.associated(cs)) return
309  if (cs%ntr < 1) return
310 
311  cs%diag => diag
312 
313  ! Establish location of source
314  do m= 1, cs%ntr
315  do j=g%jsd,g%jed ; do i=g%isd,g%ied
316  ! A dye is set dependent on the center of the cell being inside the rectangular box.
317  if (cs%dye_source_minlon(m)<g%geoLonT(i,j) .and. &
318  cs%dye_source_maxlon(m)>=g%geoLonT(i,j) .and. &
319  cs%dye_source_minlat(m)<g%geoLatT(i,j) .and. &
320  cs%dye_source_maxlat(m)>=g%geoLatT(i,j) .and. &
321  g%mask2dT(i,j) > 0.0 ) then
322  z_bot = -g%bathyT(i,j)
323  do k = gv%ke, 1, -1
324  z_center = z_bot + 0.5*h(i,j,k)*gv%H_to_m
325  if ( z_center > -cs%dye_source_maxdepth(m) .and. &
326  z_center < -cs%dye_source_mindepth(m) ) then
327  cs%tr(i,j,k,m) = 1.0
328  endif
329  z_bot = z_bot + h(i,j,k)*gv%H_to_m
330  enddo
331  endif
332  enddo; enddo
333  enddo
334 
335  do m=1,cs%ntr
336  ! Register the tracer for the restart file.
337  call query_vardesc(cs%tr_desc(m), name, units=units, longname=longname, &
338  caller="initialize_dye_tracer")
339  cs%id_tracer(m) = register_diag_field("ocean_model", trim(name), cs%diag%axesTL, &
340  day, trim(longname) , trim(units))
341  cs%id_tr_adx(m) = register_diag_field("ocean_model", trim(name)//"_adx", &
342  cs%diag%axesCuL, day, trim(longname)//" advective zonal flux" , &
343  trim(flux_units))
344  cs%id_tr_ady(m) = register_diag_field("ocean_model", trim(name)//"_ady", &
345  cs%diag%axesCvL, day, trim(longname)//" advective meridional flux" , &
346  trim(flux_units))
347  cs%id_tr_dfx(m) = register_diag_field("ocean_model", trim(name)//"_dfx", &
348  cs%diag%axesCuL, day, trim(longname)//" diffusive zonal flux" , &
349  trim(flux_units))
350  cs%id_tr_dfy(m) = register_diag_field("ocean_model", trim(name)//"_dfy", &
351  cs%diag%axesCvL, day, trim(longname)//" diffusive zonal flux" , &
352  trim(flux_units))
353  if (cs%id_tr_adx(m) > 0) call safe_alloc_ptr(cs%tr_adx(m)%p,g%IsdB,g%IedB,g%jsd,g%jed,gv%ke)
354  if (cs%id_tr_ady(m) > 0) call safe_alloc_ptr(cs%tr_ady(m)%p,g%isd,g%ied,g%JsdB,g%JedB,gv%ke)
355  if (cs%id_tr_dfx(m) > 0) call safe_alloc_ptr(cs%tr_dfx(m)%p,g%IsdB,g%IedB,g%jsd,g%jed,gv%ke)
356  if (cs%id_tr_dfy(m) > 0) call safe_alloc_ptr(cs%tr_dfy(m)%p,g%isd,g%ied,g%JsdB,g%JedB,gv%ke)
357 
358 ! Register the tracer for horizontal advection & diffusion.
359  if ((cs%id_tr_adx(m) > 0) .or. (cs%id_tr_ady(m) > 0) .or. &
360  (cs%id_tr_dfx(m) > 0) .or. (cs%id_tr_dfy(m) > 0)) &
361  call add_tracer_diagnostics(name, cs%tr_Reg, cs%tr_adx(m)%p, &
362  cs%tr_ady(m)%p,cs%tr_dfx(m)%p,cs%tr_dfy(m)%p)
363 
364  call register_z_tracer(cs%tr(:,:,:,m), trim(name), longname, units, &
365  day, g, diag_to_z_csp)
366  enddo
367 
368 end subroutine initialize_dye_tracer

370 subroutine dye_tracer_column_physics(h_old, h_new, ea, eb, fluxes, dt, G, GV, CS)
371  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h_old, h_new, ea, eb
372  type(forcing), intent(in) :: fluxes
373  real, intent(in) :: dt
374  type(ocean_grid_type), intent(in) :: G
375  type(verticalgrid_type), intent(in) :: GV
376  type(dye_tracer_cs), pointer :: CS
377 ! This subroutine applies diapycnal diffusion and any other column
378 ! tracer physics or chemistry to the tracers from this file.
379 ! This is a simple example of a set of advected passive tracers.
380 
381 ! Arguments: h_old - Layer thickness before entrainment, in m or kg m-2.
382 ! (in) h_new - Layer thickness after entrainment, in m or kg m-2.
383 ! (in) ea - an array to which the amount of fluid entrained
384 ! from the layer above during this call will be
385 ! added, in m or kg m-2.
386 ! (in) eb - an array to which the amount of fluid entrained
387 ! from the layer below during this call will be
388 ! added, in m or kg m-2.
389 ! (in) fluxes - A structure containing pointers to any possible
390 ! forcing fields. Unused fields have NULL ptrs.
391 ! (in) dt - The amount of time covered by this call, in s.
392 ! (in) G - The ocean's grid structure.
393 ! (in) GV - The ocean's vertical grid structure.
394 ! (in) CS - The control structure returned by a previous call to
395 ! register_dye_tracer.
396 !
397 ! The arguments to this subroutine are redundant in that
398 ! h_new[k] = h_old[k] + ea[k] - eb[k-1] + eb[k] - ea[k+1]
399 
400  real :: sfc_val ! The surface value for the tracers.
401  real :: Isecs_per_year ! The number of seconds in a year.
402  real :: year ! The time in years.
403  integer :: secs, days ! Integer components of the time type.
404  integer :: i, j, k, is, ie, js, je, nz, m
405  real :: z_bot, z_center
406 
407  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = gv%ke
408 
409  if (.not.associated(cs)) return
410  if (cs%ntr < 1) return
411 
412  do m=1,cs%ntr
413  call tracer_vertdiff(h_old, ea, eb, dt, cs%tr(:,:,:,m), g, gv)
414  enddo
415 
416  do m=1,cs%ntr
417  do j=g%jsd,g%jed ; do i=g%isd,g%ied
418  ! A dye is set dependent on the center of the cell being inside the rectangular box.
419  if (cs%dye_source_minlon(m)<g%geoLonT(i,j) .and. &
420  cs%dye_source_maxlon(m)>=g%geoLonT(i,j) .and. &
421  cs%dye_source_minlat(m)<g%geoLatT(i,j) .and. &
422  cs%dye_source_maxlat(m)>=g%geoLatT(i,j) .and. &
423  g%mask2dT(i,j) > 0.0 ) then
424  z_bot = -g%bathyT(i,j)
425  do k=nz,1,-1
426  z_center = z_bot + 0.5*h_new(i,j,k)*gv%H_to_m
427  if ( z_center > -cs%dye_source_maxdepth(m) .and. &
428  z_center < -cs%dye_source_mindepth(m) ) then
429  cs%tr(i,j,k,m) = 1.0
430  endif
431  z_bot = z_bot + h_new(i,j,k)*gv%H_to_m
432  enddo
433  endif
434  enddo; enddo
435  enddo
436 
437 
438  do m=1,cs%ntr
439  if (cs%id_tracer(m)>0) &
440  call post_data(cs%id_tracer(m),cs%tr(:,:,:,m),cs%diag)
441  if (cs%id_tr_adx(m)>0) &

463 ! This function calculates the mass-weighted integral of all tracer stocks,
464 ! returning the number of stocks it has calculated. If the stock_index
465 ! is present, only the stock corresponding to that coded index is returned.
466 
467 ! Arguments: h - Layer thickness, in m or kg m-2.
468 ! (out) stocks - the mass-weighted integrated amount of each tracer,

470 ! (in) G - The ocean's grid structure.

473 ! register_dye_tracer.
474 ! (out) names - the names of the stocks calculated.
475 ! (out) units - the units of the stocks calculated.

478

480  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = gv%ke
481

483  if (.not.associated(cs)) return
484  if (cs%ntr < 1) return
485 
486  if (present(stock_index)) then ; if (stock_index > 0) then
487  ! Check whether this stock is available from this routine.
488 
489  ! No stocks from this routine are being checked yet. Return 0.
490  return
491  endif ; endif

493  do m=1,cs%ntr

502  enddo
