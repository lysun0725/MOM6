366  subroutine cvmix_init_bkgnd_bryanlewis(CVmix_vars, bl1, bl2, bl3, bl4, &
367  cvmix_params_in, old_vals, &
368  cvmix_bkgnd_params_user)
369 
370 ! !DESCRIPTION:
371 ! Initialization routine for Bryan-Lewis diffusivity/viscosity calculation.
372 ! For each column, this routine sets the static viscosity \& diffusivity
373 ! based on the specified parameters. Note that the units of these parameters
374 ! must be consistent with the units of viscosity and diffusivity -- either
375 ! cgs or mks, but do not mix and match!
376 ! \\
377 ! \\
378 ! The Bryan-Lewis parameterization is based on the following:
379 ! \begin{eqnarray*}
380 ! \kappa_{BL} &=& \textrm{bl1} + \frac{\textrm{bl2}}{\pi}\tan^{-1}\bigg(
381 ! \textrm{bl3}(|z|-\textrm{bl4})\bigg)\\
382 ! \nu_{BL} &=& \textrm{Pr}\cdot\kappa_{BL}
383 ! \end{eqnarray*}
384 ! This method is based on the following paper:
385 ! \begin{quote}
386 ! \emph{A Water Mass Model of the World Ocean}\\
387 ! K. Bryan and L. J. Lewis\\
388 ! Journal of Geophysical Research, vol 84 (1979), pages 2503-2517.
389 ! \end{quote}
390 !
391 ! In that paper, they recommend the parameters
392 ! \begin{itemize}
393 ! \item[] bl1 $= 8 \cdot 10^{-5}$ m$^2/$s
394 ! \item[] bl2 $= 1.05 \cdot 10^{-4}$ m$^2/$s
395 ! \item[] bl3 $= 4.5 \cdot 10^{-3}$ m$^{-1}$
396 ! \item[] bl4 $= 2500$ m
397 ! \end{itemize}
398 ! However, more recent usage of their scheme may warrant different settings.
399 !
400 ! !USES:
401 ! Only those used by entire module.
402 
403 ! !INPUT PARAMETERS:
404  ! Contains depth and nlev
405  type(cvmix_data_type), intent(in) :: CVmix_vars
406  ! Units are first column if CVmix_data%depth is m, second if cm
407  real(cvmix_r8), intent(in) :: bl1, &! m^2/s or cm^2/s
408  bl2, &! m^2/s or cm^2/s
409  bl3, &! 1/m or 1/cm
410  bl4 ! m or cm
411  character(len=cvmix_strlen), optional, intent(in) :: old_vals
412  type(cvmix_global_params_type), intent(in) :: CVmix_params_in
413 
414 ! !OUTPUT PARAMETERS:
415  type(cvmix_bkgnd_params_type), target, optional, intent(inout) :: &
416  CVmix_bkgnd_params_user
417 !EOP
418 !BOC
419 
420  ! Pointers to parameter data type
421  type(cvmix_bkgnd_params_type), pointer :: CVmix_bkgnd_params_out
422 
423  ! Local index
424  integer :: nlev ! max number of levels
425 
426  ! Local copies to make code easier to read
427  real(cvmix_r8), dimension(CVmix_params_in%max_nlev+1) :: Mdiff, Tdiff, zw
428 
429  cvmix_bkgnd_params_out => cvmix_bkgnd_params_saved
430  if (present(cvmix_bkgnd_params_user)) then
431  cvmix_bkgnd_params_out => cvmix_bkgnd_params_user
432  end if
433 
434  nlev = cvmix_params_in%max_nlev
435 
436  ! Clean up memory in bkgnd_params_type (will be re-allocated in put call)
437  if (allocated(cvmix_bkgnd_params_out%static_Mdiff)) &
438  deallocate(cvmix_bkgnd_params_out%static_Mdiff)
439  if (allocated(cvmix_bkgnd_params_out%static_Tdiff)) &
440  deallocate(cvmix_bkgnd_params_out%static_Tdiff)
441 
442  ! Set static_[MT]diff in background_input_type
443  zw = -cvmix_vars%zw_iface
444  tdiff = bl1 + (bl2/cvmix_pi)*atan(bl3*(zw-bl4))
445  mdiff = cvmix_params_in%prandtl*tdiff
446 
447  call cvmix_put_bkgnd("static_Mdiff", mdiff, cvmix_bkgnd_params_user, &
448  nlev=nlev)
449  call cvmix_put_bkgnd("static_Tdiff", tdiff, cvmix_bkgnd_params_user, &
450  nlev=nlev)
451 
452  if (present(old_vals)) then
453  select case (trim(old_vals))
454  case ("overwrite")
455  call cvmix_put_bkgnd('handle_old_vals', cvmix_overwrite_old_val, &
456  cvmix_bkgnd_params_user)
457  case ("sum")
458  call cvmix_put_bkgnd('handle_old_vals', cvmix_sum_old_and_new_vals, &
459  cvmix_bkgnd_params_user)
460  case ("max")
461  call cvmix_put_bkgnd('handle_old_vals', cvmix_max_old_and_new_vals, &
462  cvmix_bkgnd_params_user)
463  case DEFAULT
464  print*, "ERROR: ", trim(old_vals), " is not a valid option for ", &
465  "handling old values of diff and visc."
466  stop 1
467  end select
468  else
469  call cvmix_put_bkgnd('handle_old_vals', cvmix_overwrite_old_val, &
470  cvmix_bkgnd_params_user)
471  end if
