95 ! This type is used for communication with other components via the FMS coupler.
96 ! The element names and types can be changed only with great deliberation, hence
97 ! the persistnce of things like the cutsy element name "avg_kount".
98 type, public :: ocean_public_type
99  type(domain2d) :: domain ! The domain for the surface fields.
100  logical :: is_ocean_pe ! .true. on processors that run the ocean model.
101  character(len=32) :: instance_name = '' ! A name that can be used to identify
102  ! this instance of an ocean model, for example
103  ! in ensembles when writing messages.
104  integer, pointer, dimension(:) :: pelist => null() ! The list of ocean PEs.
105  logical, pointer, dimension(:,:) :: maskmap =>null() ! A pointer to an array
106  ! indicating which logical processors are actually used for
107  ! the ocean code. The other logical processors would be all
108  ! land points and are not assigned to actual processors.
109  ! This need not be assigned if all logical processors are used.
110 
111  integer :: stagger = -999 ! The staggering relative to the tracer points
112  ! points of the two velocity components. Valid entries
113  ! include AGRID, BGRID_NE, CGRID_NE, BGRID_SW, and CGRID_SW,
114  ! corresponding to the community-standard Arakawa notation.
115  ! (These are named integers taken from mpp_parameter_mod.)
116  ! Following MOM, this is BGRID_NE by default when the ocean
117  ! is initialized, but here it is set to -999 so that a
118  ! global max across ocean and non-ocean processors can be
119  ! used to determine its value.
120  real, pointer, dimension(:,:) :: &
121  t_surf => null(), & ! SST on t-cell (degrees Kelvin)
122  s_surf => null(), & ! SSS on t-cell (psu)
123  u_surf => null(), & ! i-velocity at the locations indicated by stagger, m/s.
124  v_surf => null(), & ! j-velocity at the locations indicated by stagger, m/s.
125  sea_lev => null(), & ! Sea level in m after correction for surface pressure,
126  ! i.e. dzt(1) + eta_t + patm/rho0/grav (m)
127  frazil =>null(), & ! Accumulated heating (in Joules/m^2) from frazil
128  ! formation in the ocean.
129  area => null() ! cell area of the ocean surface, in m2.
130  type(coupler_2d_bc_type) :: fields ! A structure that may contain an
131  ! array of named tracer-related fields.
132  integer :: avg_kount ! Used for accumulating averages of this type.
133  integer, dimension(3) :: axes = 0 ! Axis numbers that are available
134  ! for I/O using this surface data.
135 end type ocean_public_type

138 type, public :: ocean_state_type ; private
139  ! This type is private, and can therefore vary between different ocean models.
140  ! All information entire ocean state may be contained here, although it is not
141  ! necessary that this is implemented with all models.
142  logical :: is_ocean_pe = .false. ! True if this is an ocean PE.
143  type(time_type) :: time ! The ocean model's time and master clock.
144  integer :: restart_control ! An integer that is bit-tested to determine whether
145  ! incremental restart files are saved and whether they
146  ! have a time stamped name. +1 (bit 0) for generic
147  ! files and +2 (bit 1) for time-stamped files. A
148  ! restart file is saved at the end of a run segment
149  ! unless Restart_control is negative.
150  type(time_type) :: energysavedays ! The interval between writing the energies
151  ! and other integral quantities of the run.
152  type(time_type) :: write_energy_time ! The next time to write to the energy file.
153 
154  integer :: nstep = 0 ! The number of calls to update_ocean.
155  logical :: use_ice_shelf ! If true, the ice shelf model is enabled.
156  type(ice_shelf_cs), pointer :: ice_shelf_csp => null()
157  logical :: restore_salinity ! If true, the coupled MOM driver adds a term to
158  ! restore salinity to a specified value.
159  real :: press_to_z ! A conversion factor between pressure and ocean
160  ! depth in m, usually 1/(rho_0*g), in m Pa-1.
161  real :: c_p ! The heat capacity of seawater, in J K-1 kg-1.
162 
163  type(directories) :: dirs ! A structure containing several relevant directory paths.
164  type(forcing) :: fluxes ! A structure containing pointers to
165  ! the ocean forcing fields.
166  type(forcing) :: flux_tmp ! A secondary structure containing pointers to the
167  ! ocean forcing fields for when multiple coupled
168  ! timesteps are taken per thermodynamic step.
169  type(surface) :: state ! A structure containing pointers to
170  ! the ocean surface state fields.
171  type(ocean_grid_type), pointer :: grid => null() ! A pointer to a grid structure
172  ! containing metrics and related information.
173  type(verticalgrid_type), pointer :: gv => null() ! A pointer to a vertical grid
174  ! structure containing metrics and related information.
175  type(mom_control_struct), pointer :: mom_csp => null()
176  type(surface_forcing_cs), pointer :: forcing_csp => null()
177  type(sum_output_cs), pointer :: sum_output_csp => null()
178 end type ocean_state_type

183 ! <SUBROUTINE NAME="ocean_model_init">
184 !
185 ! <DESCRIPTION>
186 ! Initialize the ocean model.
187 ! </DESCRIPTION>
188 !
189 subroutine ocean_model_init(Ocean_sfc, OS, Time_init, Time_in)
190  type(ocean_public_type), target, intent(inout) :: Ocean_sfc
191  type(ocean_state_type), pointer :: OS
192  type(time_type), intent(in) :: Time_init
193  type(time_type), intent(in) :: Time_in
194 ! This subroutine initializes both the ocean state and the ocean surface type.
195 ! Because of the way that indicies and domains are handled, Ocean_sfc must have
196 ! been used in a previous call to initialize_ocean_type.
197 
198 ! Arguments: Ocean_sfc - A structure containing various publicly visible ocean
199 ! surface properties after initialization, this is intent(out).
200 ! (out,private) OS - A structure whose internal contents are private
201 ! to ocean_model_mod that may be used to contain all
202 ! information about the ocean's interior state.
203 ! (in) Time_init - The start time for the coupled model's calendar.
204 ! (in) Time_in - The time at which to initialize the ocean model.
205  real :: Time_unit ! The time unit in seconds for ENERGYSAVEDAYS.
206  real :: Rho0 ! The Boussinesq ocean density, in kg m-3.
207  real :: G_Earth ! The gravitational acceleration in m s-2.
208 ! This include declares and sets the variable "version".
209 #include "version_variable.h"
210  character(len=40) :: mod = "ocean_model_init" ! This module's name.
211  character(len=48) :: stagger
212  integer :: secs, days
213  type(param_file_type) :: param_file
214 
215  call calltree_enter("ocean_model_init(), ocean_model_MOM.F90")
216  if (associated(os)) then
217  call mom_error(warning, "ocean_model_init called with an associated "// &
218  "ocean_state_type structure. Model is already initialized.")
219  return
220  endif
221  allocate(os)
222 
223  os%is_ocean_pe = ocean_sfc%is_ocean_pe
224  if (.not.os%is_ocean_pe) return
225 
226  os%state%tr_fields => ocean_sfc%fields
227  os%Time = time_in
228  call initialize_mom(os%Time, param_file, os%dirs, os%MOM_CSp, time_in)
229  os%grid => os%MOM_CSp%G ; os%GV => os%MOM_CSp%GV
230  os%C_p = os%MOM_CSp%tv%C_p
231  os%fluxes%C_p = os%MOM_CSp%tv%C_p
232 
233  ! Read all relevant parameters and write them to the model log.
234  call log_version(param_file, mod, version, "")
235  call get_param(param_file, mod, "RESTART_CONTROL", os%Restart_control, &
236  "An integer whose bits encode which restart files are \n"//&
237  "written. Add 2 (bit 1) for a time-stamped file, and odd \n"//&
238  "(bit 0) for a non-time-stamped file. A restart file \n"//&
239  "will be saved at the end of the run segment for any \n"//&
240  "non-negative value.", default=1)
241  call get_param(param_file, mod, "TIMEUNIT", time_unit, &
242  "The time unit for ENERGYSAVEDAYS.", &
243  units="s", default=86400.0)
244  call get_param(param_file, mod, "ENERGYSAVEDAYS",os%energysavedays, &
245  "The interval in units of TIMEUNIT between saves of the \n"//&
246  "energies of the run and other globally summed diagnostics.", &
247  default=set_time(0,days=1), timeunit=time_unit)
248 
249  call get_param(param_file, mod, "OCEAN_SURFACE_STAGGER", stagger, &
250  "A case-insensitive character string to indicate the \n"//&
251  "staggering of the surface velocity field that is \n"//&
252  "returned to the coupler. Valid values include \n"//&
253  "'A', 'B', or 'C'.", default="C")
254  if (uppercase(stagger(1:1)) == 'A') then ; ocean_sfc%stagger = agrid
255  elseif (uppercase(stagger(1:1)) == 'B') then ; ocean_sfc%stagger = bgrid_ne
256  elseif (uppercase(stagger(1:1)) == 'C') then ; ocean_sfc%stagger = cgrid_ne
257  else ; call mom_error(fatal,"ocean_model_init: OCEAN_SURFACE_STAGGER = "// &
258  trim(stagger)//" is invalid.") ; endif
259 
260  call get_param(param_file, mod, "RESTORE_SALINITY",os%restore_salinity, &
261  "If true, the coupled driver will add a globally-balanced \n"//&
262  "fresh-water flux that drives sea-surface salinity \n"//&
263  "toward specified values.", default=.false.)
264  call get_param(param_file, mod, "RHO_0", rho0, &
265  "The mean ocean density used with BOUSSINESQ true to \n"//&
266  "calculate accelerations and the mass for conservation \n"//&
267  "properties, or with BOUSSINSEQ false to convert some \n"//&
268  "parameters from vertical units of m to kg m-2.", &
269  units="kg m-3", default=1035.0)
270  call get_param(param_file, mod, "G_EARTH", g_earth, &
271  "The gravitational acceleration of the Earth.", &
272  units="m s-2", default = 9.80)
273 
274  call get_param(param_file, mod, "ICE_SHELF", os%use_ice_shelf, &
275  "If true, enables the ice shelf model.", default=.false.)
276 
277  os%press_to_z = 1.0/(rho0*g_earth)
278 
279  call surface_forcing_init(time_in, os%grid, param_file, os%MOM_CSp%diag, &
280  os%forcing_CSp, os%restore_salinity)
281 
282  if (os%use_ice_shelf) then
283  call initialize_ice_shelf(param_file, os%grid, os%Time, os%ice_shelf_CSp, &
284  os%MOM_CSp%diag, os%fluxes)
285  endif
286 
287  call mom_sum_output_init(os%grid, param_file, os%dirs%output_directory, &
288  os%MOM_CSp%ntrunc, time_init, os%sum_output_CSp)
289 
290  ! This call has been moved into the first call to update_ocean_model.
291 ! call write_energy(OS%MOM_CSp%u, OS%MOM_CSp%v, OS%MOM_CSp%h, OS%MOM_CSp%tv, &
292 ! OS%Time, 0, OS%grid, OS%GV, OS%sum_output_CSp, OS%MOM_CSp%tracer_flow_CSp)
293 
294  ! write_energy_time is the next integral multiple of energysavedays.
295  os%write_energy_time = time_init + os%energysavedays * &
296  (1 + (os%Time - time_init) / os%energysavedays)
297 
298  if(ASSOCIATED(os%grid%Domain%maskmap)) then
299  call initialize_ocean_public_type(os%grid%Domain%mpp_domain, ocean_sfc, &
300  maskmap=os%grid%Domain%maskmap)
301  else
302  call initialize_ocean_public_type(os%grid%Domain%mpp_domain, ocean_sfc)
303  endif
304 ! call convert_state_to_ocean_type(state, Ocean_sfc, OS%grid)
305 
306  call close_param_file(param_file)
307  call diag_mediator_close_registration(os%MOM_CSp%diag)
308 
309  if (is_root_pe()) &
310  write(*,'(/12x,a/)') '======== COMPLETED MOM INITIALIZATION ========'
311 
312  call calltree_leave("ocean_model_init(")
313 end subroutine ocean_model_init
314 ! </SUBROUTINE> NAME="ocean_model_init"

318 ! <SUBROUTINE NAME="update_ocean_model">
319 !
320 ! <DESCRIPTION>
321 ! Update in time the ocean model fields. This code wraps the call to step_MOM
322 ! with MOM4's call.
323 ! </DESCRIPTION>
324 !
325 
326 subroutine update_ocean_model(Ice_ocean_boundary, OS, Ocean_sfc, &
327  time_start_update, ocean_coupling_time_step)
328  type(ice_ocean_boundary_type), intent(inout) :: Ice_ocean_boundary
329  type(ocean_state_type), pointer :: OS
330  type(ocean_public_type), intent(inout) :: Ocean_sfc
331  type(time_type), intent(in) :: time_start_update
332  type(time_type), intent(in) :: Ocean_coupling_time_step
333 ! This subroutine uses the forcing in Ice_ocean_boundary to advance the
334 ! ocean model's state from the input value of Ocean_state (which must be for
335 ! time time_start_update) for a time interval of Ocean_coupling_time_step,
336 ! returning the publicly visible ocean surface properties in Ocean_sfc and
337 ! storing the new ocean properties in Ocean_state.
338 
339 ! Arguments: Ice_ocean_boundary - A structure containing the various forcing
340 ! fields coming from the ice. It is intent in.
341 ! (inout) Ocean_state - A structure containing the internal ocean state.
342 ! (out) Ocean_sfc - A structure containing all the publicly visible ocean
343 ! surface fields after a coupling time step.
344 ! (in) time_start_update - The time at the beginning of the update step.
345 ! (in) Ocean_coupling_time_step - The amount of time over which to advance
346 ! the ocean.
347 
348 ! Note: although several types are declared intent(inout), this is to allow for
349 ! the possibility of halo updates and to keep previously allocated memory.
350 ! In practice, Ice_ocean_boundary is intent in, Ocean_state is private to
351 ! this module and intent inout, and Ocean_sfc is intent out.
352  type(time_type) :: Master_time ! This allows step_MOM to temporarily change
353  ! the time that is seen by internal modules.
354  type(time_type) :: Time1 ! The value of the ocean model's time at the
355  ! start of a call to step_MOM.
356  integer :: index_bnds(4) ! The computational domain index bounds in the
357  ! ice-ocean boundary type.
358  real :: weight ! Flux accumulation weight
359  real :: time_step ! The time step of a call to step_MOM in seconds.
360  integer :: secs, days
361 
362  call calltree_enter("update_ocean_model(), ocean_model_MOM.F90")
363  call get_time(ocean_coupling_time_step, secs, days)
364  time_step = 86400.0*real(days) + real(secs)
365 
366  if (time_start_update /= os%Time) then
367  call mom_error(warning, "update_ocean_model: internal clock does not "//&
368  "with time_start_update argument.")
369  endif
370  if (.not.associated(os)) then
371  call mom_error(fatal, "update_ocean_model called with an unassociated "// &
372  "ocean_state_type structure. ocean_model_init must be "// &
373  "called first to allocate this structure.")
374  return
375  endif
376 
377 ! Translate Ice_ocean_boundary into fluxes.
378  call mpp_get_compute_domain(ocean_sfc%Domain, index_bnds(1), index_bnds(2), &
379  index_bnds(3), index_bnds(4))
380 
381  weight = 1.0
382 
383  if (os%fluxes%fluxes_used) then
384  call enable_averaging(time_step, os%Time + ocean_coupling_time_step, os%MOM_CSp%diag) ! Needed to allow diagnostics in convert_IOB
385  call convert_iob_to_fluxes(ice_ocean_boundary, os%fluxes, index_bnds, os%Time, &
386  os%grid, os%forcing_CSp, os%state, os%restore_salinity)
387 #ifdef _USE_GENERIC_TRACER
388  call mom_generic_tracer_fluxes_accumulate(os%fluxes, weight) !here weight=1, just saving the current fluxes
389 #endif
390 
391  ! Add ice shelf fluxes
392 
393  if (os%use_ice_shelf) then
394  call shelf_calc_flux(os%State, os%fluxes, os%Time, time_step, os%Ice_shelf_CSp)
395  endif
396 
397  ! Indicate that there are new unused fluxes.
398  os%fluxes%fluxes_used = .false.
399  os%fluxes%dt_buoy_accum = time_step
400  else
401  os%flux_tmp%C_p = os%fluxes%C_p
402  call convert_iob_to_fluxes(ice_ocean_boundary, os%flux_tmp, index_bnds, os%Time, &
403  os%grid, os%forcing_CSp, os%state, os%restore_salinity)
404 
405  if (os%use_ice_shelf) then
406  call shelf_calc_flux(os%State, os%flux_tmp, os%Time, time_step, os%Ice_shelf_CSp)
407  endif
408 
409  call forcing_accumulate(os%flux_tmp, os%fluxes, time_step, os%grid, weight)
410 #ifdef _USE_GENERIC_TRACER
411  call mom_generic_tracer_fluxes_accumulate(os%flux_tmp, weight) !weight of the current flux in the running average
412 #endif
413  endif
414 
415  if (os%nstep==0) then
416  call finish_mom_initialization(os%Time, os%dirs, os%MOM_CSp, os%fluxes)
417 
418  call write_energy(os%MOM_CSp%u, os%MOM_CSp%v, os%MOM_CSp%h, os%MOM_CSp%tv, &
419  os%Time, 0, os%grid, os%GV, os%sum_output_CSp, &
420  os%MOM_CSp%tracer_flow_CSp)
421  endif
422 
423  call disable_averaging(os%MOM_CSp%diag)
424  master_time = os%Time ; time1 = os%Time
425 
426  call step_mom(os%fluxes, os%state, time1, time_step, os%MOM_CSp)
427 
428  os%Time = master_time + ocean_coupling_time_step
429  os%nstep = os%nstep + 1
430 
431  call enable_averaging(time_step, os%Time, os%MOM_CSp%diag)
432  call mech_forcing_diags(os%fluxes, time_step, os%grid, &
433  os%MOM_CSp%diag, os%forcing_CSp%handles)
434  call disable_averaging(os%MOM_CSp%diag)
435 
436  if (os%fluxes%fluxes_used) then
437  call enable_averaging(os%fluxes%dt_buoy_accum, os%Time, os%MOM_CSp%diag)
438  call forcing_diagnostics(os%fluxes, os%state, os%fluxes%dt_buoy_accum, &
439  os%grid, os%MOM_CSp%diag, os%forcing_CSp%handles)
440  call accumulate_net_input(os%fluxes, os%state, os%fluxes%dt_buoy_accum, &
441  os%grid, os%sum_output_CSp)
442  call disable_averaging(os%MOM_CSp%diag)
443  endif
444 
445 ! See if it is time to write out the energy.
446  if ((os%Time + ((ocean_coupling_time_step)/2) > os%write_energy_time) .and. &
447  (os%MOM_CSp%dt_trans==0.0)) then
448  call write_energy(os%MOM_CSp%u, os%MOM_CSp%v, os%MOM_CSp%h, os%MOM_CSp%tv, &
449  os%Time, os%nstep, os%grid, os%GV, os%sum_output_CSp, &
450  os%MOM_CSp%tracer_flow_CSp)
451  os%write_energy_time = os%write_energy_time + os%energysavedays
452  endif
453 
454 ! Translate state into Ocean.
455 ! call convert_state_to_ocean_type(OS%state, Ocean_sfc, OS%grid, &
456 ! Ice_ocean_boundary%p, OS%press_to_z)
457  call convert_state_to_ocean_type(os%state, ocean_sfc, os%grid)
458 
459  call calltree_leave("update_ocean_model()")
460 end subroutine update_ocean_model
461 ! </SUBROUTINE> NAME="update_ocean_model"

609 subroutine convert_state_to_ocean_type(state, Ocean_sfc, G, patm, press_to_z)
610  type(surface), intent(inout) :: state
611  type(ocean_public_type), target, intent(inout) :: Ocean_sfc
612  type(ocean_grid_type), intent(inout) :: G
613  real, optional, intent(in) :: patm(:,:)
614  real, optional, intent(in) :: press_to_z
615 ! This subroutine translates the coupler's ocean_data_type into MOM's
616 ! surface state variable. This may eventually be folded into the MOM
617 ! code that calculates the surface state in the first place.
618 ! Note the offset in the arrays because the ocean_data_type has no
619 ! halo points in its arrays and always uses absolute indicies.
620  real :: IgR0
621  character(len=48) :: val_str
622  integer :: isc_bnd, iec_bnd, jsc_bnd, jec_bnd
623  integer :: i, j, i0, j0, is, ie, js, je
624 
625  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
626  call pass_vector(state%u,state%v,g%Domain)
627 
628  call mpp_get_compute_domain(ocean_sfc%Domain, isc_bnd, iec_bnd, &
629  jsc_bnd, jec_bnd)
630  if (present(patm)) then
631  ! Check that the inidicies in patm are (isc_bnd:iec_bnd,jsc_bnd:jec_bnd).
632  if (.not.present(press_to_z)) call mom_error(fatal, &
633  'convert_state_to_ocean_type: press_to_z must be present if patm is.')
634  endif
635 
636  i0 = is - isc_bnd ; j0 = js - jsc_bnd
637  do j=jsc_bnd,jec_bnd ; do i=isc_bnd,iec_bnd
638  ocean_sfc%t_surf(i,j) = state%SST(i+i0,j+j0) + celsius_kelvin_offset
639  ocean_sfc%s_surf(i,j) = state%SSS(i+i0,j+j0)
640  ocean_sfc%sea_lev(i,j) = state%sea_lev(i+i0,j+j0)
641  if (present(patm)) &
642  ocean_sfc%sea_lev(i,j) = ocean_sfc%sea_lev(i,j) + patm(i,j) * press_to_z
643  if (associated(state%frazil)) &
644  ocean_sfc%frazil(i,j) = state%frazil(i+i0,j+j0)
645  ocean_sfc%area(i,j) = g%areaT(i+i0,j+j0)
646  enddo ; enddo
647 
648  if (ocean_sfc%stagger == agrid) then
649  do j=jsc_bnd,jec_bnd ; do i=isc_bnd,iec_bnd
650  ocean_sfc%u_surf(i,j) = g%mask2dT(i+i0,j+j0)*0.5*(state%u(i+i0,j+j0)+state%u(i-1+i0,j+j0))
651  ocean_sfc%v_surf(i,j) = g%mask2dT(i+i0,j+j0)*0.5*(state%v(i+i0,j+j0)+state%v(i+i0,j-1+j0))
652  enddo ; enddo
653  elseif (ocean_sfc%stagger == bgrid_ne) then
654  do j=jsc_bnd,jec_bnd ; do i=isc_bnd,iec_bnd
655  ocean_sfc%u_surf(i,j) = g%mask2dBu(i+i0,j+j0)*0.5*(state%u(i+i0,j+j0)+state%u(i+i0,j+j0+1))
656  ocean_sfc%v_surf(i,j) = g%mask2dBu(i+i0,j+j0)*0.5*(state%v(i+i0,j+j0)+state%v(i+i0+1,j+j0))
657  enddo ; enddo
658  elseif (ocean_sfc%stagger == cgrid_ne) then
659  do j=jsc_bnd,jec_bnd ; do i=isc_bnd,iec_bnd
660  ocean_sfc%u_surf(i,j) = g%mask2dCu(i+i0,j+j0)*state%u(i+i0,j+j0)
661  ocean_sfc%v_surf(i,j) = g%mask2dCv(i+i0,j+j0)*state%v(i+i0,j+j0)
662  enddo ; enddo
663  else
664  write(val_str, '(I8)') ocean_sfc%stagger
665  call mom_error(fatal, "convert_state_to_ocean_type: "//&
666  "Ocean_sfc%stagger has the unrecognized value of "//trim(val_str))
667  endif
668 
669  if (.not.associated(state%tr_fields,ocean_sfc%fields)) &
670  call mom_error(fatal,'state%tr_fields is not pointing to Ocean_sfc%fields')
671 
672 end subroutine convert_state_to_ocean_type

766 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~!
767 ! Ocean_stock_pe - returns stocks of heat, water, etc. for conservation checks.!
768 !~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~!
769 subroutine ocean_stock_pe(OS, index, value, time_index)
770  use stock_constants_mod, only : istock_water, istock_heat,istock_salt
771  type(ocean_state_type), pointer :: OS
772  integer, intent(in) :: index
773  real, intent(out) :: value
774  integer, optional, intent(in) :: time_index
775 ! Arguments: OS - A structure containing the internal ocean state.
776 ! (in) index - Index of conservation quantity of interest.
777 ! (in) value - Sum returned for the conservation quantity of interest.
778 ! (in,opt) time_index - Index for time level to use if this is necessary.
779 
780  real :: to_heat, to_mass, to_salt, PSU_to_kg ! Conversion constants.
781  integer :: i, j, k, is, ie, js, je, nz, ind
782 
783  value = 0.0
784  if (.not.associated(os)) return
785  if (.not.os%is_ocean_pe) return
786 
787  is = os%grid%isc ; ie = os%grid%iec
788  js = os%grid%jsc ; je = os%grid%jec ; nz = os%grid%ke
789 
790  select case (index)
791  case (istock_water)
792  ! Return the mass of fresh water in the ocean on this PE in kg.
793  to_mass = os%GV%H_to_kg_m2
794  if (os%GV%Boussinesq) then
795  do k=1,nz ; do j=js,je ; do i=is,ie ; if (os%grid%mask2dT(i,j) > 0.5) then
796  value = value + to_mass*(os%MOM_CSp%h(i,j,k) * os%grid%areaT(i,j))
797  endif ; enddo ; enddo ; enddo
798  else
799  ! In non-Boussinesq mode, the mass of salt needs to be subtracted.
800  psu_to_kg = 1.0e-3
801  do k=1,nz ; do j=js,je ; do i=is,ie ; if (os%grid%mask2dT(i,j) > 0.5) then
802  value = value + to_mass * ((1.0 - psu_to_kg*os%MOM_CSp%tv%S(i,j,k))*&
803  (os%MOM_CSp%h(i,j,k) * os%grid%areaT(i,j)))
804  endif ; enddo ; enddo ; enddo
805  endif
806  case (istock_heat)
807  ! Return the heat content of the ocean on this PE in J.
808  to_heat = os%GV%H_to_kg_m2 * os%C_p
809  do k=1,nz ; do j=js,je ; do i=is,ie ; if (os%grid%mask2dT(i,j) > 0.5) then
810  value = value + (to_heat * os%MOM_CSp%tv%T(i,j,k)) * &
811  (os%MOM_CSp%h(i,j,k)*os%grid%areaT(i,j))
812  endif ; enddo ; enddo ; enddo
813  case (istock_salt)
814  ! Return the mass of the salt in the ocean on this PE in kg.
815  ! The 1000 converts salinity in PSU to salt in kg kg-1.
816  to_salt = os%GV%H_to_kg_m2 / 1000.0
817  do k=1,nz ; do j=js,je ; do i=is,ie ; if (os%grid%mask2dT(i,j) > 0.5) then
818  value = value + (to_salt * os%MOM_CSp%tv%S(i,j,k)) * &
819  (os%MOM_CSp%h(i,j,k)*os%grid%areaT(i,j))
820  endif ; enddo ; enddo ; enddo
821  case default ; value = 0.0
822  end select
823 
824 end subroutine ocean_stock_pe
