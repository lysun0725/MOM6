24 !* By Robert Hallberg and Alistair Adcroft, 2006 *
25 !* *
26 !* This file contains the subroutine that calculates the hori- *
27 !* zontal accelerations due to pressure gradients using a 2nd-order *
28 !* analitically vertically integrated finite volume form, as *
29 !* described by Adcroft, A., R. Hallberg, and M. Harrison, 2008: A *
30 !* finite volume discretization of the pressure gradient force using *
31 !* analytic integration. Ocean Modelling, 22, 106-113. This form *
32 !* eliminates the thermobaric instabilities that had been a problem *
33 !* with previous forms of the pressure gradient force calculation, *
34 !* as described by Hallberg (2005), Ocean Modelling, 8, 279-300. *
35 !* *
36 !* PressureForce takes 8 arguments, which are described below. If *
37 !* a non-split time stepping scheme is used, the last three arguments *
38 !* are ignored. *
39 !* *
40 !* Macros written all in capital letters are defined in MOM_memory.h. *
41 !* *
42 !* A small fragment of the grid is shown below: *
43 !* *
44 !* j+1 x ^ x ^ x At x: q, CoriolisBu *
45 !* j+1 > o > o > At ^: v, PFv *
46 !* j x ^ x ^ x At >: u, PFu *
47 !* j > o > o > At o: h, bathyT, M, e, p, pbce, T, S *
48 !* j-1 x ^ x ^ x *
49 !* i-1 i i+1 *
50 !* i i+1 *
51 !* *
52 !* The boundaries always run through q grid points (x). *
53 !* *

97 subroutine pressureforce_afv(h, tv, PFu, PFv, G, GV, CS, ALE_CSp, p_atm, pbce, eta)
98  type(ocean_grid_type), intent(in) :: G
99  type(verticalgrid_type), intent(in) :: GV
100  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
101  type(thermo_var_ptrs), intent(inout) :: tv
102  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out) :: PFu
103  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(out) :: PFv
104  type(pressureforce_afv_cs), pointer :: CS
105  type(ale_cs), pointer :: ALE_CSp
106  real, dimension(:,:), optional, pointer :: p_atm

123 subroutine pressureforce_afv_nonbouss(h, tv, PFu, PFv, G, GV, CS, p_atm, pbce, eta)
124  type(ocean_grid_type), intent(in) :: G
125  type(verticalgrid_type), intent(in) :: GV
126  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
127  type(thermo_var_ptrs), intent(in) :: tv
128  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out) :: PFu
129  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(out) :: PFv
130  type(pressureforce_afv_cs), pointer :: CS
131  real, dimension(:,:), optional, pointer :: p_atm
132  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), optional, intent(out) :: pbce
133  real, dimension(SZI_(G),SZJ_(G)), optional, intent(out) :: eta
134 
135 ! This subroutine determines the acceleration due to hydrostatic pressure
136 ! forces, using the analytic finite volume form of the Pressure gradient, and
137 ! does not make the Boussinesq approximation. It follows the approach described
138 ! in Adcroft, Hallberg, and Harrison, 2008, Ocean Modelling.
139 ! To work, the following fields must be set outside of the usual
140 ! ie to ie, je to je range before this subroutine is called:
141 ! h[ie+1] and h[je+1] and (if tv%eqn_of_state is set) T[ie+1], S[ie+1],
142 ! T[je+1], and S[je+1].
143 ! Arguments: h - Layer thickness, in H (probably kg m-2).
144 ! (in) tv - A structure containing pointers to any available
145 ! thermodynamic fields, including potential temperature and
146 ! salinity or mixed layer density. Absent fields have NULL ptrs.
147 ! (out) PFu - Zonal acceleration due to pressure gradients
148 ! (equal to -dM/dx) in m s-2.
149 ! (out) PFv - Meridional acceleration due to pressure
150 ! gradients (equal to -dM/dy) in m s-2.
151 ! (in) G - The ocean's grid structure.
152 ! (in) GV - The ocean's vertical grid structure.
153 ! (in) CS - The control structure returned by a previous call to
154 ! PressureForce_init.
155 ! (in) p_atm - The pressure at the ice-ocean or atmosphere-ocean
156 ! interface in Pa.
157 ! (out) pbce - The baroclinic pressure anomaly in each layer
158 ! due to eta anomalies, in m2 s-2 H-1.
159 ! pbce points to a space with nz layers or NULL.
160 ! (out) eta - The bottom mass used to calculate PFu and PFv, in H,
161 ! with any tidal contributions or compressibility compensation.
162 
163  real, dimension(SZI_(G),SZJ_(G),SZK_(G)+1) :: p ! Interface pressure in Pa.
164  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), target :: &
165  T_tmp, & ! Temporary array of temperatures where layers that are lighter
166  ! than the mixed layer have the mixed layer's properties, in C.
167  s_tmp ! Temporary array of salinities where layers that are lighter
168  ! than the mixed layer have the mixed layer's properties, in psu.
169  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: &
170  dza, & ! The change in geopotential anomaly between the top and bottom
171  ! of a layer, in m2 s-2.
172  intp_dza ! The vertical integral in depth of the pressure anomaly less
173  ! the pressure anomaly at the top of the layer, in Pa m2 s-2.
174  real, dimension(SZI_(G),SZJ_(G)) :: &
175  dp, & ! The (positive) change in pressure across a layer, in Pa.
176  SSH, & ! The sea surface height anomaly, in m.
177  e_tidal, & ! The bottom geopotential anomaly due to tidal forces from
178  ! astronomical sources and self-attraction and loading, in m.
179  dm, & ! The barotropic adjustment to the Montgomery potential to
180  ! account for a reduced gravity model, in m2 s-2.
181  za ! The geopotential anomaly (i.e. g*e + alpha_0*pressure) at the
182  ! interface atop a layer, in m2 s-2.
183  real, dimension(SZDI_(G%Block(1)),SZDJ_(G%Block(1))) :: & ! on block indices
184  dp_bk, & ! The (positive) change in pressure across a layer, in Pa.
185  za_bk ! The geopotential anomaly (i.e. g*e + alpha_0*pressure) at the
186  ! interface atop a layer, in m2 s-2.
187 
188  real, dimension(SZI_(G)) :: Rho_cv_BL ! The coordinate potential density in the deepest variable
189  ! density near-surface layer, in kg m-3.
190  real, dimension(SZDIB_(G%Block(1)),SZDJ_(G%Block(1))) :: & ! on block indices
191  intx_za_bk ! The zonal integral of the geopotential anomaly along the
192  ! interface below a layer, divided by the grid spacing, m2 s-2.
193  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)) :: &
194  intx_dza ! The change in intx_za through a layer, in m2 s-2.
195  real, dimension(SZDI_(G%Block(1)),SZDJB_(G%Block(1))) :: & ! on block indices
196  inty_za_bk ! The meridional integral of the geopotential anomaly along the
197  ! interface below a layer, divided by the grid spacing, m2 s-2.
198  real, dimension(SZI_(G),SZJB_(G),SZK_(G)) :: &
199  inty_dza ! The change in inty_za through a layer, in m2 s-2.
200  real :: p_ref(szi_(g)) ! The pressure used to calculate the coordinate
201  ! density, in Pa (usually 2e7 Pa = 2000 dbar).
202 
203  real :: dp_neglect ! A thickness that is so small it is usually lost
204  ! in roundoff and can be neglected, in Pa.
205  real :: alpha_anom ! The in-situ specific volume, averaged over a
206  ! layer, less alpha_ref, in m3 kg-1.
207  logical :: use_p_atm ! If true, use the atmospheric pressure.
208  logical :: use_EOS ! If true, density is calculated from T & S using an
209  ! equation of state.
210  type(thermo_var_ptrs) :: tv_tmp! A structure of temporary T & S.
211 
212  real :: alpha_ref ! A reference specific volume, in m3 kg-1, that is used
213  ! to reduce the impact of truncation errors.
214  real :: rho_in_situ(szi_(g)) ! The in situ density, in kg m-3.
215  real :: Pa_to_H ! A factor to convert from Pa to the thicknesss units (H).
216 ! real :: oneatm = 101325.0 ! 1 atm in Pa (kg/ms2)
217  real :: I_gEarth
218  real, parameter :: C1_6 = 1.0/6.0
219  integer :: is, ie, js, je, Isq, Ieq, Jsq, Jeq, nz, nkmb
220  integer :: is_bk, ie_bk, js_bk, je_bk, Isq_bk, Ieq_bk, Jsq_bk, Jeq_bk
221  integer :: i, j, k, n, ib, jb, ioff_bk, joff_bk
222 
223  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
224  nkmb=gv%nk_rho_varies
225  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
226 
227  use_p_atm = .false.
228  if (present(p_atm)) then ; if (associated(p_atm)) use_p_atm = .true. ; endif
229  use_eos = associated(tv%eqn_of_state)
230 
231  if (.not.associated(cs)) call mom_error(fatal, &
232  "MOM_PressureForce: Module must be initialized before it is used.")
233 
234  dp_neglect = gv%H_to_Pa * gv%H_subroundoff
235  alpha_ref = 1.0/cs%Rho0
236 
237 !$OMP parallel default(none) shared(Isq,Ieq,Jsq,Jeq,nz,use_p_atm,p,p_atm,GV,h)
238  if (use_p_atm) then
239 !$OMP do
240  do j=jsq,jeq+1 ; do i=isq,ieq+1
241  p(i,j,1) = p_atm(i,j)
242  enddo ; enddo
243  else
244 !$OMP do
245  do j=jsq,jeq+1 ; do i=isq,ieq+1
246  p(i,j,1) = 0.0 ! or oneatm
247  enddo ; enddo
248  endif
249 !$OMP do
250  do j=jsq,jeq+1 ; do k=2,nz+1 ; do i=isq,ieq+1
251  p(i,j,k) = p(i,j,k-1) + gv%H_to_Pa * h(i,j,k-1)
252  enddo ; enddo ; enddo
253 !$OMP end parallel
254 
255  i_gearth = 1.0 / g%g_Earth
256 
257  if (use_eos) then
258  ! With a bulk mixed layer, replace the T & S of any layers that are
259  ! lighter than the the buffer layer with the properties of the buffer
260  ! layer. These layers will be massless anyway, and it avoids any
261  ! formal calculations with hydrostatically unstable profiles.
262  if (nkmb>0) then
263  tv_tmp%T => t_tmp ; tv_tmp%S => s_tmp
264  tv_tmp%eqn_of_state => tv%eqn_of_state
265  do i=isq,ieq+1 ; p_ref(i) = tv%P_Ref ; enddo
266 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,nkmb,tv_tmp,tv,p_ref,GV) &
267 !$OMP private(Rho_cv_BL)
268  do j=jsq,jeq+1
269  do k=1,nkmb ; do i=isq,ieq+1
270  tv_tmp%T(i,j,k) = tv%T(i,j,k) ; tv_tmp%S(i,j,k) = tv%S(i,j,k)
271  enddo ; enddo
272  call calculate_density(tv%T(:,j,nkmb), tv%S(:,j,nkmb), p_ref, &
273  rho_cv_bl(:), isq, ieq-isq+2, tv%eqn_of_state)
274  do k=nkmb+1,nz ; do i=isq,ieq+1
275  if (gv%Rlay(k) < rho_cv_bl(i)) then
276  tv_tmp%T(i,j,k) = tv%T(i,j,nkmb) ; tv_tmp%S(i,j,k) = tv%S(i,j,nkmb)
277  else
278  tv_tmp%T(i,j,k) = tv%T(i,j,k) ; tv_tmp%S(i,j,k) = tv%S(i,j,k)
279  endif
280  enddo ; enddo
281  enddo
282  else
283  tv_tmp%T => tv%T ; tv_tmp%S => tv%S
284  tv_tmp%eqn_of_state => tv%eqn_of_state
285  endif
286  endif
287 
288 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,is,ie,js,je,tv_tmp,alpha_ref, &
289 !$OMP p,h,G,GV,tv,dza,intp_dza,intx_dza,inty_dza,use_EOS) &
290 !$OMP private(alpha_anom,dp)
291  do k=1,nz
292  ! Calculate 4 integrals through the layer that are required in the
293  ! subsequent calculation.
294  if (use_eos) then
295  call int_specific_vol_dp(tv_tmp%T(:,:,k), tv_tmp%S(:,:,k), p(:,:,k), &
296  p(:,:,k+1), alpha_ref, g%HI, tv%eqn_of_state, &
297  dza(:,:,k), intp_dza(:,:,k), intx_dza(:,:,k), &
298  inty_dza(:,:,k))
299  else
300  alpha_anom = 1.0/gv%Rlay(k) - alpha_ref
301  do j=jsq,jeq+1 ; do i=isq,ieq+1
302  dp(i,j) = gv%H_to_Pa * h(i,j,k)
303  dza(i,j,k) = alpha_anom * dp(i,j)
304  intp_dza(i,j,k) = 0.5 * alpha_anom * dp(i,j)**2
305  enddo ; enddo
306  do j=js,je ; do i=isq,ieq
307  intx_dza(i,j,k) = 0.5 * alpha_anom * (dp(i,j)+dp(i+1,j))
308  enddo ; enddo
309  do j=jsq,jeq ; do i=is,ie
310  inty_dza(i,j,k) = 0.5 * alpha_anom * (dp(i,j)+dp(i,j+1))
311  enddo ; enddo
312  endif
313  enddo
314 
315  ! The bottom geopotential anomaly is calculated first so that the increments
316  ! to the geopotential anomalies can be reused. Alternately, the surface
317  ! geopotential could be calculated directly with separate calls to
318  ! int_specific_vol_dp with alpha_ref=0, and the anomalies used going
319  ! downward, which would relieve the need for dza, intp_dza, intx_dza, and
320  ! inty_dza to be 3-D arrays.
321 
322  ! Sum vertically to determine the surface geopotential anomaly.
323 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,za,alpha_ref,p,G,dza)
324  do j=jsq,jeq+1
325  do i=isq,ieq+1
326  za(i,j) = alpha_ref*p(i,j,nz+1) - g%g_Earth*g%bathyT(i,j)
327  enddo
328  do k=nz,1,-1 ; do i=isq,ieq+1
329  za(i,j) = za(i,j) + dza(i,j,k)
330  enddo ; enddo
331  enddo
332 
333  if (cs%tides) then
334  ! Find and add the tidal geopotential anomaly.
335 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,SSH,za,alpha_ref,p,I_gEarth)
336  do j=jsq,jeq+1 ; do i=isq,ieq+1
337  ssh(i,j) = (za(i,j) - alpha_ref*p(i,j,1)) * i_gearth
338  enddo ; enddo
339  call calc_tidal_forcing(cs%Time, ssh, e_tidal, g, cs%tides_CSp)
340 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,za,G,e_tidal)
341  do j=jsq,jeq+1 ; do i=isq,ieq+1
342  za(i,j) = za(i,j) - g%g_Earth*e_tidal(i,j)
343  enddo ; enddo
344  endif
345 
346  if (cs%GFS_scale < 1.0) then
347  ! Adjust the Montgomery potential to make this a reduced gravity model.
348  if (use_eos) then
349 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,tv_tmp,p,tv,dM,CS,alpha_ref,za) &
350 !$OMP private(rho_in_situ)
351  do j=jsq,jeq+1
352  call calculate_density(tv_tmp%T(:,j,1), tv_tmp%S(:,j,1), p(:,j,1), &
353  rho_in_situ, isq, ieq-isq+2, tv%eqn_of_state)
354 
355  do i=isq,ieq+1
356  dm(i,j) = (cs%GFS_scale - 1.0) * &
357  (p(i,j,1)*(1.0/rho_in_situ(i) - alpha_ref) + za(i,j))
358  enddo
359  enddo
360  else
361 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,dM,CS,p,GV,alpha_ref,za)
362  do j=jsq,jeq+1 ; do i=isq,ieq+1
363  dm(i,j) = (cs%GFS_scale - 1.0) * &
364  (p(i,j,1)*(1.0/gv%Rlay(1) - alpha_ref) + za(i,j))
365  enddo ; enddo
366  endif
367 ! else
368 ! do j=Jsq,Jeq+1 ; do i=Isq,Ieq+1 ; dM(i,j) = 0.0 ; enddo ; enddo
369  endif
370 
371  ! This order of integrating upward and then downward again is necessary with
372  ! a nonlinear equation of state, so that the surface geopotentials will go
373  ! linearly between the values at thickness points, but the bottom
374  ! geopotentials will not now be linear at the sub-grid-scale. Doing this
375  ! ensures no motion with flat isopycnals, even with a nonlinear equation of state.
376 !$OMP parallel do default(none) shared(nz,za,G,GV,dza,intx_dza,h,PFu, &
377 !$OMP intp_dza,p,dp_neglect,inty_dza,PFv,CS,dM) &
378 !$OMP private(is_bk,ie_bk,js_bk,je_bk,Isq_bk,Ieq_bk,Jsq_bk, &
379 !$OMP Jeq_bk,ioff_bk,joff_bk,i,j,za_bk,intx_za_bk, &
380 !$OMP inty_za_bk,dp_bk)
381  do n = 1, g%nblocks
382  is_bk=g%block(n)%isc ; ie_bk=g%block(n)%iec
383  js_bk=g%block(n)%jsc ; je_bk=g%block(n)%jec
384  isq_bk=g%block(n)%IscB ; ieq_bk=g%block(n)%IecB
385  jsq_bk=g%block(n)%JscB ; jeq_bk=g%block(n)%JecB
386  ioff_bk = g%Block(n)%idg_offset - g%HI%idg_offset
387  joff_bk = g%Block(n)%jdg_offset - g%HI%jdg_offset
388  do jb=jsq_bk,jeq_bk+1 ; do ib=isq_bk,ieq_bk+1
389  i = ib+ioff_bk ; j = jb+joff_bk
390  za_bk(ib,jb) = za(i,j)
391  enddo ; enddo
392  do jb=js_bk,je_bk ; do ib=isq_bk,ieq_bk
393  i = ib+ioff_bk ; j = jb+joff_bk
394  intx_za_bk(ib,jb) = 0.5*(za_bk(ib,jb) + za_bk(ib+1,jb))
395  enddo ; enddo
396  do jb=jsq_bk,jeq_bk ; do ib=is_bk,ie_bk
397  i = ib+ioff_bk ; j = jb+joff_bk
398  inty_za_bk(ib,jb) = 0.5*(za_bk(ib,jb) + za_bk(ib,jb+1))
399  enddo ; enddo
400  do k=1,nz
401  ! These expressions for the acceleration have been carefully checked in
402  ! a set of idealized cases, and should be bug-free.
403  do jb=jsq_bk,jeq_bk+1 ; do ib=isq_bk,ieq_bk+1
404  i = ib+ioff_bk ; j = jb+joff_bk
405  dp_bk(ib,jb) = gv%H_to_Pa*h(i,j,k)
406  za_bk(ib,jb) = za_bk(ib,jb) - dza(i,j,k)
407  enddo ; enddo
408  do jb=js_bk,je_bk ; do ib=isq_bk,ieq_bk
409  i = ib+ioff_bk ; j = jb+joff_bk
410  intx_za_bk(ib,jb) = intx_za_bk(ib,jb) - intx_dza(i,j,k)
411  pfu(i,j,k) = (((za_bk(ib,jb)*dp_bk(ib,jb) + intp_dza(i,j,k)) - &
412  (za_bk(ib+1,jb)*dp_bk(ib+1,jb) + intp_dza(i+1,j,k))) + &
413  ((dp_bk(ib+1,jb) - dp_bk(ib,jb)) * intx_za_bk(ib,jb) - &
414  (p(i+1,j,k) - p(i,j,k)) * intx_dza(i,j,k))) * &
415  (2.0*g%IdxCu(i,j) / ((dp_bk(ib,jb) + dp_bk(ib+1,jb)) + &
416  dp_neglect))
417  enddo ; enddo
418  do jb=jsq_bk,jeq_bk ; do ib=is_bk,ie_bk
419  i = ib+ioff_bk ; j = jb+joff_bk
420  inty_za_bk(ib,jb) = inty_za_bk(ib,jb) - inty_dza(i,j,k)
421  pfv(i,j,k) = (((za_bk(ib,jb)*dp_bk(ib,jb) + intp_dza(i,j,k)) - &
422  (za_bk(ib,jb+1)*dp_bk(ib,jb+1) + intp_dza(i,j+1,k))) + &
423  ((dp_bk(ib,jb+1) - dp_bk(ib,jb)) * inty_za_bk(ib,jb) - &
424  (p(i,j+1,k) - p(i,j,k)) * inty_dza(i,j,k))) * &
425  (2.0*g%IdyCv(i,j) / ((dp_bk(ib,jb) + dp_bk(ib,jb+1)) + &
426  dp_neglect))
427  enddo ; enddo
428 
429  if (cs%GFS_scale < 1.0) then
430  ! Adjust the Montgomery potential to make this a reduced gravity model.
431  do j=js_bk+joff_bk,je_bk+joff_bk ; do i=isq_bk+ioff_bk,ieq_bk+ioff_bk
432  pfu(i,j,k) = pfu(i,j,k) - (dm(i+1,j) - dm(i,j)) * g%IdxCu(i,j)
433  enddo ; enddo
434  do j=jsq_bk+joff_bk,jeq_bk+joff_bk ; do i=is_bk+ioff_bk,ie_bk+ioff_bk
435  pfv(i,j,k) = pfv(i,j,k) - (dm(i,j+1) - dm(i,j)) * g%IdyCv(i,j)
436  enddo ; enddo
437  endif
438  enddo
439  enddo
440 
441  if (present(pbce)) then
442  call set_pbce_nonbouss(p, tv_tmp, g, gv, g%g_Earth, cs%GFS_scale, pbce)
443  endif
444 
445  if (present(eta)) then
446  pa_to_h = 1.0 / gv%H_to_Pa
447  if (use_p_atm) then
448 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,eta,p,p_atm,Pa_to_H)
449  do j=jsq,jeq+1 ; do i=isq,ieq+1
450  eta(i,j) = (p(i,j,nz+1) - p_atm(i,j))*pa_to_h ! eta has the same units as h.
451  enddo ; enddo
452  else
453 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,eta,p,Pa_to_H)
454  do j=jsq,jeq+1 ; do i=isq,ieq+1
455  eta(i,j) = p(i,j,nz+1)*pa_to_h ! eta has the same units as h.
456  enddo ; enddo
457  endif
458  endif
459 
460  if (cs%id_e_tidal>0) call post_data(cs%id_e_tidal, e_tidal, cs%diag)
461 
462 end subroutine pressureforce_afv_nonbouss

464 subroutine pressureforce_afv_bouss(h, tv, PFu, PFv, G, GV, CS, ALE_CSp, p_atm, pbce, eta)
465  type(ocean_grid_type), intent(in) :: G
466  type(verticalgrid_type), intent(in) :: GV
467  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
468  type(thermo_var_ptrs), intent(in) :: tv
469  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out) :: PFu
470  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(out) :: PFv
471  type(pressureforce_afv_cs), pointer :: CS
472  type(ale_cs), pointer :: ALE_CSp
473  real, dimension(:,:), optional, pointer :: p_atm
474  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), optional, intent(out) :: pbce
475  real, dimension(SZI_(G),SZJ_(G)), optional, intent(out) :: eta
476 
477 ! This subroutine determines the acceleration due to pressure forces, using
478 ! a finite volume form of the terms and analytic integrals in depth to avoid
479 ! the possibility of a thermobaric instability, as described by Adcroft et al.,
480 ! 2008.
481 ! To work, the following fields must be set outside of the usual
482 ! ie to ie, je to je range before this subroutine is called:
483 ! h[ie+1] and h[je+1] and (if tv%eqn_of_state is set) T[ie+1], S[ie+1],
484 ! T[je+1], and S[je+1].
485 ! Arguments: h - Layer thickness, in H (probably m).
486 ! (in) tv - A structure containing pointers to any available
487 ! thermodynamic fields, including potential temperature and
488 ! salinity or mixed layer density. Absent fields have NULL ptrs.
489 ! (out) PFu - Zonal acceleration due to pressure gradients
490 ! (equal to -dM/dx) in m s-2.
491 ! (out) PFv - Meridional acceleration due to pressure
492 ! gradients (equal to -dM/dy) in m s-2.
493 ! (in) G - The ocean's grid structure.
494 ! (in) GV - The ocean's vertical grid structure.
495 ! (in) CS - The control structure returned by a previous call to
496 ! PressureForce_init.
497 ! (in) p_atm - the pressure at the ice-ocean or atmosphere-ocean
498 ! interface in Pa.
499 ! (out) pbce - the baroclinic pressure anomaly in each layer
500 ! due to free surface height anomalies, in m2 H-1 s-2.
501 ! (out) eta - the free surface height used to calculate PFu and PFv, in m,
502 ! with any tidal contributions or compressibility compensation.
503 
504  real, dimension(SZI_(G),SZJ_(G),SZK_(G)+1) :: e ! Interface height in m.
505  real, dimension(SZI_(G),SZJ_(G)) :: &
506  e_tidal, & ! The bottom geopotential anomaly due to tidal forces from
507  ! astronomical sources and self-attraction and loading, in m.
508  dm ! The barotropic adjustment to the Montgomery potential to
509  ! account for a reduced gravity model, in m2 s-2.
510  real, dimension(SZI_(G)) :: &
511  Rho_cv_BL ! The coordinate potential density in the deepest variable
512  ! density near-surface layer, in kg m-3.
513  real, dimension(SZDI_(G%Block(1)),SZDJ_(G%Block(1))) :: & ! on block indices
514  dz_bk, & ! The change in geopotential thickness through a layer, m2 s-2.
515  pa_bk, & ! The pressure anomaly (i.e. pressure + g*RHO_0*e) at the
516  ! the interface atop a layer, in Pa.
517  dpa_bk, & ! The change in pressure anomaly between the top and bottom
518  ! of a layer, in Pa.
519  intz_dpa_bk ! The vertical integral in depth of the pressure anomaly less
520  ! the pressure anomaly at the top of the layer, in H Pa (m Pa).
521  real, dimension(SZDIB_(G%Block(1)),SZDJ_(G%Block(1))) :: & ! on block indices
522  intx_pa_bk, & ! The zonal integral of the pressure anomaly along the interface
523  ! atop a layer, divided by the grid spacing, in Pa.
524  intx_dpa_bk ! The change in intx_pa through a layer, in Pa.
525  real, dimension(SZDI_(G%Block(1)),SZDJB_(G%Block(1))) :: & ! on block indices
526  inty_pa_bk, & ! The meridional integral of the pressure anomaly along the
527  ! interface atop a layer, divided by the grid spacing, in Pa.
528  inty_dpa_bk ! The change in inty_pa through a layer, in Pa.
529 
530  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), target :: &
531  T_tmp, & ! Temporary array of temperatures where layers that are lighter
532  ! than the mixed layer have the mixed layer's properties, in C.
533  s_tmp ! Temporary array of salinities where layers that are lighter
534  ! than the mixed layer have the mixed layer's properties, in psu.
535  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: &
536  S_t, S_b, T_t, T_b ! Top and bottom edge values for linear reconstructions
537  ! of salinity and temperature within each layer.
538  real :: rho_in_situ(szi_(g)) ! The in situ density, in kg m-3.
539  real :: p_ref(szi_(g)) ! The pressure used to calculate the coordinate
540  ! density, in Pa (usually 2e7 Pa = 2000 dbar).
541  real :: p0(szi_(g)) ! An array of zeros to use for pressure in Pa.
542  real :: h_neglect ! A thickness that is so small it is usually lost
543  ! in roundoff and can be neglected, in m.
544  real :: I_Rho0 ! 1/Rho0.
545  real :: G_Rho0 ! G_Earth / Rho0 in m4 s-2 kg-1.
546  real :: Rho_ref ! The reference density in kg m-3.
547  logical :: use_p_atm ! If true, use the atmospheric pressure.
548  logical :: use_ALE ! If true, use an ALE pressure reconstruction.
549  logical :: use_EOS ! If true, density is calculated from T & S using an
550  ! equation of state.
551  type(thermo_var_ptrs) :: tv_tmp! A structure of temporary T & S.
552 
553  real, parameter :: C1_6 = 1.0/6.0
554  integer :: is, ie, js, je, Isq, Ieq, Jsq, Jeq, nz, nkmb
555  integer :: is_bk, ie_bk, js_bk, je_bk, Isq_bk, Ieq_bk, Jsq_bk, Jeq_bk
556  integer :: ioff_bk, joff_bk
557  integer :: i, j, k, n, ib, jb
558  integer :: PRScheme
559 
560  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
561  nkmb=gv%nk_rho_varies
562  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
563 
564  if (.not.associated(cs)) call mom_error(fatal, &
565  "MOM_PressureForce: Module must be initialized before it is used.")
566 
567  use_p_atm = .false.
568  if (present(p_atm)) then ; if (associated(p_atm)) use_p_atm = .true. ; endif
569  use_eos = associated(tv%eqn_of_state)
570  do i=isq,ieq+1 ; p0(i) = 0.0 ; enddo
571  use_ale = .false.
572  if (associated(ale_csp)) use_ale = usepressurereconstruction(ale_csp) .and. use_eos
573 
574  prscheme = pressurereconstructionscheme(ale_csp)
575  h_neglect = gv%H_subroundoff
576  i_rho0 = 1.0/gv%Rho0
577  g_rho0 = g%g_Earth/gv%Rho0
578  rho_ref = cs%Rho0
579 
580  if (cs%tides) then
581  ! Determine the surface height anomaly for calculating self attraction
582  ! and loading. This should really be based on bottom pressure anomalies,
583  ! but that is not yet implemented, and the current form is correct for
584  ! barotropic tides.
585 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,e,G,GV,h)
586  do j=jsq,jeq+1
587  do i=isq,ieq+1
588  e(i,j,1) = -1.0*g%bathyT(i,j)
589  enddo
590  do k=1,nz ; do i=isq,ieq+1
591  e(i,j,1) = e(i,j,1) + h(i,j,k)*gv%H_to_m
592  enddo ; enddo
593  enddo
594  call calc_tidal_forcing(cs%Time, e(:,:,1), e_tidal, g, cs%tides_CSp)
595  endif
596 
597 ! Here layer interface heights, e, are calculated.
598 !$OMP parallel default(none) shared(Isq,Ieq,Jsq,Jeq,nz,e,G,GV,h,CS,e_tidal)
599  if (cs%tides) then
600 !$OMP do
601  do j=jsq,jeq+1 ; do i=isq,ieq+1
602  e(i,j,nz+1) = -1.0*g%bathyT(i,j) - e_tidal(i,j)
603  enddo ; enddo
604  else
605 !$OM do
606  do j=jsq,jeq+1 ; do i=isq,ieq+1
607  e(i,j,nz+1) = -1.0*g%bathyT(i,j)
608  enddo ; enddo
609  endif
610 !$OMP do
611  do j=jsq,jeq+1; do k=nz,1,-1 ; do i=isq,ieq+1
612  e(i,j,k) = e(i,j,k+1) + h(i,j,k)*gv%H_to_m
613  enddo ; enddo ; enddo
614 !$OMP end parallel
615 
616 
617  if (use_eos) then
618 ! With a bulk mixed layer, replace the T & S of any layers that are
619 ! lighter than the the buffer layer with the properties of the buffer
620 ! layer. These layers will be massless anyway, and it avoids any
621 ! formal calculations with hydrostatically unstable profiles.
622 
623  if (nkmb>0) then
624  tv_tmp%T => t_tmp ; tv_tmp%S => s_tmp
625  tv_tmp%eqn_of_state => tv%eqn_of_state
626 
627  do i=isq,ieq+1 ; p_ref(i) = tv%P_Ref ; enddo
628 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nkmb,nz,GV,tv_tmp,tv,p_ref) &
629 !$OMP private(Rho_cv_BL)
630  do j=jsq,jeq+1
631  do k=1,nkmb ; do i=isq,ieq+1
632  tv_tmp%T(i,j,k) = tv%T(i,j,k) ; tv_tmp%S(i,j,k) = tv%S(i,j,k)
633  enddo ; enddo
634  call calculate_density(tv%T(:,j,nkmb), tv%S(:,j,nkmb), p_ref, &
635  rho_cv_bl(:), isq, ieq-isq+2, tv%eqn_of_state)
636 
637  do k=nkmb+1,nz ; do i=isq,ieq+1
638  if (gv%Rlay(k) < rho_cv_bl(i)) then
639  tv_tmp%T(i,j,k) = tv%T(i,j,nkmb) ; tv_tmp%S(i,j,k) = tv%S(i,j,nkmb)
640  else
641  tv_tmp%T(i,j,k) = tv%T(i,j,k) ; tv_tmp%S(i,j,k) = tv%S(i,j,k)
642  endif
643  enddo ; enddo
644  enddo
645  else
646  tv_tmp%T => tv%T ; tv_tmp%S => tv%S
647  tv_tmp%eqn_of_state => tv%eqn_of_state
648  endif
649  endif
650 
651 !$OMP parallel default(none) shared(Jsq,Jeq,Isq,Ieq,tv_tmp,p_atm,rho_in_situ,tv, &
652 !$OMP p0,dM,CS,G_Rho0,e,use_p_atm,use_EOS,GV, &
653 !$OMP rho_ref,js,je,is,ie)
654  if (cs%GFS_scale < 1.0) then
655  ! Adjust the Montgomery potential to make this a reduced gravity model.
656  if (use_eos) then
657 !$OMP do
658  do j=jsq,jeq+1
659  if (use_p_atm) then
660  call calculate_density(tv_tmp%T(:,j,1), tv_tmp%S(:,j,1), p_atm(:,j), &
661  rho_in_situ, isq, ieq-isq+2, tv%eqn_of_state)
662  else
663  call calculate_density(tv_tmp%T(:,j,1), tv_tmp%S(:,j,1), p0, &
664  rho_in_situ, isq, ieq-isq+2, tv%eqn_of_state)
665  endif
666  do i=isq,ieq+1
667  dm(i,j) = (cs%GFS_scale - 1.0) * (g_rho0 * rho_in_situ(i)) * e(i,j,1)
668  enddo
669  enddo
670  else
671 !$OMP do
672  do j=jsq,jeq+1 ; do i=isq,ieq+1
673  dm(i,j) = (cs%GFS_scale - 1.0) * (g_rho0 * gv%Rlay(1)) * e(i,j,1)
674  enddo ; enddo
675  endif
676  endif
677 !$OMP end parallel
678 
679 ! Have checked that rho_0 drops out and that the 1-layer case is right. RWH.
680 
681  ! If regridding is activated, do a linear reconstruction of salinity
682  ! and temperature across each layer. The subscripts 't' and 'b' refer
683  ! to top and bottom values within each layer (these are the only degrees
684  ! of freedeom needed to know the linear profile).
685  if ( use_ale ) then
686  if ( prscheme == pressure_reconstruction_plm ) then
687  call pressure_gradient_plm (ale_csp, s_t, s_b, t_t, t_b, g, gv, tv, h);
688  elseif ( prscheme == pressure_reconstruction_ppm ) then
689  call pressure_gradient_ppm (ale_csp, s_t, s_b, t_t, t_b, g, gv, tv, h);
690  endif
691  endif
692 
693 !$OMP parallel do default(none) shared(use_p_atm,rho_ref,G,GV,e, &
694 !$OMP p_atm,nz,use_EOS,use_ALE,PRScheme,T_t,T_b,S_t, &
695 !$OMP S_b,CS,tv,tv_tmp,h,PFu,I_Rho0,h_neglect,PFv,dM)&
696 !$OMP private(is_bk,ie_bk,js_bk,je_bk,Isq_bk,Ieq_bk,Jsq_bk, &
697 !$OMP Jeq_bk,ioff_bk,joff_bk,pa_bk, &
698 !$OMP intx_pa_bk,inty_pa_bk,dpa_bk,intz_dpa_bk, &
699 !$OMP intx_dpa_bk,inty_dpa_bk,dz_bk,i,j)
700  do n = 1, g%nblocks
701  is_bk=g%Block(n)%isc ; ie_bk=g%Block(n)%iec
702  js_bk=g%Block(n)%jsc ; je_bk=g%Block(n)%jec
703  isq_bk=g%Block(n)%IscB ; ieq_bk=g%Block(n)%IecB
704  jsq_bk=g%Block(n)%JscB ; jeq_bk=g%Block(n)%JecB
705  ioff_bk = g%Block(n)%idg_offset - g%HI%idg_offset
706  joff_bk = g%Block(n)%jdg_offset - g%HI%jdg_offset
707 
708  ! Set the surface boundary conditions on pressure anomaly and its horizontal
709  ! integrals, assuming that the surface pressure anomaly varies linearly
710  ! in x and y.
711  if (use_p_atm) then
712  do jb=jsq_bk,jeq_bk+1 ; do ib=isq_bk,ieq_bk+1
713  i = ib+ioff_bk ; j = jb+joff_bk
714  pa_bk(ib,jb) = (rho_ref*g%g_Earth)*e(i,j,1) + p_atm(i,j)
715  enddo ; enddo
716  else
717  do jb=jsq_bk,jeq_bk+1 ; do ib=isq_bk,ieq_bk+1
718  i = ib+ioff_bk ; j = jb+joff_bk
719  pa_bk(ib,jb) = (rho_ref*g%g_Earth)*e(i,j,1)
720  enddo ; enddo
721  endif
722  do jb=js_bk,je_bk ; do ib=isq_bk,ieq_bk
723  intx_pa_bk(ib,jb) = 0.5*(pa_bk(ib,jb) + pa_bk(ib+1,jb))
724  enddo ; enddo
725  do jb=jsq_bk,jeq_bk ; do ib=is_bk,ie_bk
726  inty_pa_bk(ib,jb) = 0.5*(pa_bk(ib,jb) + pa_bk(ib,jb+1))
727  enddo ; enddo
728 
729  do k=1,nz
730  ! Calculate 4 integrals through the layer that are required in the
731  ! subsequent calculation.
732 
733  if (use_eos) then
734  ! The following routine computes the integrals that are needed to
735  ! calculate the pressure gradient force. Linear profiles for T and S are
736  ! assumed when regridding is activated. Otherwise, the previous version
737  ! is used, whereby densities within each layer are constant no matter
738  ! where the layers are located.
739  if ( use_ale ) then
740  if ( prscheme == pressure_reconstruction_plm ) then
741  call int_density_dz_generic_plm ( t_t(:,:,k), t_b(:,:,k), &
742  s_t(:,:,k), s_b(:,:,k), e(:,:,k), e(:,:,k+1), &
743  rho_ref, cs%Rho0, g%g_Earth, &
744  gv%H_subroundoff, g%bathyT, g%HI, g%Block(n), &
745  tv%eqn_of_state, dpa_bk, intz_dpa_bk, intx_dpa_bk, inty_dpa_bk, &
746  usemasswghtinterp = cs%useMassWghtInterp)
747  elseif ( prscheme == pressure_reconstruction_ppm ) then
748  call int_density_dz_generic_ppm ( tv%T(:,:,k), t_t(:,:,k), t_b(:,:,k), &
749  tv%S(:,:,k), s_t(:,:,k), s_b(:,:,k), e(:,:,k), e(:,:,k+1), &
750  rho_ref, cs%Rho0, g%G_Earth, &
751  g%HI, g%Block(n), tv%eqn_of_state, dpa_bk, intz_dpa_bk, &
752  intx_dpa_bk, inty_dpa_bk)
753  endif
754  else
755  call int_density_dz(tv_tmp%T(:,:,k), tv_tmp%S(:,:,k), &
756  e(:,:,k), e(:,:,k+1), &
757  rho_ref, cs%Rho0, g%g_Earth, g%HI, g%Block(n), tv%eqn_of_state, &
758  dpa_bk, intz_dpa_bk, intx_dpa_bk, inty_dpa_bk )
759  endif
760  intz_dpa_bk(:,:) = intz_dpa_bk(:,:)*gv%m_to_H
761  else
762  do jb=jsq_bk,jeq_bk+1 ; do ib=isq_bk,ieq_bk+1
763  i = ib+ioff_bk ; j = jb+joff_bk
764  dz_bk(ib,jb) = g%g_Earth*gv%H_to_m*h(i,j,k)
765  dpa_bk(ib,jb) = (gv%Rlay(k) - rho_ref)*dz_bk(ib,jb)
766  intz_dpa_bk(ib,jb) = 0.5*(gv%Rlay(k) - rho_ref)*dz_bk(ib,jb)*h(i,j,k)
767  enddo ; enddo
768  do jb=js_bk,je_bk ; do ib=isq_bk,ieq_bk
769  intx_dpa_bk(ib,jb) = 0.5*(gv%Rlay(k) - rho_ref) * (dz_bk(ib,jb)+dz_bk(ib+1,jb))
770  enddo ; enddo
771  do jb=jsq_bk,jeq_bk ; do ib=is_bk,ie_bk
772  inty_dpa_bk(ib,jb) = 0.5*(gv%Rlay(k) - rho_ref) * (dz_bk(ib,jb)+dz_bk(ib,jb+1))
773  enddo ; enddo
774  endif
775 
776  ! Compute pressure gradient in x direction
777  do jb=js_bk,je_bk ; do ib=isq_bk,ieq_bk
778  i = ib+ioff_bk ; j = jb+joff_bk
779  pfu(i,j,k) = (((pa_bk(ib,jb)*h(i,j,k) + intz_dpa_bk(ib,jb)) - &
780  (pa_bk(ib+1,jb)*h(i+1,j,k) + intz_dpa_bk(ib+1,jb))) + &
781  ((h(i+1,j,k) - h(i,j,k)) * intx_pa_bk(ib,jb) - &
782  (e(i+1,j,k+1) - e(i,j,k+1)) * intx_dpa_bk(ib,jb) * gv%m_to_H)) * &
783  ((2.0*i_rho0*g%IdxCu(i,j)) / &
784  ((h(i,j,k) + h(i+1,j,k)) + h_neglect))
785  intx_pa_bk(ib,jb) = intx_pa_bk(ib,jb) + intx_dpa_bk(ib,jb)
786  enddo ; enddo
787  ! Compute pressure gradient in y direction
