44 !> This subroutine sets up the DOME topography
45 subroutine dome_initialize_topography(D, G, param_file, max_depth)
46  type(ocean_grid_type), intent(in) :: G !< The ocean's grid structure.
47  real, intent(out), dimension(SZI_(G),SZJ_(G)) :: D !< The bottom depth in m.
48  type(param_file_type), intent(in) :: param_file !< A structure indicating the open file
49  !! to parse for model parameter values.
50  real, intent(in) :: max_depth !< Maximum depth.
51 
52  real :: min_depth ! The minimum and maximum depths in m.
53 ! This include declares and sets the variable "version".
54 #include "version_variable.h"
55  character(len=40) :: mod = "DOME_initialize_topography" ! This subroutine's name.
56  integer :: i, j, is, ie, js, je, isd, ied, jsd, jed
57  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
58  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
59 
60  call mom_mesg(" DOME_initialization.F90, DOME_initialize_topography: setting topography", 5)
61 
62  call log_version(param_file, mod, version, "")
63  call get_param(param_file, mod, "MINIMUM_DEPTH", min_depth, &
64  "The minimum depth of the ocean.", units="m", default=0.0)
65 
66  do j=js,je ; do i=is,ie
67  if (g%geoLatT(i,j) < 600.0) then
68  if (g%geoLatT(i,j) < 300.0) then
69  d(i,j)=max_depth
70  else
71  d(i,j)=max_depth-10.0*(g%geoLatT(i,j)-300.0)
72  endif
73  else
74  if ((g%geoLonT(i,j) > 1000.0).AND.(g%geoLonT(i,j) < 1100.0)) then
75  d(i,j)=600.0
76  else
77  d(i,j)=0.5*min_depth
78  endif
79  endif
80 
81  if (d(i,j) > max_depth) d(i,j) = max_depth
82  if (d(i,j) < min_depth) d(i,j) = 0.5*min_depth
83  enddo ; enddo
84 
85 end subroutine dome_initialize_topography

89 !> This subroutine initializes layer thicknesses for the DOME experiment
90 subroutine dome_initialize_thickness(h, G, GV, param_file)
91  type(ocean_grid_type), intent(in) :: G !< The ocean's grid structure.
92  type(verticalgrid_type), intent(in) :: GV !< The ocean's vertical grid structure.
93  real, intent(out), dimension(SZI_(G),SZJ_(G), SZK_(G)) :: h !< The thickness that is being initialized.
94  type(param_file_type), intent(in) :: param_file !< A structure indicating the open file
95  !! to parse for model parameter values.
96 
97  real :: e0(szk_(g)+1) ! The resting interface heights, in m, usually !
98  ! negative because it is positive upward. !
99  real :: eta1D(szk_(g)+1) ! Interface height relative to the sea surface !
100  ! positive upward, in m. !
101  character(len=40) :: mod = "DOME_initialize_thickness" ! This subroutine's name.
102  integer :: i, j, k, is, ie, js, je, nz
103 
104  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
105 
106  call mom_mesg(" DOME_initialization.F90, DOME_initialize_thickness: setting thickness", 5)
107 
108  e0(1)=0.0
109  do k=2,nz
110  e0(k) = -g%max_depth * (real(k-1)-0.5)/real(nz-1)
111  enddo
112 
113  do j=g%jsc,g%jec ; do i=g%isc,g%iec
114 ! This sets the initial thickness (in m) of the layers. The !
115 ! thicknesses are set to insure that: 1. each layer is at least an !
116 ! Angstrom thick, and 2. the interfaces are where they should be !
117 ! based on the resting depths and interface height perturbations, !
118 ! as long at this doesn't interfere with 1. !
119  eta1d(nz+1) = -1.0*g%bathyT(i,j)
120  do k=nz,1,-1
121  eta1d(k) = e0(k)
122  if (eta1d(k) < (eta1d(k+1) + gv%Angstrom_z)) then
123  eta1d(k) = eta1d(k+1) + gv%Angstrom_z
124  h(i,j,k) = gv%Angstrom_z
125  else
126  h(i,j,k) = eta1d(k) - eta1d(k+1)
127  endif
128  enddo
129  enddo ; enddo
130 
131 end subroutine dome_initialize_thickness

134 ! -----------------------------------------------------------------------------
135 !> This subroutine sets the inverse restoration time (Idamp), and !
136 !! the values towards which the interface heights and an arbitrary !
137 !! number of tracers should be restored within each sponge. The !
138 !! interface height is always subject to damping, and must always be !
139 !! the first registered field. !
140 subroutine dome_initialize_sponges(G, GV, tv, PF, CSp)
141  type(ocean_grid_type), intent(in) :: G !< The ocean's grid structure.
142  type(verticalgrid_type), intent(in) :: GV !< The ocean's vertical grid structure.
143  type(thermo_var_ptrs), intent(in) :: tv !< A structure containing pointers to any available
144  !! thermodynamic fields, including potential temperature and
145  !! salinity or mixed layer density. Absent fields have NULL ptrs.
146  type(param_file_type), intent(in) :: PF !< A structure indicating the open file to
147  !! parse for model parameter values.
148  type(sponge_cs), pointer :: CSp !< A pointer that is set to point to the control
149  !! structure for this module.
150 
151  real :: eta(szi_(g),szj_(g),szk_(g)+1) ! A temporary array for eta.
152  real :: temp(szi_(g),szj_(g),szk_(g)) ! A temporary array for other variables. !
153  real :: Idamp(szi_(g),szj_(g)) ! The inverse damping rate, in s-1.
154 
155  real :: H0(szk_(g))
156  real :: min_depth
157  real :: damp, e_dense, damp_new
158  character(len=40) :: mod = "DOME_initialize_sponges" ! This subroutine's name.
159  integer :: i, j, k, is, ie, js, je, isd, ied, jsd, jed, nz
160 
161  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
162  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
163 
164  eta(:,:,:) = 0.0 ; temp(:,:,:) = 0.0 ; idamp(:,:) = 0.0
165 
166 ! Here the inverse damping time, in s-1, is set. Set Idamp to 0 !
167 ! wherever there is no sponge, and the subroutines that are called !
168 ! will automatically set up the sponges only where Idamp is positive!
169 ! and mask2dT is 1. !
170 
171 ! Set up sponges for DOME configuration
172  call get_param(pf, mod, "MINIMUM_DEPTH", min_depth, &
173  "The minimum depth of the ocean.", units="m", default=0.0)
174 
175  h0(1) = 0.0
176  do k=2,nz ; h0(k) = -(real(k-1)-0.5)*g%max_depth/real(nz-1) ; enddo
177  do i=is,ie; do j=js,je
178  if (g%geoLonT(i,j) < 100.0) then ; damp = 10.0
179  elseif (g%geoLonT(i,j) < 200.0) then
180  damp = 10.0*(200.0-g%geoLonT(i,j))/100.0
181  else ; damp=0.0
182  endif
183 
184  if (g%geoLonT(i,j) > 1400.0) then ; damp_new = 10.0
185  elseif (g%geoLonT(i,j) > 1300.0) then
186  damp_new = 10.0*(g%geoLonT(i,j)-1300.0)/100.0
187  else ; damp_new = 0.0
188  endif
189 
190  if (damp <= damp_new) damp=damp_new
191 
192  ! These will be stretched inside of apply_sponge, so they can be in
193  ! depth space for Boussinesq or non-Boussinesq models.
194  eta(i,j,1) = 0.0
195  do k=2,nz
196 ! eta(i,j,K)=max(H0(k), -G%bathyT(i,j), GV%Angstrom_z*(nz-k+1)-G%bathyT(i,j))
197  e_dense = -g%bathyT(i,j)
198  if (e_dense >= h0(k)) then ; eta(i,j,k) = e_dense
199  else ; eta(i,j,k) = h0(k) ; endif
200  if (eta(i,j,k) < gv%Angstrom_z*(nz-k+1)-g%bathyT(i,j)) &
201  eta(i,j,k) = gv%Angstrom_z*(nz-k+1)-g%bathyT(i,j)
202  enddo
203  eta(i,j,nz+1) = -g%bathyT(i,j)
204 
205  if (g%bathyT(i,j) > min_depth) then
206  idamp(i,j) = damp/86400.0
207  else ; idamp(i,j) = 0.0 ; endif
208  enddo ; enddo
209 
210 ! This call sets up the damping rates and interface heights.
211 ! This sets the inverse damping timescale fields in the sponges. !
212  call initialize_sponge(idamp, eta, g, pf, csp)
213 
214 ! Now register all of the fields which are damped in the sponge. !
215 ! By default, momentum is advected vertically within the sponge, but !
216 ! momentum is typically not damped within the sponge. !
217 
218 ! At this point, the DOME configuration is done. The following are here as a
219 ! template for other configurations.
220 
221 ! The remaining calls to set_up_sponge_field can be in any order. !
222  if ( associated(tv%T) ) then
223  call mom_error(fatal,"DOME_initialize_sponges is not set up for use with"//&
224  " a temperatures defined.")
225  ! This should use the target values of T in temp.
226  call set_up_sponge_field(temp, tv%T, g, nz, csp)
227  ! This should use the target values of S in temp.
228  call set_up_sponge_field(temp, tv%S, g, nz, csp)
229  endif
230 
231 end subroutine dome_initialize_sponges

235 !> This subroutine sets the properties of flow at open boundary conditions.
236 !! This particular example is for the DOME inflow describe in Legg et al. 2006.
237 subroutine dome_set_open_bdry_conds(OBC, tv, G, GV, param_file, tr_Reg)
238  type(ocean_obc_type), pointer :: OBC !< This open boundary condition type specifies
239  !! whether, where, and what open boundary
240  !! conditions are used.
241  type(thermo_var_ptrs), intent(in) :: tv !< A structure containing pointers to any
242  !! available thermodynamic fields, including potential
243  !! temperature and salinity or mixed layer density. Absent
244  !! fields have NULL ptrs.
245  type(ocean_grid_type), intent(in) :: G !< The ocean's grid structure.
246  type(verticalgrid_type), intent(in) :: GV !< The ocean's vertical grid structure.
247  type(param_file_type), intent(in) :: param_file !< A structure indicating the open file
248  !! to parse for model parameter values.
249  type(tracer_registry_type), pointer :: tr_Reg !< Tracer registry.
250 
251  logical :: any_OBC ! Set to true if any points in this subdomain use
252  ! open boundary conditions.
253  logical, pointer, dimension(:,:) :: &
254  OBC_mask_u => null(), & ! These arrays are true at zonal or meridional
255  obc_mask_v => null() ! velocity points that have prescribed open boundary
256  ! conditions.
257  real, pointer, dimension(:,:,:) :: &
258  OBC_T_u => null(), & ! These arrays should be allocated and set to
259  obc_t_v => null(), & ! specify the values of T and S that should come
260  obc_s_u => null(), & ! in through u- and v- points through the open
261  obc_s_v => null() ! boundary conditions, in C and psu.
262  logical :: apply_OBC_u, apply_OBC_v
263  ! The following variables are used to set the target temperature and salinity.
264  real :: T0(szk_(g)), S0(szk_(g))
265  real :: pres(szk_(g)) ! An array of the reference pressure in Pa.
266  real :: drho_dT(szk_(g)) ! Derivative of density with temperature in kg m-3 K-1. !
267  real :: drho_dS(szk_(g)) ! Derivative of density with salinity in kg m-3 PSU-1. !
268  real :: rho_guess(szk_(g)) ! Potential density at T0 & S0 in kg m-3.
269  ! The following variables are used to set up the transport in the DOME example.
270  real :: tr_0, y1, y2, tr_k, rst, rsb, rc, v_k, lon_im1
271  real :: D_edge ! The thickness in m of the dense fluid at the
272  ! inner edge of the inflow.
273  real :: g_prime_tot ! The reduced gravity across all layers, m s-2.
274  real :: Def_Rad ! The deformation radius, based on fluid of
275  ! thickness D_edge, in the same units as lat.
276  real :: Ri_trans ! The shear Richardson number in the transition
277  ! region of the specified shear profile.
278  character(len=40) :: mod = "DOME_set_Open_Bdry_Conds" ! This subroutine's name.
279  integer :: i, j, k, itt, is, ie, js, je, isd, ied, jsd, jed, nz
280  integer :: IsdB, IedB, JsdB, JedB
281 
282  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
283  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
284  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
285 
286  ! The following variables should be transformed into runtime parameters.
287  d_edge = 300.0 ! The thickness of dense fluid in the inflow.
288  ri_trans = 1.0/3.0 ! The shear Richardson number in the transition region
289  ! region of the specified shear profile.
290 
291  call get_param(param_file, mod, "APPLY_OBC_U", apply_obc_u, &
292  "If true, open boundary conditions may be set at some \n"//&
293  "u-points, with the configuration controlled by OBC_CONFIG", &
294  default=.false.)
295  call get_param(param_file, mod, "APPLY_OBC_V", apply_obc_v, &
296  "If true, open boundary conditions may be set at some \n"//&
297  "v-points, with the configuration controlled by OBC_CONFIG", &
298  default=.false.)
299 
300  if (apply_obc_u) then
301  ! Determine where u points are applied.
302  allocate(obc_mask_u(isdb:iedb,jsd:jed)) ; obc_mask_u(:,:) = .false.
303  any_obc = .false.
304  do j=jsd,jed ; do i=isdb,iedb
305  ! if (SOME_TEST_FOR_U_OPEN_BCS) then
306  ! OBC_mask_u(I,j) = .true. ; any_OBC = .true.
307  ! endif
308  enddo ; enddo
309  if (.not.any_obc) then
310  ! This processor has no u points at which open boundary conditions are
311  ! to be applied.
312  apply_obc_u = .false.
313  deallocate(obc_mask_u)
314  endif
315  endif
316  if (apply_obc_v) then
317  ! Determine where v points are applied.
318  allocate(obc_mask_v(isd:ied,jsdb:jedb)) ; obc_mask_v(:,:) = .false.
319  any_obc = .false.
320  do j=jsdb,jedb ; do i=isd,ied
321  if ((g%geoLonCv(i,j) > 1000.0) .and. (g%geoLonCv(i,j) < 1100.0) .and. &
322  (abs(g%geoLatCv(i,j) - g%gridLatB(g%JegB)) < 0.1)) then
323  obc_mask_v(i,j) = .true. ; any_obc = .true.
324  endif
325  enddo ; enddo
326  if (.not.any_obc) then
327  ! This processor has no v points at which open boundary conditions are
328  ! to be applied.
329  apply_obc_v = .false.
330  deallocate(obc_mask_v)
331  endif
332  endif
333 
334  if (.not.(apply_obc_u .or. apply_obc_v)) return
335 
336  if (.not.associated(obc)) allocate(obc)
337 
338  if (apply_obc_u) then
339  obc%apply_OBC_u = .true.
340  obc%OBC_mask_u => obc_mask_u
341  allocate(obc%u(isdb:iedb,jsd:jed,nz)) ; obc%u(:,:,:) = 0.0
342  allocate(obc%uh(isdb:iedb,jsd:jed,nz)) ; obc%uh(:,:,:) = 0.0
343  allocate(obc%OBC_kind_u(isdb:iedb,jsd:jed)) ; obc%OBC_kind_u(:,:) = obc_none
344  do j=jsd,jed ; do i=isdb,iedb
345  if (obc%OBC_mask_u(i,j)) obc%OBC_kind_u(i,j) = obc_simple
346  enddo ; enddo
347  endif
348  if (apply_obc_v) then
349  obc%apply_OBC_v = .true.
350  obc%OBC_mask_v => obc_mask_v
351  allocate(obc%v(isd:ied,jsdb:jedb,nz)) ; obc%v(:,:,:) = 0.0
352  allocate(obc%vh(isd:ied,jsdb:jedb,nz)) ; obc%vh(:,:,:) = 0.0
353  allocate(obc%OBC_kind_v(isd:ied,jsdb:jedb)) ; obc%OBC_kind_v(:,:) = obc_none
354  do j=jsdb,jedb ; do i=isd,ied
355  if (obc%OBC_mask_v(i,j)) obc%OBC_kind_v(i,j) = obc_simple
356  enddo ; enddo
357  endif
358 
359  if (apply_obc_v) then
360  g_prime_tot = (g%g_Earth/gv%Rho0)*2.0
361  def_rad = sqrt(d_edge*g_prime_tot) / (1.0e-4*1000.0)
362  tr_0 = (-d_edge*sqrt(d_edge*g_prime_tot)*0.5e3*def_rad) * gv%m_to_H
363 
364  do k=1,nz
365  rst = -1.0
366  if (k>1) rst = -1.0 + (real(k-1)-0.5)/real(nz-1)
367 
368  rsb = 0.0
369  if (k<nz) rsb = -1.0 + (real(k-1)+0.5)/real(nz-1)
370  rc = -1.0 + real(k-1)/real(nz-1)
371 
372  ! These come from assuming geostrophy and a constant Ri profile.
373  y1 = (2.0*ri_trans*rst + ri_trans + 2.0)/(2.0 - ri_trans)
374  y2 = (2.0*ri_trans*rsb + ri_trans + 2.0)/(2.0 - ri_trans)
375  tr_k = tr_0 * (2.0/(ri_trans*(2.0-ri_trans))) * &
376  ((log(y1)+1.0)/y1 - (log(y2)+1.0)/y2)
377  v_k = -sqrt(d_edge*g_prime_tot)*log((2.0 + ri_trans*(1.0 + 2.0*rc)) / &
378  (2.0 - ri_trans))
379  if (k == nz) tr_k = tr_k + tr_0 * (2.0/(ri_trans*(2.0+ri_trans))) * &
380  log((2.0+ri_trans)/(2.0-ri_trans))
381  do j=jsdb,jedb ; do i=isd,ied
382  if (obc_mask_v(i,j)) then
383  ! This needs to be unneccesarily complicated without symmetric memory.
384  lon_im1 = 2.0*g%geoLonCv(i,j) - g%geoLonBu(i,j)
385  ! if (isd > IsdB) lon_im1 = G%geoLonBu(I-1,J)
386  obc%vh(i,j,k) = tr_k * (exp(-2.0*(lon_im1 - 1000.0)/def_rad) -&
387  exp(-2.0*(g%geoLonBu(i,j) - 1000.0)/def_rad))
388  obc%v(i,j,k) = v_k * exp(-2.0*(g%geoLonCv(i,j) - 1000.0)/def_rad)
389  else
390  obc%vh(i,j,k) = 0.0 ; obc%v(i,j,k) = 0.0
391  endif
392  enddo ; enddo
393  enddo
394  endif
395 
396  if (apply_obc_u) then
397  do k=1,nz ; do j=jsd,jed ; do i=isdb,iedb
398  if (obc_mask_u(i,j)) then
399  ! An appropriate expression for the zonal inflow velocities and
400  ! transports should go here.
401  obc%uh(i,j,k) = 0.0 * gv%m_to_H ; obc%u(i,j,k) = 0.0
402  else
403  obc%uh(i,j,k) = 0.0 ; obc%u(i,j,k) = 0.0
404  endif
405  enddo ; enddo ; enddo
406  endif
407 
408  ! The inflow values of temperature and salinity also need to be set here if
409  ! these variables are used. The following code is just a naive example.
410  if (apply_obc_u .or. apply_obc_v) then
411  if (associated(tv%S)) then
412  ! In this example, all S inflows have values of 35 psu.
413  call add_tracer_obc_values("S", tr_reg, obc_inflow=35.0)
414  endif
415  if (associated(tv%T)) then
416  ! In this example, the T values are set to be consistent with the layer
417  ! target density and a salinity of 35 psu. This code is taken from
418  ! USER_initialize_temp_sal.
419  pres(:) = tv%P_Ref ; s0(:) = 35.0 ; t0(1) = 25.0
420  call calculate_density(t0(1),s0(1),pres(1),rho_guess(1),tv%eqn_of_state)
421  call calculate_density_derivs(t0,s0,pres,drho_dt,drho_ds,1,1,tv%eqn_of_state)
422 
423  do k=1,nz ; t0(k) = t0(1) + (gv%Rlay(k)-rho_guess(1)) / drho_dt(1) ; enddo
424  do itt=1,6
425  call calculate_density(t0,s0,pres,rho_guess,1,nz,tv%eqn_of_state)
426  call calculate_density_derivs(t0,s0,pres,drho_dt,drho_ds,1,nz,tv%eqn_of_state)
427  do k=1,nz ; t0(k) = t0(k) + (gv%Rlay(k)-rho_guess(k)) / drho_dt(k) ; enddo
428  enddo
429 
430  if (apply_obc_u) then
431  allocate(obc_t_u(isdb:iedb,jsd:jed,nz))
432  do k=1,nz ; do j=jsd,jed ; do i=isdb,iedb
433  obc_t_u(i,j,k) = t0(k)
434  enddo ; enddo ; enddo
435  endif
436  if (apply_obc_v) then
437  allocate(obc_t_v(isd:ied,jsdb:jedb,nz))
438  do k=1,nz ; do j=jsdb,jedb ; do i=isd,ied
439  obc_t_v(i,j,k) = t0(k)
