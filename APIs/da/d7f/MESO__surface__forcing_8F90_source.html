26 !* This file contains the subroutines that a user should modify to *
27 !* to set the surface wind stresses and fluxes of buoyancy or *
28 !* temperature and fresh water. They are called when the run-time *
29 !* parameters WIND_CONFIG or BUOY_CONFIG are set to "USER". The *
30 !* standard version has simple examples, along with run-time error *
31 !* messages that will cause the model to abort if this code has not *
32 !* been modified. This code is intended for use with relatively *
33 !* simple specifications of the forcing. For more complicated forms, *
34 !* it is probably a good idea to read the forcing from input files *
35 !* using "file" for WIND_CONFIG and BUOY_CONFIG. *
36 !* *
37 !* MESO_wind_forcing should set the surface wind stresses (taux and *
38 !* tauy) perhaps along with the surface friction velocity (ustar). *
39 !* *
40 !* MESO_buoyancy forcing is used to set the surface buoyancy *
41 !* forcing, which may include a number of fresh water flux fields *
42 !* (evap, liq_precip, froz_precip, liq_runoff, froz_runoff, and *
43 !* vprec) and the surface heat fluxes (sw, lw, latent and sens) *
44 !* if temperature and salinity are state variables, or it may simply *
45 !* be the buoyancy flux if it is not. This routine also has coded a *
46 !* restoring to surface values of temperature and salinity. *
47 !* *
48 !* Macros written all in capital letters are defined in MOM_memory.h. *
49 !* *
50 !* A small fragment of the grid is shown below: *
51 !* *
52 !* j+1 x ^ x ^ x At x: q *
53 !* j+1 > o > o > At ^: v, tauy *
54 !* j x ^ x ^ x At >: u, taux *
55 !* j > o > o > At o: h, fluxes. *
56 !* j-1 x ^ x ^ x *
57 !* i-1 i i+1 At x & ^: *
58 !* i i+1 At > & o: *
59 !* *
60 !* The boundaries always run through q grid points (x). *
61 !* *

80 type, public :: meso_surface_forcing_cs ; private
81  ! This control structure should be used to store any run-time variables
82  ! associated with the user-specified forcing. It can be readily modified
83  ! for a specific case, and because it is private there will be no changes
84  ! needed in other code (although they will have to be recompiled).
85  ! The variables in the cannonical example are used for some common
86  ! cases, but do not need to be used.

92  ! approximation, in kg m-3.
93  real :: g_earth ! The gravitational acceleration in m s-2.
94  real :: flux_const ! The restoring rate at the surface, in m s-1.
95  real :: gust_const ! A constant unresolved background gustiness
96  ! that contributes to ustar, in Pa.

104  ! heat flux into the ocean, in W m-2.

106  character(len=200) :: salinityrestore_file, sstrestore_file

116 subroutine meso_wind_forcing(state, fluxes, day, G, CS)
117  type(surface), intent(inout) :: state
118  type(forcing), intent(inout) :: fluxes
119  type(time_type), intent(in) :: day
120  type(ocean_grid_type), intent(inout) :: G
121  type(meso_surface_forcing_cs), pointer :: CS
122 
123 ! This subroutine sets the surface wind stresses, fluxes%taux and fluxes%tauy.
124 ! These are the stresses in the direction of the model grid (i.e. the same
125 ! direction as the u- and v- velocities.) They are both in Pa.
126 ! In addition, this subroutine can be used to set the surface friction
127 ! velocity, fluxes%ustar, in m s-1. This is needed with a bulk mixed layer.
128 !
129 ! Arguments: state - A structure containing fields that describe the
130 ! surface state of the ocean.
131 ! (out) fluxes - A structure containing pointers to any possible
132 ! forcing fields. Unused fields have NULL ptrs.
133 ! (in) day - Time of the fluxes.
134 ! (in) G - The ocean's grid structure.
135 ! (in) CS - A pointer to the control structure returned by a previous
136 ! call to MESO_surface_forcing_init
137 
138  integer :: i, j, is, ie, js, je, Isq, Ieq, Jsq, Jeq
139  integer :: isd, ied, jsd, jed, IsdB, IedB, JsdB, JedB
140 
141  ! When modifying the code, comment out this error message. It is here
142  ! so that the original (unmodified) version is not accidentally used.
143  call mom_error(fatal, "MESO_wind_surface_forcing: " // &
144  "User forcing routine called without modification." )
145 
146  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
147  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
148  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
149  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
150 
151  ! Allocate the forcing arrays, if necessary.
152  call allocate_forcing_type(g, fluxes, stress=.true., ustar=.true.)
153 
154  ! Set the surface wind stresses, in units of Pa. A positive taux
155  ! accelerates the ocean to the (pseudo-)east.
156 
157  ! The i-loop extends to is-1 so that taux can be used later in the
158  ! calculation of ustar - otherwise the lower bound would be Isq.
159  do j=js,je ; do i=is-1,ieq
160  fluxes%taux(i,j) = g%mask2dCu(i,j) * 0.0 ! Change this to the desired expression.
161  enddo ; enddo
162  do j=js-1,jeq ; do i=is,ie
163  fluxes%tauy(i,j) = g%mask2dCv(i,j) * 0.0 ! Change this to the desired expression.
164  enddo ; enddo
165 
166  ! Set the surface friction velocity, in units of m s-1. ustar
167  ! is always positive.
168  if (associated(fluxes%ustar)) then ; do j=js,je ; do i=is,ie
169  ! This expression can be changed if desired, but need not be.
170  fluxes%ustar(i,j) = g%mask2dT(i,j) * sqrt(cs%gust_const/cs%Rho0 + &
171  sqrt(0.5*(fluxes%taux(i-1,j)**2 + fluxes%taux(i,j)**2) + &
172  0.5*(fluxes%tauy(i,j-1)**2 + fluxes%tauy(i,j)**2))/cs%Rho0)
173  enddo ; enddo ; endif
174 
175 end subroutine meso_wind_forcing

177 subroutine meso_buoyancy_forcing(state, fluxes, day, dt, G, CS)
178  type(surface), intent(inout) :: state
179  type(forcing), intent(inout) :: fluxes
180  type(time_type), intent(in) :: day
181  real, intent(in) :: dt
182  type(ocean_grid_type), intent(in) :: G
183  type(meso_surface_forcing_cs), pointer :: CS
184 
185 ! This subroutine specifies the current surface fluxes of buoyancy or
186 ! temperature and fresh water. It may also be modified to add
187 ! surface fluxes of user provided tracers.
188 
189 ! When temperature is used, there are long list of fluxes that need to be
190 ! set - essentially the same as for a full coupled model, but most of these
191 ! can be simply set to zero. The net fresh water flux should probably be
192 ! set in fluxes%evap and fluxes%lprec, with any salinity restoring
193 ! appearing in fluxes%vprec, and the other water flux components
194 ! (fprec, lrunoff and frunoff) left as arrays full of zeros.
195 ! Evap is usually negative and precip is usually positive. All heat fluxes
196 ! are in W m-2 and positive for heat going into the ocean. All fresh water
197 ! fluxes are in kg m-2 s-1 and positive for water moving into the ocean.
198 
199 ! Arguments: state - A structure containing fields that describe the
200 ! surface state of the ocean.
201 ! (out) fluxes - A structure containing pointers to any possible
202 ! forcing fields. Unused fields have NULL ptrs.
203 ! (in) day_start - Start time of the fluxes.
204 ! (in) day_interval - Length of time over which these fluxes
205 ! will be applied.
206 ! (in) G - The ocean's grid structure.
207 ! (in) CS - A pointer to the control structure returned by a previous
208 ! call to MESO_surface_forcing_init
209 
210  real :: Temp_restore ! The temperature that is being restored toward, in C.
211  real :: Salin_restore ! The salinity that is being restored toward, in PSU.
212  real :: density_restore ! The potential density that is being restored
213  ! toward, in kg m-3.
214  real :: rhoXcp ! The mean density times the heat capacity, in J m-3 K-1.
215  real :: buoy_rest_const ! A constant relating density anomalies to the
216  ! restoring buoyancy flux, in m5 s-3 kg-1.
217 
218  integer :: i, j, is, ie, js, je
219  integer :: isd, ied, jsd, jed
220 
221  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
222  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
223 
224  ! When modifying the code, comment out this error message. It is here
225  ! so that the original (unmodified) version is not accidentally used.
226 
227  ! Allocate and zero out the forcing arrays, as necessary. This portion is
228  ! usually not changed.
229  if (cs%use_temperature) then
230  call alloc_if_needed(fluxes%evap, isd, ied, jsd, jed)
231  call alloc_if_needed(fluxes%lprec, isd, ied, jsd, jed)
232  call alloc_if_needed(fluxes%fprec, isd, ied, jsd, jed)
233  call alloc_if_needed(fluxes%lrunoff, isd, ied, jsd, jed)
234  call alloc_if_needed(fluxes%frunoff, isd, ied, jsd, jed)
235  call alloc_if_needed(fluxes%vprec, isd, ied, jsd, jed)
236 
237  call alloc_if_needed(fluxes%sw, isd, ied, jsd, jed)
238  call alloc_if_needed(fluxes%lw, isd, ied, jsd, jed)
239  call alloc_if_needed(fluxes%latent, isd, ied, jsd, jed)
240  call alloc_if_needed(fluxes%sens, isd, ied, jsd, jed)
241  call alloc_if_needed(fluxes%heat_content_lprec, isd, ied, jsd, jed)
242  else ! This is the buoyancy only mode.
243  call alloc_if_needed(fluxes%buoy, isd, ied, jsd, jed)
244  endif
245 
246 
247  ! MODIFY THE CODE IN THE FOLLOWING LOOPS TO SET THE BUOYANCY FORCING TERMS.
248  if (cs%restorebuoy .and. first_call) then !### .or. associated(CS%ctrl_forcing_CSp)) then
249  call alloc_if_needed(cs%T_Restore, isd, ied, jsd, jed)
250  call alloc_if_needed(cs%S_Restore, isd, ied, jsd, jed)
251  call alloc_if_needed(cs%Heat, isd, ied, jsd, jed)
252  call alloc_if_needed(cs%PmE, isd, ied, jsd, jed)
253  call alloc_if_needed(cs%Solar, isd, ied, jsd, jed)
254 
255  call read_data(trim(cs%inputdir)//trim(cs%SSTrestore_file), "SST", &
256  cs%T_Restore(:,:), domain=g%Domain%mpp_domain)
257  call read_data(trim(cs%inputdir)//trim(cs%salinityrestore_file), "SAL", &
258  cs%S_Restore(:,:), domain=g%Domain%mpp_domain)
259  call read_data(trim(cs%inputdir)//trim(cs%heating_file), "Heat", &
260  cs%Heat(:,:), domain=g%Domain%mpp_domain)
261  call read_data(trim(cs%inputdir)//trim(cs%PmE_file), "PmE", &
262  cs%PmE(:,:), domain=g%Domain%mpp_domain)
263  call read_data(trim(cs%inputdir)//trim(cs%Solar_file), "NET_SOL", &
264  cs%Solar(:,:), domain=g%Domain%mpp_domain)
265  first_call = .false.
266  endif
267 
268  if ( cs%use_temperature ) then
269  ! Set whichever fluxes are to be used here. Any fluxes that
270  ! are always zero do not need to be changed here.
271  do j=js,je ; do i=is,ie
272  ! Fluxes of fresh water through the surface are in units of kg m-2 s-1
273  ! and are positive downward - i.e. evaporation should be negative.
274  fluxes%evap(i,j) = -0.0 * g%mask2dT(i,j)
275  fluxes%lprec(i,j) = cs%PmE(i,j) * cs%Rho0 * g%mask2dT(i,j)
276 
277  ! vprec will be set later, if it is needed for salinity restoring.
278  fluxes%vprec(i,j) = 0.0
279 
280  ! Heat fluxes are in units of W m-2 and are positive into the ocean.
281  fluxes%lw(i,j) = 0.0 * g%mask2dT(i,j)
282  fluxes%latent(i,j) = 0.0 * g%mask2dT(i,j)
283  fluxes%sens(i,j) = cs%Heat(i,j) * g%mask2dT(i,j)
284  fluxes%sw(i,j) = cs%Solar(i,j) * g%mask2dT(i,j)
285  enddo ; enddo
286  else ! This is the buoyancy only mode.
287  do j=js,je ; do i=is,ie
288  ! fluxes%buoy is the buoyancy flux into the ocean in m2 s-3. A positive
289  ! buoyancy flux is of the same sign as heating the ocean.
290  fluxes%buoy(i,j) = 0.0 * g%mask2dT(i,j)
291  enddo ; enddo
292  endif
293 
294  if (cs%restorebuoy) then
295  if (cs%use_temperature) then
296  call alloc_if_needed(fluxes%heat_restore, isd, ied, jsd, jed)
297  ! When modifying the code, comment out this error message. It is here
298  ! so that the original (unmodified) version is not accidentally used.
299 ! call MOM_error(FATAL, "MESO_buoyancy_surface_forcing: " // &
300 ! "Temperature and salinity restoring used without modification." )
301 
302  rhoxcp = cs%Rho0 * fluxes%C_p
303  do j=js,je ; do i=is,ie
304  ! Set Temp_restore and Salin_restore to the temperature (in C) and
305  ! salinity (in PSU) that are being restored toward.
306  if (g%mask2dT(i,j) > 0) then
307  fluxes%heat_restore(i,j) = g%mask2dT(i,j) * &
308  ((cs%T_Restore(i,j) - state%SST(i,j)) * rhoxcp * cs%Flux_const)
309  fluxes%vprec(i,j) = - (cs%Rho0*cs%Flux_const) * &
310  (cs%S_Restore(i,j) - state%SSS(i,j)) / &
311  (0.5*(state%SSS(i,j) + cs%S_Restore(i,j)))
312  else
313  fluxes%heat_restore(i,j) = 0.0
314  fluxes%vprec(i,j) = 0.0
315  endif
316  enddo ; enddo
317  else
318  ! When modifying the code, comment out this error message. It is here
319  ! so that the original (unmodified) version is not accidentally used.
320  call mom_error(fatal, "MESO_buoyancy_surface_forcing: " // &
321  "Buoyancy restoring used without modification." )
322 
323  ! The -1 is because density has the opposite sign to buoyancy.
324  buoy_rest_const = -1.0 * (cs%G_Earth * cs%Flux_const) / cs%Rho0
325  do j=js,je ; do i=is,ie
326  ! Set density_restore to an expression for the surface potential
327  ! density in kg m-3 that is being restored toward.
328  density_restore = 1030.0
329 
330  fluxes%buoy(i,j) = g%mask2dT(i,j) * buoy_rest_const * &
331  (density_restore - state%sfc_density(i,j))
332  enddo ; enddo
333  endif
334  endif ! end RESTOREBUOY
335 
336 end subroutine meso_buoyancy_forcing

338 subroutine alloc_if_needed(ptr, isd, ied, jsd, jed)
339  ! If ptr is not associated, this routine allocates it with the given size
340  ! and zeros out its contents. This is equivalent to safe_alloc_ptr in
341  ! MOM_diag_mediator, but is here so as to be completely transparent.
342  real, pointer :: ptr(:,:)
343  integer :: isd, ied, jsd, jed
344  if (.not.ASSOCIATED(ptr)) then
345  allocate(ptr(isd:ied,jsd:jed))
346  ptr(:,:) = 0.0
347  endif
348 end subroutine alloc_if_needed

350 subroutine meso_surface_forcing_init(Time, G, param_file, diag, CS)

355  type(meso_surface_forcing_cs), pointer :: CS
356 ! Arguments: Time - The current model time.
357 ! (in) G - The ocean's grid structure.
358 ! (in) param_file - A structure indicating the open file to parse for

403  "to the relative surface anomalies (akin to a piston \n"//&

406  ! Convert CS%Flux_const from m day-1 to m s-1.

410  "The file with the SST toward which to restore in \n"//&

413  "The file with the surface salinity toward which to \n"//&
414  "restore in variable SALT.", fail_if_missing=.true.)
415  call get_param(param_file, mod, "SENSIBLEHEAT_FILE", cs%heating_file, &
416  "The file with the non-shortwave heat flux in \n"//&
417  "variable Heat.", fail_if_missing=.true.)
418  call get_param(param_file, mod, "PRECIP_FILE", cs%PmE_file, &
419  "The file with the net precipiation minus evaporation \n"//&
420  "in variable PmE.", fail_if_missing=.true.)
421  call get_param(param_file, mod, "SHORTWAVE_FILE", cs%Solar_file, &
422  "The file with the shortwave heat flux in \n"//&
423  "variable NET_SOL.", fail_if_missing=.true.)
424  call get_param(param_file, mod, "INPUTDIR", cs%inputdir, default=".")
425  cs%inputdir = slasher(cs%inputdir)
426 
427  endif
