57 !> Initialization of topography.
58 subroutine seamount_initialize_topography ( D, G, param_file, max_depth )
59  ! Arguments
60  type(ocean_grid_type), intent(in) :: G !< The ocean's grid structure.
61  real, intent(out), dimension(SZI_(G),SZJ_(G)) :: D !< The bottom depth in m.
62  type(param_file_type), intent(in) :: param_file !< A structure indicating the
63  !! open file to parse for model
64  !! parameter values.
65  real, intent(in) :: max_depth !< Maximum depth.
66 
67  ! Local variables
68  integer :: i, j
69  real :: x, y, delta, Lx, rLx, Ly, rLy
70 
71  call get_param(param_file,mod,"SEAMOUNT_DELTA",delta, &
72  "Non-dimensional height of seamount.", &
73  units="non-dim", default=0.5)
74  call get_param(param_file,mod,"SEAMOUNT_X_LENGTH_SCALE",lx, &
75  "Length scale of seamount in x-direction.\n"//&
76  "Set to zero make topography uniform in the x-direction.", &
77  units="Same as x,y", default=20.)
78  call get_param(param_file,mod,"SEAMOUNT_Y_LENGTH_SCALE",ly, &
79  "Length scale of seamount in y-direction.\n"//&
80  "Set to zero make topography uniform in the y-direction.", &
81  units="Same as x,y", default=0.)
82 
83  lx = lx / g%len_lon
84  ly = ly / g%len_lat
85  rlx = 0. ; if (lx>0.) rlx = 1. / lx
86  rly = 0. ; if (ly>0.) rly = 1. / ly
87  do i=g%isc,g%iec
88  do j=g%jsc,g%jec
89  ! Compute normalized zonal coordinates (x,y=0 at center of domain)
90  x = ( g%geoLonT(i,j) - g%west_lon ) / g%len_lon - 0.5
91  y = ( g%geoLatT(i,j) - g%south_lat ) / g%len_lat - 0.5
92  d(i,j) = g%max_depth * ( 1.0 - delta * exp(-(rlx*x)**2 -(rly*y)**2) )
93  enddo
94  enddo
95 
96 end subroutine seamount_initialize_topography

98 !> Initialization of thicknesses.
99 !! This subroutine initializes the layer thicknesses to be uniform.
100 subroutine seamount_initialize_thickness ( h, G, GV, param_file )
101  type(ocean_grid_type), intent(in) :: G !< The ocean's grid structure.
102  real, intent(out), dimension(SZI_(G),SZJ_(G), SZK_(G)) :: h !< The thicknesses being
103  !! initialized.
104  type(verticalgrid_type), intent(in) :: GV !< The ocean's vertical grid structure.
105  type(param_file_type), intent(in) :: param_file !< A structure indicating the
106  !! open file to parse for model
107  !! parameter values.
108 
109  real :: e0(szk_(g)+1) ! The resting interface heights, in m, usually !
110  ! negative because it is positive upward. !
111  real :: eta1D(szk_(g)+1)! Interface height relative to the sea surface !
112  ! positive upward, in m. !
113  integer :: i, j, k, is, ie, js, je, nz
114  real :: x
115  real :: delta_h
116  real :: min_thickness, S_surf, S_range, S_ref, S_light, S_dense
117  character(len=20) :: verticalCoordinate
118 
119  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
120 
121  call mom_mesg("MOM_initialization.F90, initialize_thickness_uniform: setting thickness")
122 
123  call get_param(param_file,mod,"MIN_THICKNESS",min_thickness,'Minimum thickness for layer',units='m',default=1.0e-3)
124  call get_param(param_file,mod,"REGRIDDING_COORDINATE_MODE",verticalcoordinate, &
125  default=default_coordinate_mode)
126 
127  ! WARNING: this routine specifies the interface heights so that the last layer
128  ! is vanished, even at maximum depth. In order to have a uniform
129  ! layer distribution, use this line of code within the loop:
130  ! e0(k) = -G%max_depth * real(k-1) / real(nz)
131  ! To obtain a thickness distribution where the last layer is
132  ! vanished and the other thicknesses uniformly distributed, use:
133  ! e0(k) = -G%max_depth * real(k-1) / real(nz-1)
134  !do k=1,nz+1
135  ! e0(k) = -G%max_depth * real(k-1) / real(nz)
136  !enddo
137 
138  select case ( coordinatemode(verticalcoordinate) )
139 
140  case ( regridding_layer, regridding_rho ) ! Initial thicknesses for isopycnal coordinates
141  call get_param(param_file,mod,"INITIAL_SSS", s_surf, default=34., do_not_log=.true.)
142  call get_param(param_file,mod,"INITIAL_S_RANGE", s_range, default=2., do_not_log=.true.)
143  call get_param(param_file, mod, "S_REF", s_ref, default=35.0, do_not_log=.true.)
144  call get_param(param_file, mod, "TS_RANGE_S_LIGHT", s_light, default = s_ref, do_not_log=.true.)
145  call get_param(param_file, mod, "TS_RANGE_S_DENSE", s_dense, default = s_ref, do_not_log=.true.)
146  do k=1,nz+1
147  ! Salinity of layer k is S_light + (k-1)/(nz-1) * (S_dense - S_light)
148  ! Salinity of interface K is S_light + (K-3/2)/(nz-1) * (S_dense - S_light)
149  ! Salinity at depth z should be S(z) = S_surf - S_range * z/max_depth
150  ! Equating: S_surf - S_range * z/max_depth = S_light + (K-3/2)/(nz-1) * (S_dense - S_light)
151  ! Equating: - S_range * z/max_depth = S_light - S_surf + (K-3/2)/(nz-1) * (S_dense - S_light)
152  ! Equating: z/max_depth = - ( S_light - S_surf + (K-3/2)/(nz-1) * (S_dense - S_light) ) / S_range
153  e0(k) = - g%max_depth * ( ( s_light - s_surf ) + ( s_dense - s_light ) * ( (real(k)-1.5) / real(nz-1) ) ) / s_range
154  e0(k) = nint(2048.*e0(k))/2048. ! Force round numbers ... the above expression has irrational factors ...
155  e0(k) = min(real(1-k)*GV%Angstrom_z, e0(k)) ! Bound by surface
156  e0(k) = max(-g%max_depth, e0(k)) ! Bound by bottom
157  enddo
158  do j=js,je ; do i=is,ie
159  eta1d(nz+1) = -1.0*g%bathyT(i,j)
160  do k=nz,1,-1
161  eta1d(k) = e0(k)
162  if (eta1d(k) < (eta1d(k+1) + gv%Angstrom_z)) then
163  eta1d(k) = eta1d(k+1) + gv%Angstrom_z
164  h(i,j,k) = gv%Angstrom_z
165  else
166  h(i,j,k) = eta1d(k) - eta1d(k+1)
167  endif
168  enddo
169  enddo ; enddo
170 
171  case ( regridding_zstar ) ! Initial thicknesses for z coordinates
172  do j=js,je ; do i=is,ie
173  eta1d(nz+1) = -1.0*g%bathyT(i,j)
174  do k=nz,1,-1
175  eta1d(k) = -g%max_depth * real(k-1) / real(nz)
176  if (eta1d(k) < (eta1d(k+1) + min_thickness)) then
177  eta1d(k) = eta1d(k+1) + min_thickness
178  h(i,j,k) = min_thickness
179  else
180  h(i,j,k) = eta1d(k) - eta1d(k+1)
181  endif
182  enddo
183  enddo ; enddo
184 
185  case ( regridding_sigma ) ! Initial thicknesses for sigma coordinates
186  do j=js,je ; do i=is,ie
187  delta_h = g%bathyT(i,j) / dfloat(nz)
188  h(i,j,:) = delta_h
189  end do ; end do
190 end select
191 
192 end subroutine seamount_initialize_thickness

194 !> Initial values for temperature and salinity
195 subroutine seamount_initialize_temperature_salinity ( T, S, h, G, GV, param_file, eqn_of_state)
196  type(ocean_grid_type), intent(in) :: G !< Ocean grid structure
197  type(verticalgrid_type), intent(in) :: GV !< Vertical grid structure
198  real, dimension(SZI_(G),SZJ_(G), SZK_(G)), intent(out) :: T !< Potential temperature (degC)
199  real, dimension(SZI_(G),SZJ_(G), SZK_(G)), intent(out) :: S !< Salinity (ppt)
200  real, dimension(SZI_(G),SZJ_(G), SZK_(G)), intent(in) :: h !< Layer thickness (m or Pa)
201  type(param_file_type), intent(in) :: param_file !< Parameter file structure
202  type(eos_type), pointer :: eqn_of_state !< Equation of state structure
203  ! Local variables
204  integer :: i, j, k, is, ie, js, je, nz, k_light
205  real :: xi0, xi1, dxi, r, S_surf, T_surf, S_range, T_range
206  real :: T_ref, T_Light, T_Dense, S_ref, S_Light, S_Dense, a1, frac_dense, k_frac, res_rat
207  character(len=20) :: verticalCoordinate, density_profile
208 
209  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
210 
211  call get_param(param_file, mod, "REGRIDDING_COORDINATE_MODE", verticalcoordinate, &
212  default=default_coordinate_mode)
213  call get_param(param_file,mod,"INITIAL_DENSITY_PROFILE", density_profile, &
214  'Initial profile shape. Valid values are "linear", "parabolic"\n'// &
215  'and "exponential".', default='linear')
216  call get_param(param_file,mod,"INITIAL_SSS", s_surf, &
217  'Initial surface salinity', units='1e-3', default=34.)
218  call get_param(param_file,mod,"INITIAL_SST", t_surf, &
219  'Initial surface temperature', units='C', default=0.)
220  call get_param(param_file,mod,"INITIAL_S_RANGE", s_range, &
221  'Initial salinity range (bottom - surface)', units='1e-3', default=2.)
222  call get_param(param_file,mod,"INITIAL_T_RANGE", t_range, &
223  'Initial temperature range (bottom - surface)', units='C', default=0.)
224 
225  select case ( coordinatemode(verticalcoordinate) )
226  case ( regridding_layer ) ! Initial thicknesses for layer isopycnal coordinates
227  ! These parameters are used in MOM_fixed_initialization.F90 when CONFIG_COORD="ts_range"
228  call get_param(param_file, mod, "T_REF", t_ref, default=10.0, do_not_log=.true.)
229  call get_param(param_file, mod, "TS_RANGE_T_LIGHT", t_light, default=t_ref, do_not_log=.true.)
230  call get_param(param_file, mod, "TS_RANGE_T_DENSE", t_dense, default=t_ref, do_not_log=.true.)
231  call get_param(param_file, mod, "S_REF", s_ref, default=35.0, do_not_log=.true.)
232  call get_param(param_file, mod, "TS_RANGE_S_LIGHT", s_light, default = s_ref, do_not_log=.true.)
233  call get_param(param_file, mod, "TS_RANGE_S_DENSE", s_dense, default = s_ref, do_not_log=.true.)
234  call get_param(param_file, mod, "TS_RANGE_RESOLN_RATIO", res_rat, default=1.0, do_not_log=.true.)
235  ! Emulate the T,S used in the "ts_range" coordinate configuration code
236  k_light = gv%nk_rho_varies + 1
237  do j=js,je ; do i=is,ie
238  t(i,j,k_light) = t_light ; s(i,j,k_light) = s_light
239  enddo ; enddo
240  a1 = 2.0 * res_rat / (1.0 + res_rat)
241  do k=k_light+1,nz
242  k_frac = real(k-k_light)/real(nz-k_light)
243  frac_dense = a1 * k_frac + (1.0 - a1) * k_frac**2
244  do j=js,je ; do i=is,ie
245  t(i,j,k) = frac_dense * (t_dense - t_light) + t_light
246  s(i,j,k) = frac_dense * (s_dense - s_light) + s_light
247  enddo ; enddo
248  enddo
249  case ( regridding_sigma, regridding_zstar, regridding_rho ) ! All other coordinate use FV initialization
250  do j=js,je ; do i=is,ie
251  xi0 = 0.0
252  do k = 1,nz
253  xi1 = xi0 + h(i,j,k) / g%max_depth
254  select case ( trim(density_profile) )
255  case ('linear')
256  !S(i,j,k) = S_surf + S_range * 0.5 * (xi0 + xi1)
257  s(i,j,k) = s_surf + ( 0.5 * s_range ) * (xi0 + xi1) ! Coded this way to reproduce old hard-coded answers
258  t(i,j,k) = t_surf + t_range * 0.5 * (xi0 + xi1)
259  case ('parabolic')
260  s(i,j,k) = s_surf + s_range * (2.0 / 3.0) * (xi1**3 - xi0**3) / (xi1 - xi0)
261  t(i,j,k) = t_surf + t_range * (2.0 / 3.0) * (xi1**3 - xi0**3) / (xi1 - xi0)
262  case ('exponential')
263  s(i,j,k) = s_surf + s_range * (exp(xi1/r)-exp(xi0/r)) / (xi1 - xi0)
264  t(i,j,k) = t_surf + t_range * (exp(xi1/r)-exp(xi0/r)) / (xi1 - xi0)
265  case default
266  call mom_error(fatal, 'Unknown value for "INITIAL_DENSITY_PROFILE"')
267  end select
268  xi0 = xi1
269  enddo
270  enddo ; enddo
271  end select
272 
273 end subroutine seamount_initialize_temperature_salinity
