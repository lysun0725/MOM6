185 subroutine benchmark_init_topog_local(D, G, param_file, max_depth)
186  type(ocean_grid_type), intent(in) :: G
187  real, intent(out), dimension(SZI_(G),SZJ_(G)) :: D
188  type(param_file_type), intent(in) :: param_file
189  real, intent(in) :: max_depth
190 ! Arguments: D - the bottom depth in m. Intent out.
191 ! (in) G - The ocean's grid structure.
192 ! (in) param_file - A structure indicating the open file to parse for
193 ! model parameter values.
194 
195 ! This subroutine sets up the benchmark test case topography
196  real :: min_depth ! The minimum and maximum depths in m.
197  real :: PI ! 3.1415926... calculated as 4*atan(1)
198  real :: D0 ! A constant to make the maximum !
199  ! basin depth MAXIMUM_DEPTH. !
200  real :: x, y
201 ! This include declares and sets the variable "version".
202 #include "version_variable.h"
203  character(len=40) :: mod = "benchmark_initialize_topography" ! This subroutine's name.
204  integer :: i, j, is, ie, js, je, isd, ied, jsd, jed
205  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
206  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
207 
208  call mom_mesg(" benchmark_initialization.F90, benchmark_initialize_topography: setting topography", 5)
209 
210  call log_version(param_file, mod, version)
211  call get_param(param_file, mod, "MINIMUM_DEPTH", min_depth, &
212  "The minimum depth of the ocean.", units="m", default=0.0)
213 
214  pi = 4.0*atan(1.0)
215  d0 = max_depth / 0.5;
216 
217 ! Calculate the depth of the bottom.
218  do i=is,ie ; do j=js,je
219  x=(g%geoLonT(i,j)-g%west_lon)/g%len_lon
220  y=(g%geoLatT(i,j)-g%south_lat)/g%len_lat
221 ! This sets topography that has a reentrant channel to the south.
222  d(i,j) = -d0 * ( y*(1.0 + 0.6*cos(4.0*pi*x)) &
223  + 0.75*exp(-6.0*y) &
224  + 0.05*cos(10.0*pi*x) - 0.7 )
225  if (d(i,j) > max_depth) d(i,j) = max_depth
226  if (d(i,j) < min_depth) d(i,j) = 0.
227  enddo ; enddo
228 
229 end subroutine benchmark_init_topog_local
