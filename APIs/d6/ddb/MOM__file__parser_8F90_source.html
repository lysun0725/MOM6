26 !* The subroutines here parse a set of input files for the value *
27 !* a named parameter and sets that parameter at run time. Currently *
28 !* these files use use one of several formats: *
29 !* #define VAR ! To set the logical VAR to true. *
30 !* VAR = True ! To set the logical VAR to true. *
31 !* #undef VAR ! To set the logical VAR to false. *
32 !* VAR = False ! To set the logical VAR to false. *
33 !* #define VAR 999 ! To set the real or integer VAR to 999. *
34 !* VAR = 999 ! To set the real or integer VAR to 999. *
35 !* #override VAR = 888 ! To override a previously set value. *
36 !* VAR = 1.1, 2.2, 3.3 ! To set an array of real values. *
37 !* *
38 !* In addition, when set by the get_param interface, the values of *
39 !* parameters are automatically logged, along with defaults, units, *
40 !* and a description. It is an error for a variable to be overridden *
41 !* more than once, and MOM6 has a facility to check for unused lines *
42 !* to set variables, which may indicate miss-spelled or archaic *
43 !* parameters. Parameter names are case-specific, and lines may use *
44 !* a F90 or C++ style comment, starting with ! or //. *
45 !* *

142 subroutine open_param_file(filename, CS, checkable, component, doc_file_dir)
143  character(len=*), intent(in) :: filename
144  type(param_file_type), intent(inout) :: CS
145  logical, optional, intent(in) :: checkable
146  character(len=*), optional, intent(in) :: component
147  character(len=*), optional, intent(in) :: doc_file_dir
148 
149  logical :: file_exists, unit_in_use, Netcdf_file, may_check
150  integer :: ios, iounit, strlen, i
151  character(len=240) :: doc_path
152  type(parameter_block), pointer :: block
153 
154  may_check = .true. ; if (present(checkable)) may_check = checkable
155 
156  ! Check for non-blank filename
157  strlen = len_trim(filename)
158  if (strlen == 0) then
159  call mom_error(fatal, "open_param_file: Input file has not been specified.")
160  endif
161 
162  ! Check that this file has not already been opened
163  if (cs%nfiles > 0) then
164  inquire(file=trim(filename), number=iounit)
165  if (iounit /= -1) then
166  do i = 1, cs%nfiles
167  if (cs%iounit(i) == iounit) then
168  if (trim(cs%filename(1)) /= trim(filename)) then
169  call mom_error(fatal, &
170  "open_param_file: internal inconsistency! "//trim(filename)// &
171  " is registered as open but has the wrong unit number!")
172  else
173  call mom_error(warning, &
174  "open_param_file: file "//trim(filename)// &
175  " has already been opened. This should NOT happen!"// &
176  " Did you specify the same file twice in a namelist?")
177  return
178  endif ! filenames
179  endif ! unit numbers
180  enddo ! i
181  endif
182  endif
183 
184  ! Check that the file exists to readstdlog
185  inquire(file=trim(filename), exist=file_exists)
186  if (.not.file_exists) call mom_error(fatal, &
187  "open_param_file: Input file "// trim(filename)//" does not exist.")
188 
189  netcdf_file = .false.
190  if (strlen > 3) then
191  if (filename(strlen-2:strlen) == ".nc") netcdf_file = .true.
192  endif
193 
194  if (netcdf_file) &
195  call mom_error(fatal,"open_param_file: NetCDF files are not yet supported.")
196 
197  if (all_pes_read .or. is_root_pe()) then
198  ! Find an unused unit number.
199  do iounit=10,512
200  INQUIRE(iounit,opened=unit_in_use) ; if (.not.unit_in_use) exit
201  enddo
202  if (iounit >= 512) call mom_error(fatal, &
203  "open_param_file: No unused file unit could be found.")
204 
205  ! Open the parameter file.
206  open(iounit, file=trim(filename), access='SEQUENTIAL', &
207  form='FORMATTED', action='READ', position='REWIND', iostat=ios)
208  if (ios /= 0) call mom_error(fatal, "open_param_file: Error opening "// &
209  trim(filename))
210  else
211  iounit = 1
212  endif
213 
214  ! Store/register the unit and details
215  i = cs%nfiles + 1
216  cs%nfiles = i
217  cs%iounit(i) = iounit
218  cs%filename(i) = filename
219  cs%NetCDF_file(i) = netcdf_file
220  allocate(block) ; block%name = '' ; cs%blockName => block
221 
222  call mom_mesg("open_param_file: "// trim(filename)// &
223  " has been opened successfully.", 5)
224 
225  call populate_param_data(iounit, filename, cs%param_data(i))
226 
227  call read_param(cs,"SEND_LOG_TO_STDOUT",cs%log_to_stdout)
228  call read_param(cs,"REPORT_UNUSED_PARAMS",cs%report_unused)
229  call read_param(cs,"FATAL_UNUSED_PARAMS",cs%unused_params_fatal)
230  cs%doc_file = "MOM_parameter_doc"
231  if (present(component)) cs%doc_file = trim(component)//"_parameter_doc"
232  call read_param(cs,"DOCUMENT_FILE", cs%doc_file)
233  if (.not.may_check) then
234  cs%report_unused = .false.
235  cs%unused_params_fatal = .false.
236  endif
237 
238  ! Open the log file.
239  cs%stdlog = stdlog() ; cs%stdout = stdout()
240  cs%log_open = (stdlog() > 0)
241 
242  doc_path = cs%doc_file
243  if (len_trim(cs%doc_file) > 0) then
244  cs%complete_doc = complete_doc_default
245  call read_param(cs, "COMPLETE_DOCUMENTATION", cs%complete_doc)
246  cs%minimal_doc = minimal_doc_default
247  call read_param(cs, "MINIMAL_DOCUMENTATION", cs%minimal_doc)
248  if (present(doc_file_dir)) then ; if (len_trim(doc_file_dir) > 0) then
249  doc_path = trim(slasher(doc_file_dir))//trim(cs%doc_file)
250  endif ; endif
251  else
252  cs%complete_doc = .false.
253  cs%minimal_doc = .false.
254  endif
255  call doc_init(doc_path, cs%doc, cs%minimal_doc, cs%complete_doc)
256 
257 end subroutine open_param_file

259 subroutine close_param_file(CS, quiet_close, component)
260  type(param_file_type), intent(inout) :: CS
261  logical, optional, intent(in) :: quiet_close
262  character(len=*), optional, intent(in) :: component
263 ! Arguments: CS - the param_file_type to close
264 ! (in,opt) quiet_close - if present and true, do not do any logging with this
265 ! call.
266 ! This include declares and sets the variable "version".
267 #include "version_variable.h"
268  character(len=128) :: docfile_default
269  character(len=40) :: mod ! This module's name.
270  integer :: i, n, num_unused
271 
272  if (present(quiet_close)) then ; if (quiet_close) then
273  do i = 1, cs%nfiles
274  cs%iounit(i) = -1
275  cs%filename(i) = ''
276  cs%NetCDF_file(i) = .false.
277  deallocate (cs%param_data(i)%line)
278  deallocate (cs%param_data(i)%line_used)
279  enddo
280  cs%log_open = .false.
281  call doc_end(cs%doc)
282  return
283  endif ; endif
284 
285  ! Log the parameters for the parser.
286  mod = "MOM_file_parser"
287  call log_version(cs, mod, version)
288  call log_param(cs, mod, "SEND_LOG_TO_STDOUT", &
289  cs%log_to_stdout, &
290  "If true, all log messages are also sent to stdout.", &
291  default=log_to_stdout_default)
292  call log_param(cs, mod, "REPORT_UNUSED_PARAMS", &
293  cs%report_unused, &
294  "If true, report any parameter lines that are not used \n"//&
295  "in the run.", default=report_unused_default)
296  call log_param(cs, mod, "FATAL_UNUSED_PARAMS", &
297  cs%unused_params_fatal, &
298  "If true, kill the run if there are any unused \n"//&
299  "parameters.", default=unused_params_fatal_default)
300  docfile_default = "MOM_parameter_doc"
301  if (present(component)) docfile_default = trim(component)//"_parameter_doc"
302  call log_param(cs, mod, "DOCUMENT_FILE", cs%doc_file, &
303  "The basename for files where run-time parameters, their\n"//&
304  "settings, units and defaults are documented. Blank will\n"//&
305  "disable all parameter documentation.", default=docfile_default)
306  if (len_trim(cs%doc_file) > 0) then
307  call log_param(cs, mod, "COMPLETE_DOCUMENTATION", &
308  cs%complete_doc, &
309  "If true, all run-time parameters are\n"//&
310  "documented in "//trim(cs%doc_file)//&
311  ".all .", default=complete_doc_default)
312  call log_param(cs, mod, "MINIMAL_DOCUMENTATION", &
313  cs%minimal_doc, &
314  "If true, non-default run-time parameters are\n"//&
315  "documented in "//trim(cs%doc_file)//&
316  ".short .", default=minimal_doc_default)
317  endif
318 
319  num_unused = 0
320  do i = 1, cs%nfiles
321  ! only root pe has the file open
322  if (all_pes_read .or. is_root_pe()) close(cs%iounit(i))
323  call mom_mesg("close_param_file: "// trim(cs%filename(i))// &
324  " has been closed successfully.", 5)
325 
326  ! Check for unused lines.
327  if (is_root_pe() .and. (cs%report_unused .or. &
328  cs%unused_params_fatal)) then
329  do n=1,cs%param_data(i)%num_lines
330  if (.not.cs%param_data(i)%line_used(n)) then
331  num_unused = num_unused + 1
332  if (cs%report_unused) &
333  call mom_error(warning, "Unused line in "//trim(cs%filename(i))// &
334  " : "//trim(cs%param_data(i)%line(n)))
335  endif
336  enddo
337  endif
338 
339  cs%iounit(i) = -1
340  cs%filename(i) = ''
341  cs%NetCDF_file(i) = .false.
342  deallocate (cs%param_data(i)%line)
343  deallocate (cs%param_data(i)%line_used)
344  enddo
345 
346  if (is_root_pe() .and. (num_unused>0) .and. cs%unused_params_fatal) &
347  call mom_error(fatal, "Run stopped because of unused parameter lines.")
348 
349  cs%log_open = .false.
350  call doc_end(cs%doc)
351 
352 end subroutine close_param_file

354 subroutine populate_param_data(iounit, filename, param_data)
355  integer, intent(in) :: iounit
356  character(len=*), intent(in) :: filename
357  type(file_data_type), intent(inout) :: param_data
358 
359  character(len=INPUT_STR_LENGTH) :: line
360  integer :: num_lines
361  logical :: inMultiLineComment
362 
363 ! Find the number of keyword lines in a parameter file
364 ! Allocate the space to hold the lines in param_data%line
365 ! Populate param_data%line with the keyword lines from parameter file
366 
367  if (iounit <= 0) return
368 
369  if (all_pes_read .or. is_root_pe()) then
370  ! rewind the parameter file
371  rewind(iounit)
372 
373  ! count the number of valid entries in the parameter file
374  num_lines = 0
375  inmultilinecomment = .false.
376  do while(.true.)
377  read(iounit, '(a)', end=8, err=9) line
378  line = replacetabs(line)
379  if (inmultilinecomment) then
380  if (closemultilinecomment(line)) inmultilinecomment=.false.
381  else
382  if (lastnoncommentnonblank(line)>0) num_lines = num_lines + 1
383  if (openmultilinecomment(line)) inmultilinecomment=.true.
384  endif
385  enddo ! while (.true.)
386  8 continue ! get here when read() reaches EOF
387 
388  if (inmultilinecomment .and. is_root_pe()) &
389  call mom_error(fatal, 'MOM_file_parser : A C-style multi-line comment '// &
390  '(/* ... */) was not closed before the end of '//trim(filename))
391 
392  ! allocate space to hold contents of the parameter file
393  param_data%num_lines = num_lines
394  endif ! (is_root_pe())
395 
396  ! Broadcast the number of valid entries in parameter file
397  if (.not. all_pes_read) then
398  call broadcast(param_data%num_lines, root_pe())
399  endif
400 
401  ! Set up the space for storing the actual lines.
402  num_lines = param_data%num_lines
403  allocate (param_data%line(num_lines))
404  allocate (param_data%line_used(num_lines))
405  param_data%line(:) = ' '
406  param_data%line_used(:) = .false.
407 
408  ! Read the actual lines.
409  if (all_pes_read .or. is_root_pe()) then
410  ! rewind the parameter file
411  rewind(iounit)
412 
413  ! Populate param_data%line
414  num_lines = 0
415  do while(.true.)
416  read(iounit, '(a)', end=18, err=9) line
417  line = replacetabs(line)
418  if (inmultilinecomment) then
419  if (closemultilinecomment(line)) inmultilinecomment=.false.
420  else
421  if (lastnoncommentnonblank(line)>0) then
422  line = removecomments(line)
423  line = simplifywhitespace(line(:len_trim(line)))
424  num_lines = num_lines + 1
425  param_data%line(num_lines) = line
426  endif
427  if (openmultilinecomment(line)) inmultilinecomment=.true.
428  endif
429  enddo ! while (.true.)
430 18 continue ! get here when read() reaches EOF
431 
432  if (num_lines /= param_data%num_lines) &
433  call mom_error(fatal, 'MOM_file_parser : Found different number of '// &
434  'valid lines on second reading of '//trim(filename))
435  endif ! (is_root_pe())
436 
437  ! Broadcast the populated array param_data%line
438  if (.not. all_pes_read) then
439  call broadcast(param_data%line, input_str_length, root_pe())
440  endif
441 
442  return
443 
444 9 call mom_error(fatal, "MOM_file_parser : "//&
445  "Error while reading file "//trim(filename))
446 
447 end subroutine populate_param_data

449 function openmultilinecomment(string)
450  character(len=*), intent(in) :: string
451  logical :: openMultiLineComment
452 ! True if a /* appears on this line without a closing */
453  integer :: icom, last
454  openmultilinecomment = .false.

457  if (icom > 0) then
458  openmultilinecomment=.true.
459  last = last+icom+1
460  endif

516 function simplifywhitespace(string)
517  character(len=*), intent(in) :: string
518  character(len=len(string)+16) :: simplifyWhiteSpace
519 ! Constructs a string with all repeated whitespace replaced with single blanks
520 ! and insert white space where it helps delineate tokens (e.g. around =)
521  integer :: i,j
522  logical :: nonBlank = .false., insidestring = .false.
523  character(len=1) :: quoteChar=" "
524  nonblank = .false.; insidestring = .false. ! NOTE: For some reason this line is needed??
525  i=0
526  simplifywhitespace=repeat(" ",len(string)+16)
527  do j=1,len_trim(string)
528  if (insidestring) then ! Do not change formatting inside strings
529  i=i+1
530  simplifywhitespace(i:i)=string(j:j)
531  if (string(j:j)==quotechar) insidestring=.false. ! End of string
532  else ! The following is outside of string delimiters
533  if (string(j:j)==" " .or. string(j:j)==achar(9)) then ! Space or tab
534  if (nonblank) then ! Only copy a blank if the preceeding character was non-blank
535  i=i+1
536  simplifywhitespace(i:i)=" " ! Not string(j:j) so that tabs are replace by blanks
537  nonblank=.false.
538  endif
539  elseif (string(j:j)=='"' .or. string(j:j)=="'") then ! Start a sting
540  i=i+1
541  simplifywhitespace(i:i)=string(j:j)
542  insidestring=.true.
543  quotechar=string(j:j) ! Keep copy of starting quote
544  nonblank=.true. ! For exit from string
545  elseif (string(j:j)=='=') then
546  ! Insert spaces if this character is "=" so that line contains " = "
547  if (nonblank) then
548  i=i+1
549  simplifywhitespace(i:i)=" "
550  endif
551  i=i+2
552  simplifywhitespace(i-1:i)=string(j:j)//" "
553  nonblank=.false.
554  else ! All other characters
555  i=i+1
556  simplifywhitespace(i:i)=string(j:j)

838 subroutine get_variable_line(CS, varname, found, defined, value_string, paramIsLogical)
839  type(param_file_type), intent(in) :: CS

842  character(len=*), intent(out) :: value_string(:)
843  logical, optional, intent(in) :: paramIsLogical
844 
845  character(len=INPUT_STR_LENGTH) :: val_str, lname, origLine
846  character(len=INPUT_STR_LENGTH) :: line, continuationBuffer, blockName
847  character(len=FILENAME_LENGTH) :: filename
848  integer :: is, id, isd, isu, ise, iso, verbose, ipf
849  integer :: last, last1, ival, oval, max_vals, count, contBufSize
850  character(len=52) :: set
851  logical :: found_override, found_equals
852  logical :: found_define, found_undef
853  logical :: force_cycle, defined_in_line, continuedLine

858  continuationbuffer = repeat(" ",input_str_length)
859  contbufsize = 0
860  verbose = 1
861 
862  variablekindislogical=.false.
863  if (present(paramislogical)) variablekindislogical = paramislogical
864 
865  ! Find the first instance (if any) where the named variable is found, and
866  ! return variables indicating whether this variable is defined and the string
867  ! that contains the value of this variable.
868  found = .false.
869  oval = 0; ival = 0;
870  max_vals = SIZE(value_string)
871  do is=1,max_vals ; value_string(is) = " " ; enddo
872 
873  paramfile_loop: do ipf = 1, cs%nfiles
874  filename = cs%filename(ipf)
875  continuedline = .false.
876  blockname = ''
877 
878  ! Scan through each line of the file
879  do count = 1, cs%param_data(ipf)%num_lines

882 
883  last1 = max(1,last)
884  ! Check if line ends in continuation character (either & or \)
885  ! Note achar(92) is a backslash
886  if (line(last1:last1) == achar(92).or.line(last1:last1) == "&") then
887  continuationbuffer(contbufsize+1:contbufsize+len_trim(line))=line(:last-1)
888  contbufsize=contbufsize + len_trim(line)-1
889  continuedline = .true.
890  if (count==cs%param_data(ipf)%num_lines .and. is_root_pe()) &
891  call mom_error(fatal, "MOM_file_parser : the last line"// &
892  " of the file ends in a continuation character but"// &
893  " there are no more lines to read. "// &
894  " Line: '"//trim(line(:last))//"'"//&
895  " in file "//trim(filename)//".")
896  cycle ! cycle inorder to append the next line of the file
897  elseif (continuedline) then
898  ! If we reached this point then this is the end of line continuation
899  continuationbuffer(contbufsize+1:contbufsize+len_trim(line))=line(:last)
900  line = continuationbuffer
901  continuationbuffer=repeat(" ",input_str_length) ! Clear for next use
902  contbufsize = 0
903  continuedline = .false.
904  last = len_trim(line)
905  endif
906 
907  origline = trim(line) ! Keep original for error messages
908 
909  ! Check for '#override' at start of line
910  found_override = .false.; found_define = .false.; found_undef = .false.
911  iso = index(line(:last), "#override " )!; if (is > 0) found_override = .true.
912  if (iso>1) call mom_error(fatal, "MOM_file_parser : #override was found "// &
913  " but was not the first keyword."// &
914  " Line: '"//trim(line(:last))//"'"//&
915  " in file "//trim(filename)//".")
916  if (iso==1) then
917  found_override = .true.
918  if (index(line(:last), "#override define ")==1) found_define = .true.
919  if (index(line(:last), "#override undef ")==1) found_undef = .true.
920  line = trim(adjustl(line(iso+10:last))); last = len_trim(line)
921  endif
922 
923  ! Check for start of fortran namelist, ie. '&namelist'
924  if (index(line(:last),'&')==1) then
925  iso=index(line(:last),' ')
926  if (iso>0) then ! possibly simething else on this line
927  blockname = pushblocklevel(blockname,line(2:iso-1))
928  line=trim(adjustl(line(iso:last)))
929  last=len_trim(line)
930  if (last==0) cycle ! nothing else on this line
931  else ! just the namelist on this line
932  if (len_trim(blockname)>0) then
933  blockname = trim(blockname) // '%' //trim(line(2:last))
934  else
935  blockname = trim(line(2:last))
936  endif
937  call flag_line_as_read(cs%param_data(ipf)%line_used,count)
938  cycle
939  endif
940  endif
941 
942  ! Newer form of parameter block, block%, %block or block%param or
943  iso=index(line(:last),'%')
944  fullpathparameter = .false.
945  if (iso==1) then ! % is first character means this is a close
946  if (len_trim(blockname)==0 .and. is_root_pe()) call mom_error(fatal, &
947  'get_variable_line: An extra close block was encountered. Line="'// &

951  ' block did not match the open block. Line="'//trim(line(:last))//'"' )

953  call mom_error(fatal, 'get_variable_line: A named close for a parameter'// &

955  blockname = popblocklevel(blockname)
956  call flag_line_as_read(cs%param_data(ipf)%line_used,count)
957  elseif (iso==last) then ! This is a new block if % is last character
958  blockname = pushblocklevel(blockname, line(:iso-1))
959  call flag_line_as_read(cs%param_data(ipf)%line_used,count)
960  else ! This is of the form block%parameter = ... (full path parameter)
961  iso=index(line(:last),'%',.true.)
962  ! Check that the parameter block names on the line matches the state set by the caller
963  if (iso>0 .and. trim(cs%blockName%name)==trim(line(:iso-1))) then
964  fullpathparameter = .true.
965  line = trim(line(iso+1:last)) ! Strip away the block name for subsequent processing
966  last = len_trim(line)
967  endif
968  endif
969

971  inwrongblock = .false.
972  if (len_trim(blockname)>0) then ! In a namelist block in file
973  if (trim(cs%blockName%name)/=trim(blockname)) inwrongblock = .true. ! Not in the required block
974  endif
975  if (len_trim(cs%blockName%name)>0) then ! In a namelist block in the model
976  if (trim(cs%blockName%name)/=trim(blockname)) inwrongblock = .true. ! Not in the required block
977  endif
978 
979  ! Check for termination of a fortran namelist (with a '/')
980  if (line(last:last)=='/') then
981  if (len_trim(blockname)==0 .and. is_root_pe()) call mom_error(fatal, &
982  'get_variable_line: An extra namelist/block end was encountered. Line="'// &
983  trim(line(:last))//'"' )
984  blockname = popblocklevel(blockname)
985  last = last - 1 ! Ignore the termination character from here on
986  endif
987  if (inwrongblock .and. .not. fullpathparameter) then
988  if (index(" "//line(:last+1), " "//trim(varname)//" ")>0) &
989  call mom_error(warning,"MOM_file_parser : "//trim(varname)// &
990  ' found outside of block '//trim(cs%blockName%name)//'%. Ignoring.')
991  cycle
992  endif
993 
994  ! Determine whether this line mentions the named parameter or not
995  if (index(" "//line(:last)//" ", " "//trim(varname)//" ") == 0) cycle
996 
997  ! Detect keywords
998  found_equals = .false.
999  isd = index(line(:last), "define" )!; if (isd > 0) found_define = .true.
1000  isu = index(line(:last), "undef" )!; if (isu > 0) found_undef = .true.
1001  ise = index(line(:last), " = " ); if (ise > 1) found_equals = .true.
1002  if (index(line(:last), "#define ")==1) found_define = .true.
1003  if (index(line(:last), "#undef ")==1) found_undef = .true.
1004 
1005  ! Check for missing, mutually exclusive or incomplete keywords
1006  if (is_root_pe()) then
1007  if (.not. (found_define .or. found_undef .or. found_equals)) &
1008  call mom_error(fatal, "MOM_file_parser : the parameter name '"// &
1009  trim(varname)//"' was found without define or undef."// &
1010  " Line: '"//trim(line(:last))//"'"//&
1011  " in file "//trim(filename)//".")
1012  if (found_define .and. found_undef) call mom_error(fatal, &
1013  "MOM_file_parser : Both 'undef' and 'define' occur."// &
1014  " Line: '"//trim(line(:last))//"'"//&
1015  " in file "//trim(filename)//".")
1016  if (found_equals .and. (found_define .or. found_undef)) &
1017  call mom_error(fatal, &
1018  "MOM_file_parser : Both 'a=b' and 'undef/define' syntax occur."// &
1019  " Line: '"//trim(line(:last))//"'"//&
1020  " in file "//trim(filename)//".")
1021  if (found_override .and. .not. (found_define .or. found_undef .or. found_equals)) &
1022  call mom_error(fatal, "MOM_file_parser : override was found "// &
1023  " without a define or undef."// &
1024  " Line: '"//trim(line(:last))//"'"//&
1025  " in file "//trim(filename)//".")
1026  endif
1027 
1028  ! Interpret the line and collect values, if any
1029  if (found_define) then
1030  ! Move starting pointer to first letter of defined name.
1031  is = isd + 5 + scan(line(isd+6:last), set)
1032 
1033  id = scan(line(is:last), ' ') ! Find space between name and value
1034  if ( id == 0 ) then
1035  ! There is no space so the name is simply being defined.
1036  lname = trim(line(is:last))
1037  if (trim(lname) /= trim(varname)) cycle
1038  val_str = " "
1039  else
1040  ! There is a string or number after the name.
1041  lname = trim(line(is:is+id-1))
1042  if (trim(lname) /= trim(varname)) cycle
1043  val_str = trim(adjustl(line(is+id:last)))
1044  endif
1045  found = .true. ; defined_in_line = .true.
1046  elseif (found_undef) then
