26 !* This program contains the subroutine that advects layer *
27 !* thickness. The scheme here uses a Piecewise-Parabolic method with *
28 !* a positive definite limiter. *
29 !* *
30 !* Macros written all in capital letters are defined in MOM_memory.h. *
31 !* *
32 !* A small fragment of the grid is shown below: *
33 !* *
34 !* j+1 x ^ x ^ x At x: q *
35 !* j+1 > o > o > At ^: v, vh *
36 !* j x ^ x ^ x At >: u, uh *
37 !* j > o > o > At o: h, hin *
38 !* j-1 x ^ x ^ x *
39 !* i-1 i i+1 At x & ^: *
40 !* i i+1 At > & o: *
41 !* *
42 !* The boundaries always run through q grid points (x). *

107 subroutine continuity_ppm(u, v, hin, h, uh, vh, dt, G, GV, CS, uhbt, vhbt, OBC, &
108  visc_rem_u, visc_rem_v, u_cor, v_cor, &
109  uhbt_aux, vhbt_aux, u_cor_aux, v_cor_aux, bt_cont)
110  type(ocean_grid_type), intent(inout) :: G
111  type(continuity_ppm_cs), pointer :: CS
112  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in) :: u
113  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(in) :: v
114  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: hin
115  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(inout) :: h
116  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out) :: uh
117  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(out) :: vh
118  real, intent(in) :: dt
119  type(verticalgrid_type), intent(in) :: GV
120  real, dimension(SZIB_(G),SZJ_(G)), intent(in), optional :: uhbt
121  real, dimension(SZI_(G),SZJB_(G)), intent(in), optional :: vhbt
122  type(ocean_obc_type), pointer, optional :: OBC
123  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in), optional :: visc_rem_u
124  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(in), optional :: visc_rem_v
125  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out), optional :: u_cor
126  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(out), optional :: v_cor
127  real, dimension(SZIB_(G),SZJ_(G)), intent(in), optional :: uhbt_aux
128  real, dimension(SZI_(G),SZJB_(G)), intent(in), optional :: vhbt_aux
129  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out), optional :: u_cor_aux
130  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(out), optional :: v_cor_aux
131  type(bt_cont_type), pointer, optional :: BT_cont
132 ! This subroutine time steps the layer thicknesses, using a monotonically
133 ! limit, directionally split PPM scheme, based on Lin (1994). In the following
134 ! documentation, H is used for the units of thickness (usually m or kg m-2.)
135 
136 ! Arguments: u - Zonal velocity, in m s-1.
137 ! (in) v - Meridional velocity, in m s-1.
138 ! (in) hin - Initial layer thickness, in H.
139 ! (out) h - Final layer thickness, in H.
140 ! (out) uh - Volume flux through zonal faces = u*h*dy, H m2 s-1.
141 ! (out) vh - Volume flux through meridional faces = v*h*dx,
142 ! in H m2 s-1.
143 ! (in) dt - Time increment in s.
144 ! (in) G - The ocean's grid structure.
145 ! (in) GV - The ocean's vertical grid structure.
146 ! (in) CS - The control structure returned by a previous call to
147 ! continuity_PPM_init.
148 ! (in, opt) uhbt - The summed volume flux through zonal faces, H m2 s-1.
149 ! (in, opt) vhbt - The summed volume flux through meridional faces, H m2 s-1.
150 ! (in, opt) OBC - This open boundary condition type specifies whether, where,
151 ! and what open boundary conditions are used.
152 ! (in, opt) visc_rem_u - Both the fraction of the momentum originally in a
153 ! (in, opt) visc_rem_v - layer that remains after a time-step of viscosity,
154 ! and the fraction of a time-step's worth of a
155 ! barotropic acceleration that a layer experiences
156 ! after viscosity is applied, in the zonal (_u) and
157 ! meridional (_v) directions. Nondimensional between
158 ! 0 (at the bottom) and 1 (far above the bottom).
159 ! (out, opt) u_cor - The zonal velocities that give uhbt as the depth-
160 ! integrated transport, in m s-1.
161 ! (out, opt) v_cor - The meridional velocities that give vhbt as the
162 ! depth-integrated transport, in m s-1.
163 ! (in, opt) uhbt_aux - A second set of summed volume fluxes through zonal
164 ! (in, opt) vhbt_aux - and meridional faces, both in H m2 s-1.
165 ! (out, opt) u_cor_aux - The zonal and meridional velocities that give uhbt_aux
166 ! (out, opt) v_cor_aux - and vhbt_aux as the depth-integrated transports,
167 ! both in m s-1.
168 ! (out, opt) BT_cont - A structure with elements that describe the effective
169 ! open face areas as a function of barotropic flow.
170 
171  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: &
172  h_input ! Left and right face thicknesses, in H.
173  real :: h_min
174  type(loop_bounds_type) :: LB
175  integer :: is, ie, js, je, nz, stensil
176  integer :: i, j, k
177 
178  logical :: apply_OBC_u, apply_OBC_v, x_first
179  logical :: apply_OBC_u_flather_east, apply_OBC_u_flather_west
180  logical :: apply_OBC_v_flather_north, apply_OBC_v_flather_south
181  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
182 
183  h_min = gv%Angstrom
184 
185  if (.not.associated(cs)) call mom_error(fatal, &
186  "MOM_continuity_PPM: Module must be initialized before it is used.")
187  x_first = (mod(g%first_direction,2) == 0)
188  if (present(obc)) then ; if (associated(obc)) then
189  apply_obc_u = obc%apply_OBC_u ; apply_obc_v = obc%apply_OBC_v
190  endif ; endif
191 
192  apply_obc_u_flather_east = .false. ; apply_obc_u_flather_west = .false.
193  apply_obc_v_flather_north = .false. ; apply_obc_v_flather_south = .false.
194  if (present(obc)) then ; if (associated(obc)) then
195  apply_obc_u_flather_east = obc%apply_OBC_u_flather_east
196  apply_obc_u_flather_west = obc%apply_OBC_u_flather_west
197  apply_obc_v_flather_north = obc%apply_OBC_v_flather_north
198  apply_obc_v_flather_south = obc%apply_OBC_v_flather_south
199  ! If an OBC is being applied, copy the input thicknesses so that the
200  ! OBC code works even if hin == h.
201  if (apply_obc_u_flather_east .or. apply_obc_u_flather_west .or. &
202  apply_obc_v_flather_north .or. apply_obc_v_flather_south) &
203  h_input(:,:,:) = hin(:,:,:)
204  endif ; endif
205 
206  if (present(visc_rem_u) .neqv. present(visc_rem_v)) call mom_error(fatal, &
207  "MOM_continuity_PPM: Either both visc_rem_u and visc_rem_v or neither"// &
208  " one must be present in call to continuity_PPM.")
209 
210  stensil = 3 ; if (cs%simple_2nd) stensil = 2 ; if (cs%upwind_1st) stensil = 1
211 
212  if (x_first) then
213  ! First, advect zonally.
214  lb%ish = g%isc ; lb%ieh = g%iec
215  lb%jsh = g%jsc-stensil ; lb%jeh = g%jec+stensil
216  call zonal_mass_flux(u, hin, uh, dt, g, gv, cs, lb, uhbt, obc, visc_rem_u, &
217  u_cor, uhbt_aux, u_cor_aux, bt_cont)
218 
219  call cpu_clock_begin(id_clock_update)
220 !$OMP parallel do default(none) shared(LB,nz,G,uh,hin,dt,h)
221  do k=1,nz ; do j=lb%jsh,lb%jeh ; do i=lb%ish,lb%ieh
222  h(i,j,k) = hin(i,j,k) - dt* g%IareaT(i,j) * (uh(i,j,k) - uh(i-1,j,k))
223  ! Uncomment this line to prevent underflow.
224  ! if (h(i,j,k) < h_min) h(i,j,k) = h_min
225  enddo ; enddo ; enddo
226  call cpu_clock_end(id_clock_update)
227 
228  if (apply_obc_u_flather_east .or. apply_obc_u_flather_west) then
229  do k=1,nz ; do j=lb%jsh,lb%jeh
230  do i=lb%ish,lb%ieh+1
231  if (obc%OBC_mask_u(i-1,j) .and. (obc%OBC_kind_u(i-1,j) == obc_flather_e)) &
232  h(i,j,k) = h_input(i-1,j,k)
233  enddo
234  do i=lb%ish-1,lb%ieh
235  if (obc%OBC_mask_u(i,j) .and. (obc%OBC_kind_u(i,j) == obc_flather_w)) &
236  h(i,j,k) = h_input(i+1,j,k)
237  enddo
238  enddo ; enddo
239  endif
240  lb%ish = g%isc ; lb%ieh = g%iec ; lb%jsh = g%jsc ; lb%jeh = g%jec
241 
242  ! Now advect meridionally, using the updated thicknesses to determine
243  ! the fluxes.
244  call meridional_mass_flux(v, h, vh, dt, g, gv, cs, lb, vhbt, obc, visc_rem_v, &
245  v_cor, vhbt_aux, v_cor_aux, bt_cont)
246 
247  call cpu_clock_begin(id_clock_update)
248 !$OMP parallel do default(none) shared(h_min,nz,LB,h,dt,G,vh)
249  do k=1,nz ; do j=lb%jsh,lb%jeh ; do i=lb%ish,lb%ieh
250  h(i,j,k) = h(i,j,k) - dt*g%IareaT(i,j) * (vh(i,j,k) - vh(i,j-1,k))
251  ! This line prevents underflow.
252  if (h(i,j,k) < h_min) h(i,j,k) = h_min
253  enddo ; enddo ; enddo
254  call cpu_clock_end(id_clock_update)
255 
256  if (apply_obc_v_flather_north .or. apply_obc_v_flather_south) then
257  do k=1,nz
258  do j=lb%jsh,lb%jeh+1 ; do i=lb%ish-1,lb%ieh+1
259  if (obc%OBC_mask_v(i,j-1) .and. (obc%OBC_kind_v(i,j-1) == obc_flather_n)) &
260  h(i,j,k) = h_input(i,j-1,k)
261  enddo ; enddo
262  do j=lb%jsh-1,lb%jeh ; do i=lb%ish-1,lb%ieh+1
263  if (obc%OBC_mask_v(i,j) .and. (obc%OBC_kind_v(i,j) == obc_flather_s)) &
264  h(i,j,k) = h_input(i,j+1,k)
265  enddo ; enddo
266  enddo
267  endif
268  else ! .not. x_first
269  ! First, advect meridionally, so set the loop bounds accordingly.
270  lb%ish = g%isc-stensil ; lb%ieh = g%iec+stensil
271  lb%jsh = g%jsc ; lb%jeh = g%jec
272 
273  call meridional_mass_flux(v, hin, vh, dt, g, gv, cs, lb, vhbt, obc, visc_rem_v, &
274  v_cor, vhbt_aux, v_cor_aux, bt_cont)
275 
276  call cpu_clock_begin(id_clock_update)
277 !$OMP parallel do default(none) shared(nz,LB,h,hin,dt,G,vh)
278  do k=1,nz ; do j=lb%jsh,lb%jeh ; do i=lb%ish,lb%ieh
279  h(i,j,k) = hin(i,j,k) - dt*g%IareaT(i,j) * (vh(i,j,k) - vh(i,j-1,k))
280  enddo ; enddo ; enddo
281  call cpu_clock_end(id_clock_update)
282 
283  if (apply_obc_v_flather_north .or. apply_obc_v_flather_south) then
284  do k=1,nz
285  do j=lb%jsh,lb%jeh+1 ; do i=lb%ish-1,lb%ieh+1
286  if (obc%OBC_mask_v(i,j-1) .and. (obc%OBC_kind_v(i,j-1) == obc_flather_n)) &
287  h(i,j,k) = h_input(i,j-1,k)
288  enddo ; enddo
289  do j=lb%jsh-1,lb%jeh ; do i=lb%ish-1,lb%ieh+1
290  if (obc%OBC_mask_v(i,j) .and. (obc%OBC_kind_v(i,j) == obc_flather_s)) &
291  h(i,j,k) = h_input(i,j+1,k)
292  enddo ; enddo
293  enddo
294  endif
295 
296  ! Now advect zonally, using the updated thicknesses to determine
297  ! the fluxes.
298  lb%ish = g%isc ; lb%ieh = g%iec ; lb%jsh = g%jsc ; lb%jeh = g%jec
299  call zonal_mass_flux(u, h, uh, dt, g, gv, cs, lb, uhbt, obc, visc_rem_u, &
300  u_cor, uhbt_aux, u_cor_aux, bt_cont)
301 
302  call cpu_clock_begin(id_clock_update)
303 !$OMP parallel do default(none) shared(h_min,nz,LB,h,dt,G,uh)
304  do k=1,nz ; do j=lb%jsh,lb%jeh ; do i=lb%ish,lb%ieh
305  h(i,j,k) = h(i,j,k) - dt* g%IareaT(i,j) * (uh(i,j,k) - uh(i-1,j,k))
306  ! This line prevents underflow.
307  if (h(i,j,k) < h_min) h(i,j,k) = h_min
308  enddo ; enddo ; enddo
309  call cpu_clock_end(id_clock_update)
310 
311  if (apply_obc_u_flather_east .or. apply_obc_u_flather_west) then
312  do k=1,nz ; do j=lb%jsh,lb%jeh
313  do i=lb%ish,lb%ieh+1
314  if (obc%OBC_mask_u(i-1,j) .and. (obc%OBC_kind_u(i-1,j) == obc_flather_e)) &
315  h(i,j,k) = h_input(i-1,j,k)
316  enddo
317  do i=lb%ish-1,lb%ieh
318  if (obc%OBC_mask_u(i,j) .and. (obc%OBC_kind_u(i,j) == obc_flather_w)) &
319  h(i,j,k) = h_input(i+1,j,k)
320  enddo
321  enddo ; enddo
322  endif
323  endif
324 
325 end subroutine continuity_ppm

327 subroutine zonal_mass_flux(u, h_in, uh, dt, G, GV, CS, LB, uhbt, OBC, &
328  visc_rem_u, u_cor, uhbt_aux, u_cor_aux, bt_cont)
329  type(ocean_grid_type), intent(inout) :: G
330  type(verticalgrid_type), intent(in) :: GV
331  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in) :: u
332  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h_in
333  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out) :: uh
334  real, intent(in) :: dt
335  type(continuity_ppm_cs), pointer :: CS
336  type(loop_bounds_type), intent(in) :: LB
337  type(ocean_obc_type), pointer, optional :: OBC
338  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in), optional :: visc_rem_u
339  real, dimension(SZIB_(G),SZJ_(G)), intent(in), optional :: uhbt, uhbt_aux
340  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out), optional :: u_cor, u_cor_aux
341  type(bt_cont_type), pointer, optional :: BT_cont
342 ! This subroutine calculates the mass or volume fluxes through the zonal
343 ! faces, and other related quantities.
344 ! Arguments: u - Zonal velocity, in m s-1.
345 ! (in) h_in - Layer thickness used to calculate the fluxes, in H.
346 ! (out) uh - Volume flux through zonal faces = u*h*dy, H m2 s-1.
347 ! (in) dt - Time increment in s.
348 ! (in) G - The ocean's grid structure.
349 ! (in) CS - The control structure returned by a previous call to
350 ! continuity_PPM_init.
351 ! (in) LB - A structure with the active loop bounds.
352 ! (in, opt) uhbt - The summed volume flux through zonal faces, H m2 s-1.
353 ! (in, opt) OBC - This open boundary condition type specifies whether, where,
354 ! and what open boundary conditions are used.
355 ! (in, opt) visc_rem_u - Both the fraction of the momentum originally in a
356 ! layer that remains after a time-step of viscosity,
357 ! and the fraction of a time-step's worth of a
358 ! barotropic acceleration that a layer experiences
359 ! after viscosity is applied. Nondimensional between
360 ! 0 (at the bottom) and 1 (far above the bottom).
361 ! (out, opt) u_cor - The zonal velocitiess (u with a barotropic correction)
362 ! that give uhbt as the depth-integrated transport, m s-1.
363 ! (in, opt) uhbt_aux - A second set of summed volume fluxes through zonal
364 ! faces, in H m2 s-1.
365 ! (out, opt) u_cor_aux - The zonal velocities (u with a barotropic correction)
366 ! that give uhbt_aux as the depth-integrated transports,
367 ! in m s-1.
368 ! (out, opt) BT_cont - A structure with elements that describe the effective
369 ! open face areas as a function of barotropic flow.
370 
371  real, dimension(SZIB_(G),SZK_(G)) :: &
372  duhdu ! Partial derivative of uh with u, in H m.
373  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: &
374  hl, hr ! Left and right face thicknesses, in H.
375  real, dimension(SZIB_(G)) :: &
376  du, & ! Corrective barotropic change in the velocity, in m s-1.
377  du_min_CFL, & ! Min/max limits on du correction
378  du_max_CFL, & ! to avoid CFL violations
379  duhdu_tot_0, & ! Summed partial derivative of uh with u, in H m.
380  uh_tot_0, & ! Summed transport with no barotropic correction in H m2 s-1.
381  visc_rem_max ! The column maximum of visc_rem.
382  logical, dimension(SZIB_(G)) :: do_i
383  real, dimension(SZIB_(G),SZK_(G)) :: &
384  visc_rem ! A 2-D copy of visc_rem_u or an array of 1's.
385  real :: I_vrm ! 1.0 / visc_rem_max, nondim.
386  real :: CFL_dt ! The maximum CFL ratio of the adjusted velocities divided by
387  ! the time step, in s-1.
388  real :: I_dt ! 1.0 / dt, in s-1.
389  real :: du_lim ! The velocity change that give a relative CFL of 1, in m s-1.
390  real :: dx_E, dx_W ! Effective x-grid spacings to the east and west, in m.
391  integer :: i, j, k, ish, ieh, jsh, jeh, nz
392  logical :: do_aux, apply_OBC_u, use_visc_rem, set_BT_cont, any_simple_OBC
393 
394  do_aux = (present(uhbt_aux) .and. present(u_cor_aux))
395  use_visc_rem = present(visc_rem_u)
396  apply_obc_u = .false. ; set_bt_cont = .false.
397  if (present(bt_cont)) set_bt_cont = (associated(bt_cont))
398  if (present(obc)) then ; if (associated(obc)) then
399  apply_obc_u = obc%apply_OBC_u
400  endif ; endif
401  ish = lb%ish ; ieh = lb%ieh ; jsh = lb%jsh ; jeh = lb%jeh ; nz = g%ke
402 
403  cfl_dt = cs%CFL_limit_adjust / dt
404  i_dt = 1.0 / dt
405  if (cs%aggress_adjust) cfl_dt = i_dt

609 subroutine zonal_flux_layer(u, h, hL, hR, uh, duhdu, visc_rem, dt, G, j, &
610  ish, ieh, do_i, vol_cfl)
611  type(ocean_grid_type), intent(inout) :: G
612  real, dimension(SZIB_(G)), intent(in) :: u, visc_rem
613  real, dimension(SZI_(G)), intent(in) :: h, hL, hR
614  real, dimension(SZIB_(G)), intent(inout) :: uh, duhdu
615  real, intent(in) :: dt
616  integer, intent(in) :: j, ish, ieh
617  logical, dimension(SZIB_(G)), intent(in) :: do_i
618  logical, intent(in) :: vol_CFL
619 ! This subroutines evaluates the zonal mass or volume fluxes in a layer.
620 !
621 ! Arguments: u - Zonal velocity, in m s-1.
622 ! (in) h - Layer thickness used to calculate the fluxes, in H.
623 ! (in) hL, hR - Left- and right- thicknesses in the reconstruction, in H.
624 ! (out) uh - The zonal mass or volume transport, in H m2 s-1.
625 ! (out) duhdu - The partial derivative of uh with u, in H m.
626 ! (in) dt - Time increment in s.
627 ! (in) G - The ocean's grid structure.
628 ! (in) visc_rem - Both the fraction of the momentum originally in a
629 ! layer that remains after a time-step of viscosity,
630 ! and the fraction of a time-step's worth of a
631 ! barotropic acceleration that a layer experiences
632 ! after viscosity is applied. Nondimensional between
633 ! 0 (at the bottom) and 1 (far above the bottom).
634 ! (in) j, ish, ieh - The index range to work on.
635 ! (in) do_i - A logical flag indiciating which I values to work on.
636 ! (in) vol_CFL - If true, rescale the ratio of face areas to the cell
637 ! areas when estimating the CFL number.
638  real :: CFL ! The CFL number based on the local velocity and grid spacing, ND.
639  real :: curv_3 ! A measure of the thickness curvature over a grid length,
640  ! with the same units as h_in.
641  real :: h_marg ! The marginal thickness of a flux, in H.
642  integer :: i
643 
644  do i=ish-1,ieh ; if (do_i(i)) then
645  ! Set new values of uh and duhdu.
646  if (u(i) > 0.0) then
647  if (vol_cfl) then ; cfl = (u(i) * dt) * (g%dy_Cu(i,j) * g%IareaT(i,j))
648  else ; cfl = u(i) * dt * g%IdxT(i,j) ; endif
649  curv_3 = hl(i) + hr(i) - 2.0*h(i)
650  uh(i) = g%dy_Cu(i,j) * u(i) * &
651  (hr(i) + cfl * (0.5*(hl(i) - hr(i)) + curv_3*(cfl - 1.5)))
652  h_marg = hr(i) + cfl * ((hl(i) - hr(i)) + 3.0*curv_3*(cfl - 1.0))
653  elseif (u(i) < 0.0) then
654  if (vol_cfl) then ; cfl = (-u(i) * dt) * (g%dy_Cu(i,j) * g%IareaT(i+1,j))
655  else ; cfl = -u(i) * dt * g%IdxT(i+1,j) ; endif
656  curv_3 = hl(i+1) + hr(i+1) - 2.0*h(i+1)
657  uh(i) = g%dy_Cu(i,j) * u(i) * &
658  (hl(i+1) + cfl * (0.5*(hr(i+1)-hl(i+1)) + curv_3*(cfl - 1.5)))
659  h_marg = hl(i+1) + cfl * ((hr(i+1)-hl(i+1)) + 3.0*curv_3*(cfl - 1.0))
660  else
661  uh(i) = 0.0
662  h_marg = 0.5 * (hl(i+1) + hr(i))
663  endif
664  duhdu(i) = g%dy_Cu(i,j) * h_marg * visc_rem(i)
665  endif ; enddo
666 
667 end subroutine zonal_flux_layer

669 subroutine zonal_face_thickness(u, h, hL, hR, h_u, dt, G, LB, vol_CFL, &
670  marginal, visc_rem_u)
671  type(ocean_grid_type), intent(inout) :: G
672  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in) :: u
673  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h, hL, hR
674  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(inout) :: h_u
675  real, intent(in) :: dt
676  type(loop_bounds_type), intent(in) :: LB
677  logical, intent(in) :: vol_CFL
678  logical, intent(in) :: marginal
679  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in), optional :: visc_rem_u
680 ! This subroutines sets the effective interface thickness at each zonal
681 ! velocity point.
682 !
683 ! Arguments: u - Zonal velocity, in m s-1.
684 ! (in) h - Layer thickness used to calculate the fluxes, in H.
685 ! (in) hL, hR - Left- and right- thicknesses in the reconstruction, in H.
686 ! (out) h_u - The thickness at zonal faces, in H.
687 ! (in) dt - Time increment in s.
688 ! (in) G - The ocean's grid structure.
689 ! (in) LB - A structure with the active thickness loop bounds.
690 ! (in) vol_CFL - If true, rescale the ratio of face areas to the cell
691 ! areas when estimating the CFL number.
692 ! (in) marginal - If true, report the marginal face thicknesses; otherwise
693 ! report transport-averaged thicknesses.
694 ! (in, opt) visc_rem_u - Both the fraction of the momentum originally in a
695 ! layer that remains after a time-step of viscosity,
696 ! and the fraction of a time-step's worth of a
697 ! barotropic acceleration that a layer experiences
698 ! after viscosity is applied. Nondimensional between
699 ! 0 (at the bottom) and 1 (far above the bottom).
700  real :: CFL ! The CFL number based on the local velocity and grid spacing, ND.
701  real :: curv_3 ! A measure of the thickness curvature over a grid length,
702  ! with the same units as h_in.
703  real :: h_avg ! The average thickness of a flux, in H.
704  real :: h_marg ! The marginal thickness of a flux, in H.
705  integer :: i, j, k, ish, ieh, jsh, jeh, nz
706  ish = lb%ish ; ieh = lb%ieh ; jsh = lb%jsh ; jeh = lb%jeh ; nz = g%ke
707 
708 !$OMP parallel default(none) shared(ish,ieh,jsh,jeh,nz,u,vol_CFL,dt,G, &
709 !$OMP hL,hR,h,h_u,visc_rem_u,marginal) &
710 !$OMP private(CFL,curv_3,h_marg,h_avg)
711 !$OMP do
712  do k=1,nz ; do j=jsh,jeh ; do i=ish-1,ieh
713  if (u(i,j,k) > 0.0) then
714  if (vol_cfl) then ; cfl = (u(i,j,k) * dt) * (g%dy_Cu(i,j) * g%IareaT(i,j))
715  else ; cfl = u(i,j,k) * dt * g%IdxT(i,j) ; endif
716  curv_3 = hl(i,j,k) + hr(i,j,k) - 2.0*h(i,j,k)
717  h_avg = hr(i,j,k) + cfl * (0.5*(hl(i,j,k) - hr(i,j,k)) + curv_3*(cfl - 1.5))
718  h_marg = hr(i,j,k) + cfl * ((hl(i,j,k) - hr(i,j,k)) + 3.0*curv_3*(cfl - 1.0))
719  elseif (u(i,j,k) < 0.0) then
720  if (vol_cfl) then ; cfl = (-u(i,j,k)*dt) * (g%dy_Cu(i,j) * g%IareaT(i+1,j))
721  else ; cfl = -u(i,j,k) * dt * g%IdxT(i+1,j) ; endif
722  curv_3 = hl(i+1,j,k) + hr(i+1,j,k) - 2.0*h(i+1,j,k)
723  h_avg = hl(i+1,j,k) + cfl * (0.5*(hr(i+1,j,k)-hl(i+1,j,k)) + curv_3*(cfl - 1.5))
724  h_marg = hl(i+1,j,k) + cfl * ((hr(i+1,j,k)-hl(i+1,j,k)) + &
725  3.0*curv_3*(cfl - 1.0))
726  else
727  h_avg = 0.5 * (hl(i+1,j,k) + hr(i,j,k))
728  ! The choice to use the arithmetic mean here is somewhat arbitrariy, but
729  ! it should be noted that hl(i+1,j,k) and hr(i,j,k) are usually the same.
730  h_marg = 0.5 * (hl(i+1,j,k) + hr(i,j,k))
731  ! h_marg = (2.0 * hl(i+1,j,k) * hr(i,j,k)) / &
732  ! (hl(i+1,j,k) + hr(i,j,k) + GV%H_subroundoff)
733  endif
734 
735  if (marginal) then ; h_u(i,j,k) = h_marg
736  else ; h_u(i,j,k) = h_avg ; endif
737  enddo; enddo ; enddo
738  if (present(visc_rem_u)) then
739 !$OMP do
740  do k=1,nz ; do j=jsh,jeh ; do i=ish-1,ieh
741  h_u(i,j,k) = h_u(i,j,k) * visc_rem_u(i,j,k)
742  enddo ; enddo ; enddo
743  endif
744 !$OMP end parallel
745 
746 end subroutine zonal_face_thickness

1655  endif
1656  enddo ! itt-loop
