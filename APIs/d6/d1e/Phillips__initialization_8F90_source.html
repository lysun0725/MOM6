51 !> Initialize thickness field.
52 subroutine phillips_initialize_thickness(h, G, GV, param_file)
53  type(ocean_grid_type), intent(in) :: G !< The ocean's grid structure.
54  real, intent(out), dimension(SZI_(G),SZJ_(G), SZK_(G)) :: h !< The thickness that is
55  !! being initialized.
56  type(verticalgrid_type), intent(in) :: GV !< The ocean's vertical grid structure.
57  type(param_file_type), intent(in) :: param_file !< A structure indicating the
58  !! open file to parse for model
59  !! parameter values.
60 
61  real :: eta0(szk_(g)+1) ! The 1-d nominal positions of the interfaces.
62  real :: eta_im(szj_(g),szk_(g)+1) ! A temporary array for zonal-mean eta, m.
63  real :: eta1D(szk_(g)+1) ! Interface height relative to the sea surface
64  ! positive upward, in m.
65  real :: damp_rate, jet_width, jet_height, y_2
66  real :: half_strat, half_depth
67  character(len=40) :: mod = "Phillips_initialize_thickness" ! This subroutine's name.
68  integer :: i, j, k, is, ie, js, je, isd, ied, jsd, jed, nz
69 
70  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
71  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
72 
73  eta_im(:,:) = 0.0
74 
75  call log_version(param_file, mod, version)
76  call get_param(param_file, mod, "HALF_STRAT_DEPTH", half_strat, &
77  "The maximum depth of the ocean.", units="nondim", &
78  default = 0.5)
79  call get_param(param_file, mod, "JET_WIDTH", jet_width, &
80  "The width of the zonal-mean jet.", units="km", &
81  fail_if_missing=.true.)
82  call get_param(param_file, mod, "JET_HEIGHT", jet_height, &
83  "The interface height scale associated with the \n"//&
84  "zonal-mean jet.", units="m", &
85  fail_if_missing=.true.)
86 
87  half_depth = g%max_depth*half_strat
88  eta0(1) = 0.0 ; eta0(nz+1) = -g%max_depth
89  do k=2,1+nz/2 ; eta0(k) = -half_depth*(2.0*(k-1)/real(nz)) ; enddo
90  do k=2+nz/2,nz+1
91  eta0(k) = -g%max_depth - 2.0*(g%max_depth-half_depth) * ((k-(nz+1))/real(nz))
92  enddo
93 
94  do j=js,je
95  eta_im(j,1) = 0.0 ; eta_im(j,nz+1) = -g%max_depth
96  enddo
97  do k=2,nz ; do j=js,je
98  y_2 = g%geoLatT(is,j) - g%south_lat - 0.5*g%len_lat
99  eta_im(j,k) = eta0(k) + &
100  jet_height * tanh(y_2 / jet_width)
101 ! jet_height * atan(y_2 / jet_width)
102  if (eta_im(j,k) > 0.0) eta_im(j,k) = 0.0
103  if (eta_im(j,k) < -g%max_depth) eta_im(j,k) = -g%max_depth
104  enddo ; enddo
105 
106  do j=js,je ; do i=is,ie
107 ! This sets the initial thickness (in m) of the layers. The !
108 ! thicknesses are set to insure that: 1. each layer is at least an !
109 ! Angstrom thick, and 2. the interfaces are where they should be !
110 ! based on the resting depths and interface height perturbations, !
111 ! as long at this doesn't interfere with 1. !
112  eta1d(nz+1) = -1.0*g%bathyT(i,j)
113  do k=nz,1,-1
114  eta1d(k) = eta_im(j,k)
115  if (eta1d(k) < (eta1d(k+1) + gv%Angstrom_z)) then
116  eta1d(k) = eta1d(k+1) + gv%Angstrom_z
117  h(i,j,k) = gv%Angstrom_z
118  else
119  h(i,j,k) = eta1d(k) - eta1d(k+1)
120  endif
121  enddo
122  enddo ; enddo
123 
124 end subroutine phillips_initialize_thickness

126 !> Initialize velocity fields.
127 subroutine phillips_initialize_velocity(u, v, G, GV, param_file)
128  type(ocean_grid_type), intent(in) :: G !< Grid structure
129  type(verticalgrid_type), intent(in) :: GV !< Vertical grid structure
130  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out) :: u !< i-component of velocity [m/s]
131  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(out) :: v !< j-component of velocity [m/s]
132  type(param_file_type), intent(in) :: param_file !< A structure indicating
133  !! the open file to parse for model
134  !! parameter values.
135 
136  real :: damp_rate, jet_width, jet_height, x_2, y_2
137  real :: velocity_amplitude, pi
138  integer :: i, j, k, is, ie, js, je, nz, m
139  character(len=40) :: mod = "Phillips_initialize_velocity" ! This subroutine's name.
140  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
141 
142  u(:,:,:) = 0.0
143  v(:,:,:) = 0.0
144 
145  pi = 4.0*atan(1.0)
146 
147  call log_version(param_file, mod, version)
148  call get_param(param_file, mod, "VELOCITY_IC_PERTURB_AMP", velocity_amplitude, &
149  "The magnitude of the initial velocity perturbation.", &
150  units="m s-1", default=0.001)
151  call get_param(param_file, mod, "JET_WIDTH", jet_width, &
152  "The width of the zonal-mean jet.", units="km", &
153  fail_if_missing=.true.)
154  call get_param(param_file, mod, "JET_HEIGHT", jet_height, &
155  "The interface height scale associated with the \n"//&
156  "zonal-mean jet.", units="m", &
157  fail_if_missing=.true.)
158 
159  ! Use thermal wind shear to give a geostrophically balanced flow.
160  do k=nz-1,1 ; do j=js,je ; do i=is-1,ie
161  y_2 = g%geoLatCu(i,j) - g%south_lat - 0.5*g%len_lat
162 ! This uses d/d y_2 atan(y_2 / jet_width)
163 ! u(I,j,k) = u(i,j,k+1) + (1e-3 * jet_height / &
164 ! (jet_width * (1.0 + (y_2 / jet_width)**2))) * &
165 ! (2.0 * GV%g_prime(K+1) / (G%CoriolisBu(I,J) + G%CoriolisBu(I,J-1)))
166 ! This uses d/d y_2 tanh(y_2 / jet_width)
167  u(i,j,k) = u(i,j,k+1) + (1e-3 * (jet_height / jet_width) * &
168  (sech(y_2 / jet_width))**2 ) * &
169  (2.0 * gv%g_prime(k+1) / (g%CoriolisBu(i,j) + g%CoriolisBu(i,j-1)))
170  enddo ; enddo ; enddo
171 
172  do k=1,nz ; do j=js,je ; do i=is-1,ie
173  y_2 = (g%geoLatCu(i,j) - g%south_lat - 0.5*g%len_lat) / g%len_lat
174  x_2 = (g%geoLonCu(i,j) - g%west_lon - 0.5*g%len_lon) / g%len_lon
175  u(i,j,k) = u(i,j,k) + velocity_amplitude * ((real(k)-0.5)/real(nz)) * &
176  (0.5 - abs(2.0*x_2) + 0.1*abs(cos(10.0*pi*x_2)) - abs(sin(5.0*pi*y_2)))
177  do m=1,10
178  u(i,j,k) = u(i,j,k) + 0.2*velocity_amplitude * ((real(k)-0.5)/real(nz)) * &
179  cos(2.0*m*pi*x_2 + 2*m) * cos(6.0*pi*y_2)
180  enddo
181  enddo ; enddo ; enddo
182 
183 end subroutine phillips_initialize_velocity

277 !> Initialize topography.
278 subroutine phillips_initialize_topography(D, G, param_file)
279  type(ocean_grid_type), intent(in) :: G !< The ocean's grid structure.
280  real, intent(out), dimension(SZI_(G),SZJ_(G)) :: D !< The bottom depth in m.
281  type(param_file_type), intent(in) :: param_file !< A structure indicating the
282  !! open file to parse for model
283  !! parameter values.
284 
285  real :: PI, Htop, Wtop, Ltop, offset, dist, &
286  x1, x2, x3, x4, y1, y2
287  integer :: i,j,is,ie,js,je
288  character(len=40) :: mod = "Phillips_initialize_topography" ! This subroutine's name.
289 
290  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
291 
292  pi = 4.0*atan(1.0)
293 
294  call get_param(param_file, mod, "PHILLIPS_HTOP", htop, &
295  "The maximum height of the topography.", units="m", &
296  fail_if_missing=.true.)
297 ! Htop=0.375*G%max_depth ! max height of topog. above max_depth
298  wtop=0.5*g%len_lat ! meridional width of drake and mount
299  ltop=0.25*g%len_lon ! zonal width of topographic features
300  offset=0.1*g%len_lat ! meridional offset from center
301  dist=0.333*g%len_lon ! distance between drake and mount
302  ! should be longer than Ltop/2
303 
304  y1=g%south_lat+0.5*g%len_lat+offset-0.5*wtop; y2=y1+wtop
305  x1=g%west_lon+0.1*g%len_lon; x2=x1+ltop; x3=x1+dist; x4=x3+3.0/2.0*ltop
306 
307  do i=is,ie ; do j=js,je
308  d(i,j)=0.0
309  if (g%geoLonT(i,j)>x1 .and. g%geoLonT(i,j)<x2) then
310  d(i,j) = htop*sin(pi*(g%geoLonT(i,j)-x1)/(x2-x1))**2
311  if (g%geoLatT(i,j)>y1 .and. g%geoLatT(i,j)<y2) then
312  d(i,j)=d(i,j)*(1-sin(pi*(g%geoLatT(i,j)-y1)/(y2-y1))**2)
313  end if
314  else if (g%geoLonT(i,j)>x3 .and. g%geoLonT(i,j)<x4 .and. &
315  g%geoLatT(i,j)>y1 .and. g%geoLatT(i,j)<y2) then
316  d(i,j) = 2.0/3.0*htop*sin(pi*(g%geoLonT(i,j)-x3)/(x4-x3))**2 &
317  *sin(pi*(g%geoLatT(i,j)-y1)/(y2-y1))**2
318  end if
319  d(i,j)=g%max_depth-d(i,j)
320  enddo; enddo
321 
322 end subroutine phillips_initialize_topography

324 !> \class Phillips_initialization
325 !!
326 !! By Robert Hallberg, April 1994 - June 2002 *
327 !! *
328 !! This subroutine initializes the fields for the simulations. *
329 !! The one argument passed to initialize, Time, is set to the *
330 !! current time of the simulation. The fields which are initialized *
331 !! here are: *
332 !! u - Zonal velocity in m s-1. *
333 !! v - Meridional velocity in m s-1. *
334 !! h - Layer thickness in m. (Must be positive.) *
335 !! D - Basin depth in m. (Must be positive.) *
336 !! f - The Coriolis parameter, in s-1. *
337 !! g - The reduced gravity at each interface, in m s-2. *
338 !! Rlay - Layer potential density (coordinate variable) in kg m-3. *
339 !! If ENABLE_THERMODYNAMICS is defined: *
340 !! T - Temperature in C. *
341 !! S - Salinity in psu. *
342 !! If SPONGE is defined: *
343 !! A series of subroutine calls are made to set up the damping *
344 !! rates and reference profiles for all variables that are damped *
345 !! in the sponge. *
346 !! Any user provided tracer code is also first linked through this *
347 !! subroutine. *
348 !! *
349 !! Forcing-related fields (taux, tauy, buoy, ustar, etc.) are set *
350 !! in MOM_surface_forcing.F90. *
351 !! *
352 !! These variables are all set in the set of subroutines (in this *
353 !! file) USER_initialize_bottom_depth, USER_initialize_thickness, *
354 !! USER_initialize_velocity, USER_initialize_temperature_salinity, *
355 !! USER_initialize_mixed_layer_density, USER_initialize_sponges, *
356 !! USER_set_coord, and USER_set_ref_profile. *
357 !! *
358 !! The names of these subroutines should be self-explanatory. They *
359 !! start with "USER_" to indicate that they will likely have to be *
360 !! modified for each simulation to set the initial conditions and *
361 !! boundary conditions. Most of these take two arguments: an integer *
362 !! argument specifying whether the fields are to be calculated *
363 !! internally or read from a NetCDF file; and a string giving the *
364 !! path to that file. If the field is initialized internally, the *
365 !! path is ignored. *
366 !! *
367 !! Macros written all in capital letters are defined in MOM_memory.h. *
368 !! *
369 !! A small fragment of the grid is shown below: *
370 !! *
371 !! j+1 x ^ x ^ x At x: q, f *
372 !! j+1 > o > o > At ^: v, tauy *
373 !! j x ^ x ^ x At >: u, taux *
374 !! j > o > o > At o: h, D, buoy, tr, T, S, ustar *
375 !! j-1 x ^ x ^ x *
376 !! i-1 i i+1 At x & ^: *
377 !! i i+1 At > & o: *
378 !! *
379 !! The boundaries always run through q grid points (x). *
380 end module phillips_initialization
