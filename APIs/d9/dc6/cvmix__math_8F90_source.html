59  subroutine cvmix_math_poly_interp(coeffs, interp_type, x, y, x0, y0)
60 
61 ! !DESCRIPTION:
62 ! Given (x(1), y(1)), (x(2), y(2)), and possibly (x0, y0), compute coeffs =
63 ! $(/a_0, a_1, a_2, a_3/)$ such that, for $f(x) = \sum a_nx^n$, the following
64 ! hold: $f(x(1)) = y(1)$ and $f(x(2)) = y(2)$. For both quadratic and cubic
65 ! interpolation, $f'(x(1)) = (y(1)-y0)/(x(1)-x0)$ as well, and for cubic splines
66 ! $f'(x(2)) = (y(2) - y(1))/(x(2) - x(1))$.
67 ! \\
68 ! \\
69 
70 ! !INPUT PARAMETERS:
71  integer, intent(in) :: interp_type
72  real(cvmix_r8), dimension(2), intent(in) :: x, y
73  real(cvmix_r8), optional, intent(in) :: x0, y0
74 ! !OUTPUT PARAMETERS:
75  real(cvmix_r8), dimension(4), intent(inout) :: coeffs
76 
77 !EOP
78 !BOC
79 
80  ! Local variables
81  real(cvmix_r8) :: det
82  integer :: k, k2
83  real(kind=cvmix_r8), dimension(4,4) :: Minv
84  real(kind=cvmix_r8), dimension(4) :: rhs
85 
86  ! All interpolation assumes form of
87  ! y = dx^3 + cx^2 + bx + a
88  ! linear => c = d = 0
89  ! quad => d = 0
90  coeffs(1:4) = 0.0_cvmix_r8
91  select case (interp_type)
92  case (cvmix_math_interp_linear)
93  ! Match y(1) and y(2)
94 ! print*, "Linear interpolation"
95  coeffs(2) = (y(2)-y(1))/(x(2)-x(1))
96  coeffs(1) = y(1)-coeffs(2)*x(1)
97  case (cvmix_math_interp_quad)
98  ! Match y(1), y(2), and y'(1) [requires x(0)]
99 ! print*, "Quadratic interpolation"
100  ! [ x2^2 x2 1 ][ c ] [ y2 ]
101  ! [ x1^2 x1 1 ][ b ] = [ y1 ]
102  ! [ 2x1 1 0 ][ a ] [ slope ]
103  ! ^^^
104  ! M
105  det = -((x(2)-x(1))**2)
106  ! only using 3x3 block of Minv and first 3 elements of rhs
107  rhs(1) = y(2)
108  rhs(2) = y(1)
109  if (present(x0).and.present(y0)) then
110  rhs(3) = (y(1)-y0)/(x(1)-x0)
111  else
112  rhs(3) = 0.0_cvmix_r8
113  end if
114 
115  minv(1,1) = -cvmix_one/det
116  minv(1,2) = cvmix_one/det
117  minv(1,3) = -cvmix_one/(x(2)-x(1))
118  minv(2,1) = real(2, cvmix_r8)*x(1)/det
119  minv(2,2) = -real(2, cvmix_r8)*x(1)/det
120  minv(2,3) = (x(2)+x(1))/(x(2)-x(1))
121  minv(3,1) = -(x(1)**2)/det
122  minv(3,2) = x(2)*(real(2, cvmix_r8)*x(1)-x(2))/det
123  minv(3,3) = -x(2)*x(1)/(x(2)-x(1))
124 
125  do k=1,3
126  do k2=1,3
127  ! Note: weird "4-k2" term is used because I switched from
128  ! y= 0x^3 + bx^2 + cx + d to
129  ! y = a + bx + cx^2 + 0x^3
130  coeffs(k2) = coeffs(k2)+minv(4-k2,k)*rhs(k)
131  end do
132  end do
133  case (cvmix_math_interp_cube_spline)
134  ! Match y(1), y(2), y'(1), and y'(2)
135 ! print*, "Cubic spline interpolation"
136  ! [ x2^3 x2^2 x2 1 ][ d ] [ y2 ]
137  ! [ x1^3 x1^2 x1 1 ][ c ] = [ y1 ]
138  ! [ 3x1 2x1 1 0 ][ b ] [ slope1 ]
139  ! [ 3x2 2x2 1 0 ][ a ] [ slope2 ]
140  ! ^^^
141  ! M
142  det = -((x(2)-x(1))**3)
143  rhs(1) = y(2)
144  rhs(2) = y(1)
145  if (present(x0).and.present(y0)) then
146  rhs(3) = (y(1)-y0)/(x(1)-x0)
147  else
148  rhs(3) = 0.0_cvmix_r8
149  end if
150  rhs(4) = (y(2)-y(1))/(x(2)-x(1))
151 
152  minv(1,1) = real(2, cvmix_r8)/det
153  minv(1,2) = -real(2, cvmix_r8)/det
154  minv(1,3) = (x(1)-x(2))/det
155  minv(1,4) = (x(1)-x(2))/det
156  minv(2,1) = -real(3, cvmix_r8)*(x(2)+x(1))/det
157  minv(2,2) = real(3, cvmix_r8)*(x(2)+x(1))/det
158  minv(2,3) = (x(2)-x(1))*(real(2, cvmix_r8)*x(2)+x(1))/det
159  minv(2,4) = (x(2)-x(1))*(real(2, cvmix_r8)*x(1)+x(2))/det
160  minv(3,1) = real(6, cvmix_r8)*x(2)*x(1)/det
161  minv(3,2) = -real(6, cvmix_r8)*x(2)*x(1)/det
162  minv(3,3) = -x(2)*(x(2)-x(1))*(real(2, cvmix_r8)*x(1)+x(2))/det
163  minv(3,4) = -x(1)*(x(2)-x(1))*(real(2, cvmix_r8)*x(2)+x(1))/det
164  minv(4,1) = -(x(1)**2)*(real(3, cvmix_r8)*x(2)-x(1))/det
165  minv(4,2) = -(x(2)**2)*(-real(3, cvmix_r8)*x(1)+x(2))/det
166  minv(4,3) = x(1)*(x(2)**2)*(x(2)-x(1))/det
167  minv(4,4) = x(2)*(x(1)**2)*(x(2)-x(1))/det
168 
169  do k=1,4
170  do k2=1,4
171  ! Note: weird "5-k2" term is used because I switched from
172  ! y = a + bx + cx^2 + dx^3 to
173  ! y= ax^3 + bx^2 + cx + d
174  coeffs(k2) = coeffs(k2)+minv(5-k2,k)*rhs(k)
175  end do
176  end do
177  end select
178 
179 !EOC
180 
181  end subroutine cvmix_math_poly_interp

183  function cvmix_math_cubic_root_find(coeffs, x0)
184 
185  real(cvmix_r8), dimension(4), intent(in) :: coeffs
186  real(cvmix_r8), intent(in) :: x0
187 
188  real(cvmix_r8) :: cvmix_math_cubic_root_find
189  real(cvmix_r8) :: fun_val, root, slope
190  integer :: it_cnt
191 
192  root = x0
193  fun_val = coeffs(4)*(root**3)+coeffs(3)*(root**2)+coeffs(2)*root+coeffs(1)
194  do it_cnt = 1, cvmix_math_max_newton_iters
195  if (abs(fun_val).lt.cvmix_math_newton_tol) &
196  exit
197  slope = 3.0_cvmix_r8*coeffs(4)*(root**2)+2.0_cvmix_r8*coeffs(3)*root+coeffs(2)
198  root = root - fun_val/slope
199  fun_val = coeffs(4)*(root**3)+coeffs(3)*(root**2)+coeffs(2)*root+coeffs(1)
200  end do
201  cvmix_math_cubic_root_find = root
202 
203  end function cvmix_math_cubic_root_find

210  function cvmix_math_evaluate_cubic(coeffs, x_in, fprime)
211 
212 ! !DESCRIPTION:
213 ! Computes $f(x) = a_0 + a_1x + a_2x^2 + a_3x^3$ at $x = $\verb|x_in|, where
214 ! \verb|coeffs|$ = (/a_0, a_1, a_2, a_3/)$. If requested, can also return
215 ! $f'(x)$
216 !\\
217 !\\
218 
219 ! !USES:
220 ! Only those used by entire module.
221 
222 ! !INPUT PARAMETERS:
223  real(cvmix_r8), dimension(4), intent(in) :: coeffs
224  real(cvmix_r8), intent(in) :: x_in
225 
226 ! !OUTPUT PARAMETERS:
227  real(cvmix_r8) :: cvmix_math_evaluate_cubic
228  real(cvmix_r8), optional, intent(out) :: fprime
229 
230 !EOP
231 !BOC
232 
233  ! Local Variables
234  integer :: i
235 
236  ! Initialize both the cubic and its derivative to its constant term and
237  ! then add the powers of x_in via a do-loop. This both reduces the number
238  ! of arithmetic steps in the algorithm and avoids possible compiler issues
239  ! if x_in = 0 (because 0*0 is undefined in some compilers)
240  cvmix_math_evaluate_cubic = coeffs(1)
241  if (present(fprime)) &
242  fprime = coeffs(2)
243  do i=2,4
244  cvmix_math_evaluate_cubic = cvmix_math_evaluate_cubic + &
245  coeffs(i)*(x_in**(i-1))
246  if (present(fprime).and.(i.gt.2)) &
247  fprime = fprime + real(i-1,cvmix_r8)*(x_in**(i-2))
248  end do
249 
250  end function cvmix_math_evaluate_cubic
