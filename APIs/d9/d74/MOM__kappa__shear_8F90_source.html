25 !* *
26 !* This file contains the subroutines that determine the diapycnal *
27 !* diffusivity driven by resolved shears, as specified by the *
28 !* parameterizations described in Jackson and Hallberg (JPO, 2008). *
29 !* *
30 !* The technique by which the 6 equations (for kappa, TKE, u, v, T, *
31 !* and S) are solved simultaneously has been dramatically revised *
32 !* from the previous version. The previous version was not converging *
33 !* in some cases, especially near the surface mixed layer, while the *
34 !* revised version does. The revised version solves for kappa and *
35 !* TKE with shear and stratification fixed, then marches the density *
36 !* and velocities forward with an adaptive (and aggressive) time step *
37 !* in a predictor-corrector-corrector emulation of a trapezoidal *
38 !* scheme. Run-time-settable parameters determine the tolerence to *
39 !* which the kappa and TKE equations are solved and the minimum time *
40 !* step that can be taken. *
41 !* *

999  sal(k) = sal(k) + c1(k+1)*sal(k+1)
1000  enddo
1001  else ! dt <= 0.0

1004  enddo
1005  endif

1013  enddo
1014  if (ke<nz) &

1016  endif
1017

1019  n2(1) = 0.0 ; n2(nz+1) = 0.0
1020  if (ks > 1) &
1021  n2(ks) = max(0.0, i_dz_int(ks) * &
1022  (dbuoy_dt(ks) * (t0(ks-1)-t(ks)) + dbuoy_ds(ks) * (s0(ks-1)-sal(ks))))
1023  do k=ks+1,ke
1024  n2(k) = max(0.0, i_dz_int(k) * &
1025  (dbuoy_dt(k) * (t(k-1)-t(k)) + dbuoy_ds(k) * (sal(k-1)-sal(k))))
1026  enddo
1027  if (ke<nz) &
1028  n2(ke+1) = max(0.0, i_dz_int(ke+1) * &
1029  (dbuoy_dt(ke+1) * (t(ke)-t0(ke+1)) + dbuoy_ds(ke+1) * (sal(ke)-s0(ke+1))))
1030  endif
1031
