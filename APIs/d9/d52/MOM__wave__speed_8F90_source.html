67 subroutine wave_speed(h, tv, G, GV, cg1, CS, full_halos)
68  type(ocean_grid_type), intent(in) :: G
69  type(verticalgrid_type), intent(in) :: GV
70  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
71  type(thermo_var_ptrs), intent(in) :: tv
72  real, dimension(SZI_(G),SZJ_(G)), intent(out) :: cg1
73  type(wave_speed_cs), optional, pointer :: CS
74  logical, optional, intent(in) :: full_halos
75 ! This subroutine determines the first mode internal wave speed.
76 ! Arguments: h - Layer thickness, in m or kg m-2.
77 ! (in) tv - A structure containing the thermobaric variables.
78 ! (out) cg1 - The first mode internal gravity wave speed, in m s-1.
79 ! (in) G - The ocean's grid structure.
80 ! (in) GV - The ocean's vertical grid structure.
81 ! (in) CS - The control structure returned by a previous call to
82 ! wave_speed_init.
83 ! (in,opt) full_halos - If true, do the calculation over the entire
84 ! computational domain.
85 
86 ! This subroutine solves for the first baroclinic mode wave speed. (It could
87 ! solve for all the wave speeds, but the iterative approach taken here means
88 ! that this is not particularly efficient.)
89 
90 ! If e(k) is the perturbation interface height, this means solving for the
91 ! smallest eigenvalue (lam = 1/c^2) of the system
92 !
93 ! -Igu(k)*e(k-1) + (Igu(k)+Igl(k)-lam)*e(k) - Igl(k)*e(k+1) = 0.0
94 !
95 ! with rigid lid boundary conditions e(1) = e(nz+1) = 0.0 giving
96 !
97 ! (Igu(2)+Igl(2)-lam)*e(2) - Igl(2)*e(3) = 0.0
98 ! -Igu(nz)*e(nz-1) + (Igu(nz)+Igl(nz)-lam)*e(nz) = 0.0
99 !
100 ! Here
101 ! Igl(k) = 1.0/(gprime(k)*H(k)) ; Igu(k) = 1.0/(gprime(k)*H(k-1))
102 !
103 ! Alternately, these same eigenvalues can be found from the second smallest
104 ! eigenvalue of the Montgomery potential (M(k)) calculation:
105 !
106 ! -Igl(k)*M(k-1) + (Igl(k)+Igu(k+1)-lam)*M(k) - Igu(k+1)*M(k+1) = 0.0
107 !
108 ! with rigid lid and flat bottom boundary conditions
109 !
110 ! (Igu(2)-lam)*M(1) - Igu(2)*M(2) = 0.0
111 ! -Igl(nz)*M(nz-1) + (Igl(nz)-lam)*M(nz) = 0.0
112 !
113 ! Note that the barotropic mode has been eliminated from the rigid lid
114 ! interface height equations, hence the matrix is one row smaller. Without
115 ! the rigid lid, the top boundary condition is simpler to implement with
116 ! the M equations.
117 
118  real, dimension(SZK_(G)+1) :: &
119  dRho_dT, dRho_dS, &
120  pres, T_int, S_int, &
121  gprime ! The reduced gravity across each interface, in m s-2.
122  real, dimension(SZK_(G)) :: &
123  Igl, Igu ! The inverse of the reduced gravity across an interface times
124  ! the thickness of the layer below (Igl) or above (Igu) it,
125  ! in units of s2 m-2.
126  real, dimension(SZK_(G),SZI_(G)) :: &
127  Hf, Tf, Sf, Rf
128  real, dimension(SZK_(G)) :: &
129  Hc, Tc, Sc, Rc, &
130  det, ddet
131  real :: lam, dlam, lam0
132  real :: min_h_frac
133  real :: H_to_pres
134  real :: H_to_m ! Local copy of a unit conversion factor.
135  real, dimension(SZI_(G)) :: &
136  htot, hmin, & ! Thicknesses in m.
137  H_here, HxT_here, HxS_here, HxR_here
138  real :: speed2_tot
139  real :: I_Hnew, drxh_sum
140  real, parameter :: tol1 = 0.0001, tol2 = 0.001
141  real, pointer, dimension(:,:,:) :: T, S
142  real :: g_Rho0 ! G_Earth/Rho0 in m4 s-2 kg-1.
143  real :: rescale, I_rescale
144  integer :: kf(szi_(g))
145  integer, parameter :: max_itt = 10
146  real :: lam_it(max_itt), det_it(max_itt), ddet_it(max_itt)
147  logical :: use_EOS ! If true, density is calculated from T & S using an
148  ! equation of state.
149  integer :: kc
150  integer :: i, j, k, k2, itt, is, ie, js, je, nz
151 
152  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
153 
154  if (present(cs)) then
155  if (.not. associated(cs)) call mom_error(fatal, "MOM_wave_speed: "// &
156  "Module must be initialized before it is used.")
157  endif
158  if (present(full_halos)) then ; if (full_halos) then
159  is = g%isd ; ie = g%ied ; js = g%jsd ; je = g%jed
160  endif ; endif
161 
162  s => tv%S ; t => tv%T
163  g_rho0 = g%g_Earth/gv%Rho0
164  use_eos = associated(tv%eqn_of_state)
165 
166  h_to_pres = g%g_Earth * gv%Rho0
167  h_to_m = gv%H_to_m
168  rescale = 1024.0**4 ; i_rescale = 1.0/rescale
169 
170  min_h_frac = tol1 / real(nz)
171 !$OMP parallel do default(none) shared(is,ie,js,je,nz,h,G,GV,min_h_frac,use_EOS,T,S,tv,&
172 !$OMP H_to_pres,H_to_m,cg1,g_Rho0,rescale,I_rescale) &
173 !$OMP private(htot,hmin,kf,H_here,HxT_here,HxS_here,HxR_here, &
174 !$OMP Hf,Tf,Sf,Rf,pres,T_int,S_int,drho_dT, &
175 !$OMP drho_dS,drxh_sum,kc,Hc,Tc,Sc,I_Hnew,gprime, &
176 !$OMP Rc,speed2_tot,Igl,Igu,lam0,lam,lam_it,dlam, &
177 !$OMP det,ddet,det_it,ddet_it)
178  do j=js,je
179  ! First merge very thin layers with the one above (or below if they are
180  ! at the top). This also transposes the row order so that columns can
181  ! be worked upon one at a time.
182  do i=is,ie ; htot(i) = 0.0 ; enddo
183  do k=1,nz ; do i=is,ie ; htot(i) = htot(i) + h(i,j,k)*h_to_m ; enddo ; enddo
184 
185  do i=is,ie
186  hmin(i) = htot(i)*min_h_frac ; kf(i) = 1 ; h_here(i) = 0.0
187  hxt_here(i) = 0.0 ; hxs_here(i) = 0.0 ; hxr_here(i) = 0.0
188  enddo
189  if (use_eos) then
190  do k=1,nz ; do i=is,ie
191  if ((h_here(i) > hmin(i)) .and. (h(i,j,k)*h_to_m > hmin(i))) then
192  hf(kf(i),i) = h_here(i)
193  tf(kf(i),i) = hxt_here(i) / h_here(i)
194  sf(kf(i),i) = hxs_here(i) / h_here(i)
195  kf(i) = kf(i) + 1
196 
197  ! Start a new layer
198  h_here(i) = h(i,j,k)*h_to_m
199  hxt_here(i) = (h(i,j,k)*h_to_m)*t(i,j,k)
200  hxs_here(i) = (h(i,j,k)*h_to_m)*s(i,j,k)
201  else
202  h_here(i) = h_here(i) + h(i,j,k)*h_to_m
203  hxt_here(i) = hxt_here(i) + (h(i,j,k)*h_to_m)*t(i,j,k)
204  hxs_here(i) = hxs_here(i) + (h(i,j,k)*h_to_m)*s(i,j,k)
205  endif
206  enddo ; enddo
207  do i=is,ie ; if (h_here(i) > 0.0) then
208  hf(kf(i),i) = h_here(i)
209  tf(kf(i),i) = hxt_here(i) / h_here(i)
210  sf(kf(i),i) = hxs_here(i) / h_here(i)
211  endif ; enddo
212  else
213  do k=1,nz ; do i=is,ie
214  if ((h_here(i) > hmin(i)) .and. (h(i,j,k)*h_to_m > hmin(i))) then
215  hf(kf(i),i) = h_here(i) ; rf(kf(i),i) = hxr_here(i) / h_here(i)
216  kf(i) = kf(i) + 1
217 
218  ! Start a new layer
219  h_here(i) = h(i,j,k)*h_to_m
220  hxr_here(i) = (h(i,j,k)*h_to_m)*gv%Rlay(k)
221  else
222  h_here(i) = h_here(i) + h(i,j,k)*h_to_m
223  hxr_here(i) = hxr_here(i) + (h(i,j,k)*h_to_m)*gv%Rlay(k)
224  endif
225  enddo ; enddo
226  do i=is,ie ; if (h_here(i) > 0.0) then
227  hf(kf(i),i) = h_here(i) ; rf(kf(i),i) = hxr_here(i) / h_here(i)
228  endif ; enddo
229  endif
230 
231  ! From this point, we can work on individual columns without causing memory
232  ! to have page faults.
233  do i=is,ie ; if (g%mask2dT(i,j) > 0.5) then
234  if (use_eos) then
235  pres(1) = 0.0
236  do k=2,kf(i)
237  pres(k) = pres(k-1) + h_to_pres*hf(k-1,i)
238  t_int(k) = 0.5*(tf(k,i)+tf(k-1,i))
239  s_int(k) = 0.5*(sf(k,i)+sf(k-1,i))
240  enddo
241  call calculate_density_derivs(t_int, s_int, pres, drho_dt, drho_ds, 2, &
242  kf(i)-1, tv%eqn_of_state)
243 
244  ! Sum the reduced gravities to find out how small a density difference
245  ! is negligibly small.
246  drxh_sum = 0.0
247  do k=2,kf(i)
248  drxh_sum = drxh_sum + 0.5*(hf(k-1,i)+hf(k,i)) * &
249  max(0.0,drho_dt(k)*(tf(k,i)-tf(k-1,i)) + &
250  drho_ds(k)*(sf(k,i)-sf(k-1,i)))
251  enddo
252  else
253  drxh_sum = 0.0
254  do k=2,kf(i)
255  drxh_sum = drxh_sum + 0.5*(hf(k-1,i)+hf(k,i)) * &
256  max(0.0,rf(k,i)-rf(k-1,i))
257  enddo
258  endif
259  ! Find gprime across each internal interface, taking care of convective
260  ! instabilities by merging layers.
261  if (drxh_sum <= 0.0) then
262  cg1(i,j) = 0.0
263  else
264  ! Merge layers to eliminate convective instabilities or exceedingly
265  ! small reduced gravities.
266  if (use_eos) then
267  kc = 1
268  hc(1) = hf(1,i) ; tc(1) = tf(1,i) ; sc(1) = sf(1,i)
269  do k=2,kf(i)
270  if ((drho_dt(k)*(tf(k,i)-tc(kc)) + drho_ds(k)*(sf(k,i)-sc(kc))) * &
271  (hc(kc) + hf(k,i)) < 2.0 * tol2*drxh_sum) then
272  ! Merge this layer with the one above and backtrack.
273  i_hnew = 1.0 / (hc(kc) + hf(k,i))
274  tc(kc) = (hc(kc)*tc(kc) + hf(k,i)*tf(k,i)) * i_hnew
275  sc(kc) = (hc(kc)*sc(kc) + hf(k,i)*sf(k,i)) * i_hnew
276  hc(kc) = (hc(kc) + hf(k,i))
277  ! Backtrack to remove any convective instabilities above... Note
278  ! that the tolerance is a factor of two larger, to avoid limit how
279  ! far back we go.
280  do k2=kc,2,-1
281  if ((drho_dt(k2)*(tc(k2)-tc(k2-1)) + drho_ds(k2)*(sc(k2)-sc(k2-1))) * &
282  (hc(k2) + hc(k2-1)) < tol2*drxh_sum) then
283  ! Merge the two bottommost layers. At this point kc = k2.
284  i_hnew = 1.0 / (hc(kc) + hc(kc-1))
285  tc(kc-1) = (hc(kc)*tc(kc) + hc(kc-1)*tc(kc-1)) * i_hnew
286  sc(kc-1) = (hc(kc)*sc(kc) + hc(kc-1)*sc(kc-1)) * i_hnew
287  hc(kc-1) = (hc(kc) + hc(kc-1))
288  kc = kc - 1
289  else ; exit ; endif
290  enddo
291  else
292  ! Add a new layer to the column.
293  kc = kc + 1
294  drho_ds(kc) = drho_ds(k) ; drho_dt(kc) = drho_dt(k)
295  tc(kc) = tf(k,i) ; sc(kc) = sf(k,i) ; hc(kc) = hf(k,i)
296  endif
297  enddo
298  ! At this point there are kc layers and the gprimes should be positive.
299  do k=2,kc ! Revisit this if non-Boussinesq.
300  gprime(k) = g_rho0 * (drho_dt(k)*(tc(k)-tc(k-1)) + &
301  drho_ds(k)*(sc(k)-sc(k-1)))
302  enddo
303  else ! .not.use_EOS
304  ! Do the same with density directly...
305  kc = 1
306  hc(1) = hf(1,i) ; rc(1) = rf(1,i)
307  do k=2,kf(i)
308  if ((rf(k,i) - rc(kc)) * (hc(kc) + hf(k,i)) < 2.0*tol2*drxh_sum) then
309  ! Merge this layer with the one above and backtrack.
310  rc(kc) = (hc(kc)*rc(kc) + hf(k,i)*rf(k,i)) / (hc(kc) + hf(k,i))
311  hc(kc) = (hc(kc) + hf(k,i))
312  ! Backtrack to remove any convective instabilities above... Note
313  ! that the tolerance is a factor of two larger, to avoid limit how
314  ! far back we go.
315  do k2=kc,2,-1
316  if ((rc(k2)-rc(k2-1)) * (hc(k2)+hc(k2-1)) < tol2*drxh_sum) then
317  ! Merge the two bottommost layers. At this point kc = k2.
318  rc(kc-1) = (hc(kc)*rc(kc) + hc(kc-1)*rc(kc-1)) / (hc(kc) + hc(kc-1))
319  hc(kc-1) = (hc(kc) + hc(kc-1))
320  kc = kc - 1
321  else ; exit ; endif
322  enddo
323  else
324  ! Add a new layer to the column.
325  kc = kc + 1
326  rc(kc) = rf(k,i) ; hc(kc) = hf(k,i)
327  endif
328  enddo
329  ! At this point there are kc layers and the gprimes should be positive.
330  do k=2,kc ! Revisit this if non-Boussinesq.
331  gprime(k) = g_rho0 * (rc(k)-rc(k-1))
332  enddo
333  endif ! use_EOS
334 
335  ! Sum the contributions from all of the interfaces to give an over-estimate
336  ! of the first-mode wave speed.
337  if (kc >= 2) then
338  speed2_tot = 0.0
339  do k=2,kc
340  igl(k) = 1.0/(gprime(k)*hc(k)) ; igu(k) = 1.0/(gprime(k)*hc(k-1))
341  speed2_tot = speed2_tot + gprime(k)*(hc(k-1)+hc(k))
342  enddo
343 
344  ! Overestimate the speed to start with.
345  lam0 = 1.0 / speed2_tot ; lam = lam0
346  ! Find the determinant and its derivative with lam.
347  do itt=1,max_itt
348  lam_it(itt) = lam
349  det(1) = 1.0 ; ddet(1) = 0.0
350  det(2) = (igu(2)+igl(2)-lam) ; ddet(2) = -1.0
351  do k=3,kc
352  det(k) = (igu(k)+igl(k)-lam)*det(k-1) - (igu(k)*igl(k-1))*det(k-2)
353  ddet(k) = (igu(k)+igl(k)-lam)*ddet(k-1) - (igu(k)*igl(k-1))*ddet(k-2) - &
354  det(k-1)
355 
356  ! Rescale det & ddet if det is getting too large.
357  if (abs(det(k)) > rescale) then
358  det(k) = i_rescale*det(k) ; det(k-1) = i_rescale*det(k-1)
359  ddet(k) = i_rescale*ddet(k) ; ddet(k-1) = i_rescale*ddet(k-1)
360  endif
361  enddo
362  ! Use Newton's method iteration to find a new estimate of lam.
363  det_it(itt) = det(kc) ; ddet_it(itt) = ddet(kc)
364 
365  if ((ddet(kc) >= 0.0) .or. (-det(kc) > -0.5*lam*ddet(kc))) then
366  ! lam was not an under-estimate, as intended, so Newton's method
367  ! may not be reliable; lam must be reduced, but not by more
368  ! than half.
369  lam = 0.5 * lam
370  else ! Newton's method is OK.
371  dlam = - det(kc) / ddet(kc)
372  lam = lam + dlam
373  if (abs(dlam) < tol2*lam) exit
374  endif
375  enddo
376 
377  cg1(i,j) = 0.0
378  if (lam > 0.0) cg1(i,j) = 1.0 / sqrt(lam)
379  else
380  cg1(i,j) = 0.0
381  endif
382 
383  endif ! cg1 /= 0.0
384  else
385  cg1(i,j) = 0.0 ! This is a land point.
386  endif ; enddo ! i-loop
387  enddo ! j-loop
388 
389 end subroutine wave_speed

391 subroutine wave_speeds(h, tv, G, GV, nmodes, cn, CS, full_halos)
392  type(ocean_grid_type), intent(in) :: G
393  type(verticalgrid_type), intent(in) :: GV
394  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
395  type(thermo_var_ptrs), intent(in) :: tv
396  integer, intent(in) :: nmodes
397  real, dimension(G%isd:G%ied,G%jsd:G%jed,nmodes), intent(out) :: cn
398  type(wave_speed_cs), optional, pointer :: CS
399  logical, optional, intent(in) :: full_halos
400 ! This subroutine determines the first mode internal wave speed.
401 ! Arguments: h - Layer thickness, in m or kg m-2.
402 ! (in) tv - A structure containing the thermobaric variables.
403 ! (out) cn - The internal gravity wave mode speeds, in m s-1.
404 ! (in) G - The ocean's grid structure.
405 ! (in) GV - The ocean's vertical grid structure.
406 ! (in) CS - The control structure returned by a previous call to
407 ! wave_speed_init.
408 ! (in,opt) full_halos - If true, do the calculation over the entire
409 ! computational domain.
410 !
411 ! This subroutine solves for the baroclinic mode wave speed for the first
412 ! few modes. (It is an iterative approach so is not particularly efficient.)
413 
414 ! If e(k) is the perturbation interface height, this means solving for the
415 ! eigenvalues (lam = 1/c^2) of the system
416 !
417 ! -Igu(k)*e(k-1) + (Igu(k)+Igl(k)-lam)*e(k) - Igl(k)*e(k+1) = 0.0
418 !
419 ! with rigid lid boundary conditions e(1) = e(nz+1) = 0.0 giving
420 !
421 ! (Igu(2)+Igl(2)-lam)*e(2) - Igl(2)*e(3) = 0.0
422 ! -Igu(nz)*e(nz-1) + (Igu(nz)+Igl(nz)-lam)*e(nz) = 0.0
423 !
424 ! Here
425 ! Igl(k) = 1.0/(gprime(k)*H(k)) ; Igu(k) = 1.0/(gprime(k)*H(k-1))
426 !
427 ! Alternately, these same eigenvalues can be found from the second smallest
428 ! eigenvalue of the Montgomery potential (M(k)) calculation:
429 !
430 ! -Igl(k)*M(k-1) + (Igl(k)+Igu(k+1)-lam)*M(k) - Igu(k+1)*M(k+1) = 0.0
431 !
432 ! with rigid lid and flat bottom boundary conditions
433 !
434 ! (Igu(2)-lam)*M(1) - Igu(2)*M(2) = 0.0
435 ! -Igl(nz)*M(nz-1) + (Igl(nz)-lam)*M(nz) = 0.0
436 !
437 ! Note that the barotropic mode has been eliminated from the rigid lid
438 ! interface height equations, hence the matrix is one row smaller. Without
439 ! the rigid lid, the top boundary condition is simpler to implement with
440 ! the M equations.
441 
442  real, dimension(SZK_(G)+1) :: &
443  dRho_dT, dRho_dS, &
444  pres, T_int, S_int, &
445  gprime ! The reduced gravity across each interface, in m s-2.
446  real, dimension(SZK_(G)) :: &
447  Igl, Igu ! The inverse of the reduced gravity across an interface times
448  ! the thickness of the layer below (Igl) or above (Igu) it,
449  ! in units of s2 m-2.
450  real, dimension(SZK_(G)-1) :: &
451  a_diag, b_diag, c_diag
452  ! diagonals of tridiagonal matrix; one value for each
453  ! interface (excluding surface and bottom)
454  real, dimension(SZK_(G),SZI_(G)) :: &
455  Hf, Tf, Sf, Rf
456  real, dimension(SZK_(G)) :: &
457  Hc, Tc, Sc, Rc
458  real, parameter :: c1_thresh = 0.01
459  ! if c1 is below this value, don't bother calculating
460  ! cn values for higher modes
461  real :: det, ddet ! determinant & its derivative of eigen system
462  real :: lam_1 ! approximate mode-1 eigen value
463  real :: lam_n ! approximate mode-n eigen value
464  real :: dlam ! increment in lam for Newton's method
465  real :: lamMin ! minimum lam value for root searching range
466  real :: lamMax ! maximum lam value for root searching range
467  real :: lamInc ! width of moving window for root searching
468  real :: det_l,det_r ! determinant value at left and right of window
469  real :: ddet_l,ddet_r ! derivative of determinant at left and right of window
470  real :: det_sub,ddet_sub! derivative of determinant at subinterval endpoint
471  real :: xl,xr ! lam guesses at left and right of window
472  real :: xl_sub ! lam guess at left of subinterval window
473  real,dimension(nmodes) :: &
474  xbl,xbr ! lam guesses bracketing a zero-crossing (root)
475  integer :: numint ! number of widows (intervals) in root searching range
476  integer :: nrootsfound ! number of extra roots found (not including 1st root)
477  real :: min_h_frac
478  real :: H_to_pres
479  real :: H_to_m ! Local copy of a unit conversion factor.
480  real, dimension(SZI_(G)) :: &
481  htot, hmin, & ! Thicknesses in m.
482  H_here, HxT_here, HxS_here, HxR_here
483  real :: speed2_tot ! overestimate of the mode-1 speed squared, m2 s-2
484  real :: speed2_min ! minimum mode speed (squared) to consider in root searching
485  real, parameter :: reduct_factor = 0.5
486  ! factor used in setting speed2_min
487  real :: I_Hnew, drxh_sum
488  real, parameter :: tol1 = 0.0001, tol2 = 0.001
489  real, pointer, dimension(:,:,:) :: T, S
490  real :: g_Rho0 ! G_Earth/Rho0 in m4 s-2 kg-1.
491  integer :: kf(szi_(g))
492  integer, parameter :: max_itt = 10
493  logical :: use_EOS ! If true, density is calculated from T & S using an
494  ! equation of state.
495  real, dimension(SZK_(G)+1) :: z_int, N2
496  integer :: nsub ! number of subintervals used for root finding
497  integer, parameter :: sub_it_max = 4
498  ! maximum number of times to subdivide interval
499  ! for root finding (# intervals = 2**sub_it_max)
500  logical :: sub_rootfound ! if true, subdivision has located root
501  integer :: kc, nrows
502  integer :: sub, sub_it
503  integer :: i, j, k, k2, itt, is, ie, js, je, nz, row, iint, m, ig, jg
504  integer :: ig_need_sub, jg_need_sub ! for debugging (BDM)
505 
506  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
507 
508  if (present(cs)) then
509  if (.not. associated(cs)) call mom_error(fatal, "MOM_wave_speed: "// &
510  "Module must be initialized before it is used.")
511  endif
512 
513  if (present(full_halos)) then ; if (full_halos) then
514  is = g%isd ; ie = g%ied ; js = g%jsd ; je = g%jed
515  endif ; endif
516 
517  s => tv%S ; t => tv%T
518  g_rho0 = g%g_Earth/gv%Rho0
519  use_eos = associated(tv%eqn_of_state)
520 
521  h_to_pres = g%g_Earth * gv%Rho0
522  h_to_m = gv%H_to_m
523 
524  min_h_frac = tol1 / real(nz)
525 !$OMP parallel do default(none) shared(is,ie,js,je,nz,h,G,GV,min_h_frac,use_EOS,T,S, &
526 !$OMP H_to_pres,H_to_m,tv,cn,g_Rho0,nmodes) &
527 !$OMP private(htot,hmin,kf,H_here,HxT_here,HxS_here,HxR_here, &
528 !$OMP Hf,Tf,Sf,Rf,pres,T_int,S_int,drho_dT, &
529 !$OMP drho_dS,drxh_sum,kc,Hc,Tc,Sc,I_Hnew,gprime, &
530 !$OMP Rc,speed2_tot,Igl,Igu,dlam, &
531 !$OMP det,ddet,ig,jg,z_int,N2,row,nrows,lam_1, &
532 !$OMP lamMin,speed2_min,lamMax,lamInc,numint,det_l, &
533 !$OMP ddet_l,xr,xl,det_r,xbl,xbr,ddet_r,xl_sub, &
534 !$OMP ig_need_sub,jg_need_sub,sub_rootfound,nsub, &
535 !$OMP det_sub,ddet_sub,lam_n, &
536 !$OMP a_diag,b_diag,c_diag,nrootsfound)
537  do j=js,je
538  ! First merge very thin layers with the one above (or below if they are
539  ! at the top). This also transposes the row order so that columns can
540  ! be worked upon one at a time.
541  do i=is,ie ; htot(i) = 0.0 ; enddo
542  do k=1,nz ; do i=is,ie ; htot(i) = htot(i) + h(i,j,k)*h_to_m ; enddo ; enddo
543 
544  do i=is,ie
545  hmin(i) = htot(i)*min_h_frac ; kf(i) = 1 ; h_here(i) = 0.0
546  hxt_here(i) = 0.0 ; hxs_here(i) = 0.0 ; hxr_here(i) = 0.0
547  enddo
548  if (use_eos) then
549  do k=1,nz ; do i=is,ie
550  if ((h_here(i) > hmin(i)) .and. (h(i,j,k)*h_to_m > hmin(i))) then
551  hf(kf(i),i) = h_here(i)
552  tf(kf(i),i) = hxt_here(i) / h_here(i)
553  sf(kf(i),i) = hxs_here(i) / h_here(i)
554  kf(i) = kf(i) + 1
555 
556  ! Start a new layer
557  h_here(i) = h(i,j,k)*h_to_m
558  hxt_here(i) = (h(i,j,k)*h_to_m)*t(i,j,k)
559  hxs_here(i) = (h(i,j,k)*h_to_m)*s(i,j,k)
560  else
561  h_here(i) = h_here(i) + h(i,j,k)*h_to_m
562  hxt_here(i) = hxt_here(i) + (h(i,j,k)*h_to_m)*t(i,j,k)
563  hxs_here(i) = hxs_here(i) + (h(i,j,k)*h_to_m)*s(i,j,k)
564  endif
565  enddo ; enddo
566  do i=is,ie ; if (h_here(i) > 0.0) then
567  hf(kf(i),i) = h_here(i)
568  tf(kf(i),i) = hxt_here(i) / h_here(i)
569  sf(kf(i),i) = hxs_here(i) / h_here(i)
570  endif ; enddo
571  else
572  do k=1,nz ; do i=is,ie
573  if ((h_here(i) > hmin(i)) .and. (h(i,j,k)*h_to_m > hmin(i))) then
574  hf(kf(i),i) = h_here(i) ; rf(kf(i),i) = hxr_here(i) / h_here(i)
575  kf(i) = kf(i) + 1
576 
577  ! Start a new layer
578  h_here(i) = h(i,j,k)*h_to_m
579  hxr_here(i) = (h(i,j,k)*h_to_m)*gv%Rlay(k)
580  else
581  h_here(i) = h_here(i) + h(i,j,k)*h_to_m
582  hxr_here(i) = hxr_here(i) + (h(i,j,k)*h_to_m)*gv%Rlay(k)
583  endif
584  enddo ; enddo
585  do i=is,ie ; if (h_here(i) > 0.0) then
586  hf(kf(i),i) = h_here(i) ; rf(kf(i),i) = hxr_here(i) / h_here(i)
587  endif ; enddo
588  endif
589 
590  ! From this point, we can work on individual columns without causing memory
591  ! to have page faults.
592  do i=is,ie
593  if (g%mask2dT(i,j) > 0.5) then
594  if (use_eos) then
595  pres(1) = 0.0
596  do k=2,kf(i)
597  pres(k) = pres(k-1) + h_to_pres*hf(k-1,i)
598  t_int(k) = 0.5*(tf(k,i)+tf(k-1,i))
599  s_int(k) = 0.5*(sf(k,i)+sf(k-1,i))
600  enddo
601  call calculate_density_derivs(t_int, s_int, pres, drho_dt, drho_ds, 2, &
602  kf(i)-1, tv%eqn_of_state)
603 
604  ! Sum the reduced gravities to find out how small a density difference
605  ! is negligibly small.
606  drxh_sum = 0.0
607  do k=2,kf(i)
608  drxh_sum = drxh_sum + 0.5*(hf(k-1,i)+hf(k,i)) * &
609  max(0.0,drho_dt(k)*(tf(k,i)-tf(k-1,i)) + &
610  drho_ds(k)*(sf(k,i)-sf(k-1,i)))
611  enddo
612  else
613  drxh_sum = 0.0
614  do k=2,kf(i)
615  drxh_sum = drxh_sum + 0.5*(hf(k-1,i)+hf(k,i)) * &
616  max(0.0,rf(k,i)-rf(k-1,i))
617  enddo
618  endif
619  ! Find gprime across each internal interface, taking care of convective
620  ! instabilities by merging layers.
621  if (drxh_sum <= 0.0) then
622  cn(i,j,:) = 0.0
623  else
624  ! Merge layers to eliminate convective instabilities or exceedingly
625  ! small reduced gravities.
626  if (use_eos) then
627  kc = 1
628  hc(1) = hf(1,i) ; tc(1) = tf(1,i) ; sc(1) = sf(1,i)
629  do k=2,kf(i)
630  if ((drho_dt(k)*(tf(k,i)-tc(kc)) + drho_ds(k)*(sf(k,i)-sc(kc))) * &
631  (hc(kc) + hf(k,i)) < 2.0 * tol2*drxh_sum) then
632  ! Merge this layer with the one above and backtrack.
633  i_hnew = 1.0 / (hc(kc) + hf(k,i))
634  tc(kc) = (hc(kc)*tc(kc) + hf(k,i)*tf(k,i)) * i_hnew
635  sc(kc) = (hc(kc)*sc(kc) + hf(k,i)*sf(k,i)) * i_hnew
636  hc(kc) = (hc(kc) + hf(k,i))
637  ! Backtrack to remove any convective instabilities above... Note
638  ! that the tolerance is a factor of two larger, to avoid limit how
639  ! far back we go.
640  do k2=kc,2,-1
641  if ((drho_dt(k2)*(tc(k2)-tc(k2-1)) + drho_ds(k2)*(sc(k2)-sc(k2-1))) * &
642  (hc(k2) + hc(k2-1)) < tol2*drxh_sum) then
643  ! Merge the two bottommost layers. At this point kc = k2.
644  i_hnew = 1.0 / (hc(kc) + hc(kc-1))
645  tc(kc-1) = (hc(kc)*tc(kc) + hc(kc-1)*tc(kc-1)) * i_hnew
646  sc(kc-1) = (hc(kc)*sc(kc) + hc(kc-1)*sc(kc-1)) * i_hnew
647  hc(kc-1) = (hc(kc) + hc(kc-1))
648  kc = kc - 1
649  else ; exit ; endif
650  enddo
651  else
652  ! Add a new layer to the column.
653  kc = kc + 1
654  drho_ds(kc) = drho_ds(k) ; drho_dt(kc) = drho_dt(k)
655  tc(kc) = tf(k,i) ; sc(kc) = sf(k,i) ; hc(kc) = hf(k,i)
656  endif
657  enddo
658  ! At this point there are kc layers and the gprimes should be positive.
659  do k=2,kc ! Revisit this if non-Boussinesq.
660  gprime(k) = g_rho0 * (drho_dt(k)*(tc(k)-tc(k-1)) + &
661  drho_ds(k)*(sc(k)-sc(k-1)))
662  enddo
663  else ! .not.use_EOS
664  ! Do the same with density directly...
665  kc = 1
666  hc(1) = hf(1,i) ; rc(1) = rf(1,i)
667  do k=2,kf(i)
668  if ((rf(k,i) - rc(kc)) * (hc(kc) + hf(k,i)) < 2.0*tol2*drxh_sum) then
669  ! Merge this layer with the one above and backtrack.
670  rc(kc) = (hc(kc)*rc(kc) + hf(k,i)*rf(k,i)) / (hc(kc) + hf(k,i))
671  hc(kc) = (hc(kc) + hf(k,i))
672  ! Backtrack to remove any convective instabilities above... Note
673  ! that the tolerance is a factor of two larger, to avoid limit how
674  ! far back we go.
675  do k2=kc,2,-1
676  if ((rc(k2)-rc(k2-1)) * (hc(k2)+hc(k2-1)) < tol2*drxh_sum) then
677  ! Merge the two bottommost layers. At this point kc = k2.
678  rc(kc-1) = (hc(kc)*rc(kc) + hc(kc-1)*rc(kc-1)) / (hc(kc) + hc(kc-1))
679  hc(kc-1) = (hc(kc) + hc(kc-1))
680  kc = kc - 1
681  else ; exit ; endif
682  enddo
683  else
684  ! Add a new layer to the column.
685  kc = kc + 1
686  rc(kc) = rf(k,i) ; hc(kc) = hf(k,i)
687  endif
688  enddo
689  ! At this point there are kc layers and the gprimes should be positive.
690  do k=2,kc ! Revisit this if non-Boussinesq.
691  gprime(k) = g_rho0 * (rc(k)-rc(k-1))
692  enddo
693  endif ! use_EOS
694 
695  !-----------------NOW FIND WAVE SPEEDS---------------------------------------
696  ig = i + g%idg_offset ; jg = j + g%jdg_offset
697  ! Sum the contributions from all of the interfaces to give an over-estimate
698  ! of the first-mode wave speed.
699  if (kc >= 2) then
700  ! Set depth at surface
701  z_int(1) = 0.0
702  ! initialize speed2_tot
703  speed2_tot = 0.0
704  ! Calculate Igu, Igl, depth, and N2 at each interior interface
705  ! [excludes surface (K=1) and bottom (K=kc+1)]
706  do k=2,kc
707  igl(k) = 1.0/(gprime(k)*hc(k)) ; igu(k) = 1.0/(gprime(k)*hc(k-1))
708  z_int(k) = z_int(k-1) + hc(k-1)
709  n2(k) = gprime(k)/(0.5*(hc(k)+hc(k-1)))
710  speed2_tot = speed2_tot + gprime(k)*(hc(k-1)+hc(k))
711  enddo
712  ! Set stratification for surface and bottom (setting equal to nearest interface for now)
713  n2(1) = n2(2) ; n2(kc+1) = n2(kc)
714  ! Calcualte depth at bottom
715  z_int(kc+1) = z_int(kc)+hc(kc)
716  ! check that thicknesses sum to total depth
717  if (abs(z_int(kc+1)-htot(i)) > 1.e-10) then
718  call mom_error(warning, "wave_structure: mismatch in total depths")
719  print *, "kc=", kc
720  print *, "z_int(kc+1)=", z_int(kc+1)
721  print *, "htot(i)=", htot(i)
722  endif
723 
724  ! Define the diagonals of the tridiagonal matrix
725  ! First, populate interior rows
726  do k=3,kc-1
727  row = k-1 ! indexing for TD matrix rows
728  a_diag(row) = (-igu(k))
729  b_diag(row) = (igu(k)+igl(k))
730  c_diag(row) = (-igl(k))
731  enddo
732  ! Populate top row of tridiagonal matrix
733  k=2 ; row = k-1
734  a_diag(row) = 0.0
735  b_diag(row) = (igu(k)+igl(k))
736  c_diag(row) = (-igl(k))
737  ! Populate bottom row of tridiagonal matrix
738  k=kc ; row = k-1
739  a_diag(row) = (-igu(k))
740  b_diag(row) = (igu(k)+igl(k))
741  c_diag(row) = 0.0
742  ! Total number of rows in the matrix = number of interior interfaces
743  nrows = kc-1
744 
745  ! Under estimate the first eigen value to start with.
746  lam_1 = 1.0 / speed2_tot
747 
748  ! Find the first eigen value
749  do itt=1,max_itt
750  ! calculate the determinant of (A-lam_1*I)
751  call tridiag_det(a_diag(1:nrows),b_diag(1:nrows),c_diag(1:nrows), &
752  nrows,lam_1,det,ddet)
753  ! Use Newton's method iteration to find a new estimate of lam_1
754  !det = det_it(itt) ; ddet = ddet_it(itt)
755  if ((ddet >= 0.0) .or. (-det > -0.5*lam_1*ddet)) then
756  ! lam_1 was not an under-estimate, as intended, so Newton's method
757  ! may not be reliable; lam_1 must be reduced, but not by more
758  ! than half.
759  lam_1 = 0.5 * lam_1
760  else ! Newton's method is OK.
761  dlam = - det / ddet
762  lam_1 = lam_1 + dlam
763  if (abs(dlam) < tol2*lam_1) then
764  ! calculate 1st mode speed
765  if (lam_1 > 0.0) cn(i,j,1) = 1.0 / sqrt(lam_1)
766  exit
767  endif
768  endif
769  enddo
770 
771  ! print resutls (for debugging only)
772  !if(ig .eq. 83 .and. jg .eq. 2) then
773  ! if(nmodes>1)then
774  ! print *, "Results after finding first mode:"
775  ! print *, "first guess at lam_1=", 1./speed2_tot
776  ! print *, "final guess at lam_1=", lam_1
777  ! print *, "det value after iterations, det=", det
778  ! print *, "ddet value after iterations, det=", ddet
779  ! print *, "final guess at c1=", cn(i,j,1)
780  ! print *, "a_diag=",a_diag(1:nrows)
781  ! print *, "b_diag=",b_diag(1:nrows)
782  ! print *, "c_diag=",c_diag(1:nrows)
783  ! !stop
784  ! endif
785  !endif
786 
787  ! Find other eigen values if c1 is of significant magnitude, > cn_thresh
788  nrootsfound = 0 ! number of extra roots found (not including 1st root)
789  if (nmodes>1 .and. kc>=nmodes+1 .and. cn(i,j,1)>c1_thresh) then
790  ! Set the the range to look for the other desired eigen values
791  ! set min value just greater than the 1st root (found above)
792  lammin = lam_1*(1.0 + tol2)
793  ! set max value based on a low guess at wavespeed for highest mode
794  speed2_min = (reduct_factor*cn(i,j,1)/real(nmodes))**2
795  lammax = 1.0 / speed2_min
796  ! set width of interval (not sure about this - BDM)
797  laminc = 0.5*lam_1
798  ! set number of intervals within search range
799  numint = nint((lammax - lammin)/laminc)
800 
801  !if(ig .eq. 144 .and. jg .eq. 5) then
802  ! print *, 'Looking for other eigenvalues at', ig, jg
803  ! print *, 'Wave_speed: lamMin=', lamMin
804  ! print *, 'Wave_speed: cnMax=', 1/sqrt(lamMin)
805  ! print *, 'Wave_speed: lamMax=', lamMax
806  ! print *, 'Wave_speed: cnMin=', 1/sqrt(lamMax)
807  ! print *, 'Wave_speed: lamInc=', lamInc
808  !endif
809 
810  ! Find intervals containing zero-crossings (roots) of the determinant
811  ! that are beyond the first root
812 
813  ! find det_l of first interval (det at left endpoint)
814  call tridiag_det(a_diag(1:nrows),b_diag(1:nrows),c_diag(1:nrows), &
815  nrows,lammin,det_l,ddet_l)
816  ! move interval window looking for zero-crossings************************
817  do iint=1,numint
818  xr = lammin + laminc * iint
819  xl = xr - laminc
820  call tridiag_det(a_diag(1:nrows),b_diag(1:nrows),c_diag(1:nrows), &
821  nrows,xr,det_r,ddet_r)
822  !if(ig .eq. 83 .and. jg .eq. 2) then
823  ! print *, "Move interval"
824  ! print *, "iint=",iint
825  ! print *, "@ xr=",xr
826  ! print *, "det_r=",det_r
827  ! print *, "ddet_r=",ddet_r
828  !endif
829  if (det_l*det_r < 0.0) then ! if function changes sign
830  if (det_l*ddet_l < 0.0) then ! if function at left is headed to zero
831  nrootsfound = nrootsfound + 1
832  xbl(nrootsfound) = xl
833  xbr(nrootsfound) = xr
834  !if(ig .eq. 144 .and. jg .eq. 5) then
835  ! print *, "Root located without subdivision!"
836  ! print *, "between xbl=",xl,"and xbr=",xr
837  !endif
838  else
839  ! function changes sign but has a local max/min in interval,
840  ! try subdividing interval as many times as necessary (or sub_it_max).
841  ! loop that increases number of subintervals:
842  !call MOM_error(WARNING, "determinant changes sign"// &
843  ! "but has a local max/min in interval;"//&
844  ! " reduce increment in lam.")
845  ig_need_sub = i + g%idg_offset ; jg_need_sub = j + g%jdg_offset
846  ! begin subdivision loop -------------------------------------------
847  !print *, "subdividing interval at ig=",ig_need_sub,"jg=",jg_need_sub
848  sub_rootfound = .false. ! initialize
849  do sub_it=1,sub_it_max
850  nsub = 2**sub_it ! number of subintervals; nsub=2,4,8,...
851  ! loop over each subinterval:
852  do sub=1,nsub-1,2 ! only check odds; sub = 1; 1,3; 1,3,5,7;...
853  xl_sub = xl + laminc/(nsub)*sub
854  call tridiag_det(a_diag(1:nrows),b_diag(1:nrows),c_diag(1:nrows), &
855  nrows,xl_sub,det_sub,ddet_sub)
856  if (det_sub*det_r < 0.0) then ! if function changes sign
857  if (det_sub*ddet_sub < 0.0) then ! if function at left is headed to zero
858  sub_rootfound = .true.
859  nrootsfound = nrootsfound + 1
860  xbl(nrootsfound) = xl_sub
861  xbr(nrootsfound) = xr
862  !if(ig .eq. 144 .and. jg .eq. 5) then
863  ! print *, "Root located after subdiving",sub_it," times!"
864  ! print *, "between xbl=",xl_sub,"and xbr=",xr
865  !endif
866  exit ! exit sub loop
867  endif ! headed toward zero
868  endif ! sign change
869  enddo ! sub-loop
870  if (sub_rootfound) exit ! root has been found, exit sub_it loop
871  ! Otherwise, function changes sign but has a local max/min in one of the
872  ! sub intervals, try subdividing again unless sub_it_max has been reached.
873  if (sub_it == sub_it_max) then
874  call mom_error(warning, "wave_speed: root not found "// &
875  " after sub_it_max subdivisions of original"// &
876  " interval.")
877  !if(ig .eq. 144 .and. jg .eq. 5) then
878  !print *, "xbl=",xbl
879  !print *, "xbr=",xbr
880  !print *, "Wave_speed: kc=",kc

972 subroutine tridiag_det(a,b,c,nrows,lam,det_out,ddet_out)
973  real, dimension(:), intent(in) :: a
974  real, dimension(:), intent(in) :: b
975  real, dimension(:), intent(in) :: c
976  integer, intent(in) :: nrows
977  real, intent(in) :: lam
978  real, intent(out):: det_out
979  real, intent(out):: ddet_out
980  ! Arguments:
981  ! (in) a - lower diagonal with first entry equal to zero
982  ! (in) b - middle diagonal
983  ! (in) c - upper diagonal with last entry equal to zero
984  ! (in) nrows - number of rows
985  ! (in) lam - value to subtract from middle diagonal
986  ! (out) det_out - determinate evaluated for given lam value
987  ! (out) ddet_out - derivative of determinate with respect to lam
988  ! evaluated for given lam value
989  real, dimension(nrows) :: &
990  det ! value of recursion function
991  real, dimension(nrows) :: &
992  ddet ! value of recursion function for derivative
993  real, parameter:: &
994  rescale = 1024.0**4 ! max value of determinant allowed before rescaling
995  real :: I_rescale ! inverse of rescale
996  integer :: n ! row (layer interface) index
997 
998  if (size(b) .ne. nrows) call mom_error(warning, "Diagonal b must be same length as nrows.")
999  if (size(a) .ne. nrows) call mom_error(warning, "Diagonal a must be same length as nrows.")
1000  if (size(c) .ne. nrows) call mom_error(warning, "Diagonal c must be same length as nrows.")
1001 
1002  i_rescale = 1.0/rescale
1003 
1004  det(1) = 1.0 ; ddet(1) = 0.0
1005  det(2) = b(2)-lam ; ddet(2) = -1.0
1006  do n=3,nrows
1007  det(n) = (b(n)-lam)*det(n-1) - (a(n)*c(n-1))*det(n-2)
1008  ddet(n) = (b(n)-lam)*ddet(n-1) - (a(n)*c(n-1))*ddet(n-2) - det(n-1)
1009  ! Rescale det & ddet if det is getting too large.
1010  if (abs(det(n)) > rescale) then
1011  det(n) = i_rescale*det(n) ; det(n-1) = i_rescale*det(n-1)
1012  ddet(n) = i_rescale*ddet(n) ; ddet(n-1) = i_rescale*ddet(n-1)
1013  endif
1014  enddo
1015  det_out = det(nrows)
1016  ddet_out = ddet(nrows)
