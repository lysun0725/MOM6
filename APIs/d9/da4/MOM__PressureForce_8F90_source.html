26 !* This file contains the subroutine that directs the model to use *
27 !* the code that determines the horizontal accelerations due to *
28 !* pressure gradients. The two options currently available are a *
29 !* traditional Montgomery potential form, and the analytic finite *
30 !* volume form described in Adcroft, Hallberg and Harrison, 2008, *
31 !* Ocean Modelling, 22, 106-113. *
32 !* *
33 !* PressureForce takes 9 arguments, which are described below. If *
34 !* a non-split time stepping scheme is used, the last three arguments *
35 !* are ignored. *
36 !* *
37 !* Macros written all in capital letters are defined in MOM_memory.h. *
38 !* *
39 !* A small fragment of the grid is shown below: *
40 !* *
41 !* j+1 x ^ x ^ x At x: q, CoriolisBu *
42 !* j+1 > o > o > At ^: v, PFv *
43 !* j x ^ x ^ x At >: u, PFu *
44 !* j > o > o > At o: h, bathyT, M, e, p, pbce, T, S *
45 !* j-1 x ^ x ^ x *
46 !* i-1 i i+1 *
47 !* i i+1 *
48 !* *
49 !* The boundaries always run through q grid points (x). *
50 !* *
51 !********+*********+*********+*********+*********+*********+*********+**

80 subroutine pressureforce(h, tv, PFu, PFv, G, GV, CS, ALE_CSp, p_atm, pbce, eta)
81  type(ocean_grid_type), intent(in) :: G
82  type(verticalgrid_type), intent(in) :: GV
83  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
84  type(thermo_var_ptrs), intent(in) :: tv
85  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out) :: PFu
86  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(out) :: PFv
87  type(pressureforce_cs), pointer :: CS
88  type(ale_cs), pointer :: ALE_CSp
89  real, dimension(:,:), optional, pointer :: p_atm
90  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), optional, intent(out) :: pbce
91  real, dimension(SZI_(G),SZJ_(G)), optional, intent(out) :: eta
92 
93 ! This subroutine works as a temporary interface between the model and the
94 ! Boussinesq and non-Boussinesq pressure force routines.
95 ! Descriptions of the variables are in each of the routines called in the
96 ! following conditional block.
97 
98  if (cs%Analytic_FV_PGF) then
99  if (gv%Boussinesq) then
100  call pressureforce_afv_bouss(h, tv, pfu, pfv, g, gv, cs%PressureForce_AFV_CSp, &
101  ale_csp, p_atm, pbce, eta)
102  else
103  call pressureforce_afv_nonbouss(h, tv, pfu, pfv, g, gv, cs%PressureForce_AFV_CSp, &
104  p_atm, pbce, eta)
105  endif
106  else
107  if (gv%Boussinesq) then
108  call pressureforce_mont_bouss(h, tv, pfu, pfv, g, gv, cs%PressureForce_Mont_CSp, &
109  p_atm, pbce, eta)
110  else
111  call pressureforce_mont_nonbouss(h, tv, pfu, pfv, g, gv, cs%PressureForce_Mont_CSp, &
112  p_atm, pbce, eta)
113  endif
114  endif
115 
116 end subroutine pressureforce

118 subroutine pressureforce_init(Time, G, GV, param_file, diag, CS, tides_CSp)
119  type(time_type), target, intent(in) :: Time
120  type(ocean_grid_type), intent(in) :: G
121  type(verticalgrid_type), intent(in) :: GV
122  type(param_file_type), intent(in) :: param_file
123  type(diag_ctrl), target, intent(inout) :: diag
124  type(pressureforce_cs), pointer :: CS
125  type(tidal_forcing_cs), optional, pointer :: tides_CSp
126 ! Arguments: Time - The current model time.
127 ! (in) G - The ocean's grid structure.
128 ! (in) GV - The ocean's vertical grid structure.
129 ! (in) param_file - A structure indicating the open file to parse for
130 ! model parameter values.
131 ! (in) diag - A structure that is used to regulate diagnostic output.
132 ! (in/out) CS - A pointer that is set to point to the control structure
133 ! for this module.
134 ! (in) tides_CSp - a pointer to the control structure of the tide module.
135 
136 ! This include declares and sets the variable "version".
137 #include "version_variable.h"
138  character(len=40) :: mod = "MOM_PressureForce" ! This module's name.
139 
140  if (associated(cs)) then
141  call mom_error(warning, "PressureForce_init called with an associated "// &
142  "control structure.")
143  return
144  else ; allocate(cs) ; endif
145 
146  ! Read all relevant parameters and write them to the model log.
147  call log_version(param_file, mod, version)
148  call get_param(param_file, mod, "ANALYTIC_FV_PGF", cs%Analytic_FV_PGF, &
149  "If true the pressure gradient forces are calculated \n"//&
150  "with a finite volume form that analytically integrates \n"//&
151  "the equations of state in pressure to avoid any \n"//&
152  "possibility of numerical thermobaric instability, as \n"//&
153  "described in Adcroft et al., O. Mod. (2008).", default=.true.)
154 
155  if (cs%Analytic_FV_PGF) then
156  call pressureforce_afv_init(time, g, gv, param_file, diag, &
157  cs%PressureForce_AFV_CSp, tides_csp)
158  else
159  call pressureforce_mont_init(time, g, gv, param_file, diag, &
160  cs%PressureForce_Mont_CSp, tides_csp)
161  endif
162 
163 end subroutine pressureforce_init
