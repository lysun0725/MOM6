153 !> MOM_grid_init initializes the ocean grid array sizes and grid memory.
154 subroutine mom_grid_init(G, param_file)
155  type(ocean_grid_type), intent(inout) :: G !< The horizontal grid type
156  type(param_file_type), intent(in) :: param_file !< Parameter file handle
157 ! Arguments: G - The ocean's grid structure.
158 ! (in) param_file - A structure indicating the open file to parse for
159 ! model parameter values.
160 
161 ! This include declares and sets the variable "version".
162 #include "version_variable.h"
163  integer :: isd, ied, jsd, jed, nk
164  integer :: IsdB, IedB, JsdB, JedB
165  integer :: ied_max, jed_max
166  integer :: niblock, njblock, nihalo, njhalo, nblocks, n, i, j
167  logical :: global_indexing ! If true use global index values instead of having
168  ! the data domain on each processor start at 1.
169  integer, allocatable, dimension(:) :: ibegin, iend, jbegin, jend
170  character(len=40) :: mod_nm = "MOM_grid" ! This module's name.
171 
172 
173  ! Read all relevant parameters and write them to the model log.
174  call log_version(param_file, mod_nm, version, &
175  "Parameters providing information about the lateral grid.")
176  call get_param(param_file, "MOM", "G_EARTH", g%g_Earth, &
177  "The gravitational acceleration of the Earth.", &
178  units="m s-2", default = 9.80)
179  call get_param(param_file, mod_nm, "GLOBAL_INDEXING", global_indexing, &
180  "If true, use a global lateral indexing convention, so \n"//&
181  "that corresponding points on different processors have \n"//&
182  "the same index. This does not work with static memory.", &
183  default=.false., layoutparam=.true.)
184 #ifdef STATIC_MEMORY_
185  if (global_indexing) call mom_error(fatal, "MOM_grid_init : "//&
186  "GLOBAL_INDEXING can not be true with STATIC_MEMORY.")
187 #endif
188  call get_param(param_file, mod_nm, "FIRST_DIRECTION", g%first_direction, &
189  "An integer that indicates which direction goes first \n"//&
190  "in parts of the code that use directionally split \n"//&
191  "updates, with even numbers (or 0) used for x- first \n"//&
192  "and odd numbers used for y-first.", default=0)
193 
194  call get_param(param_file, mod_nm, "BATHYMETRY_AT_VEL", g%bathymetry_at_vel, &
195  "If true, there are separate values for the basin depths \n"//&
196  "at velocity points. Otherwise the effects of of \n"//&
197  "topography are entirely determined from thickness points.", &
198  default=.false.)
199 
200  call get_param(param_file, mod_nm, "NIBLOCK", niblock, "The number of blocks "// &
201  "in the x-direction on each processor (for openmp).", default=1, &
202  layoutparam=.true.)
203  call get_param(param_file, mod_nm, "NJBLOCK", njblock, "The number of blocks "// &
204  "in the y-direction on each processor (for openmp).", default=1, &
205  layoutparam=.true.)
206 
207  call hor_index_init(g%Domain, g%HI, param_file, &
208  local_indexing=.not.global_indexing)
209 
210  ! get_domain_extent ensures that domains start at 1 for compatibility between
211  ! static and dynamically allocated arrays, unless global_indexing is true.
212  call get_domain_extent(g%Domain, g%isc, g%iec, g%jsc, g%jec, &
213  g%isd, g%ied, g%jsd, g%jed, &
214  g%isg, g%ieg, g%jsg, g%jeg, &
215  g%idg_offset, g%jdg_offset, g%symmetric, &
216  local_indexing=.not.global_indexing)
217  g%isd_global = g%isd+g%idg_offset ; g%jsd_global = g%jsd+g%jdg_offset
218 
219  g%nonblocking_updates = g%Domain%nonblocking_updates
220 
221  ! Set array sizes for fields that are discretized at tracer cell boundaries.
222  g%IscB = g%isc ; g%JscB = g%jsc
223  g%IsdB = g%isd ; g%JsdB = g%jsd
224  g%IsgB = g%isg ; g%JsgB = g%jsg
225  if (g%symmetric) then
226  g%IscB = g%isc-1 ; g%JscB = g%jsc-1
227  g%IsdB = g%isd-1 ; g%JsdB = g%jsd-1
228  g%IsgB = g%isg-1 ; g%JsgB = g%jsg-1
229  endif
230  g%IecB = g%iec ; g%JecB = g%jec
231  g%IedB = g%ied ; g%JedB = g%jed
232  g%IegB = g%ieg ; g%JegB = g%jeg
233 
234  call mom_mesg(" MOM_grid.F90, MOM_grid_init: allocating metrics", 5)
235 
236  call allocate_metrics(g)
237 
238  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
239  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
240  if (g%bathymetry_at_vel) then
241  alloc_(g%Dblock_u(isdb:iedb, jsd:jed)) ; g%Dblock_u(:,:) = 0.0
242  alloc_(g%Dopen_u(isdb:iedb, jsd:jed)) ; g%Dopen_u(:,:) = 0.0
243  alloc_(g%Dblock_v(isd:ied, jsdb:jedb)) ; g%Dblock_v(:,:) = 0.0
244  alloc_(g%Dopen_v(isd:ied, jsdb:jedb)) ; g%Dopen_v(:,:) = 0.0
245  endif
246 
247 ! setup block indices.
248  nihalo = g%Domain%nihalo
249  njhalo = g%Domain%njhalo
250  nblocks = niblock * njblock
251  if (nblocks < 1) call mom_error(fatal, "MOM_grid_init: " // &
252  "nblocks(=NI_BLOCK*NJ_BLOCK) must be no less than 1")
253 
254  allocate(ibegin(niblock), iend(niblock), jbegin(njblock), jend(njblock))
255  call compute_block_extent(g%HI%isc,g%HI%iec,niblock,ibegin,iend)
256  call compute_block_extent(g%HI%jsc,g%HI%jec,njblock,jbegin,jend)
257  !-- make sure the last block is the largest.
258  do i = 1, niblock-1
259  if (iend(i)-ibegin(i) > iend(niblock)-ibegin(niblock) ) call mom_error(fatal, &
260  "MOM_grid_init: the last block size in x-direction is not the largest")
261  enddo
262  do j = 1, njblock-1
263  if (jend(j)-jbegin(j) > jend(njblock)-jbegin(njblock) ) call mom_error(fatal, &
264  "MOM_grid_init: the last block size in y-direction is not the largest")
265  enddo
266 
267  g%nblocks = nblocks
268  allocate(g%Block(nblocks))
269  ied_max = 1 ; jed_max = 1
270  do n = 1,nblocks
271  ! Copy all information from the array index type describing the local grid.
272  g%Block(n) = g%HI
273 
274  i = mod((n-1), niblock) + 1
275  j = (n-1)/niblock + 1
276  !--- isd and jsd are always 1 for each block to permit array reuse.
277  g%Block(n)%isd = 1 ; g%Block(n)%jsd = 1
278  g%Block(n)%isc = g%Block(n)%isd+nihalo
279  g%Block(n)%jsc = g%Block(n)%jsd+njhalo
280  g%Block(n)%iec = g%Block(n)%isc + iend(i) - ibegin(i)
281  g%Block(n)%jec = g%Block(n)%jsc + jend(j) - jbegin(j)
282  g%Block(n)%ied = g%Block(n)%iec + nihalo
283  g%Block(n)%jed = g%Block(n)%jec + njhalo
284  g%Block(n)%IscB = g%Block(n)%isc; g%Block(n)%IecB = g%Block(n)%iec
285  g%Block(n)%JscB = g%Block(n)%jsc; g%Block(n)%JecB = g%Block(n)%jec
286  ! For symmetric memory domains, the first block will have the extra point
287  ! at the lower boundary of its computational domain.
288  if (g%symmetric) then
289  if (i==1) g%Block(n)%IscB = g%Block(n)%IscB-1
290  if (j==1) g%Block(n)%JscB = g%Block(n)%JscB-1
291  endif
292  g%Block(n)%IsdB = g%Block(n)%isd; g%Block(n)%IedB = g%Block(n)%ied
293  g%Block(n)%JsdB = g%Block(n)%jsd; g%Block(n)%JedB = g%Block(n)%jed
294  !--- For symmetric memory domain, every block will have an extra point
295  !--- at the lower boundary of its data domain.
296  if (g%symmetric) then
297  g%Block(n)%IsdB = g%Block(n)%IsdB-1
298  g%Block(n)%JsdB = g%Block(n)%JsdB-1
299  endif
300  g%Block(n)%idg_offset = (ibegin(i) - g%Block(n)%isc) + g%HI%idg_offset
301  g%Block(n)%jdg_offset = (jbegin(j) - g%Block(n)%jsc) + g%HI%jdg_offset
302  ! Find the largest values of ied and jed so that all blocks will have the
303  ! same size in memory.
304  ied_max = max(ied_max, g%Block(n)%ied)
305  jed_max = max(jed_max, g%Block(n)%jed)
306  enddo
307 
308  ! Reset all of the data domain sizes to match the largest for array reuse,
309  ! recalling that all block have isd=jed=1 for array reuse.
310  do n = 1,nblocks
311  g%Block(n)%ied = ied_max ; g%Block(n)%IedB = ied_max
312  g%Block(n)%jed = jed_max ; g%Block(n)%JedB = jed_max
313  enddo
314 
315  !-- do some bounds error checking
316  if ( g%block(nblocks)%ied+g%block(nblocks)%idg_offset > g%HI%ied + g%HI%idg_offset ) &
317  call mom_error(fatal, "MOM_grid_init: G%ied_bk > G%ied")
318  if ( g%block(nblocks)%jed+g%block(nblocks)%jdg_offset > g%HI%jed + g%HI%jdg_offset ) &
319  call mom_error(fatal, "MOM_grid_init: G%jed_bk > G%jed")
320 
321 end subroutine mom_grid_init

324 !> set_derived_metrics calculates metric terms that are derived from other metrics.
325 subroutine set_derived_metrics(G)
326  type(ocean_grid_type), intent(inout) :: G !< The horizontal grid structure
327 ! Various inverse grid spacings and derived areas are calculated within this
328 ! subroutine.
329  integer :: i, j, isd, ied, jsd, jed
330  integer :: IsdB, IedB, JsdB, JedB
331 
332  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
333  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
334 
335  do j=jsd,jed ; do i=isd,ied
336  if (g%dxT(i,j) < 0.0) g%dxT(i,j) = 0.0
337  if (g%dyT(i,j) < 0.0) g%dyT(i,j) = 0.0
338  g%IdxT(i,j) = adcroft_reciprocal(g%dxT(i,j))
339  g%IdyT(i,j) = adcroft_reciprocal(g%dyT(i,j))
340  g%IareaT(i,j) = adcroft_reciprocal(g%areaT(i,j))
341  enddo ; enddo
342 
343  do j=jsd,jed ; do i=isdb,iedb
344  if (g%dxCu(i,j) < 0.0) g%dxCu(i,j) = 0.0
345  if (g%dyCu(i,j) < 0.0) g%dyCu(i,j) = 0.0
346  g%IdxCu(i,j) = adcroft_reciprocal(g%dxCu(i,j))
347  g%IdyCu(i,j) = adcroft_reciprocal(g%dyCu(i,j))
348  enddo ; enddo
349 
350  do j=jsdb,jedb ; do i=isd,ied
351  if (g%dxCv(i,j) < 0.0) g%dxCv(i,j) = 0.0
352  if (g%dyCv(i,j) < 0.0) g%dyCv(i,j) = 0.0
353  g%IdxCv(i,j) = adcroft_reciprocal(g%dxCv(i,j))
354  g%IdyCv(i,j) = adcroft_reciprocal(g%dyCv(i,j))
355  enddo ; enddo
356 
357  do j=jsdb,jedb ; do i=isdb,iedb
358  if (g%dxBu(i,j) < 0.0) g%dxBu(i,j) = 0.0
359  if (g%dyBu(i,j) < 0.0) g%dyBu(i,j) = 0.0
360 
361  g%IdxBu(i,j) = adcroft_reciprocal(g%dxBu(i,j))
362  g%IdyBu(i,j) = adcroft_reciprocal(g%dyBu(i,j))
363  ! areaBu has usually been set to a positive area elsewhere.
364  if (g%areaBu(i,j) <= 0.0) g%areaBu(i,j) = g%dxBu(i,j) * g%dyBu(i,j)
365  g%IareaBu(i,j) = adcroft_reciprocal(g%areaBu(i,j))
366  enddo ; enddo
367 end subroutine set_derived_metrics

377 !> Returns true if the coordinates (x,y) are within the h-cell (i,j)
378 logical function ispointincell(G, i, j, x, y)
379  type(ocean_grid_type), intent(in) :: G !< Grid type
380  integer, intent(in) :: i, j !< i,j indices of cell to test
381  real, intent(in) :: x, y !< x,y coordinates of point
382 ! This is a crude calculation that assume a geographic coordinate system
383  real :: xNE, xNW, xSE, xSW, yNE, yNW, ySE, ySW
384  real :: p0, p1, p2, p3, l0, l1, l2, l3
385  ispointincell = .false.
386  xne = g%geoLonBu(i ,j ) ; yne = g%geoLatBu(i ,j )
387  xnw = g%geoLonBu(i-1,j ) ; ynw = g%geoLatBu(i-1,j )
388  xse = g%geoLonBu(i ,j-1) ; yse = g%geoLatBu(i ,j-1)
389  xsw = g%geoLonBu(i-1,j-1) ; ysw = g%geoLatBu(i-1,j-1)
390  if (x<min(xne,xnw,xse,xsw) .or. x>max(xne,xnw,xse,xsw) .or. &
391  y<min(yne,ynw,yse,ysw) .or. y>max(yne,ynw,yse,ysw) ) then
392  return ! Avoid the more complicated calculation
393  endif
394  l0 = (x-xsw)*(yse-ysw) - (y-ysw)*(xse-xsw)
395  l1 = (x-xse)*(yne-yse) - (y-yse)*(xne-xse)
396  l2 = (x-xne)*(ynw-yne) - (y-yne)*(xnw-xne)
397  l3 = (x-xnw)*(ysw-ynw) - (y-ynw)*(xsw-xnw)
398 
399  p0 = sign(1., l0) ; if (l0 == 0.) p0=0.
400  p1 = sign(1., l1) ; if (l1 == 0.) p1=0.
401  p2 = sign(1., l2) ; if (l2 == 0.) p2=0.
