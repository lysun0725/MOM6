40 !> This subroutine initializes layer thicknesses for the circle_obcs experiment.
41 subroutine circle_obcs_initialize_thickness(h, G, GV, param_file)
42  type(ocean_grid_type), intent(in) :: G !< The ocean's grid structure.
43  type(verticalgrid_type), intent(in) :: GV !< The ocean's vertical grid structure.
44  real, intent(out), dimension(SZI_(G),SZJ_(G), SZK_(G)) :: h !< The thickness that is being initialized.
45  type(param_file_type), intent(in) :: param_file !< A structure indicating the open
46  !! file to parse for model parameter values.
47 
48  real :: e0(szk_(g)+1) ! The resting interface heights, in m, usually !
49  ! negative because it is positive upward. !
50  real :: eta1D(szk_(g)+1)! Interface height relative to the sea surface !
51  ! positive upward, in m. !
52 ! This include declares and sets the variable "version".
53 #include "version_variable.h"
54  character(len=40) :: mod = "circle_obcs_initialize_thickness" ! This module's name.
55  integer :: i, j, k, is, ie, js, je, nz
56  real :: diskrad, rad, xCenter, xRadius, lonC, latC
57 
58  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
59 
60  call mom_mesg(" circle_obcs_initialization.F90, circle_obcs_initialize_thickness: setting thickness", 5)
61 
62  ! Parameters read by cartesian grid initialization
63  call log_version(param_file, mod, version, "")
64  call get_param(param_file, mod, "DISK_RADIUS", diskrad, &
65  "The radius of the initially elevated disk in the \n"//&
66  "circle_obcs test case.", units=g%x_axis_units, &
67  fail_if_missing=.true.)
68 
69  do k=1,nz
70  e0(k) = -g%max_depth * real(k-1) / real(nz)
71  enddo
72 
73  ! Uniform thicknesses for base state
74  do j=js,je ; do i=is,ie !
75  eta1d(nz+1) = -1.0*g%bathyT(i,j)
76  do k=nz,1,-1
77  eta1d(k) = e0(k)
78  if (eta1d(k) < (eta1d(k+1) + gv%Angstrom_z)) then
79  eta1d(k) = eta1d(k+1) + gv%Angstrom_z
80  h(i,j,k) = gv%Angstrom_z
81  else
82  h(i,j,k) = eta1d(k) - eta1d(k+1)
83  endif
84  enddo
85  enddo ; enddo
86 
87  ! Perturb base state by circular anomaly in center
88  k=nz
89  latc = g%south_lat + 0.5*g%len_lat
90  lonc = g%west_lon + 0.5*g%len_lon
91  do j=js,je ; do i=is,ie
92  rad = sqrt((g%geoLonT(i,j)-lonc)**2+(g%geoLatT(i,j)-latc)**2)/(diskrad)
93  ! if (rad <= 6.*diskrad) h(i,j,k) = h(i,j,k)+10.0*exp( -0.5*( rad**2 ) )
94  rad = min( rad, 1. ) ! Flatten outside radius of diskrad
95  rad = rad*(2.*asin(1.)) ! Map 0-1 to 0-pi
96  h(i,j,k) = h(i,j,k) + 10.0*0.5*(1.+cos(rad)) ! cosine bell
97  h(i,j,k-1) = h(i,j,k-1) - 10.0*0.5*(1.+cos(rad)) ! cosine bell
98  enddo ; enddo
99 
100 end subroutine circle_obcs_initialize_thickness
