87 subroutine step_forward_meke(MEKE, h, SN_u, SN_v, visc, dt, G, GV, CS, hu, hv)

89  type(ocean_grid_type), intent(inout) :: G !< Ocean grid.
90  type(verticalgrid_type), intent(in) :: GV !< Ocean vertical grid structure.
91  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h !< Layer thickness (m or kg m-2).
92  real, dimension(SZIB_(G),SZJ_(G)), intent(in) :: SN_u !< Eady growth rate at u-points (s-1).
93  real, dimension(SZI_(G),SZJB_(G)), intent(in) :: SN_v !< Eady growth rate at u-points (s-1).
94  type(vertvisc_type), intent(in) :: visc !< The vertical viscosity type.
95  real, intent(in) :: dt !< Model(baroclinic) time-step (s).
96  type(meke_cs), pointer :: CS !< MEKE control structure.
97  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in) :: hu !< Zonal flux flux (m3).
98  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(in) :: hv !< Meridional mass flux (m3).
99  ! Local variables
100  real, dimension(SZI_(G),SZJ_(G)) :: &
101  mass, & ! The total mass of the water column, in kg m-2.
102  I_mass, & ! The inverse of mass, in m2 kg-1.
103  src, & ! The sum of all MEKE sources, in m2 s-3.
104  MEKE_decay, & ! The MEKE decay timescale, in s-1.
105  MEKE_GM_src, & ! The MEKE source from thickness mixing, in m2 s-3.
106  MEKE_mom_src, & ! The MEKE source from momentum, in m2 s-3.
107  drag_rate_visc, &
108  drag_rate, & ! The MEKE spindown timescale due to bottom drag, in s-1.
109  LmixScale, & ! Square of eddy mixing length, in m2.
110  barotrFac2, & ! Ratio of EKE_barotropic / EKE (nondim)/
111  bottomFac2 ! Ratio of EKE_bottom / EKE (nondim)/
112  real, dimension(SZIB_(G),SZJ_(G)) :: &
113  MEKE_uflux, & ! The zonal diffusive flux of MEKE, in kg m2 s-3.
114  Kh_u, & ! The zonal diffusivity that is actually used, in m2 s-1.
115  baroHu, & ! Depth integrated zonal mass flux (m3).
116  drag_vel_u ! A (vertical) viscosity associated with bottom drag at
117  ! u-points, in m s-1.
118  real, dimension(SZIB_(G),SZJB_(G)) :: &
119  MEKE_vflux, & ! The meridional diffusive flux of MEKE, in kg m2 s-3.
120  Kh_v, & ! The meridional diffusivity that is actually used, in m2 s-1.
121  baroHv, & ! Depth integrated meridional mass flux (m3).
122  drag_vel_v ! A (vertical) viscosity associated with bottom drag at
123  ! v-points, in m s-1.
124  real :: Kh_here, Inv_Kh_max, K4_here
125  real :: cdrag2
126  real :: advFac
127  real :: mass_neglect ! A negligible mass, in kg m-2.
128  real :: ldamping ! The MEKE damping rate in s-1.
129  real :: Rho0 ! A density used to convert mass to distance, in kg m-3.
130  real :: sdt ! dt to use locally (could be scaled to accelerate)
131  real :: sdt_damp ! dt for damping (sdt could be split).
132  logical :: use_drag_rate ! Flag to indicate drag_rate is finite
133  integer :: i, j, k, is, ie, js, je, Isq, Ieq, Jsq, Jeq, nz
134 
135  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
136  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
137 
138  if (.not.associated(cs)) call mom_error(fatal, &
139  "MOM_MEKE: Module must be initialized before it is used.")
140  if (.not.associated(meke)) call mom_error(fatal, &
141  "MOM_MEKE: MEKE must be initialized before it is used.")
142 
143  rho0 = gv%H_to_kg_m2 * gv%m_to_H
144  mass_neglect = gv%H_to_kg_m2 * gv%H_subroundoff
145  sdt = dt*cs%MEKE_dtScale ! Scaled dt to use for time-stepping
146  if (cs%MEKE_damping + cs%MEKE_Cd_scale > 0.0 .or. cs%MEKE_Cb>0. &
147  .or. cs%visc_drag) then
148  use_drag_rate = .true.
149  else
150  use_drag_rate = .false.
151  endif
152 
153  ! Only integrate the MEKE equations if MEKE is required.
154  if (associated(meke%MEKE)) then
155 
156  if (cs%debug) then
157  if (associated(meke%mom_src)) call hchksum(meke%mom_src, 'MEKE mom_src',g)
158  if (associated(meke%GM_src)) call hchksum(meke%GM_src, 'MEKE GM_src',g)
159  if (associated(meke%MEKE)) call hchksum(meke%MEKE, 'MEKE MEKE',g)
160  endif
161 
162  ! Why are these 3 lines repeated from above?
163  sdt = dt*cs%MEKE_dtScale ! Scaled dt to use for time-stepping
164  rho0 = gv%H_to_kg_m2 * gv%m_to_H
165  mass_neglect = gv%H_to_kg_m2 * gv%H_subroundoff
166  cdrag2 = cs%cdrag**2
167 
168  ! With a depth-dependent (and possibly strong) damping, it seems
169  ! advisable to use Strang splitting between the damping and diffusion.
170  sdt_damp = sdt ; if (cs%MEKE_KH >= 0.0 .or. cs%MEKE_K4 >= 0.) sdt_damp = 0.5*sdt
171 
172  ! Calculate depth integrated mass flux if doing advection
173  if (cs%MEKE_advection_factor>0.) then
174  do j=js,je ; do i=is-1,ie
175  barohu(i,j) = 0.
176  enddo ; enddo
177  do k=1,nz
178  do j=js,je ; do i=is-1,ie
179  barohu(i,j) = hu(i,j,k)
180  enddo ; enddo
181  enddo
182  do j=js-1,je ; do i=is,ie
183  barohv(i,j) = 0.
184  enddo ; enddo
185  do k=1,nz
186  do j=js-1,je ; do i=is,ie
187  barohv(i,j) = hv(i,j,k)
188  enddo ; enddo
189  enddo
190  endif

192 !$OMP parallel default(none) shared(MEKE,CS,is,ie,js,je,nz,src,mass,G,GV,h,I_mass, &
193 !$OMP sdt,drag_vel_u,visc,drag_vel_v,drag_rate_visc, &
194 !$OMP drag_rate,Rho0,MEKE_decay,sdt_damp,cdrag2, &
195 !$OMP bottomFac2) &
196 !$OMP private(ldamping)
197 
198  if (cs%MEKE_Cd_scale == 0.0 .and. .not. cs%visc_drag) then
199 !$OMP do
200  do j=js,je ; do i=is,ie
201  drag_rate(i,j) = 0.
202  enddo ; enddo
203  endif
204 
205  ! Calculate drag_rate_visc(i,j) which accounts for the model bottom mean flow
206  if (cs%visc_drag) then
207 !$OMP do
208  do j=js,je ; do i=is-1,ie
209  drag_vel_u(i,j) = 0.0
210  if ((g%mask2dCu(i,j) > 0.0) .and. (visc%bbl_thick_u(i,j) > 0.0)) &
211  drag_vel_u(i,j) = visc%kv_bbl_u(i,j) / visc%bbl_thick_u(i,j)
212  enddo ; enddo
213 !$OMP do
214  do j=js-1,je ; do i=is,ie
215  drag_vel_v(i,j) = 0.0
216  if ((g%mask2dCu(i,j) > 0.0) .and. (visc%bbl_thick_v(i,j) > 0.0)) &
217  drag_vel_v(i,j) = visc%kv_bbl_v(i,j) / visc%bbl_thick_v(i,j)
218  enddo ; enddo
219 
220 !$OMP do
221  do j=js,je ; do i=is,ie
222  drag_rate_visc(i,j) = (0.25*g%IareaT(i,j) * &
223  ((g%areaCu(i-1,j)*drag_vel_u(i-1,j) + &
224  g%areaCu(i,j)*drag_vel_u(i,j)) + &
225  (g%areaCv(i,j-1)*drag_vel_v(i,j-1) + &
226  g%areaCv(i,j)*drag_vel_v(i,j)) ) )
227  enddo ; enddo
228  else
229 !$OMP do
230  do j=js,je ; do i=is,ie
231  drag_rate_visc(i,j) = 0.
232  enddo ; enddo
233  endif
234 
235 !$OMP do
236  do j=js-1,je+1
237  do i=is-1,ie+1 ; mass(i,j) = 0.0 ; enddo
238  do k=1,nz ; do i=is-1,ie+1
239  mass(i,j) = mass(i,j) + gv%H_to_kg_m2 * h(i,j,k)
240  enddo ; enddo
241  do i=is-1,ie+1
242  i_mass(i,j) = 0.0
243  if (mass(i,j) > 0.0) i_mass(i,j) = 1.0 / mass(i,j)
244  enddo
245  enddo
246 !$OMP end parallel
247 
248  if (cs%initialize) then
249  call meke_equilibrium(cs, meke, g, gv, sn_u, sn_v, drag_rate_visc, i_mass)
250  cs%initialize = .false.
251  endif
252 
253  ! Calculates bottomFac2, barotrFac2 and LmixScale
254  call meke_lengthscales(cs, meke, g, sn_u, sn_v, meke%MEKE, bottomfac2, barotrfac2, lmixscale)
255 
256 !$OMP parallel default(none) shared(MEKE,CS,is,ie,js,je,nz,src,mass,G,h,I_mass, &
257 !$OMP sdt,drag_vel_u,visc,drag_vel_v,drag_rate_visc, &
258 !$OMP drag_rate,Rho0,MEKE_decay,sdt_damp,cdrag2, &
259 !$OMP bottomFac2,barotrFac2,use_drag_rate) &
260 !$OMP private(ldamping)
261 
262  ! Aggregate sources of MEKE (background, frictional and GM)
263 !$OMP do
264  do j=js,je ; do i=is,ie
265  src(i,j) = cs%MEKE_BGsrc
266  enddo ; enddo
267 
268  if (associated(meke%mom_src)) then
269 !$OMP do
270  do j=js,je ; do i=is,ie
271  src(i,j) = src(i,j) - cs%MEKE_FrCoeff*i_mass(i,j)*meke%mom_src(i,j)
272  enddo ; enddo
273  endif
274 
275  if (associated(meke%GM_src)) then
276 !$OMP do
277  do j=js,je ; do i=is,ie
278  src(i,j) = src(i,j) - cs%MEKE_GMcoeff*i_mass(i,j)*meke%GM_src(i,j)
279  enddo ; enddo
280  endif
281 
282  ! Increase EKE by a full time-steps worth of source
283 !$OMP do
284  do j=js,je ; do i=is,ie
285  meke%MEKE(i,j) = (meke%MEKE(i,j) + sdt*src(i,j) )*g%mask2dT(i,j)
286  enddo ; enddo
287 
288  if (use_drag_rate) then
289  ! Calculate a viscous drag rate (includes BBL contributions from mean flow and eddies)
290 !$OMP do
291  do j=js,je ; do i=is,ie
292  drag_rate(i,j) = (rho0 * i_mass(i,j)) * sqrt( drag_rate_visc(i,j)**2 &
293  + cdrag2 * ( max(0.0, 2.0*bottomfac2(i,j)*meke%MEKE(i,j)) + cs%MEKE_Uscale**2 ) )
294  enddo ; enddo
295  endif
296 
297  ! First stage of Strang splitting
298 !$OMP do
299  do j=js,je ; do i=is,ie
300  ldamping = cs%MEKE_damping + drag_rate(i,j) * bottomfac2(i,j)
301  if (meke%MEKE(i,j)<0.) ldamping = 0.
302  ! notice that the above line ensures a damping only if MEKE is positive,
303  ! while leaving MEKE unchanged if it is negative
304  meke%MEKE(i,j) = meke%MEKE(i,j) / (1.0 + sdt_damp*ldamping)
305  meke_decay(i,j) = ldamping*g%mask2dT(i,j)
306  enddo ; enddo
307 !$OMP end parallel
308 
309  if (cs%MEKE_KH >= 0.0 .or. cs%MEKE_K4 >= 0.0) then
310  ! Update halos for lateral or bi-harmonic diffusion
311  call cpu_clock_begin(cs%id_clock_pass)
312  call do_group_pass(cs%pass_MEKE, g%Domain)
313  call cpu_clock_end(cs%id_clock_pass)
314  endif
315 
316  if (cs%MEKE_K4 >= 0.0) then
317  ! Calculate Laplacian of MEKE
318 !$OMP parallel default(none) shared(is,ie,js,je,MEKE_uflux,G,MEKE,MEKE_vflux,CS)
319 !$OMP do
320  do j=js,je ; do i=is-1,ie
321  meke_uflux(i,j) = ((g%dy_Cu(i,j)*g%IdxCu(i,j)) * g%mask2dCu(i,j)) * &
322  (meke%MEKE(i+1,j) - meke%MEKE(i,j))
323  ! MEKE_uflux(I,j) = ((G%dy_Cu(I,j)*G%IdxCu(I,j)) * &
324  ! ((2.0*mass(i,j)*mass(i+1,j)) / ((mass(i,j)+mass(i+1,j)) + mass_neglect)) ) * &
325  ! (MEKE%MEKE(i+1,j) - MEKE%MEKE(i,j))
326  enddo ; enddo
327 !$OMP do
328  do j=js-1,je ; do i=is,ie
329  meke_vflux(i,j) = ((g%dx_Cv(i,j)*g%IdyCv(i,j)) * g%mask2dCv(i,j)) * &
330  (meke%MEKE(i,j+1) - meke%MEKE(i,j))
331  ! MEKE_vflux(i,J) = ((G%dx_Cv(i,J)*G%IdyCv(i,J)) * &
332  ! ((2.0*mass(i,j)*mass(i,j+1)) / ((mass(i,j)+mass(i,j+1)) + mass_neglect)) ) * &
333  ! (MEKE%MEKE(i,j+1) - MEKE%MEKE(i,j))
334  enddo ; enddo
335 !$OMP do
336  do j=js,je ; do i=is,ie
337  cs%del2MEKE(i,j) = g%IareaT(i,j) * &
338  ((meke_uflux(i,j) - meke_uflux(i-1,j)) + (meke_vflux(i,j) - meke_vflux(i,j-1)))
339  ! CS%del2MEKE(i,j) = (G%IareaT(i,j)*I_mass(i,j)) * &
340  ! ((MEKE_uflux(I,j) - MEKE_uflux(I-1,j)) + (MEKE_vflux(i,J) - MEKE_vflux(i,J-1)))
341  enddo ; enddo
342 !$OMP end parallel
343  call cpu_clock_begin(cs%id_clock_pass)
344  call do_group_pass(cs%pass_del2MEKE, g%Domain)
345  call cpu_clock_end(cs%id_clock_pass)
346 
347  ! Bi-harmonic diffusion of MEKE
348 !$OMP parallel default(none) shared(is,ie,js,je,MEKE_uflux,G,CS,sdt,mass, &
349 !$OMP mass_neglect,MEKE_vflux,I_mass) &
350 !$OMP private(K4_here,Inv_Kh_max)
351 !$OMP do
352  do j=js,je ; do i=is-1,ie
353  k4_here = cs%MEKE_K4
354  ! Limit Kh to avoid CFL violations.
355  inv_kh_max = 64.0*sdt * (((g%dy_Cu(i,j)*g%IdxCu(i,j)) * &
356  max(g%IareaT(i,j),g%IareaT(i+1,j))))**2.0
357  if (k4_here*inv_kh_max > 0.3) k4_here = 0.3 / inv_kh_max
358 
359  meke_uflux(i,j) = ((k4_here * (g%dy_Cu(i,j)*g%IdxCu(i,j))) * &
360  ((2.0*mass(i,j)*mass(i+1,j)) / ((mass(i,j)+mass(i+1,j)) + mass_neglect)) ) * &
361  (cs%del2MEKE(i+1,j) - cs%del2MEKE(i,j))
362  enddo ; enddo
363 !$OMP do
364  do j=js-1,je ; do i=is,ie
365  k4_here = cs%MEKE_K4
366  inv_kh_max = 64.0*sdt * (((g%dx_Cv(i,j)*g%IdyCv(i,j)) * &
367  max(g%IareaT(i,j),g%IareaT(i,j+1))))**2.0
368  if (k4_here*inv_kh_max > 0.3) k4_here = 0.3 / inv_kh_max
369 
370  meke_vflux(i,j) = ((k4_here * (g%dx_Cv(i,j)*g%IdyCv(i,j))) * &
371  ((2.0*mass(i,j)*mass(i,j+1)) / ((mass(i,j)+mass(i,j+1)) + mass_neglect)) ) * &
372  (cs%del2MEKE(i,j+1) - cs%del2MEKE(i,j))
373  enddo ; enddo
374 !$OMP do
375  ! Store tendency of bi-harmonic in del2MEKE
376  do j=js,je ; do i=is,ie
377  cs%del2MEKE(i,j) = (sdt*(g%IareaT(i,j)*i_mass(i,j))) * &
378  ((meke_uflux(i-1,j) - meke_uflux(i,j)) + &
379  (meke_vflux(i,j-1) - meke_vflux(i,j)))
380  enddo ; enddo
381 !$OMP end parallel
382  endif !
383 
384 !$OMP parallel default(none) shared(is,ie,js,je,MEKE,CS,sdt,G,Kh_u,MEKE_uflux, &
385 !$OMP mass,mass_neglect,Kh_v,MEKE_vflux,I_mass, &
386 !$OMP sdt_damp,drag_rate,Rho0,drag_rate_visc, &
387 !$OMP cdrag2,bottomFac2,MEKE_decay,barotrFac2, &
388 !$OMP use_drag_rate,dt,baroHu,baroHv) &
389 !$OMP private(Kh_here,Inv_Kh_max,ldamping,advFac)
390  if (cs%MEKE_KH >= 0.0 .or. cs%MEKE_advection_factor >0.) then
391  ! Lateral diffusion of MEKE
392  kh_here = max(0.,cs%MEKE_Kh)
393 !$OMP do
394  do j=js,je ; do i=is-1,ie
395  ! Limit Kh to avoid CFL violations.
396  if (associated(meke%Kh)) &
397  kh_here = max(0.,cs%MEKE_Kh) + cs%KhMEKE_Fac*0.5*(meke%Kh(i,j)+meke%Kh(i+1,j))
398  inv_kh_max = 2.0*sdt * ((g%dy_Cu(i,j)*g%IdxCu(i,j)) * &
399  max(g%IareaT(i,j),g%IareaT(i+1,j)))
400  if (kh_here*inv_kh_max > 0.25) kh_here = 0.25 / inv_kh_max
401  kh_u(i,j) = kh_here
402

404  ((2.0*mass(i,j)*mass(i+1,j)) / ((mass(i,j)+mass(i+1,j)) + mass_neglect)) ) * &
405  (meke%MEKE(i,j) - meke%MEKE(i+1,j))
406  enddo ; enddo

408  do j=js-1,je ; do i=is,ie

411  inv_kh_max = 2.0*sdt * ((g%dx_Cv(i,j)*g%IdyCv(i,j)) * &
412  max(g%IareaT(i,j),g%IareaT(i,j+1)))

414  kh_v(i,j) = kh_here

417  ((2.0*mass(i,j)*mass(i,j+1)) / ((mass(i,j)+mass(i,j+1)) + mass_neglect)) ) * &
418  (meke%MEKE(i,j) - meke%MEKE(i,j+1))

426  elseif (barohu(i,j)<0.) then
427  meke_uflux(i,j) = meke_uflux(i,j) + barohu(i,j)*meke%MEKE(i+1,j)*advfac

433  meke_vflux(i,j) = meke_vflux(i,j) + barohv(i,j)*meke%MEKE(i,j)*advfac
434  elseif (barohv(i,j)<0.) then
435  meke_vflux(i,j) = meke_vflux(i,j) + barohv(i,j)*meke%MEKE(i,j+1)*advfac
436  endif
437  enddo ; enddo
438  endif
439 !$OMP do
440  do j=js,je ; do i=is,ie
441  meke%MEKE(i,j) = meke%MEKE(i,j) + (sdt*(g%IareaT(i,j)*i_mass(i,j))) * &

452  enddo ; enddo
453  endif

461  do j=js,je ; do i=is,ie
462  drag_rate(i,j) = (rho0 * i_mass(i,j)) * sqrt( drag_rate_visc(i,j)**2 &
463  + cdrag2 * ( max(0.0, 2.0*bottomfac2(i,j)*meke%MEKE(i,j)) + cs%MEKE_Uscale**2 ) )

475  endif
476  endif ! MEKE_KH>=0

491  * min(meke%Rd_dx_h(i,j), 1.0)

496  meke%Kh(i,j) = cs%MEKE_KhCoeff*sqrt(2.*max(0.,barotrfac2(i,j)*meke%MEKE(i,j))*g%areaT(i,j))
497  enddo ; enddo
498  endif
499  else

502  meke%Kh(i,j) = (cs%MEKE_KhCoeff*sqrt(2.*max(0.,barotrfac2(i,j)*meke%MEKE(i,j)))*lmixscale(i,j))

544 !> Calculates the equilibrium solutino where the source depends only on MEKE diffusivity
545 !! and there is no lateral diffusion of MEKE.
546 !! Results is in MEKE%MEKE.
547 subroutine meke_equilibrium(CS, MEKE, G, GV, SN_u, SN_v, drag_rate_visc, I_mass)
548  type(ocean_grid_type), intent(inout) :: G !< Ocean grid.
549  type(verticalgrid_type), intent(in) :: GV !< Ocean vertical grid structure.
550  type(meke_cs), pointer :: CS !< MEKE control structure.
551  type(meke_type), pointer :: MEKE !< MEKE data.
552  real, dimension(SZIB_(G),SZJ_(G)), intent(in) :: SN_u !< Eady growth rate at u-points (s-1).
553  real, dimension(SZI_(G),SZJB_(G)), intent(in) :: SN_v !< Eady growth rate at u-points (s-1).
554  real, dimension(SZI_(G),SZJ_(G)), intent(in) :: drag_rate_visc !< Mean flow contrib. to drag rate
555  real, dimension(SZI_(G),SZJ_(G)), intent(in) :: I_mass !< Inverse of column mass.
556  ! Local variables
557  real :: beta, SN, bottomFac2, barotrFac2, LmixScale, Lrhines, Leady
558  real :: I_H, KhCoeff, Kh, Ubg2, cd2, drag_rate, ldamping, src
559  real :: EKE, EKEmin, EKEmax, resid, ResMin, ResMax, EKEerr
560  integer :: i, j, is, ie, js, je, n1, n2
561  real, parameter :: tolerance = 1.e-12 ! Width of EKE bracket in m^2 s^-2.
562  logical :: useSecant, debugIteration
563 
564  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
565 
566  debugiteration = .false.
567  khcoeff = cs%MEKE_KhCoeff
568  ubg2 = cs%MEKE_Uscale**2
569  cd2 = cs%cdrag**2
570 
571 !$OMP do
572  do j=js,je ; do i=is,ie
573  !SN = 0.25*max( (SN_u(I,j) + SN_u(I-1,j)) + (SN_v(i,J) + SN_v(i,J-1)), 0.)
574  ! This avoids extremes values in equilibrium solution due to bad values in SN_u, SN_v
575  sn = min( min(sn_u(i,j) , sn_u(i-1,j)) , min(sn_v(i,j), sn_v(i,j-1)) )
576  beta = sqrt( g%dF_dx(i,j)**2 + g%dF_dy(i,j)**2 )
577  i_h = gv%Rho0 * i_mass(i,j)
578 
579  if (khcoeff*sn*i_h>0.) then
580  ! Solve resid(E) = 0, where resid = Kh(E) * (SN)^2 - damp_rate(E) E
581  ekemin = 0. ! Use the trivial root as the left bracket
582  resmin = 0. ! Need to detect direction of left residual
583  ekemax = 0.01 ! First guess at right bracket
584  usesecant = .false. ! Start using a bisection method
585 
586  ! First find right bracket for which resid<0
587  resid = 1. ; n1 = 0
588  do while (resid>0.)
589  n1 = n1 + 1
590  eke = ekemax
591  call meke_lengthscales_0d(cs, g%areaT(i,j), beta, g%bathyT(i,j), &
592  meke%Rd_dx_h(i,j), sn, eke, &
593  bottomfac2, barotrfac2, lmixscale, &
594  lrhines, leady)
595  ! TODO: Should include resolution function in Kh
596  kh = (khcoeff * sqrt(2.*barotrfac2*eke) * lmixscale)
597  src = kh * (sn * sn)
598  drag_rate = i_h * sqrt( drag_rate_visc(i,j)**2 + cd2 * ( 2.0*bottomfac2*eke + ubg2 ) )
599  ldamping = cs%MEKE_damping + drag_rate * bottomfac2
600  resid = src - ldamping * eke
601  if (debugiteration) then
602  write(0,*) n1, 'EKE=',eke,'resid=',resid
603  write(0,*) 'EKEmin=',ekemin,'ResMin=',resmin
604  write(0,*) 'src=',src,'ldamping=',ldamping
605  write(0,*) 'gamma-b=',bottomfac2,'gamma-t=',barotrfac2
606  write(0,*) 'drag_visc=',drag_rate_visc(i,j),'Ubg2=',ubg2
607  endif
608  if (resid>0.) then ! EKE is to the left of the root
609  ekemin = eke ! so we move the left bracket here
610  ekemax = 10. * eke ! and guess again for the right bracket
611  if (resid<resmin) usesecant = .true.
612  resmin = resid
613  if (ekemax > 2.e17) then
614  if (debugiteration) stop 'Something has gone very wrong'
615  debugiteration = .true.
616  resid = 1. ; n1 = 0
617  ekemin = 0. ; resmin = 0.
618  ekemax = 0.01
619  usesecant = .false.
620  endif
621  endif
622  enddo ! while(resid>0.) searching for right bracket
623  resmax = resid
624 
625  ! Bisect the bracket
626  n2 = 0 ; ekeerr = ekemax - ekemin
627  do while (ekeerr>tolerance)
628  n2 = n2 + 1
629  if (usesecant) then
630  eke = ekemin + (ekemax - ekemin) * (resmin / (resmin - resmax))
631  else
632  eke = 0.5 * (ekemin + ekemax)
633  endif
634  ekeerr = min( eke-ekemin, ekemax-eke )
635  ! TODO: Should include resolution function in Kh
636  kh = (khcoeff * sqrt(2.*barotrfac2*eke) * lmixscale)
637  src = kh * (sn * sn)
638  drag_rate = i_h * sqrt( drag_rate_visc(i,j)**2 + cd2 * ( 2.0*bottomfac2*eke + ubg2 ) )
639  ldamping = cs%MEKE_damping + drag_rate * bottomfac2
640  resid = src - ldamping * eke
641  if (usesecant.and.resid>resmin) usesecant = .false.
642  if (resid>0.) then ! EKE is to the left of the root
643  ekemin = eke ! so we move the left bracket here
644  if (resid<resmin) usesecant = .true.
645  resmin = resid ! Save this for the secant method
646  elseif (resid<0.) then ! EKE is to the right of the root
647  ekemax = eke ! so we move the right bracket here
648  resmax = resid ! Save this for the secant method
649  else
650  exit ! resid=0 => EKE is exactly at the root
651  endif
652  if (n2>200) stop 'Failing to converge?'
653  enddo ! while(EKEmax-EKEmin>tolerance)
654 
655  else
656  eke = 0.
657  endif
658  meke%MEKE(i,j) = eke
659  enddo ; enddo
660 
661 end subroutine meke_equilibrium

706 !> Calculates the eddy mixing length scale and \f$\gamma_b\f$ and \f$\gamma_t\f$
707 !! functions that are ratios of either bottom or barotropic eddy energy to the
708 !! column eddy energy, respectively. See \ref section_MEKE_equations.
709 subroutine meke_lengthscales_0d(CS, area, beta, depth, Rd_dx, SN, &
710  eke, bottomfac2, barotrfac2, lmixscale, lrhines, leady)
711  type(meke_cs), pointer :: CS !< MEKE control structure.
712  real, intent(in) :: area !< Grid cell area (m2)
713  real, intent(in) :: beta !< Planetary beta = |grad F| (s-1 m-1)
714  real, intent(in) :: depth !< Ocean depth (m)
715  real, intent(in) :: Rd_dx !< Resolution Ld/dx (nondim).
716  real, intent(in) :: SN !< Eady growth rate (s-1).
717  real, intent(in) :: EKE !< Eddy kinetic energy (m s-1).
718  real, intent(out) :: bottomFac2 !< gamma_b^2
719  real, intent(out) :: barotrFac2 !< gamma_t^2
720  real, intent(out) :: LmixScale !< Eddy mixing length (m).
721  real, intent(out) :: Lrhines !< Rhines length scale (m).
722  real, intent(out) :: Leady !< Eady length scale (m).
723  ! Local variables
724  real :: Lgrid, Ldeform, LdeformLim, Ue, Lfrict
725 
726  ! Length scale for MEKE derived diffusivity
727  lgrid = sqrt(area) ! Grid scale
728  ldeform = lgrid * rd_dx ! Deformation scale
729  lfrict = depth / cs%cdrag ! Frictional arrest scale
730  ! gamma_b^2 is the ratio of bottom eddy energy to mean column eddy energy
731  ! used in calculating bottom drag
732  bottomfac2 = cs%MEKE_CD_SCALE**2
733  if (lfrict*cs%MEKE_Cb>0.) bottomfac2 = bottomfac2 + 1./( 1. + cs%MEKE_Cb*(ldeform/lfrict) )**0.8
734  bottomfac2 = max(bottomfac2, cs%MEKE_min_gamma)
735  ! gamma_t^2 is the ratio of barotropic eddy energy to mean column eddy energy
736  ! used in the velocity scale for diffusivity
737  barotrfac2 = 1.
738  if (lfrict*cs%MEKE_Ct>0.) barotrfac2 = 1./( 1. + cs%MEKE_Ct*(ldeform/lfrict) )**0.25
739  barotrfac2 = max(barotrfac2, cs%MEKE_min_gamma)
740  if (cs%use_old_lscale) then
741  if (cs%Rd_as_max_scale) then
742  lmixscale = min(ldeform, lgrid) ! The smaller of Ld or dx
743  else
744  lmixscale = lgrid
745  endif
746  else
747  ue = sqrt( 2.0 * max( 0., barotrfac2*eke ) ) ! Barotropic eddy flow scale
748  lrhines = sqrt( ue / max( beta, 1.e-30 ) ) ! Rhines scale
749  if (cs%aEady > 0.) then
750  leady = ue / max( sn, 1.e-15 ) ! Bound Eady time-scale < 1e15 seconds
751  else
752  leady = 0.
753  endif
754  lmixscale = 0.
755  if (cs%aDeform*ldeform > 0.) lmixscale = lmixscale + 1./(cs%aDeform*ldeform)
756  if (cs%aFrict *lfrict > 0.) lmixscale = lmixscale + 1./(cs%aFrict *lfrict)
757  if (cs%aRhines*lrhines > 0.) lmixscale = lmixscale + 1./(cs%aRhines*lrhines)
758  if (cs%aEady *leady > 0.) lmixscale = lmixscale + 1./(cs%aEady *leady)
759  if (cs%aGrid *lgrid > 0.) lmixscale = lmixscale + 1./(cs%aGrid *lgrid)
760  if (cs%Lfixed > 0.) lmixscale = lmixscale + 1./cs%Lfixed
761  if (lmixscale > 0.) lmixscale = 1. / lmixscale
762  endif
763 
764 end subroutine meke_lengthscales_0d

1101 !> \namespace mom_meke
1102 !!
1103 !! \section section_MEKE The Mesoscale Eddy Kinetic Energy (MEKE) framework
1104 !!
1105 !! The MEKE framework accounts for the mean potential energy removed by
1106 !! the first order closures used to parameterize mesoscale eddies.
1107 !! It requires closure at the second order, namely dissipation and transport
1108 !! of eddy energy.
1109 !!
1110 !! Monitoring the sub-grid scale eddy energy budget provides a means to predict
1111 !! a sub-grid eddy-velocity scale which can be used in the lower order closures.
1112 !!
1113 !! \subsection section_MEKE_equations MEKE equations
1114 !!
1115 !! The eddy kinetic energy equation is:
1116 !! \f[ \partial_\tilde{t} E =
1117 !! \overbrace{ \dot{E}_b + \gamma_\eta \dot{E}_\eta + \gamma_v \dot{E}_v
1118 !! }^\text{sources}
1119 !! - \overbrace{ ( \lambda + C_d | U_d | \gamma_b^2 ) E
1120 !! }^\text{local dissipation}
1121 !! + \overbrace{ \nabla \cdot ( ( \kappa_E + \gamma_M \kappa_M ) \nabla E
1122 !! - \kappa_4 \nabla^3 E )
1123 !! }^\text{smoothing}
1124 !! \f]
1125 !! where \f$ E \f$ is the eddy kinetic energy (variable <code>MEKE</code>) with units of
1126 !! m<sup>2</sup>s<sup>-2</sup>,
1127 !! and \f$\tilde{t} = a t\f$ is a scaled time. The non-dimensional factor
1128 !! \f$ a\geq 1 \f$ is used to accelerate towards equilibrium.
1129 !!
1130 !! The MEKE equation is two-dimensional and obtained by depth averaging the
1131 !! the three-dimensional eddy energy equation. In the following expressions
1132 !! \f$ \left< \phi \right> = \frac{1}{H} \int^\eta_{-D} \phi \, dz \f$ maps
1133 !! three dimensional terms into the two-dimensional quantities needed.
1134 !!
1135 !! \subsubsection section_MEKE_source_terms MEKE source terms
1136 !!
1137 !! The source term \f$ \dot{E}_b \f$ is a constant background source
1138 !! of energy intended to avoid the limit \f$E\rightarrow 0\f$.
1139 !!
1140 !! The "GM" source term
1141 !! \f[ \dot{E}_\eta = - \left< \overline{w^\prime b^\prime} \right>
1142 !! = \left< \kappa_h N^2S^2 \right>
1143 !! \approx \left< \kappa_h g\prime |\nabla_\sigma \eta|^2 \right>\f]
1144 !! equals the mean potential energy removed by the Gent-McWilliams closure,
1145 !! and is excluded/included in the MEKE budget by the efficiency parameter
1146 !! \f$ \gamma_\eta \in [0,1] \f$.
1147 !!
1148 !! The "frictional" source term
1149 !! \f[ \dot{E}_{v} = \left< u \cdot \tau_h \right> \f]
1150 !! equals the mean kinetic energy removed by lateral viscous fluxes, and
1151 !! is excluded/included in the MEKE budget by the efficiency parameter
1152 !! \f$ \gamma_v \in [0,1] \f$.
1153 !!
1154 !! \subsubsection section_MEKE_dissipation_terms MEKE dissipation terms
1155 !!
1156 !! The local dissipation of \f$ E \f$ is parameterized through a linear
1157 !! damping, \f$\lambda\f$, and bottom drag, \f$ C_d | U_d | \gamma_b^2 \f$.
1158 !! The \f$ \gamma_b \f$ accounts for the weak projection of the column-mean
1159 !! eddy velocty to the bottom. In other words, the bottom velocity is
1160 !! estimated as \f$ \gamma_b U_e \f$.
1161 !! The bottom drag coefficient, \f$ C_d \f$ is the same as that used in the bottom
1162 !! friction in the mean model equations.
1163 !!
1164 !! The bottom drag velocity scale, \f$ U_d \f$, has contributions from the
1165 !! resolved state and \f$ E \f$:
1166 !! \f[ U_d = \sqrt{ U_b^2 + |u|^2_{z=-D} + |\gamma_b U_e|^2 } .\f]
1167 !! where the eddy velocity scale, \f$ U_e \f$, is given by:
1168 !! \f[ U_e = \sqrt{ 2 E } .\f]
1169 !! \f$ U_b \f$ is a constant background bottom velocity scale and is
1170 !! typically not used (i.e. set to zero).
1171 !!
1172 !! Following Jansen et al., 2015, the projection of eddy energy on to the bottom
1173 !! is given by the ratio of bottom energy to column mean energy:
1174 !! \f[
1175 !! \gamma_b^2 = \frac{E_b}{E} = \gamma_{d0}
1176 !! + \left( 1 + c_{b} \frac{L_d}{L_f} \right)^{-\frac{4}{5}}
1177 !! \f]
1178 !!
1179 !! \subsection section_MEKE_smoothing MEKE smoothing terms
1180 !!
1181 !! \f$ E \f$ is laterally diffused by a diffusivity \f$ \kappa_E + \gamma_M
