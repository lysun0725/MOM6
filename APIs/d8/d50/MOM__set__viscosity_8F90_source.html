27 !* This file contains the subroutine that calculates various values *
28 !* related to the bottom boundary layer, such as the viscosity and *
29 !* thickness of the BBL (set_viscous_BBL). This would also be the *
30 !* module in which other viscous quantities that are flow-independent *
31 !* might be set. This information is transmitted to other modules *
32 !* via a vertvisc type structure. *
33 !* *
34 !* The same code is used for the two velocity components, by *
35 !* indirectly referencing the velocities and defining a handful of *
36 !* direction-specific defined variables. *
37 !* *
38 !* Macros written all in capital letters are defined in MOM_memory.h. *
39 !* *
40 !* A small fragment of the grid is shown below: *
41 !* *
42 !* j+1 x ^ x ^ x At x: q *
43 !* j+1 > o > o > At ^: v, frhatv, tauy *
44 !* j x ^ x ^ x At >: u, frhatu, taux *
45 !* j > o > o > At o: h *
46 !* j-1 x ^ x ^ x *
47 !* i-1 i i+1 At x & ^: *
48 !* i i+1 At > & o: *
49 !* *
50 !* The boundaries always run through q grid points (x). *

143 ! The following subroutine calculates the thickness of the bottom
144 ! boundary layer and the viscosity within that layer. A drag law is
145 ! used, either linearized about an assumed bottom velocity or using
146 ! the actual near-bottom velocities combined with an assumed
147 ! unresolved velocity. The bottom boundary layer thickness is
148 ! limited by a combination of stratification and rotation, as in the
149 ! paper of Killworth and Edwards, JPO 1999. It is not necessary to
150 ! calculate the thickness and viscosity every time step; instead
151 ! previous values may be used.
152 !
153 ! Arguments: u - Zonal velocity, in m s-1.
154 ! (in) v - Meridional velocity, in m s-1.
155 ! (in) h - Layer thickness, in m or kg m-2. In the comments below,
156 ! the units of h are denoted as H.
157 ! (in) tv - A structure containing pointers to any available
158 ! thermodynamic fields. Absent fields have NULL ptrs.
159 ! (out) visc - A structure containing vertical viscosities and related
160 ! fields.
161 ! (in) G - The ocean's grid structure.
162 ! (in) GV - The ocean's vertical grid structure.
163 ! (in) CS - The control structure returned by a previous call to
164 ! vertvisc_init.

166  ustar, & ! The bottom friction velocity, in m s-1.
167  T_EOS, & ! The temperature used to calculate the partial derivatives
168  ! of density with T and S, in deg C.
169  s_eos, & ! The salinity used to calculate the partial derivatives
170  ! of density with T and S, in PSU.
171  dr_dt, & ! Partial derivative of the density in the bottom boundary
172  ! layer with temperature, in units of kg m-3 K-1.
173  dr_ds, & ! Partial derivative of the density in the bottom boundary
174  ! layer with salinity, in units of kg m-3 psu-1.
175  press ! The pressure at which dR_dT and dR_dS are evaluated, in Pa.

177  ! point, in H (i.e., m or kg m-2).
178  real :: htot_vel ! Sum of the layer thicknesses up to some
179  ! point, in H (i.e., m or kg m-2).
180 
181  real :: Rhtot ! Running sum of thicknesses times the
182  ! layer potential densities in H kg m-3.

184  ! using an upwind-biased second order
185  ! accurate estimate based on the previous
186  ! velocity direction, in H.

188  ! in H.

190  ! Rho0 divided by G_Earth and the conversion

197  ! the layer, in H kg m-3.
198  real :: Dh ! The increment in layer thickness from
199  ! the present layer, in H.
200  real :: bbl_thick ! The thickness of the bottom boundary layer in m.
201  real :: Rl ! The potential density of the current layer and the
202  real :: Rla ! layer above, in kg m-3.
203  real :: TLay, Tabove ! The temperature and salinity of the current layer
204  real :: SLay, Sabove ! and the layer above, in deg C and PSU.
205  real :: C2f ! C2f = 2*f at velocity points.
206 
207  real :: U_bg_sq ! The square of an assumed background
208  ! velocity, for calculating the mean
209  ! magnitude near the bottom for use in the
210  ! quadratic bottom drag, in m2.
211  real :: hwtot ! Sum of the thicknesses used to calculate
212  ! the near-bottom velocity magnitude, in H.
213  real :: hutot ! Running sum of thicknesses times the
214  ! velocity magnitudes, in H m s-1.
215  real :: Thtot ! Running sum of thickness times temperature, in H C.
216  real :: Shtot ! Running sum of thickness times salinity, in H psu.
217  real :: hweight ! The thickness of a layer that is within Hbbl
218  ! of the bottom, in H.
219  real :: v_at_u, u_at_v ! v at a u point or vice versa, m s-1.
220  real :: Rho0x400_G ! 400*Rho0/G_Earth, in kg s2 m-4. The 400 is a
221  ! constant proposed by Killworth and Edwards, 1999.
222  real, dimension(SZI_(G),SZJ_(G),max(GV%nk_rho_varies,1)) :: &
223  Rml ! The mixed layer coordinate density, in kg m-3.
224  real :: p_ref(szi_(g)) ! The pressure used to calculate the coordinate
225  ! density, in Pa (usually set to 2e7 Pa = 2000 dbar).
226

229  real :: Dp, Dm ! The depths at the edges of a velocity cell, in H.
230  real :: a ! a is the curvature of the bottom depth across a

233  real :: slope ! The absolute value of the bottom depth slope across

254  real :: L_direct ! The value of L above volume Vol_direct, nondim.

257  real :: Vol_err_min ! the correct value for L, in H.

263  real :: h_neglect ! A thickness that is so small it is usually lost
264  ! in roundoff and can be neglected, in H.
265  real :: ustH ! ustar converted to units of H s-1.
266  real :: root ! A temporary variable with units of H s-1.
267  real :: H_to_m, m_to_H ! Local copies of unit conversion factors.
268 
269  real :: Cell_width ! The transverse width of the velocity cell, in m.
270  real :: Rayleigh ! A nondimensional value that is multiplied by the
271  ! layer's velocity magnitude to give the Rayleigh
272  ! drag velocity.
273  real :: gam ! The ratio of the change in the open interface width
274  ! to the open interface width atop a cell, nondim.

276  ! viscous bottom boundary layer, nondim.
277  real :: BBL_visc_frac ! The fraction of all the drag that is expressed as
278  ! a viscous bottom boundary layer, nondim.

280  real :: C2pi_3 ! An irrational constant, 2/3 pi.
281  real :: tmp ! A temporary variable.
282  logical :: use_BBL_EOS, do_i(szib_(g))
283  integer :: i, j, k, is, ie, js, je, Isq, Ieq, Jsq, Jeq, nz, m, K2, nkmb, nkml
284  integer :: itt, maxitt=20
285  real :: tmp_val_m1_to_p1
286  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
287  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
288  nkmb = gv%nk_rho_varies ; nkml = gv%nkml
289  h_neglect = gv%H_subroundoff
290  rho0x400_g = 400.0*(gv%Rho0/g%g_Earth)*gv%m_to_H
291  vol_quit = 0.9*gv%Angstrom + h_neglect
292  h_to_m = gv%H_to_m ; m_to_h = gv%m_to_H
293  c2pi_3 = 8.0*atan(1.0)/3.0
294 
295  if (.not.associated(cs)) call mom_error(fatal,"MOM_vert_friction(BBL): "//&
296  "Module must be initialized before it is used.")
297  if (.not.cs%bottomdraglaw) return
298 
299  use_bbl_eos = associated(tv%eqn_of_state) .and. cs%BBL_use_EOS
300 
301  u_bg_sq = cs%drag_bg_vel * cs%drag_bg_vel
302  cdrag_sqrt=sqrt(cs%cdrag)
303  k2 = max(nkmb+1, 2)
304 
305 ! With a linear drag law, the friction velocity is already known.
306 ! if (CS%linear_drag) ustar(:) = cdrag_sqrt*CS%drag_bg_vel
307 
308  if ((nkml>0) .and. .not.use_bbl_eos) then
309  do i=g%IscB,g%IecB+1 ; p_ref(i) = tv%P_ref ; enddo
310 !$OMP parallel do default(none) shared(Jsq,Jeq,Isq,Ieq,nkmb,tv,p_ref,Rml)
311  do j=jsq,jeq+1 ; do k=1,nkmb
312  call calculate_density(tv%T(:,j,k), tv%S(:,j,k), p_ref, &
313  rml(:,j,k), isq, ieq-isq+2, tv%eqn_of_state)
314  enddo ; enddo

321 !$OMP private(do_i,h_at_vel,htot_vel,hwtot,hutot,Thtot,Shtot, &
322 !$OMP hweight,v_at_u,u_at_v,ustar,T_EOS,S_EOS,press, &

376

389  else
390  u_at_v = set_u_at_v(u, h, g, i, j, k)

392  u_at_v*u_at_v + u_bg_sq)
393  endif ; endif
394

397  shtot = shtot + hweight * 0.5 * (tv%S(i,j,k) + tv%S(i+1,j,k))
398  else
399  thtot = thtot + hweight * 0.5 * (tv%T(i,j,k) + tv%T(i,j+1,k))
400  shtot = shtot + hweight * 0.5 * (tv%S(i,j,k) + tv%S(i,j+1,k))
401  endif ; endif
402  enddo ! end of k loop
403

405  ustar(i) = cdrag_sqrt*hutot/hwtot
406  else
407  ustar(i) = cdrag_sqrt*cs%drag_bg_vel
408  endif
409 
410  if (use_bbl_eos) then ; if (hwtot > 0.0) then
411  t_eos(i) = thtot/hwtot ; s_eos(i) = shtot/hwtot
412  else
413  t_eos(i) = 0.0 ; s_eos(i) = 0.0
414  endif ; endif
415  endif ; enddo
416  else
417  do i=is,ie ; ustar(i) = cdrag_sqrt*cs%drag_bg_vel ; enddo
418  endif ! Not linear_drag
419

421  do i=is,ie
422  press(i) = 0.0 ! or = fluxes%p_surf(i,j)
423  if (.not.do_i(i)) then ; t_eos(i) = 0.0 ; s_eos(i) = 0.0 ; endif
424  enddo
425  if (m==1) then ; do k=1,nz ; do i=is,ie
426  press(i) = press(i) + gv%H_to_Pa * 0.5 * (h(i,j,k) + h(i+1,j,k))
427  enddo ; enddo ; else ; do k=1,nz ; do i=is,ie
428  press(i) = press(i) + gv%H_to_Pa * 0.5 * (h(i,j,k) + h(i,j+1,k))
429  enddo ; enddo ; endif

431  is-g%IsdB+1, ie-is+1, tv%eqn_of_state)
432  endif

443 ! 2000 (unpublished and lost manuscript).

459  sabove = 0.5 * (tv%S(i,j,k-1) + tv%S(i,j+1,k-1))

480  slay = 0.5 * (tv%S(i,j,1) + tv%S(i+1,j,1))
481  else ! Work on a v-point

498  else

541 ! equation as in Killworth and Edwards: (h/h_f)^2 + h/h_N = 1.

544  else ; c2f = (g%CoriolisBu(i-1,j)+g%CoriolisBu(i,j)) ; endif

557  ((ustar(i)*ustar(i)) * (m_to_h**2) )))

571  ! directly to the layers in question as a Rayleigh drag term.

580  tmp = g%mask2dCv(i+1,j) * 0.5*(g%bathyT(i+1,j) + g%bathyT(i+1,j+1))

582  tmp = g%mask2dCv(i,j-1) * 0.5*(g%bathyT(i-1,j) + g%bathyT(i-1,j+1))
583  dm = 2.0 * d_vel * tmp / (d_vel + tmp)
584  endif

593  ! a uniformly sloping or flat bottom.

596  ! given by D(x) = a*x^2 + b*x + D - a/12.

599  ! other volumes at which the equation that is solved for L changes.

603  else
604  tmp = slope/a
605  vol_open = 0.25*slope*tmp + c1_12*a
606  vol_2_reg = 0.5*tmp**2 * (a - c1_3*slope)
607  endif

622  l_direct = 1.0 + slope/a ! L_direct < 1 because a < 0
623  vol_direct = -c1_6*a*l_direct**3
624  endif
625  ibma_2 = 2.0 / (slope - a)
626  endif

633  else ; h_vel = 0.5*(h(i,j,k) + h(i,j+1,k)) ; endif

635  h_vel = h_vel + h_neglect

638  elseif (a == 0) then ! The bottom has no curvature.
639  l(k) = sqrt(2.0*vol*iapb)
640  elseif (a > 0) then
641  ! There may be a minimum depth, and there are
642  ! analytic expressions for L for all cases.
643  if (vol < vol_2_reg) then
644  ! In this case, there is a contiguous open region and
645  ! vol = 0.5*L^2*(slope + a/3*(3-4L)).
646  if (a2x48_apb3*vol < 1e-8) then ! Could be 1e-7?
647  ! There is a very good approximation here for massless layers.
648  l0 = sqrt(2.0*vol*iapb) ; l(k) = l0*(1.0 + ax2_3apb*l0)
649  else
650  l(k) = apb_4a * (1.0 - &
651  2.0 * cos(c1_3*acos(a2x48_apb3*vol - 1.0) - c2pi_3))
652  endif
653  ! To check the answers.
654  ! Vol_err = 0.5*(L(K)*L(K))*(slope + a_3*(3.0-4.0*L(K))) - vol
655  else ! There are two separate open regions.
656  ! vol = slope^2/4a + a/12 - (a/12)*(1-L)^2*(1+2L)
657  ! At the deepest volume, L = slope/a, at the top L = 1.
658  !L(K) = 0.5 - cos(C1_3*acos(1.0 - C24_a*(Vol_open - vol)) - C2pi_3)
659  tmp_val_m1_to_p1 = 1.0 - c24_a*(vol_open - vol)
660  tmp_val_m1_to_p1 = max(-1., min(1., tmp_val_m1_to_p1))
661  l(k) = 0.5 - cos(c1_3*acos(tmp_val_m1_to_p1) - c2pi_3)
662  ! To check the answers.
663  ! Vol_err = Vol_open - a_12*(1.0+2.0*L(K)) * (1.0-L(K))**2 - vol

666  if (vol <= vol_direct) then
667  ! Both edges of the cell are bounded by walls.
668  l(k) = (-0.25*c24_a*vol)**c1_3
669  else
670  ! x_R is at 1/2 but x_L is in the interior & L is found by solving
671  ! vol = 0.5*L^2*(slope + a/3*(3-4L))
672 
673  ! Vol_err = 0.5*(L(K+1)*L(K+1))*(slope + a_3*(3.0-4.0*L(K+1))) - vol_below
674  ! Change to ...
675  ! if (min(Vol_below + Vol_err, vol) <= Vol_direct) then ?
676  if (vol_below + vol_err <= vol_direct) then
677  l0 = l_direct ; vol_0 = vol_direct
678  else
679  l0 = l(k+1) ; vol_0 = vol_below + vol_err

681  endif
682 
683  ! Try a relatively simple solution that usually works well
684  ! for massless layers.
685  dv_dl2 = 0.5*(slope+a) - a*l0 ; dvol = (vol-vol_0)
686  ! dV_dL2 = 0.5*(slope+a) - a*L0 ; dVol = max(vol-Vol_0, 0.0)
687 
688  !### The following code is more robust when GV%Angstrom=0, but it
689  !### changes answers.
690  ! Vol_tol = max(0.5*GV%Angstrom + GV%H_subroundoff, 1e-14*vol)

694  ! L(K) = L0
695  ! Vol_err = 0.5*(L(K)*L(K))*(slope + a_3*(3.0-4.0*L(K))) - vol

697  ! (dV_dL2*Vol_tol - 2.0*a*L0*dVol)) then

699  (0.25*dv_dl2*gv%Angstrom - a*l0*dvol)) then
700  ! One iteration of Newton's method should give an estimate
701  ! that is accurate to within Vol_tol.
702  l(k) = sqrt(l0*l0 + dvol / dv_dl2)
703  vol_err = 0.5*(l(k)*l(k))*(slope + a_3*(3.0-4.0*l(k))) - vol
704  else
705  if (dv_dl2*(1.0-l0*l0) < dvol + &
706  dv_dl2 * (vol_open - vol)*ibma_2) then
707  l_max = sqrt(1.0 - (vol_open - vol)*ibma_2)
708  else

710  endif

720  (vol_err_max - vol_err_min))

725  ! robust enough not to need bracketing.
726  l(k) = l(k) - vol_err / (l(k)* (slope + a - 2.0*a*l(k)))
727  ! This would be a Newton's method iteration for L^2:
728  ! L(K) = sqrt(L(K)*L(K) - Vol_err / (0.5*(slope+a) - a*L(K)))
729  enddo
730  endif ! end of iterative solver
731  endif ! end of 1-boundary alternatives.
732  endif ! end of a<0 cases.
733  endif
734

736  ! and the Raleigh drag that acting on each layer.
737  if (l(k) > l(k+1)) then

741  else
742  bbl_frac = 0.0
743  endif
744

746  else ; cell_width = g%dx_Cv(i,j) ; endif

749  (12.0*cs%c_Smag*h_vel) / (12.0*cs%c_Smag*h_vel + m_to_h * &
750  cs%cdrag * gam*(1.0-gam)*(1.0-1.5*gam) * l(k)**2 * cell_width)
751  else ! This layer feels no drag.
752  rayleigh = 0.0
753  endif
754 
755  if (m==1) then
756  if (rayleigh > 0.0) then

759  v_at_u*v_at_u + u_bg_sq)
760  else ; visc%Ray_u(i,j,k) = 0.0 ; endif

765  u_at_v*u_at_v + u_bg_sq)
766  else ; visc%Ray_v(i,j,k) = 0.0 ; endif
767  endif
768 
769  enddo ! k loop to determine L(K).
770 
771  bbl_thick = bbl_thick * h_to_m
772  if (m==1) then
773  visc%kv_bbl_u(i,j) = max(cs%KV_BBL_min, &
774  cdrag_sqrt*ustar(i)*bbl_thick*bbl_visc_frac)
775  visc%bbl_thick_u(i,j) = bbl_thick
776  else
777  visc%kv_bbl_v(i,j) = max(cs%KV_BBL_min, &
778  cdrag_sqrt*ustar(i)*bbl_thick*bbl_visc_frac)
779  visc%bbl_thick_v(i,j) = bbl_thick
780  endif
781 
782  else ! Not Channel_drag.
783 ! Here the near-bottom viscosity is set to a value which will give
784 ! the correct stress when the shear occurs over bbl_thick.
785  bbl_thick = bbl_thick * h_to_m
786  if (m==1) then
787  visc%kv_bbl_u(i,j) = max(cs%KV_BBL_min, cdrag_sqrt*ustar(i)*bbl_thick)
788  visc%bbl_thick_u(i,j) = bbl_thick
789  else
790  visc%kv_bbl_v(i,j) = max(cs%KV_BBL_min, cdrag_sqrt*ustar(i)*bbl_thick)
791  visc%bbl_thick_v(i,j) = bbl_thick
792  endif
793  endif
794  endif ; enddo ! end of i loop
795  enddo ; enddo ! end of m & j loops
796 
797 ! Offer diagnostics for averaging
798  if (cs%id_bbl_thick_u > 0) &
799  call post_data(cs%id_bbl_thick_u, visc%bbl_thick_u, cs%diag)
800  if (cs%id_kv_bbl_u > 0) &
801  call post_data(cs%id_kv_bbl_u, visc%kv_bbl_u, cs%diag)
802  if (cs%id_bbl_thick_v > 0) &
803  call post_data(cs%id_bbl_thick_v, visc%bbl_thick_v, cs%diag)
804  if (cs%id_kv_bbl_v > 0) &
805  call post_data(cs%id_kv_bbl_v, visc%kv_bbl_v, cs%diag)
806  if (cs%id_Ray_u > 0) &
807  call post_data(cs%id_Ray_u, visc%Ray_u, cs%diag)
808  if (cs%id_Ray_v > 0) &
809  call post_data(cs%id_Ray_v, visc%Ray_v, cs%diag)
810

812  if (associated(visc%Ray_u)) call uchksum(visc%Ray_u,"Ray u",g,haloshift=0)
813  if (associated(visc%Ray_v)) call vchksum(visc%Ray_v,"Ray v",g,haloshift=0)

815  if (associated(visc%kv_bbl_v)) call vchksum(visc%kv_bbl_v,"kv_bbl_v",g,haloshift=0)
816  if (associated(visc%bbl_thick_u)) call uchksum(visc%bbl_thick_u,"bbl_thick_u",g,haloshift=0)
817  if (associated(visc%bbl_thick_v)) call vchksum(visc%bbl_thick_v,"bbl_thick_v",g,haloshift=0)

822 function set_v_at_u(v, h, G, i, j, k)
823  type(ocean_grid_type), intent(in) :: G
824  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(in) :: v
825  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
826  integer, intent(in) :: i, j, k
827  real :: set_v_at_u
828  ! This subroutine finds a thickness-weighted value of v at the u-points.
829  real :: hwt(4) ! Masked weights used to average v onto u, in H.
830  real :: hwt_tot ! The sum of the masked thicknesses, in H.
831 
832  hwt(1) = (h(i,j-1,k) + h(i,j,k)) * g%mask2dCv(i,j-1)
833  hwt(2) = (h(i+1,j-1,k) + h(i+1,j,k)) * g%mask2dCv(i+1,j-1)

864 subroutine set_viscous_ml(u, v, h, tv, fluxes, visc, dt, G, GV, CS)

942  real :: S_lay ! The layer salinity at velocity points, in PSU.

944  real :: Rlb ! The potential density of the layer below, in kg m-3.

946  real :: u_at_v ! The zonal velocity at a meridonal velocity point in m s-1.
