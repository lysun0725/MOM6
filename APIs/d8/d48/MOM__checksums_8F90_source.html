177 !> chksum_h_2d performs checksums on a 2d array staggered at tracer points.
178 subroutine chksum_h_2d(array, mesg, HI, haloshift)
179  type(hor_index_type), intent(in) :: HI !< A horizontal index type
180  real, dimension(HI%isd:,HI%jsd:), intent(in) :: array !< The array to be checksummed
181  character(len=*), intent(in) :: mesg !< An identifying message
182  integer, optional, intent(in) :: haloshift !< The width of halos to check (default 0)
183 
184  integer :: bc0,bcSW,bcSE,bcNW,bcNE,hshift
185 
186  if (checkfornans) then
187  if (is_nan(array(hi%isc:hi%iec,hi%jsc:hi%jec))) &
188  call chksum_error(fatal, 'NaN detected: '//trim(mesg))
189 ! if (is_NaN(array)) &
190 ! call chksum_error(FATAL, 'NaN detected in halo: '//trim(mesg))
191  endif
192 
193  if (calculatestatistics) then
194  call substats(hi, array, mesg); return
195  endif
196 
197  hshift=default_shift
198  if (present(haloshift)) hshift=haloshift
199  if (hshift<0) hshift=hi%ied-hi%iec
200 
201  if ( hi%isc-hshift<hi%isd .or. &
202  hi%iec+hshift>hi%ied .or. &
203  hi%jsc-hshift<hi%jsd .or. &
204  hi%jec+hshift>hi%jed ) then
205  write(0,*) 'chksum_h_2d: haloshift =',hshift
206  write(0,*) 'chksum_h_2d: isd,isc,iec,ied=',hi%isd,hi%isc,hi%iec,hi%ied
207  write(0,*) 'chksum_h_2d: jsd,jsc,jec,jed=',hi%jsd,hi%jsc,hi%jec,hi%jed
208  call chksum_error(fatal,'Error in chksum_h_2d '//trim(mesg))
209  endif
210 
211  bc0=subchk(array, hi, 0, 0)
212 
213  if (hshift==0) then
214  if (is_root_pe()) call chk_sum_msg("h-point:",bc0,mesg)
215  return
216  endif
217 
218  bcsw=subchk(array, hi, -hshift, -hshift)
219  bcse=subchk(array, hi, hshift, -hshift)
220  bcnw=subchk(array, hi, -hshift, hshift)
221  bcne=subchk(array, hi, hshift, hshift)
222 
223  if (is_root_pe()) call chk_sum_msg("h-point:",bc0,bcsw,bcse,bcnw,bcne,mesg)

227  integer function subchk(array, HI, di, dj)
228  type(hor_index_type), intent(in) :: HI
229  real, dimension(HI%isd:,HI%jsd:), intent(in) :: array
230  integer, intent(in) :: di, dj
231  integer :: bitcount, i, j, bc
232  subchk = 0
233  do j=hi%jsc+dj,hi%jec+dj; do i=hi%isc+di,hi%iec+di
234  bc = bitcount(abs(array(i,j)))
235  subchk = subchk + bc
236  enddo; enddo
237  call sum_across_pes(subchk)
238  subchk=mod(subchk,1000000000)
239  end function subchk
