85 !> Allocates and initializes the model's vertical grid structure.
86 subroutine verticalgridinit( param_file, GV )
87 ! This routine initializes the verticalGrid_type structure (GV).
88 ! All memory is allocated but not necessarily set to meaningful values until later.
89  type(param_file_type), intent(in) :: param_file ! Parameter file handle/type
90  type(verticalgrid_type), pointer :: GV ! The container for vertical grid data
91 ! This include declares and sets the variable "version".
92 #include "version_variable.h"
93  integer :: nk
94  character(len=16) :: mod = 'MOM_verticalGrid'
95 
96  if (associated(gv)) call mom_error(fatal, &
97  'verticalGridInit: called with an associated GV pointer.')
98  allocate(gv)
99 
100  ! Read all relevant parameters and write them to the model log.
101  call log_version(param_file, mod, version, &
102  "Parameters providing information about the vertical grid.")
103  call get_param(param_file, "MOM", "G_EARTH", gv%g_Earth, &
104  "The gravitational acceleration of the Earth.", &
105  units="m s-2", default = 9.80)
106  call get_param(param_file, mod, "RHO_0", gv%Rho0, &
107  "The mean ocean density used with BOUSSINESQ true to \n"//&
108  "calculate accelerations and the mass for conservation \n"//&
109  "properties, or with BOUSSINSEQ false to convert some \n"//&
110  "parameters from vertical units of m to kg m-2.", &
111  units="kg m-3", default=1035.0)
112  call get_param(param_file, mod, "BOUSSINESQ", gv%Boussinesq, &
113  "If true, make the Boussinesq approximation.", default=.true.)
114  call get_param(param_file, mod, "ANGSTROM", gv%Angstrom_z, &
115  "The minumum layer thickness, usually one-Angstrom.", &
116  units="m", default=1.0e-10)
117  if (.not.gv%Boussinesq) then
118  call get_param(param_file, mod, "H_TO_KG_M2", gv%H_to_kg_m2,&
119  "A constant that translates thicknesses from the model's \n"//&
120  "internal units of thickness to kg m-2.", units="kg m-2 H-1", &
121  default=1.0)
122  else
123  call get_param(param_file, mod, "H_TO_M", gv%H_to_m, &
124  "A constant that translates the model's internal \n"//&
125  "units of thickness into m.", units="m H-1", default=1.0)
126  endif
127 #ifdef STATIC_MEMORY_
128  ! Here NK_ is a macro, while nk is a variable.
129  call get_param(param_file, mod, "NK", nk, &
130  "The number of model layers.", units="nondim", &
131  static_value=nk_)
132  if (nk /= nk_) call mom_error(fatal, "verticalGridInit: " // &
133  "Mismatched number of layers NK_ between MOM_memory.h and param_file")
134 
135 #else
136  call get_param(param_file, mod, "NK", nk, &
137  "The number of model layers.", units="nondim", fail_if_missing=.true.)
138 #endif
139  gv%ke = nk
140 
141  if (gv%Boussinesq) then
142  gv%H_to_kg_m2 = gv%Rho0 * gv%H_to_m
143  gv%kg_m2_to_H = 1.0 / gv%H_to_kg_m2
144  gv%m_to_H = 1.0 / gv%H_to_m
145  gv%Angstrom = gv%Angstrom_z
146  else
147  gv%kg_m2_to_H = 1.0 / gv%H_to_kg_m2
148  gv%m_to_H = gv%Rho0 * gv%kg_m2_to_H
149  gv%H_to_m = gv%H_to_kg_m2 / gv%Rho0
150  gv%Angstrom = gv%Angstrom_z*1000.0*gv%kg_m2_to_H
151  endif
152  gv%H_subroundoff = 1e-20 * max(gv%Angstrom,gv%m_to_H*1e-17)
153  gv%H_to_Pa = gv%g_Earth * gv%H_to_kg_m2
154 
155 ! Log derivative values.
156  call log_param(param_file, mod, "M to THICKNESS", gv%m_to_H)
157 
158  alloc_( gv%sInterface(nk+1) )
159  alloc_( gv%sLayer(nk) )
160  alloc_( gv%g_prime(nk+1) ) ; gv%g_prime(:) = 0.0
161  ! The extent of Rlay should be changed to nk?
162  alloc_( gv%Rlay(nk+1) ) ; gv%Rlay(:) = 0.0
163 
164 end subroutine verticalgridinit

202 !> Returns the model's tracer flux units.
203 function get_tr_flux_units(GV, tr_units, tr_vol_conc_units, tr_mass_conc_units)
204  character(len=48) :: get_tr_flux_units
205  type(verticalgrid_type), intent(in) :: GV
206  character(len=*), optional, intent(in) :: tr_units
207  character(len=*), optional, intent(in) :: tr_vol_conc_units
208  character(len=*), optional, intent(in) :: tr_mass_conc_units
209 ! This subroutine returns the appropriate units for thicknesses and fluxes,
210 ! depending on whether the model is Boussinesq or not and the scaling for
211 ! the vertical thickness.
212 
213 ! Arguments: G - The ocean's grid structure.
214 ! One of the following three arguments must be present.
215 ! (in,opt) tr_units - Units for a tracer, for example Celsius or PSU.
216 ! (in,opt) tr_vol_conc_units - The concentration units per unit volume, for
217 ! example if the units are umol m-3,
218 ! tr_vol_conc_units would be umol.
219 ! (in,opt) tr_mass_conc_units - The concentration units per unit mass of sea
220 ! water, for example if the units are mol kg-1,
221 ! tr_vol_conc_units would be mol.
222 ! (ret) get_tr_flux_units - The model's flux units for a tracer.
223  integer :: cnt
224 
225  cnt = 0
226  if (present(tr_units)) cnt = cnt+1
227  if (present(tr_vol_conc_units)) cnt = cnt+1
228  if (present(tr_mass_conc_units)) cnt = cnt+1
229 
230  if (cnt == 0) call mom_error(fatal, "get_tr_flux_units: One of the three "//&
231  "arguments tr_units, tr_vol_conc_units, or tr_mass_conc_units "//&
232  "must be present.")
233  if (cnt > 1) call mom_error(fatal, "get_tr_flux_units: Only one of "//&
234  "tr_units, tr_vol_conc_units, and tr_mass_conc_units may be present.")
235  if (present(tr_units)) then
236  if (gv%Boussinesq) then
237  get_tr_flux_units = trim(tr_units)//" meter3 second-1"
238  else
239  get_tr_flux_units = trim(tr_units)//" kilogram second-1"
240  endif
241  endif
242  if (present(tr_vol_conc_units)) then
243  if (gv%Boussinesq) then
244  get_tr_flux_units = trim(tr_vol_conc_units)//" second-1"
245  else
246  get_tr_flux_units = trim(tr_vol_conc_units)//" m-3 kg s-1"
247  endif
248  endif
249  if (present(tr_mass_conc_units)) then
250  if (gv%Boussinesq) then
251  get_tr_flux_units = trim(tr_mass_conc_units)//" kg-1 m3 s-1"
252  else
253  get_tr_flux_units = trim(tr_mass_conc_units)//" second-1"
254  endif
255  endif
256 
257 end function get_tr_flux_units

259 !> This sets the coordinate data for the "layer mode" of the isopycnal model.
260 subroutine setverticalgridaxes( Rlay, GV )
261  ! Arguments
262  type(verticalgrid_type), intent(inout) :: GV !< The container for vertical grid data
263  real, dimension(GV%ke), intent(in) :: Rlay !< The layer target density
264  ! Local variables
265  integer :: nk
266 
267  nk = gv%ke
268 
269  gv%zAxisLongName = 'Target Potential Density'
270  gv%zAxisUnits = 'kg m-3'
271  gv%sLayer(:) = rlay(:)
272  gv%sInterface(1) = 1.5*rlay(1) - 0.5*rlay(2)
273  gv%sInterface(2:nk) = 0.5*( rlay(1:nk-1) + rlay(2:nk) )
274  gv%sInterface(nk+1) = 1.5*rlay(nk) - 0.5*rlay(nk-1)
