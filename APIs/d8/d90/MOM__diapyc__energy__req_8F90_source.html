52 subroutine diapyc_energy_req_test(h_3d, dt, tv, G, GV)
53  type(ocean_grid_type), intent(in) :: G
54  type(verticalgrid_type), intent(in) :: GV
55  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h_3d
56  type(thermo_var_ptrs), intent(inout) :: tv
57  real, intent(in) :: dt
58 ! Arguments: h_3d - Layer thickness before entrainment, in m or kg m-2.
59 ! (in/out) tv - A structure containing pointers to any available
60 ! thermodynamic fields. Absent fields have NULL ptrs.
61 ! (in) dt - The amount of time covered by this call, in s.
62 ! (in) G - The ocean's grid structure.
63 ! (in) GV - The ocean's vertical grid structure.
64 
65  real, dimension(SZK_(G)) :: &
66  T0, S0, & ! T0 & S0 are columns of initial temperatures and salinities, in degC and g/kg.
67  h_col ! h_col is a column of thicknesses h at tracer points, in H (m or kg m-2).
68  real, dimension(SZK_(G)+1) :: &
69  Kd, & ! A column of diapycnal diffusivities at interfaces, in m2 s-1.
70  h_top, h_bot ! Distances from the top or bottom, in H.
71  real :: ustar, absf, htot
72  real :: energy_Kd ! The energy used by diapycnal mixing in W m-2.
73  real :: tmp1 ! A temporary array.
74  integer :: i, j, k, is, ie, js, je, nz, itt
75  logical :: surface_BL, bottom_BL
76  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
77 
78 !$OMP do
79  do j=js,je ; do i=is,ie ; if (g%mask2dT(i,j) > 0.5) then
80  htot = 0.0 ; h_top(1) = 0.0
81  do k=1,nz
82  t0(k) = tv%T(i,j,k) ; s0(k) = tv%S(i,j,k)
83  h_col(k) = h_3d(i,j,k)
84  h_top(k+1) = h_top(k) + h_col(k)
85  enddo
86  htot = h_top(nz+1)
87  h_bot(nz+1) = 0.0
88  do k=nz,1,-1
89  h_bot(k) = h_bot(k+1) + h_col(k)
90  enddo
91 
92  ustar = 0.01 ! Change this to being an input parameter?
93  absf = 0.25*((abs(g%CoriolisBu(i-1,j-1)) + abs(g%CoriolisBu(i,j))) + &
94  (abs(g%CoriolisBu(i-1,j-1)) + abs(g%CoriolisBu(i,j))))
95  kd(1) = 0.0 ; kd(nz+1) = 0.0
96  do k=2,nz
97  tmp1 = h_top(k) * h_bot(k) * gv%H_to_m
98  kd(k) = ustar * 0.41 * (tmp1*ustar) / (absf*tmp1 + htot*ustar)
99  enddo
100  call diapyc_energy_req_calc(h_col, t0, s0, kd, energy_kd, dt, tv, g, gv)
101  endif ; enddo ; enddo
102 
103 end subroutine diapyc_energy_req_test

105 subroutine diapyc_energy_req_calc(h_in, T_in, S_in, Kd, energy_Kd, dt, tv, G, GV)
106  type(ocean_grid_type), intent(in) :: G
107  type(verticalgrid_type), intent(in) :: GV
108  real, dimension(SZK_(G)), intent(in) :: h_in, T_in, S_in
109  real, dimension(SZK_(G)+1), intent(in) :: Kd
110  real, intent(in) :: dt
111  real, intent(out) :: energy_Kd
112  type(thermo_var_ptrs), intent(inout) :: tv
113 ! Arguments: h_in - Layer thickness before entrainment, in m or kg m-2.
114 ! (in) T_in - The layer temperatures, in degC.
115 ! (in) S_in - The layer salinities, in g kg-1.
116 ! (in) Kd - The interfaces diapycnal diffusivities, in m2 s-1.
117 ! (in/out) tv - A structure containing pointers to any available
118 ! thermodynamic fields. Absent fields have NULL ptrs.
119 ! (in) dt - The amount of time covered by this call, in s.
120 ! (out) energy_Kd - The column-integrated rate of energy consumption
121 ! by diapycnal diffusion, in W m-2.
122 ! (in) G - The ocean's grid structure.
123 ! (in) GV - The ocean's vertical grid structure.
124 
125 ! This subroutine uses a substantially refactored tridiagonal equation for
126 ! diapycnal mixing of temperature and salinity to estimate the potential energy
127 ! change due to diapycnal mixing in a column of water. It does this estimate
128 ! 4 different ways, all of which should be equivalent, but reports only one.
129 ! The various estimates are taken because they will later be used as templates
130 ! for other bits of code.
131 
132  real, dimension(SZK_(G)) :: &
133  p_lay, & ! Average pressure of a layer, in Pa.
134  dSV_dT, & ! Partial derivative of specific volume with temperature, in m3 kg-1 K-1.
135  dSV_dS, & ! Partial derivative of specific volume with salinity, in m3 kg-1 / (g kg-1).
136  T0, S0, &
137  Te, Se, &
138  dTe, dSe, & ! Running (1-way) estimates of temperature and salinity change.
139  dT_to_dPE, & ! Partial derivative of column potential energy with the temperature
140  dS_to_dPE, & ! and salinity changes within a layer, in J m-2 K-1 and J m-2 / (g kg-1).
141  b_denom_1, & ! The first term in the denominator of b1, in m or kg m-2.
142  c1, & ! c1 is used by the tridiagonal solver, ND.
143  h_tr ! h_tr is h at tracer points with a h_neglect added to
144  ! ensure positive definiteness, in m or kg m-2.
145  real, dimension(SZK_(G)+1) :: &
146  pres, & ! Interface pressures in Pa.
147  dT_to_dPE_a, &
148  dS_to_dPE_a, &
149  dT_to_dPE_b, &
150  dS_to_dPE_b, &
151  PE_chg_k, & ! The integrated potential energy change within a timestep due
152  ! to the diffusivity at interface K, in J m-2.
153  pechg, &
154  kddt_h ! The diapycnal diffusivity times a timestep divided by the
155  ! average thicknesses around a layer, in m or kg m-2.
156  real :: &
157  b1 ! b1 is used by the tridiagonal solver, in m-1 or m2 kg-1.
158  real :: h_neglect ! A thickness that is so small it is usually lost
159  ! in roundoff and can be neglected, in m.
160  real :: dTe_term, dSe_term
161  real :: Kddt_h_guess
162  real :: htot
163  real :: dT_k, dT_km1, dS_k, dS_km1 ! Temporary arrays
164  real :: b1Kd ! Temporary arrays
165  real :: Kd_rat, Kdr_denom, I_Kdr_denom ! Temporary arrays
166  real :: dTe_t2, dSe_t2, dT_km1_t2, dS_km1_t2, dT_k_t2, dS_k_t2
167 
168  ! The following are a bunch of diagnostic arrays for debugging purposes.
169  real, dimension(SZK_(G)) :: &
170  Ta, Sa
171  real, dimension(SZK_(G)+1) :: &
172  dPEa_dKd, dPEa_dKd_est, dPEa_dKd_err, dPEa_dKd_trunc, dPEa_dKd_err_norm, &
173  dPEb_dKd, dPEb_dKd_est, dPEb_dKd_err, dPEb_dKd_trunc, dPEb_dKd_err_norm
174  real :: PE_chg_tot1A, PE_chg_tot2A, T_chg_totA
175  real :: PE_chg_tot1B, PE_chg_tot2B, T_chg_totB
176  real, dimension(SZK_(G)+1) :: dPEchg_dKd
177  real :: PE_chg(6)
178  real, dimension(6) :: dT_k_itt, dS_k_itt, dT_km1_itt, dS_km1_itt
179 
180  real :: I_G_Earth
181  real :: dKd_rat_dKd, ddT_k_dKd, ddS_k_dKd, ddT_km1_dKd, ddS_km1_dKd
182  integer :: k, nz, itt, max_itt
183  logical :: surface_BL, bottom_BL, debug
184  nz = g%ke
185  h_neglect = gv%H_subroundoff
186 
187  i_g_earth = 1.0 / g%G_earth
188  surface_bl = .true. ; bottom_bl = .true. ; debug = .true.
189 
190  dpea_dkd(:) = 0.0 ; dpea_dkd_est(:) = 0.0 ; dpea_dkd_err(:) = 0.0 ; dpea_dkd_err_norm(:) = 0.0 ; dpea_dkd_trunc(:) = 0.0
191  dpeb_dkd(:) = 0.0 ; dpeb_dkd_est(:) = 0.0 ; dpeb_dkd_err(:) = 0.0 ; dpeb_dkd_err_norm(:) = 0.0 ; dpeb_dkd_trunc(:) = 0.0
192 
193  htot = 0.0 ; pres(1) = 0.0
194  do k=1,nz
195  t0(k) = t_in(k) ; s0(k) = s_in(k)
196  h_tr(k) = h_in(k)
197  htot = htot + h_tr(k)
198  pres(k+1) = pres(k) + g%g_Earth * gv%H_to_kg_m2 * h_tr(k)
199  p_lay(k) = 0.5*(pres(k) + pres(k+1))
200  enddo
201  do k=1,nz
202  h_tr(k) = max(h_tr(k), 1e-15*htot)
203  enddo
204 
205  ! Introduce a diffusive flux variable, Kddt_h(K) = ea(k) = eb(k-1)
206 
207  kddt_h(1) = 0.0 ; kddt_h(nz+1) = 0.0
208  do k=2,nz
209  kddt_h(k) = min((gv%m_to_H**2*dt)*kd(k) / (0.5*(h_tr(k-1) + h_tr(k))),1e3*htot)
210  enddo
211 
212  ! Solve the tridiagonal equations for new temperatures.
213 
214  call calculate_specific_vol_derivs(t0, s0, p_lay, dsv_dt, dsv_ds, 1, nz, tv%eqn_of_state)
215 
216  do k=1,nz
217  dt_to_dpe(k) = 0.5 * i_g_earth *(pres(k+1)**2 - pres(k)**2) * dsv_dt(k)
218  ds_to_dpe(k) = 0.5 * i_g_earth *(pres(k+1)**2 - pres(k)**2) * dsv_ds(k)
219  enddo
220 
221  pe_chg_k(1) = 0.0 ; pe_chg_k(nz+1) = 0.0
222  pechg(:) = 0.0 ; dpechg_dkd(:) = 0.0
223 
224  if (surface_bl) then ! This version is appropriate for a surface boundary layer.
225 
226  b_denom_1(1) = h_tr(1)
227  dt_to_dpe_a(1) = dt_to_dpe(1)
228  ds_to_dpe_a(1) = ds_to_dpe(1)
229  do k=2,nz
230  ! At this point, Kddt_h(K) will be unknown because its value may depend
231  ! on how much energy is available.
232 
233  ! Precalculate some temporary expressions that are independent of Kddt_h_guess.
234  if (k==2) then
235  dt_km1_t2 = (t0(k)-t0(k-1))
236  ds_km1_t2 = (s0(k)-s0(k-1))
237  dte_t2 = 0.0 ; dse_t2 = 0.0
238  else
239  dte_t2 = kddt_h(k-1) * ((t0(k-2) - t0(k-1)) + dte(k-2))
240  dse_t2 = kddt_h(k-1) * ((s0(k-2) - s0(k-1)) + dse(k-2))
241  dt_km1_t2 = (t0(k)-t0(k-1)) - &
242  (kddt_h(k-1) / b_denom_1(k-1)) * ((t0(k-2) - t0(k-1)) + dte(k-2))
243  ds_km1_t2 = (s0(k)-s0(k-1)) - &
244  (kddt_h(k-1) / b_denom_1(k-1)) * ((s0(k-2) - s0(k-1)) + dse(k-2))
245  endif
246  dte_term = dte_t2 + b_denom_1(k-1) * (t0(k-1)-t0(k))
247  dse_term = dse_t2 + b_denom_1(k-1) * (s0(k-1)-s0(k))
248 
249 
250  ! Find the energy change due to a guess at the strength of diffusion at interface K.
251 
252  kddt_h_guess = kddt_h(k)
253  call find_pe_chg(kddt_h_guess, h_tr(k), b_denom_1(k-1), &
254  dte_term, dse_term, dt_km1_t2, ds_km1_t2, &
255  dt_to_dpe(k), ds_to_dpe(k), dt_to_dpe_a(k-1), ds_to_dpe_a(k-1), &
256  pe_chg_k(k), dpea_dkd(k))
257 
258  if (debug) then
259  do itt=1,5
260  kddt_h_guess = (1.0+0.01*(itt-3))*kddt_h(k)
261 
262  call find_pe_chg(kddt_h_guess, h_tr(k), b_denom_1(k-1), &
263  dte_term, dse_term, dt_km1_t2, ds_km1_t2, &
264  dt_to_dpe(k), ds_to_dpe(k), dt_to_dpe_a(k-1), ds_to_dpe_a(k-1), &
265  pe_chg=pe_chg(itt))
266  enddo
267  ! Compare with a 4th-order finite difference estimate.
268  dpea_dkd_est(k) = (4.0*(pe_chg(4)-pe_chg(2))/(0.02*kddt_h(k)) - &
269  (pe_chg(5)-pe_chg(1))/(0.04*kddt_h(k))) / 3.0
270  dpea_dkd_trunc(k) = (pe_chg(4)-pe_chg(2))/(0.02*kddt_h(k)) - &
271  (pe_chg(5)-pe_chg(1))/(0.04*kddt_h(k))
272  dpea_dkd_err(k) = (dpea_dkd_est(k) - dpea_dkd(k))
273  dpea_dkd_err_norm(k) = (dpea_dkd_est(k) - dpea_dkd(k)) / &
274  (abs(dpea_dkd_est(k)) + abs(dpea_dkd(k)) + 1e-100)
275  endif
276 
277  ! At this point, the final value of Kddt_h(K) is known, so the estimated
278  ! properties for layer k-1 can be calculated.
279 
280  b1 = 1.0 / (b_denom_1(k-1) + kddt_h(k))
281  c1(k) = kddt_h(k) * b1
282  if (k==2) then
283  te(1) = b1*(h_tr(1)*t0(1))
284  se(1) = b1*(h_tr(1)*s0(1))
285  else
286  te(k-1) = b1 * (h_tr(k-1) * t0(k-1) + kddt_h(k-1) * te(k-2))
287  se(k-1) = b1 * (h_tr(k-1) * s0(k-1) + kddt_h(k-1) * se(k-2))
288  endif
289  dte(k-1) = b1 * ( kddt_h(k)*(t0(k)-t0(k-1)) + dte_t2 )
290  dse(k-1) = b1 * ( kddt_h(k)*(s0(k)-s0(k-1)) + dse_t2 )
291 
292  b_denom_1(k) = h_tr(k) + (b_denom_1(k-1) * b1) * kddt_h(k)
293  dt_to_dpe_a(k) = dt_to_dpe(k) + c1(k)*dt_to_dpe_a(k-1)
294  ds_to_dpe_a(k) = ds_to_dpe(k) + c1(k)*ds_to_dpe_a(k-1)
295 
296  enddo
297 
298  b1 = 1.0 / (b_denom_1(nz))
299  te(nz) = b1 * (h_tr(nz) * t0(nz) + kddt_h(nz) * te(nz-1))
300  se(nz) = b1 * (h_tr(nz) * s0(nz) + kddt_h(nz) * se(nz-1))
301  dte(nz) = b1 * kddt_h(nz) * ((t0(nz-1)-t0(nz)) + dte(nz-1))
302  dse(nz) = b1 * kddt_h(nz) * ((s0(nz-1)-s0(nz)) + dse(nz-1))
303 
304  do k=nz-1,1,-1
305  te(k) = te(k) + c1(k+1)*te(k+1)
306  se(k) = se(k) + c1(k+1)*se(k+1)
307  enddo
308 
309  if (debug) then
310  do k=1,nz ; ta(k) = te(k) ; sa(k) = se(k) ; enddo
311  pe_chg_tot1a = 0.0 ; pe_chg_tot2a = 0.0 ; t_chg_tota = 0.0
312  do k=1,nz
313  pe_chg_tot1a = pe_chg_tot1a + (dt_to_dpe(k) * (te(k) - t0(k)) + &
314  ds_to_dpe(k) * (se(k) - s0(k)))
315  t_chg_tota = t_chg_tota + h_tr(k) * (te(k) - t0(k))
316  enddo
317  do k=2,nz
318  pe_chg_tot2a = pe_chg_tot2a + pe_chg_k(k)
319  enddo
320  endif
321 
322  endif
323 
324  if (bottom_bl) then ! This version is appropriate for a bottom boundary layer.
325 
326  b_denom_1(nz) = h_tr(nz)
327  dt_to_dpe_b(nz) = dt_to_dpe(nz)
328  ds_to_dpe_b(nz) = ds_to_dpe(nz)
329  do k=nz,2,-1 ! Loop over interior interfaces.
330  ! At this point, Kddt_h(K) will be unknown because its value may depend
331  ! on how much energy is available.
332 
333  ! Precalculate some temporary expressions that are independent of Kddt_h_guess.
334  if (k==nz) then
335  dt_k_t2 = (t0(k-1)-t0(k))
336  ds_k_t2 = (s0(k-1)-s0(k))
337  dte_t2 = 0.0 ; dse_t2 = 0.0
338  else
339  dte_t2 = kddt_h(k+1) * ((t0(k+1) - t0(k)) + dte(k+1))
340  dse_t2 = kddt_h(k+1) * ((s0(k+1) - s0(k)) + dse(k+1))
341  dt_k_t2 = (t0(k-1)-t0(k)) - &
342  (kddt_h(k+1)/ b_denom_1(k)) * ((t0(k+1) - t0(k)) + dte(k+1))
343  ds_k_t2 = (s0(k-1)-s0(k)) - &
344  (kddt_h(k+1)/ b_denom_1(k)) * ((s0(k+1) - s0(k)) + dse(k+1))
345  endif
346  dte_term = dte_t2 + b_denom_1(k) * (t0(k)-t0(k-1))
347  dse_term = dse_t2 + b_denom_1(k) * (s0(k)-s0(k-1))
348 
349  ! Find the energy change due to a guess at the strength of diffusion at interface K.
350 
351  kddt_h_guess = kddt_h(k)
352  call find_pe_chg(kddt_h_guess, h_tr(k-1), b_denom_1(k), &
353  dte_term, dse_term, dt_k_t2, ds_k_t2, &
354  dt_to_dpe(k-1), ds_to_dpe(k-1), dt_to_dpe_b(k), ds_to_dpe_b(k), &
355  pe_chg_k(k), dpeb_dkd(k))
356 
357 
358  if (debug) then
359  ! Compare with a 4th-order finite difference estimate.
360  do itt=1,5
361  kddt_h_guess = (1.0+0.01*(itt-3))*kddt_h(k)
362 
363  call find_pe_chg(kddt_h_guess, h_tr(k-1), b_denom_1(k), &
364  dte_term, dse_term, dt_k_t2, ds_k_t2, &
365  dt_to_dpe(k-1), ds_to_dpe(k-1), dt_to_dpe_b(k), ds_to_dpe_b(k), &
366  pe_chg=pe_chg(itt))
367  enddo
368 
369  dpeb_dkd_est(k) = (4.0*(pe_chg(4)-pe_chg(2))/(0.02*kddt_h(k)) - &
370  (pe_chg(5)-pe_chg(1))/(0.04*kddt_h(k))) / 3.0
371  dpeb_dkd_trunc(k) = (pe_chg(4)-pe_chg(2))/(0.02*kddt_h(k)) - &
372  (pe_chg(5)-pe_chg(1))/(0.04*kddt_h(k))
373  dpeb_dkd_err(k) = (dpeb_dkd_est(k) - dpeb_dkd(k))
374  dpeb_dkd_err_norm(k) = (dpeb_dkd_est(k) - dpeb_dkd(k)) / &
375  (abs(dpeb_dkd_est(k)) + abs(dpeb_dkd(k)) + 1e-100)
376  endif
377 
378  ! At this point, the final value of Kddt_h(K) is known, so the estimated
379  ! properties for layer k can be calculated.
380 
381  b1 = 1.0 / (b_denom_1(k) + kddt_h(k))
382  c1(k) = kddt_h(k) * b1
383  if (k==nz) then
384  te(nz) = b1* (h_tr(nz)*t0(nz))
385  se(nz) = b1* (h_tr(nz)*s0(nz))
386  else
387  te(k) = b1 * (h_tr(k) * t0(k) + kddt_h(k+1) * te(k+1))
388  se(k) = b1 * (h_tr(k) * s0(k) + kddt_h(k+1) * se(k+1))
389  endif
390  dte(k) = b1 * ( kddt_h(k)*(t0(k-1)-t0(k)) + dte_t2 )
391  dse(k) = b1 * ( kddt_h(k)*(s0(k-1)-s0(k)) + dse_t2 )
392 
393  b_denom_1(k-1) = h_tr(k-1) + (b_denom_1(k) * b1) * kddt_h(k)
394  dt_to_dpe_b(k-1) = dt_to_dpe(k-1) + c1(k)*dt_to_dpe_b(k)
395  ds_to_dpe_b(k-1) = ds_to_dpe(k-1) + c1(k)*ds_to_dpe_b(k)
396 
397  enddo
398 
399  b1 = 1.0 / (b_denom_1(1))
400  te(1) = b1 * (h_tr(1) * t0(1) + kddt_h(2) * te(2))
401  se(1) = b1 * (h_tr(1) * s0(1) + kddt_h(2) * se(2))
402  dte(1) = b1 * kddt_h(2) * ((t0(2)-t0(1)) + dte(2))
403  dse(1) = b1 * kddt_h(2) * ((s0(2)-s0(1)) + dse(2))
404 
405  do k=2,nz
406  te(k) = te(k) + c1(k)*te(k-1)
407  se(k) = se(k) + c1(k)*se(k-1)
408  enddo
409 
410  if (debug) then

412  do k=1,nz
413  pe_chg_tot1b = pe_chg_tot1b + (dt_to_dpe(k) * (te(k) - t0(k)) + &
414  ds_to_dpe(k) * (se(k) - s0(k)))
415  t_chg_totb = t_chg_totb + h_tr(k) * (te(k) - t0(k))
416  enddo
417  do k=2,nz
418  pe_chg_tot2b = pe_chg_tot2b + pe_chg_k(k)
419  enddo
420  endif

422  endif

424  energy_kd = 0.0 ; do k=2,nz ; energy_kd = energy_kd + pe_chg_k(k) ; enddo

430 subroutine find_pe_chg(Kddt_h, h_k, b_den_1, dTe_term, dSe_term, &
431  dt_km1_t2, ds_km1_t2, dt_to_dpe_k, ds_to_dpe_k, &
432  dt_to_dpea, ds_to_dpea, pe_chg, dpe_chg_dkd)
433  real, intent(in) :: Kddt_h, h_k, b_den_1, dTe_term, dSe_term
434  real, intent(in) :: dT_km1_t2, dS_km1_t2, dT_to_dPE_k, dS_to_dPE_k
435  real, intent(in) :: dT_to_dPEa, dS_to_dPEa
436  real, optional, intent(out) :: PE_chg
437  real, optional, intent(out) :: dPE_chg_dKd
438 
439 ! This subroutine determines the total potential energy change due to mixing
440 ! at an interface, including all of the implicit effects of the prescribed
441 ! mixing at interfaces above. Everything here is derived by careful manipulation
442 ! of the robust tridiagonal solvers used for tracers by MOM6.
443 ! The comments describing these arguments are for a downward mixing pass, but
444 ! this routine can also be used for an upward pass with the sense of direction
445 ! reversed.
446 !
447 ! Arguments: Kddt_h - The diffusivity at an interface times the time step and
448 ! divided by the average of the thicknesses around the
449 ! interface, in units of H (m or kg-2).
450 ! (in) h_k - The thickness of the layer below the interface, in H.
451 ! (in) b_den_1 - The first first term in the denominator of the pivot
452 ! for the tridiagonal solver, given by h_k plus a term that
453 ! is a fraction (determined from the tridiagonal solver) of
454 ! Kddt_h for the interface above, in H.
455 ! (in) dTe_term - A diffusivity-independent term related to the
456 ! temperature change in the layer below the interface, in K H.
457 ! (in) dSe_term - A diffusivity-independent term related to the
458 ! salinity change in the layer below the interface, in ppt H.
459 ! (in) dT_km1_t2 - A diffusivity-independent term related to the
460 ! temperature change in the layer above the interface, in K.
461 ! (in) dS_km1_t2 - A diffusivity-independent term related to the
462 ! salinity change in the layer above the interface, in ppt.
463 ! (in) dT_to_dPE_k - A factor (1/2*pres*mass_lay*dSpec_vol/dT) relating
464 ! a layer's temperature change to the change in column
465 ! potential energy, in J m-2 K-1.
466 ! (in) dS_to_dPE_k - A factor (1/2*pres*mass_lay*dSpec_vol/dS) relating
467 ! a layer's salinity change to the change in column
468 ! potential energy, in J m-2 ppt-1.
469 ! (in) dT_to_dPEa - A factor (1/2*pres*mass_lay*dSpec_vol/dT) relating
470 ! a layer's temperature change to the change in column
471 ! potential energy, including all implicit diffusive changes
472 ! in the temperatures of all the layers above, in J m-2 K-1.
473 ! (in) dS_to_dPEa - A factor (1/2*pres*mass_lay*dSpec_vol/dS) relating
474 ! a layer's salinity change to the change in column
475 ! potential energy, including all implicit diffusive changes
476 ! in the salinities of all the layers above, in J m-2 ppt-1.
477 ! (out,opt) PE_chg - The change in column potential energy from applying
478 ! Kddt_h at the present interface, in J m-2.
479 ! (out,opt) dPE_chg_dKd - The partial derivative of PE_chg with Kddt_h,
480 ! in units of J m-2 H-1.
481 
482  ! b_den_1 - The first term in the denominator of b1, in m or kg m-2.
483  real :: b1 ! b1 is used by the tridiagonal solver, in H-1.
484  real :: b1Kd ! Temporary array (nondim.)
485  real :: dT_k, dT_km1 ! Temporary arrays in K.
486  real :: dS_k, dS_km1 ! Temporary arrays in ppt.
487  real :: I_Kr_denom, dKr_dKd ! Temporary arrays in H-2 and nondim.
488  real :: ddT_k_dKd, ddT_km1_dKd ! Temporary arrays in K H-1.
489  real :: ddS_k_dKd, ddS_km1_dKd ! Temporary arrays in ppt H-1.
490 
491  b1 = 1.0 / (b_den_1 + kddt_h)
492  b1kd = kddt_h*b1
493 
494  ! Start with the temperature change in layer k-1 due to the diffusivity at
495  ! interface K without considering the effects of changes in layer k.
496 
497  ! Calculate the change in PE due to the diffusion at interface K
498  ! if Kddt_h(K+1) = 0.
499  i_kr_denom = 1.0 / (h_k*b_den_1 + (b_den_1 + h_k)*kddt_h)
500 
501  dt_k = (kddt_h*i_kr_denom) * dte_term
502  ds_k = (kddt_h*i_kr_denom) * dse_term
503 
504  if (present(pe_chg)) then
505  ! Find the change in energy due to diffusion with strength Kddt_h at this interface.
506  ! Increment the temperature changes in layer k-1 due the changes in layer k.
507  dt_km1 = b1kd * ( dt_k + dt_km1_t2 )
508  ds_km1 = b1kd * ( ds_k + ds_km1_t2 )
509 
510  pe_chg = (dt_to_dpe_k * dt_k + dt_to_dpea * dt_km1) + &
511  (ds_to_dpe_k * ds_k + ds_to_dpea * ds_km1)
512  endif
513 
514  if (present(dpe_chg_dkd)) then
515  ! Find the derivatives of the temperature and salinity changes with Kddt_h.
516  dkr_dkd = (h_k*b_den_1) * i_kr_denom**2
517 
518  ddt_k_dkd = dkr_dkd * dte_term
519  dds_k_dkd = dkr_dkd * dse_term
520  ddt_km1_dkd = (b1**2 * b_den_1) * ( dt_k + dt_km1_t2 ) + b1kd * ddt_k_dkd
521  dds_km1_dkd = (b1**2 * b_den_1) * ( ds_k + ds_km1_t2 ) + b1kd * dds_k_dkd
522 
523  ! Calculate the partial derivative of Pe_chg with Kddt_h.
524  dpe_chg_dkd = (dt_to_dpe_k * ddt_k_dkd + dt_to_dpea * ddt_km1_dkd) + &
525  (ds_to_dpe_k * dds_k_dkd + ds_to_dpea * dds_km1_dkd)
526  endif
527 
528 end subroutine find_pe_chg
