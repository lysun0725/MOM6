25 !* *
26 !* This file contains the subroutines that implement diapycnal *
27 !* mixing and advection in isopycnal layers. The main subroutine, *
28 !* calculate_entrainment, returns the entrainment by each layer *
29 !* across the interfaces above and below it. These are calculated *
30 !* subject to the constraints that no layers can be driven to neg- *
31 !* ative thickness and that the each layer maintains its target *
32 !* density, using the scheme described in Hallberg (MWR 2000). There *
33 !* may or may not be a bulk mixed layer above the isopycnal layers. *
34 !* The solution is iterated until the change in the entrainment *
35 !* between successive iterations is less than some small tolerance. *
36 !* *
37 !* The dual-stream entrainment scheme of MacDougall and Dewar *
38 !* (JPO 1997) is used for combined diapycnal advection and diffusion, *
39 !* modified as described in Hallberg (MWR 2000) to be solved *
40 !* implicitly in time. Any profile of diffusivities may be used. *
41 !* Diapycnal advection is fundamentally the residual of diapycnal *
42 !* diffusion, so the fully implicit upwind differencing scheme that *
43 !* is used is entirely appropriate. The downward buoyancy flux in *
44 !* each layer is determined from an implicit calculation based on *
45 !* the previously calculated flux of the layer above and an estim- *
46 !* ated flux in the layer below. This flux is subject to the foll- *
47 !* owing conditions: (1) the flux in the top and bottom layers are *
48 !* set by the boundary conditions, and (2) no layer may be driven *
49 !* below an Angstrom thickness. If there is a bulk mixed layer, the *
50 !* mixed and buffer layers are treated as Eulerian layers, whose *
51 !* thicknesses only change due to entrainment by the interior layers. *
52 !* *
53 !* In addition, the model may adjust the fluxes to drive the layer *
54 !* densities (sigma 2?) back toward their targer values. *
55 !* *
56 !* A small fragment of the grid is shown below: *
57 !* *
58 !* j+1 x ^ x ^ x At x: q *
59 !* j+1 > o > o > At ^: v *
60 !* j x ^ x ^ x At >: u *
61 !* j > o > o > At o: h, buoy, T, S, ea, eb, etc. *
62 !* j-1 x ^ x ^ x *
63 !* i-1 i i+1 At x & ^: *
64 !* i i+1 At > & o: *
65 !* *
66 !* The boundaries always run through q grid points (x). *
67 !* *

796  endif

801  0.5*h_guess(i,k) / (ds_kb(i) * i_dskbp1(i)))
802  else
803  ! Ensure that -ea_cor < 0.5*h_guess & F_cor < 0.25*h_guess(k+1)
804  ea_cor = -min(-ea_cor, 0.5*h_guess(i,k), &
805  0.25*h_guess(i,k+1) / (ds_kb(i) * i_dskbp1(i)))

1210  ! estimate of the thickness in the denominator of the thickness diffusion.

1212  kd_x_dt / (0.5*(h_bl(i,kmb) + h_bl(i,kmb+1))))
1213  endif
1214  else
1215  ent_bl(i,kmb+1) = 0.0
1216  endif ; enddo

1221  dskb, ddskb_de, dslay, ddslay_de, ds_anom_lim, do_i_in)

1223  type(verticalgrid_type), intent(in) :: GV

1225  real, dimension(SZI_(G)), intent(in) :: E_kb
1226  integer, intent(in) :: is, ie, kmb
1227  logical, intent(in) :: limit
1228  real, dimension(SZI_(G)), intent(inout) :: dSkb
1229  real, dimension(SZI_(G)), optional, intent(inout) :: ddSkb_dE, dSlay, ddSlay_dE

1232 ! Arguments: h_bl - Layer thickness, in m or kg m-2 (abbreviated as H below).
1233 ! (in) Sref - Reference potential vorticity (in kg m-3?)
1234 ! (in) Ent_bl - The average entrainment upward and downward across
1235 ! each interface around the buffer layers, in H.
1236 ! (in) E_kb - The entrainment by the top interior layer, in H.
1237 ! (in) is, ie - The range of i-indices to work on.
1238 ! (in) kmb - The number of mixed and buffer layers.
1239 ! (in) G - The ocean's grid structure.
1240 ! (in) GV - The ocean's vertical grid structure.
1241 ! (in) limit - If true, limit dSkb and dSlay to avoid negative values.
1242 ! (out) dSkb - The limited potential density difference across the
1243 ! interface between the bottommost buffer layer and the
1244 ! topmost interior layer. dSkb > 0.
1245 ! (out,opt) dSlay - The limited potential density difference across the

1253 ! This subroutine determines the reference density difference between the
1254 ! bottommost buffer layer and the first interior after the mixing between mixed
1255 ! and buffer layers and mixing with the layer below. Within the mixed and buffer
1256 ! layers, entrainment from the layer above is increased when it is necessary to
1257 ! keep the layers from developing a negative thickness; otherwise it equals
1258 ! Ent_bl. At each interface, the upward and downward fluxes average out to
1259 ! Ent_bl, unless entrainment by the layer below is larger than twice Ent_bl.
1260 ! The density difference across the first interior layer may also be returned.
1261 ! It could also be limited to avoid negative values or values that greatly
1262 ! exceed the density differences across an interface.
1263 ! Additionally, the partial derivatives of dSkb and dSlay with E_kb could
1264 ! also be returned.

1266  b1, c1, & ! b1 and c1 are variables used by the tridiagonal solver.
1267  S, dS_dE, & ! The coordinate density and its derivative with R.

1272  real :: src ! A source term for dS_dR.
1273  real :: h1 ! The thickness in excess of the minimum that will remain
1274  ! after exchange with the layer below, in m or kg m-2.
1275  logical, dimension(SZI_(G)) :: do_i
1276  real :: h_neglect ! A thickness that is so small it is usually lost
1277  ! in roundoff and can be neglected, in H.
1278  real :: h_tr ! h_tr is h at tracer points with a tiny thickness
1279  ! added to ensure positive definiteness, in m or kg m-2.
1280  real :: b_denom_1 ! The first term in the denominator of b1 in m or kg m-2.

1282  real :: dS_kbp1, IdS_kbp1
1283  real :: deriv_dSLay

1285  real :: f1, df1_drat ! Nondimensional temporary variables.
1286  real :: z, dz_drat, f2, df2_dz, expz ! Nondimensional temporary variables.
1287  real :: eps_dSLay, eps_dSkb ! Small nondimensional constants.
1288  integer :: i, k
1289 
1290  if (present(ddslay_de) .and. .not.present(dslay)) call mom_error(fatal, &
1291  "In deterimine_dSkb, ddSLay_dE may only be present if dSlay is.")
1292 
1293  h_neglect = gv%H_subroundoff

1296  ea(i,kmb+1) = e_kb(i) ; dea_de(i,kmb+1) = 1.0
1297  s(i,kmb+1) = sref(i,kmb+1) ; ds_de(i,kmb+1) = 0.0
1298  b1(i,kmb+1) = 0.0
1299  d1(i) = 1.0
1300  do_i(i) = .true.
1301  enddo
1302  if (present(do_i_in)) then
1303  do i=is,ie ; do_i(i) = do_i_in(i) ; enddo
1304  endif
1305  do k=kmb,1,-1 ; do i=is,ie
1306  if (do_i(i)) then
1307  ! The do_i test here is only for efficiency.
1308  ! Determine the entrainment from below for each buffer layer.
1309  if (2.0*ent_bl(i,k+1) > ea(i,k+1)) then
1310  eb(i,k) = 2.0*ent_bl(i,k+1) - ea(i,k+1) ; deb_de(i,k) = -dea_de(i,k+1)
1311  else
1312  eb(i,k) = 0.0 ; deb_de(i,k) = 0.0
1313  endif
1314 
1315  ! Determine the entrainment from above for each buffer layer.
1316  h1 = (h_bl(i,k) - gv%Angstrom) + (eb(i,k) - ea(i,k+1))
1317  if (h1 >= 0.0) then
1318  ea(i,k) = ent_bl(i,k) ; dea_de(i,k) = 0.0
1319  elseif (ent_bl(i,k) + 0.5*h1 >= 0.0) then
1320  ea(i,k) = ent_bl(i,k) - 0.5*h1
1321  dea_de(i,k) = 0.5*(dea_de(i,k+1) - deb_de(i,k))
1322  else
1323  ea(i,k) = -h1
1324  dea_de(i,k) = dea_de(i,k+1) - deb_de(i,k)
1325  endif
1326  else
1327  ea(i,k) = 0.0 ; dea_de(i,k) = 0.0 ; eb(i,k) = 0.0 ; deb_de(i,k) = 0.0
1328  endif

1331  h_tr = h_bl(i,k) + h_neglect
1332  c1(i,k) = ea(i,k+1) * b1(i,k+1)
1333  b_denom_1 = (h_tr + d1(i)*eb(i,k))
1334  b1(i,k) = 1.0 / (b_denom_1 + ea(i,k))
1335  d1(i) = b_denom_1 * b1(i,k)
1336 
1337  s(i,k) = (h_tr*sref(i,k) + eb(i,k)*s(i,k+1)) * b1(i,k)
1338  enddo ; enddo
1339  do k=2,kmb ; do i=is,ie
1340  s(i,k) = s(i,k) + c1(i,k-1)*s(i,k-1)
1341  enddo ; enddo
1342 
1343  if (present(ddskb_de) .or. present(ddslay_de)) then
1344  ! These two tridiagonal solvers cannot be combined because the solutions for
1345  ! S are required as a source for dS_dE.
1346  do k=kmb,2,-1 ; do i=is,ie
1347  if (do_i(i) .and. (dea_de(i,k) - deb_de(i,k) > 0.0)) then
1348  src = (((s(i,k+1) - sref(i,k)) * (h_bl(i,k) + h_neglect) + &
1349  (s(i,k+1) - s(i,k-1)) * ea(i,k)) * deb_de(i,k) - &
1350  ((sref(i,k) - s(i,k-1)) * h_bl(i,k) + &
1351  (s(i,k+1) - s(i,k-1)) * eb(i,k)) * dea_de(i,k)) / &
1352  ((h_bl(i,k) + h_neglect + ea(i,k)) + eb(i,k))
1353  else ; src = 0.0 ; endif
1354  ds_de(i,k) = (src + eb(i,k)*ds_de(i,k+1)) * b1(i,k)
1355  enddo ; enddo
1356  do i=is,ie
1357  if (do_i(i) .and. (deb_de(i,1) < 0.0)) then
1358  src = (((s(i,2) - sref(i,1)) * (h_bl(i,1) + h_neglect)) * deb_de(i,1)) / &
1359  (h_bl(i,1) + h_neglect + eb(i,1))
1360  else ; src = 0.0 ; endif
1361  ds_de(i,1) = (src + eb(i,1)*ds_de(i,2)) * b1(i,1)
1362  enddo
1363  do k=2,kmb ; do i=is,ie
1364  ds_de(i,k) = ds_de(i,k) + c1(i,k-1)*ds_de(i,k-1)
1365  enddo ; enddo
1366  endif
1367 
1368  ! Now, apply any limiting and return the requested variables.
1369 
1370  eps_dskb = 1.0e-6 ! Should be a small, nondimensional, positive number.
1371  if (.not.limit) then
1372  do i=is,ie ; if (do_i(i)) then
1373  dskb(i) = sref(i,kmb+1) - s(i,kmb)
1374  endif ; enddo
1375  if (present(ddskb_de)) then ; do i=is,ie ; if (do_i(i)) then
1376  ddskb_de(i) = -1.0*ds_de(i,kmb)
1377  endif ; enddo ; endif
1378 
1379  if (present(dslay)) then ; do i=is,ie ; if (do_i(i)) then
1380  dslay(i) = 0.5 * (sref(i,kmb+2) - s(i,kmb))
1381  endif ; enddo ; endif
1382  if (present(ddslay_de)) then ; do i=is,ie ; if (do_i(i)) then
1383  ddslay_de(i) = -0.5*ds_de(i,kmb)
1384  endif ; enddo ; endif
1385  else
1386  do i=is,ie ; if (do_i(i)) then
1387  ! Need to ensure that 0 < dSkb <= S_kb - Sbl
1388  if (sref(i,kmb+1) - s(i,kmb) < eps_dskb*(sref(i,kmb+2) - sref(i,kmb+1))) then
1389  dskb(i) = eps_dskb * (sref(i,kmb+2) - sref(i,kmb+1)) ; deriv_dskb(i) = 0.0
1390  else
1391  dskb(i) = sref(i,kmb+1) - s(i,kmb) ; deriv_dskb(i) = -1.0
1392  endif
1393  if (present(ddskb_de)) ddskb_de(i) = deriv_dskb(i)*ds_de(i,kmb)
1394  endif ; enddo
1395 
1396  if (present(dslay)) then
1397  dz_drat = 1000.0 ! The limit of large dz_drat the same as choosing a
1398  ! Heaviside function.
1399  eps_dslay = 1.0e-10 ! Should be ~= GV%Angstrom / sqrt(Kd*dt)
1400  do i=is,ie ; if (do_i(i)) then
1401  ds_kbp1 = sref(i,kmb+2) - sref(i,kmb+1)
1402  ids_kbp1 = 1.0 / (sref(i,kmb+2) - sref(i,kmb+1))
1403  rat = (sref(i,kmb+1) - s(i,kmb)) * ids_kbp1
1404  ! Need to ensure that 0 < dSLay <= 2*dSkb
1405  if (rat < 0.5) then
1406  ! The coefficients here are chosen so that at rat = 0.5, the value (1.5)
1407  ! and first derivative (-0.5) match with the "typical" case (next).
1408  ! The functional form here is arbitrary.
1409  ! f1 provides a reasonable profile that matches the value and derivative
1410  ! of the "typical" case at rat = 0.5, and has a maximum of less than 2.
1411  inv_term = 1.0 / (1.0-rat)
1412  f1 = 2.0 - 0.125*(inv_term**2)
1413  df1_drat = - 0.25*(inv_term**3)
1414 
1415  ! f2 ensures that dSLay goes to 0 rapidly if rat is significantly
1416  ! negative.
1417  z = dz_drat * rat + 4.0 ! The 4 here gives f2(0) = 0.982.
1418  if (z >= 18.0) then ; f2 = 1.0 ; df2_dz = 0.0
1419  elseif (z <= -58.0) then ; f2 = eps_dslay ; df2_dz = 0.0
1420  else
1421  expz = exp(z) ; inv_term = 1.0 / (1.0 + expz)
1422  f2 = (eps_dslay + expz) * inv_term
1423  df2_dz = (1.0 - eps_dslay) * expz * inv_term**2
1424  endif
1425 
1426  dslay(i) = dskb(i) * f1 * f2
1427  deriv_dslay = deriv_dskb(i) * (f1 * f2) - (dskb(i)*ids_kbp1) * &
1428  (df1_drat*f2 + f1 * dz_drat * df2_dz)
1429  elseif (dskb(i) <= 3.0*ds_kbp1) then
1430  ! This is the "typical" case.
1431  dslay(i) = 0.5 * (dskb(i) + ds_kbp1)
1432  deriv_dslay = 0.5 * deriv_dskb(i) ! = -0.5
1433  else
1434  dslay(i) = 2.0*ds_kbp1
1435  deriv_dslay = 0.0
1436  endif
1437  if (present(ddslay_de)) ddslay_de(i) = deriv_dslay*ds_de(i,kmb)
1438  endif ; enddo
1439  endif ! present(dSlay)
1440  endif ! Not limited.
1441 
1442  if (present(ds_anom_lim)) then ; do i=is,ie ; if (do_i(i)) then
1443  ds_anom_lim(i) = max(0.0, eps_dskb * (sref(i,kmb+2) - sref(i,kmb+1)) - &
1444  (sref(i,kmb+1) - s(i,kmb)) )
1445  endif ; enddo ; endif
1446 
1447 end subroutine determine_dskb

1450 subroutine f_kb_to_ea_kb(h_bl, Sref, Ent_bl, I_dSkbp1, F_kb, kmb, i, &
1451  g, gv, cs, ea_kb, tol_in)
1452  type(ocean_grid_type), intent(in) :: G
1453  type(verticalgrid_type), intent(in) :: GV
1454  real, dimension(SZI_(G),SZK_(G)), intent(in) :: h_bl, Sref, Ent_bl
1455  real, dimension(SZI_(G)), intent(in) :: I_dSkbp1, F_kb
1456  integer, intent(in) :: kmb, i
1457  type(entrain_diffusive_cs), pointer :: CS
1458  real, dimension(SZI_(G)), intent(inout) :: ea_kb
1459  real, optional, intent(in) :: tol_in
1460 
1461  ! Given an entrainment from below for layer kb, determine a consistent
1462  ! entrainment from above, such that dSkb * ea_kb = dSkbp1 * F_kb. The input
1463  ! value of ea_kb is both the maximum value that can be obtained and the first
1464  ! guess of the iterations. Also, make sure that ea_kb is an under-estimate
1465  real :: max_ea, min_ea
1466  real :: err, err_min, err_max
1467  real :: derr_dea
1468  real :: val, tolerance, tol1
1469  real :: ea_prev
1470  real :: dS_kbp1
1471  logical :: bisect_next, Newton
1472  real, dimension(SZI_(G)) :: dS_kb
1473  real, dimension(SZI_(G)) :: maxF, ent_maxF, zeros
1474  real, dimension(SZI_(G)) :: ddSkb_dE
1475  integer :: it
1476  integer, parameter :: MAXIT = 30
1477 
1478  ds_kbp1 = sref(i,kmb+2) - sref(i,kmb+1)
1479  max_ea = ea_kb(i) ; min_ea = 0.0
1480  val = ds_kbp1 * f_kb(i)
1481  err_min = -val
1482 
1483  tolerance = gv%m_to_H * cs%Tolerance_Ent
1484  if (present(tol_in)) tolerance = tol_in
1485  bisect_next = .true.
1486 
1487  call determine_dskb(h_bl, sref, ent_bl, ea_kb, i, i, kmb, g, gv, .true., &
1488  ds_kb, ddskb_de)
1489 
1490  err = ds_kb(i) * ea_kb(i) - val
1491  derr_dea = ds_kb(i) + ddskb_de(i) * ea_kb(i)
1492  ! Return if Newton's method on the first guess would give a tolerably small
1493  ! change in the value of ea_kb.
1494  if ((err <= 0.0) .and. (abs(err) <= tolerance*abs(derr_dea))) return
1495 
1496  if (err == 0.0) then ; return ! The exact solution on the first guess...
1497  elseif (err > 0.0) then ! The root is properly bracketed.
1498  max_ea = ea_kb(i) ; err_max = err
1499  ! Use Newton's method (if it stays bounded) or the false position method
1500  ! to find the next value.
1501  if ((derr_dea > 0.0) .and. (derr_dea*(ea_kb(i) - min_ea) > err) .and. &
1502  (derr_dea*(max_ea - ea_kb(i)) > -1.0*err)) then
1503  ea_kb(i) = ea_kb(i) - err / derr_dea
1504  else ! Use the bisection for the next guess.
1505  ea_kb(i) = 0.5*(max_ea+min_ea)
1506  endif
1507  else
1508  ! Try to bracket the root first. If unable to bracket the root, return
1509  ! the maximum.
1510  zeros(i) = 0.0
1511  call find_maxf_kb(h_bl, sref, ent_bl, i_dskbp1, zeros, ea_kb, &
1512  kmb, i, i, g, gv, cs, maxf, ent_maxf, f_thresh = f_kb)
1513  err_max = ds_kbp1 * maxf(i) - val
1514  ! If err_max is negative, there is no good solution, so use the maximum
1515  ! value of F in the valid range.
1516  if (err_max <= 0.0) then
1517  ea_kb(i) = ent_maxf(i) ; return
1518  else
1519  max_ea = ent_maxf(i)
1520  ea_kb(i) = 0.5*(max_ea+min_ea) ! Use bisection for the next guess.
1521  endif
1522  endif
1523 
1524  ! Exit if the range between max_ea and min_ea already acceptable.
1525  ! if (abs(max_ea - min_ea) < 0.1*tolerance) return
1526 
1527  do it = 1, maxit
1528  call determine_dskb(h_bl, sref, ent_bl, ea_kb, i, i, kmb, g, gv, .true., &
1529  ds_kb, ddskb_de)
1530 
1531  err = ds_kb(i) * ea_kb(i) - val
1532  derr_dea = ds_kb(i) + ddskb_de(i) * ea_kb(i)
1533 
1534  ea_prev = ea_kb(i)
1535  ! Use Newton's method or the false position method to find the next value.
1536  newton = .false.
1537  if (err > 0.0) then
1538  max_ea = ea_kb(i) ; err_max = err
1539  if ((derr_dea > 0.0) .and. (derr_dea*(ea_kb(i)-min_ea) > err)) newton = .true.
1540  else
1541  min_ea = ea_kb(i) ; err_min = err
1542  if ((derr_dea > 0.0) .and. (derr_dea*(ea_kb(i)-max_ea) < err)) newton = .true.
1543  endif
1544 
1545  if (newton) then
1546  ea_kb(i) = ea_kb(i) - err / derr_dea
1547  elseif (bisect_next) then ! Use bisection to reduce the range.
1548  ea_kb(i) = 0.5*(max_ea+min_ea)
1549  bisect_next = .false.
1550  else ! Use the false-position method for the next guess.
1551  ea_kb(i) = min_ea + (max_ea-min_ea) * (err_min/(err_min - err_max))
1552  bisect_next = .true.
1553  endif
1554 
1555  tol1 = tolerance ; if (err > 0.0) tol1 = 0.099*tolerance
1556  if (ds_kb(i) <= ds_kbp1) then
1557  if (abs(ea_kb(i) - ea_prev) <= tol1) return
1558  else
1559  if (ds_kbp1*abs(ea_kb(i) - ea_prev) <= ds_kb(i)*tol1) return
1560  endif
1561  enddo
1562 
1563 end subroutine f_kb_to_ea_kb
