110 subroutine call_tracer_register(HI, GV, param_file, CS, tr_Reg, restart_CS)
111  type(hor_index_type), intent(in) :: HI
112  type(verticalgrid_type), intent(in) :: GV
113  type(param_file_type), intent(in) :: param_file
114  type(tracer_flow_control_cs), pointer :: CS
115  type(tracer_registry_type), pointer :: tr_Reg
116  type(mom_restart_cs), pointer :: restart_CS
117 ! Arguments: HI - A horizontal index type structure.
118 ! (in) GV - The ocean's vertical grid structure.
119 ! (in) param_file - A structure indicating the open file to parse for
120 ! model parameter values.
121 ! (in/out) CS - A pointer that is set to point to the control structure
122 ! for this module
123 ! (in/out) tr_Reg - A pointer that is set to point to the control structure
124 ! for the tracer advection and diffusion module.
125 ! (in) restart_CS - A pointer to the restart control structure.
126 
127 ! This include declares and sets the variable "version".
128 #include "version_variable.h"
129  character(len=40) :: mod = "MOM_tracer_flow_control" ! This module's name.
130 
131  if (associated(cs)) then
132  call mom_error(warning, "call_tracer_register called with an associated "// &
133  "control structure.")
134  return
135  else ; allocate(cs) ; endif
136 
137  ! Read all relevant parameters and write them to the model log.
138  call log_version(param_file, mod, version, "")
139  call get_param(param_file, mod, "USE_USER_TRACER_EXAMPLE", &
140  cs%use_USER_tracer_example, &
141  "If true, use the USER_tracer_example tracer package.", &
142  default=.false.)
143  call get_param(param_file, mod, "USE_DOME_TRACER", cs%use_DOME_tracer, &
144  "If true, use the DOME_tracer tracer package.", &
145  default=.false.)
146  call get_param(param_file, mod, "USE_IDEAL_AGE_TRACER", cs%use_ideal_age, &
147  "If true, use the ideal_age_example tracer package.", &
148  default=.false.)
149  call get_param(param_file, mod, "USE_REGIONAL_DYES", cs%use_regional_dyes, &
150  "If true, use the regional_dyes tracer package.", &
151  default=.false.)
152  call get_param(param_file, mod, "USE_OIL_TRACER", cs%use_oil, &
153  "If true, use the oil_tracer tracer package.", &
154  default=.false.)
155  call get_param(param_file, mod, "USE_ADVECTION_TEST_TRACER", cs%use_advection_test_tracer, &
156  "If true, use the advection_test_tracer tracer package.", &
157  default=.false.)
158  call get_param(param_file, mod, "USE_OCMIP2_CFC", cs%use_OCMIP2_CFC, &
159  "If true, use the MOM_OCMIP2_CFC tracer package.", &
160  default=.false.)
161  call get_param(param_file, mod, "USE_generic_tracer", &
162  cs%use_MOM_generic_tracer, &
163  "If true and _USE_GENERIC_TRACER is defined as a \n"//&
164  "preprocessor macro, use the MOM_generic_tracer packages.", &
165  default=.false.)
166 
167 #ifndef _USE_GENERIC_TRACER
168  if (cs%use_MOM_generic_tracer) call mom_error(fatal, &
169  "call_tracer_register: use_MOM_generic_tracer=.true. BUT not compiled")
170 #endif
171 
172 ! Add other user-provided calls to register tracers for restarting here. Each
173 ! tracer package registration call returns a logical false if it cannot be run
174 ! for some reason. This then overrides the run-time selection from above.
175  if (cs%use_USER_tracer_example) cs%use_USER_tracer_example = &
176  user_register_tracer_example(hi, gv, param_file, cs%USER_tracer_example_CSp, &
177  tr_reg, restart_cs)
178  if (cs%use_DOME_tracer) cs%use_DOME_tracer = &
179  register_dome_tracer(hi, gv, param_file, cs%DOME_tracer_CSp, &
180  tr_reg, restart_cs)
181  if (cs%use_ideal_age) cs%use_ideal_age = &
182  register_ideal_age_tracer(hi, gv, param_file, cs%ideal_age_tracer_CSp, &
183  tr_reg, restart_cs)
184  if (cs%use_regional_dyes) cs%use_regional_dyes = &
185  register_dye_tracer(hi, gv, param_file, cs%dye_tracer_CSp, &
186  tr_reg, restart_cs)
187  if (cs%use_oil) cs%use_oil = &
188  register_oil_tracer(hi, gv, param_file, cs%oil_tracer_CSp, &
189  tr_reg, restart_cs)
190  if (cs%use_advection_test_tracer) cs%use_advection_test_tracer = &
191  register_advection_test_tracer(hi, gv, param_file, cs%advection_test_tracer_CSp, &
192  tr_reg, restart_cs)
193  if (cs%use_OCMIP2_CFC) cs%use_OCMIP2_CFC = &
194  register_ocmip2_cfc(hi, gv, param_file, cs%OCMIP2_CFC_CSp, &
195  tr_reg, restart_cs)
196 #ifdef _USE_GENERIC_TRACER
197  if (cs%use_MOM_generic_tracer) cs%use_MOM_generic_tracer = &
198  register_mom_generic_tracer(hi, gv, param_file, cs%MOM_generic_tracer_CSp, &
199  tr_reg, restart_cs)
200 #endif
201 
202 end subroutine call_tracer_register

204 subroutine tracer_flow_control_init(restart, day, G, GV, h, param_file, diag, OBC, &
205  cs, sponge_csp, ale_sponge_csp, diag_to_z_csp)
206  logical, intent(in) :: restart
207  type(time_type), target, intent(in) :: day
208  type(ocean_grid_type), intent(inout) :: G
209  type(verticalgrid_type), intent(in) :: GV
210  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
211  type(param_file_type), intent(in) :: param_file
212  type(diag_ctrl), target, intent(in) :: diag
213  type(ocean_obc_type), pointer :: OBC
214  type(tracer_flow_control_cs), pointer :: CS
215  type(sponge_cs), pointer :: sponge_CSp
216  type(ale_sponge_cs), pointer :: ALE_sponge_CSp
217  type(diag_to_z_cs), pointer :: diag_to_Z_CSp
218 ! This subroutine calls all registered tracer initialization
219 ! subroutines.
220 
221 ! Arguments: restart - 1 if the fields have already been read from
222 ! a restart file.
223 ! (in) day - Time of the start of the run.
224 ! (in) G - The ocean's grid structure.
225 ! (in) GV - The ocean's vertical grid structure.
226 ! (in) h - Layer thickness, in m (Boussinesq) or kg m-2 (non-Boussinesq).
227 ! (in) diag - A structure that is used to regulate diagnostic output.
228 ! (in) OBC - This open boundary condition type specifies whether, where,
229 ! and what open boundary conditions are used.
230 ! (in) CS - The control structure returned by a previous call to
231 ! call_tracer_register.
232 ! (in/out) sponge_CSp - A pointer to the control structure for the sponges, if
233 ! they are in use. Otherwise this may be unassociated.
234 ! (in/out) ALE_sponge_CSp - A pointer to the control structure for the ALE sponges, if they are in use. Otherwise this may be unassociated.
235 ! (in/out) diag_to_Z_Csp - A pointer to the control structure for diagnostics
236 ! in depth space.
237  if (.not. associated(cs)) call mom_error(fatal, "tracer_flow_control_init: "// &
238  "Module must be initialized via call_tracer_register before it is used.")
239 
240 ! Add other user-provided calls here.
241  if (cs%use_USER_tracer_example) &
242  call user_initialize_tracer(restart, day, g, gv, h, diag, obc, cs%USER_tracer_example_CSp, &
243  sponge_csp, diag_to_z_csp)
244  if (cs%use_DOME_tracer) &
245  call initialize_dome_tracer(restart, day, g, gv, h, diag, obc, cs%DOME_tracer_CSp, &
246  sponge_csp, diag_to_z_csp)
247  if (cs%use_ideal_age) &
248  call initialize_ideal_age_tracer(restart, day, g, gv, h, diag, obc, cs%ideal_age_tracer_CSp, &
249  sponge_csp, diag_to_z_csp)
250  if (cs%use_regional_dyes) &
251  call initialize_dye_tracer(restart, day, g, gv, h, diag, obc, cs%dye_tracer_CSp, &
252  sponge_csp, diag_to_z_csp)
253  if (cs%use_oil) &
254  call initialize_oil_tracer(restart, day, g, gv, h, diag, obc, cs%oil_tracer_CSp, &
255  sponge_csp, diag_to_z_csp)
256  if (cs%use_advection_test_tracer) &
257  call initialize_advection_test_tracer(restart, day, g, gv, h, diag, obc, cs%advection_test_tracer_CSp, &
258  sponge_csp, diag_to_z_csp)
259  if (cs%use_OCMIP2_CFC) &
260  call initialize_ocmip2_cfc(restart, day, g, gv, h, diag, obc, cs%OCMIP2_CFC_CSp, &
261  sponge_csp, diag_to_z_csp)
262 #ifdef _USE_GENERIC_TRACER
263  if (cs%use_MOM_generic_tracer) &
264  call initialize_mom_generic_tracer(restart, day, g, gv, h, param_file, diag, obc, &
265  cs%MOM_generic_tracer_CSp, sponge_csp, ale_sponge_csp, diag_to_z_csp)
266 #endif
267 
268 end subroutine tracer_flow_control_init

326 subroutine call_tracer_column_fns(h_old, h_new, ea, eb, fluxes, dt, G, GV, tv, optics, CS)
327  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h_old, h_new, ea, eb
328  type(forcing), intent(in) :: fluxes
329  real, intent(in) :: dt
330  type(ocean_grid_type), intent(in) :: G
331  type(verticalgrid_type), intent(in) :: GV
332  type(thermo_var_ptrs), intent(in) :: tv
333  type(optics_type), pointer :: optics
334  type(tracer_flow_control_cs), pointer :: CS
335 ! This subroutine calls all registered tracer column physics
336 ! subroutines.
337 
338 ! Arguments: h_old - Layer thickness before entrainment, in m (Boussinesq)
339 ! or kg m-2 (non-Boussinesq).
340 ! (in) h_new - Layer thickness after entrainment, in m or kg m-2.
341 ! (in) ea - an array to which the amount of fluid entrained
342 ! from the layer above during this call will be
343 ! added, in m or kg m-2, the same as h_old.
344 ! (in) eb - an array to which the amount of fluid entrained
345 ! from the layer below during this call will be
346 ! added, in m or kg m-2, the same as h_old.
347 ! (in) fluxes - A structure containing pointers to any possible
348 ! forcing fields. Unused fields have NULL ptrs.
349 ! (in) dt - The amount of time covered by this call, in s.
350 ! (in) G - The ocean's grid structure.
351 ! (in) GV - The ocean's vertical grid structure.
352 ! (in) tv - The structure containing thermodynamic variables.
353 ! (in) optics - The structure containing optical properties.
354 ! (in) CS - The control structure returned by a previous call to
355 ! call_tracer_register.
356 
357  if (.not. associated(cs)) call mom_error(fatal, "call_tracer_column_fns: "// &
358  "Module must be initialized via call_tracer_register before it is used.")
359 ! Add calls to tracer column functions here.
360  if (cs%use_USER_tracer_example) &
361  call tracer_column_physics(h_old, h_new, ea, eb, fluxes, dt, &
362  g, gv, cs%USER_tracer_example_CSp)
363  if (cs%use_DOME_tracer) &
364  call dome_tracer_column_physics(h_old, h_new, ea, eb, fluxes, dt, &
365  g, gv, cs%DOME_tracer_CSp)
366  if (cs%use_ideal_age) &
367  call ideal_age_tracer_column_physics(h_old, h_new, ea, eb, fluxes, dt, &
368  g, gv, cs%ideal_age_tracer_CSp)
369  if (cs%use_regional_dyes) &
370  call dye_tracer_column_physics(h_old, h_new, ea, eb, fluxes, dt, &
371  g, gv, cs%dye_tracer_CSp)
372  if (cs%use_oil) &
373  call oil_tracer_column_physics(h_old, h_new, ea, eb, fluxes, dt, &
374  g, gv, cs%oil_tracer_CSp, tv)
375  if (cs%use_advection_test_tracer) &
376  call advection_test_tracer_column_physics(h_old, h_new, ea, eb, fluxes, dt, &
377  g, gv, cs%advection_test_tracer_CSp)
378  if (cs%use_OCMIP2_CFC) &
379  call ocmip2_cfc_column_physics(h_old, h_new, ea, eb, fluxes, dt, &
380  g, gv, cs%OCMIP2_CFC_CSp)
381 #ifdef _USE_GENERIC_TRACER
382  if (cs%use_MOM_generic_tracer) &
383  call mom_generic_tracer_column_physics(h_old, h_new, ea, eb, fluxes, dt, &
384  g, gv, cs%MOM_generic_tracer_CSp, tv, optics)
385 #endif
386 
387 
388 end subroutine call_tracer_column_fns

391 subroutine call_tracer_stocks(h, stock_values, G, GV, CS, stock_names, stock_units, &
392  num_stocks, stock_index, got_min_max,global_min, global_max,xgmin, ygmin, zgmin, xgmax, ygmax, zgmax)
393  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
394  real, dimension(:), intent(out) :: stock_values
395  type(ocean_grid_type), intent(in) :: G
396  type(verticalgrid_type), intent(in) :: GV
397  type(tracer_flow_control_cs), pointer :: CS
398  character(len=*), dimension(:), optional, intent(out) :: stock_names
399  character(len=*), dimension(:), optional, intent(out) :: stock_units
400  integer, optional, intent(out) :: num_stocks
401  integer, optional, intent(in) :: stock_index
402  logical, dimension(:), optional, intent(inout) :: got_min_max
403  real, dimension(:), optional, intent(out) :: global_min, global_max
404  real, dimension(:), optional, intent(out) :: xgmin, ygmin, zgmin, xgmax, ygmax, zgmax
405 ! This subroutine calls all registered tracer packages to enable them to
406 ! add to the surface state returned to the coupler. These routines are optional.
407 
408 ! Arguments: h - Layer thickness, in m (Boussinesq) or kg m-2 (non-Boussinesq).
409 ! (out) stock_values - The integrated amounts of a tracer on the current
410 ! PE, usually in kg x concentration.
411 ! (in) G - The ocean's grid structure.
412 ! (in) GV - The ocean's vertical grid structure.
413 ! (in) CS - The control structure returned by a previous call to
414 ! call_tracer_register.
415 ! (out,opt) stock_names - Diagnostic names to use for each stock.
416 ! (out,opt) stock_units - Units to use in the metadata for each stock.
