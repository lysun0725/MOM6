33 !> MOM_initialize_coord sets up time-invariant quantities related to MOM6's
34 !! vertical coordinate.
35 subroutine mom_initialize_coord(GV, PF, write_geom, output_dir, tv, max_depth)
36  type(verticalgrid_type), intent(inout) :: GV !< Ocean vertical grid structure
37  type(param_file_type), intent(in) :: PF !< A structure indicating the open file
38  !! to parse for model parameter values.
39  logical, intent(in) :: write_geom !< If true, write grid geometry files.
40  character(len=*), intent(in) :: output_dir !< The directory into which to write files.
41  type(thermo_var_ptrs), intent(inout) :: tv !< The thermodynamic variable structure.
42  real, intent(in) :: max_depth !< The ocean's maximum depth, in m.
43  ! Local
44  character(len=200) :: config
45  logical :: debug
46 ! This include declares and sets the variable "version".
47 #include "version_variable.h"
48  integer :: nz
49  type(eos_type), pointer :: eos => null()
50 
51  if (associated(tv%eqn_of_state)) eos => tv%eqn_of_state
52 
53  nz = gv%ke
54 
55  call calltree_enter("MOM_initialize_coord(), MOM_coord_initialization.F90")
56 ! call log_version(PF, mod, version)
57  call get_param(pf, mod, "DEBUG", debug, default=.false.)
58 
59 ! Set-up the layer densities, GV%Rlay, and reduced gravities, GV%g_prime.
60  call get_param(pf, mod, "COORD_CONFIG", config, &
61  "This specifies how layers are to be defined: \n"//&
62  " \t file - read coordinate information from the file \n"//&
63  " \t\t specified by (COORD_FILE).\n"//&
64  " \t linear - linear based on interfaces not layers \n"//&
65  " \t layer_ref - linear based on layer densities \n"//&
66  " \t ts_ref - use reference temperature and salinity \n"//&
67  " \t ts_range - use range of temperature and salinity \n"//&
68  " \t\t (T_REF and S_REF) to determine surface density \n"//&
69  " \t\t and GINT calculate internal densities. \n"//&
70  " \t gprime - use reference density (RHO_0) for surface \n"//&
71  " \t\t density and GINT calculate internal densities. \n"//&
72  " \t ts_profile - use temperature and salinity profiles \n"//&
73  " \t\t (read from COORD_FILE) to set layer densities. \n"//&
74  " \t USER - call a user modified routine.", &
75  fail_if_missing=.true.)
76  select case ( trim(config) )
77  case ("gprime")
78  call set_coord_from_gprime(gv%Rlay, gv%g_prime, gv, pf)
79  case ("layer_ref")
80  call set_coord_from_layer_density(gv%Rlay, gv%g_prime, gv, pf)
81  case ("linear")
82  call set_coord_linear(gv%Rlay, gv%g_prime, gv, pf)
83  case ("ts_ref")
84  call set_coord_from_ts_ref(gv%Rlay, gv%g_prime, gv, pf, eos, tv%P_Ref)
85  case ("ts_profile")
86  call set_coord_from_ts_profile(gv%Rlay, gv%g_prime, gv, pf, eos, tv%P_Ref)
87  case ("ts_range")
88  call set_coord_from_ts_range(gv%Rlay, gv%g_prime, gv, pf, eos, tv%P_Ref)
89  case ("file")
90  call set_coord_from_file(gv%Rlay, gv%g_prime, gv, pf)
91  case ("USER")
92  call user_set_coord(gv%Rlay, gv%g_prime, gv, pf, eos)
93  case ("none")
94  case default ; call mom_error(fatal,"MOM_initialize_coord: "// &
95  "Unrecognized coordinate setup"//trim(config))
96  end select
97  if (debug) call chksum(gv%Rlay, "MOM_initialize_coord: Rlay ", 1, nz)
98  if (debug) call chksum(gv%g_prime, "MOM_initialize_coord: g_prime ", 1, nz)
99  call setverticalgridaxes( gv%Rlay, gv )
100 
101 ! Copy the maximum depth across from the input argument
102  gv%max_depth = max_depth
103 
104 ! Write out all of the grid data used by this run.
105  if (write_geom) call write_vertgrid_file(gv, pf, output_dir)
106 
107  call calltree_leave('MOM_initialize_coord()')
108 
109 end subroutine mom_initialize_coord

114 subroutine set_coord_from_gprime(Rlay, g_prime, GV, param_file)
115  real, dimension(:), intent(out) :: Rlay, g_prime
116  type(verticalgrid_type), intent(in) :: GV
117  type(param_file_type), intent(in) :: param_file
118 ! Arguments: Rlay - the layers' target coordinate values (potential density).
119 ! (out) g_prime - the reduced gravity across the interfaces, in m s-2.
120 ! (in) GV - The ocean's vertical grid structure.
121 ! (in) param_file - A structure indicating the open file to parse for
122 ! model parameter values.
123 
124 ! This subroutine sets the layer densities (Rlay) and the interface !
125 ! reduced gravities (g). !
126  real :: g_int ! Reduced gravities across the internal interfaces, in m s-2.
127  real :: g_fs ! Reduced gravity across the free surface, in m s-2.
128  character(len=40) :: mod = "set_coord_from_gprime" ! This subroutine's name.
129  integer :: k, nz
130  nz = gv%ke
131 
132  call calltree_enter(trim(mod)//"(), MOM_coord_initialization.F90")
133 
134  call get_param(param_file, mod, "GFS" , g_fs, &
135  "The reduced gravity at the free surface.", units="m s-2", &
136  default=gv%g_Earth)
137  call get_param(param_file, mod, "GINT", g_int, &
138  "The reduced gravity across internal interfaces.", &
139  units="m s-2", fail_if_missing=.true.)
140 
141  g_prime(1) = g_fs
142  do k=2,nz ; g_prime(k) = g_int ; enddo
143  rlay(1) = gv%Rho0
144  do k=2,nz ; rlay(k) = rlay(k-1) + g_prime(k)*(gv%Rho0/gv%g_Earth) ; enddo
145 
146  call calltree_leave(trim(mod)//'()')
147 
148 end subroutine set_coord_from_gprime

152 subroutine set_coord_from_layer_density(Rlay, g_prime, GV, param_file)
153  real, dimension(:), intent(out) :: Rlay, g_prime
154  type(verticalgrid_type), intent(in) :: GV
155  type(param_file_type), intent(in) :: param_file
156 ! Arguments: Rlay - the layers' target coordinate values (potential density).
157 ! (out) g_prime - the reduced gravity across the interfaces, in m s-2.
158 ! (in) GV - The ocean's vertical grid structure.
159 ! (in) param_file - A structure indicating the open file to parse for
160 ! model parameter values.
161 
162 ! This subroutine sets the layer densities (Rlay) and the interface !
163 ! reduced gravities (g). !
164  real :: g_fs ! Reduced gravity across the free surface, in m s-2.
165  real :: Rlay_Ref! The surface layer's target density, in kg m-3.
166  real :: RLay_range ! The range of densities, in kg m-3.
167  character(len=40) :: mod = "set_coord_from_layer_density" ! This subroutine's name.
168  integer :: k, nz
169  nz = gv%ke
170 
171  call calltree_enter(trim(mod)//"(), MOM_coord_initialization.F90")
172 
173  call get_param(param_file, mod, "GFS", g_fs, &
174  "The reduced gravity at the free surface.", units="m s-2", &
175  default=gv%g_Earth)
176  call get_param(param_file, mod, "LIGHTEST_DENSITY", rlay_ref, &
177  "The reference potential density used for layer 1.", &
178  units="kg m-3", default=gv%Rho0)
179  call get_param(param_file, mod, "DENSITY_RANGE", rlay_range, &
180  "The range of reference potential densities in the layers.", &

182 
183  g_prime(1) = g_fs
184  rlay(1) = rlay_ref
185  do k=2,nz
186  rlay(k) = rlay(k-1) + rlay_range/(real(nz-1))
187  enddo
188 ! These statements set the interface reduced gravities. !
189  do k=2,nz
190  g_prime(k) = (gv%g_Earth/gv%Rho0) * (rlay(k) - rlay(k-1))
191  enddo
192 
193  call calltree_leave(trim(mod)//'()')
194 end subroutine set_coord_from_layer_density

307 subroutine set_coord_from_ts_range(Rlay, g_prime, GV, param_file, &
308  eqn_of_state, p_ref)
309  real, dimension(:), intent(out) :: Rlay, g_prime
310  type(verticalgrid_type), intent(in) :: GV
311  type(param_file_type), intent(in) :: param_file
312  type(eos_type), pointer :: eqn_of_state
313  real, intent(in) :: P_Ref
314 ! Arguments: Rlay - the layers' target coordinate values (potential density).
315 ! (out) g_prime - the reduced gravity across the interfaces, in m s-2.
316 ! (in) GV - The ocean's vertical grid structure.
317 ! (in) param_file - A structure indicating the open file to parse for
318 ! model parameter values.
319 ! (in) eqn_of_state - integer that selects equation of state
320 ! (in) P_Ref - The coordinate-density reference pressure in Pa.
321 
322 ! This subroutine sets the layer densities (Rlay) and the interface !
323 ! reduced gravities (g). !
324  real, dimension(SZK_(GV)) :: T0, S0, Pref
325  real :: S_Ref, S_Light, S_Dense ! Salnity range parameters in PSU.
326  real :: T_Ref, T_Light, T_Dense ! Temperature range parameters in dec C.
327  real :: res_rat ! The ratio of density space resolution in the denser part
328  ! of the range to that in the lighter part of the range.
329  ! Setting this greater than 1 increases the resolution for
330  ! the denser water.
331  real :: g_fs ! Reduced gravity across the free surface, in m s-2.
332  real :: a1, frac_dense, k_frac
333  integer :: k, nz, k_light
334  character(len=40) :: mod = "set_coord_from_TS_range" ! This subroutine's name.
335  character(len=200) :: filename, coord_file, inputdir ! Strings for file/path
336  nz = gv%ke
337 
338  call calltree_enter(trim(mod)//"(), MOM_coord_initialization.F90")
339 
340  call get_param(param_file, mod, "T_REF", t_ref, &
341  "The default initial temperatures.", units="degC", default=10.0)
342  call get_param(param_file, mod, "TS_RANGE_T_LIGHT", t_light, &
343  "The initial temperature of the lightest layer when \n"//&
344  "COORD_CONFIG is set to ts_range.", units="degC", default=t_ref)
345  call get_param(param_file, mod, "TS_RANGE_T_DENSE", t_dense, &
346  "The initial temperature of the densest layer when \n"//&
347  "COORD_CONFIG is set to ts_range.", units="degC", default=t_ref)
348 
349  call get_param(param_file, mod, "S_REF", s_ref, &
350  "The default initial salinities.", units="PSU", default=35.0)
351  call get_param(param_file, mod, "TS_RANGE_S_LIGHT", s_light, &
352  "The initial lightest salinities when COORD_CONFIG \n"//&
353  "is set to ts_range.", default = s_ref, units="PSU")
354  call get_param(param_file, mod, "TS_RANGE_S_DENSE", s_dense, &
355  "The initial densest salinities when COORD_CONFIG \n"//&
356  "is set to ts_range.", default = s_ref, units="PSU")
357 
358  call get_param(param_file, mod, "TS_RANGE_RESOLN_RATIO", res_rat, &
359  "The ratio of density space resolution in the densest \n"//&
360  "part of the range to that in the lightest part of the \n"//&
361  "range when COORD_CONFIG is set to ts_range. Values \n"//&
362  "greater than 1 increase the resolution of the denser water.",&
363  default=1.0, units="nondim")
364 
365  call get_param(param_file, mod, "GFS", g_fs, &
366  "The reduced gravity at the free surface.", units="m s-2", &
367  default=gv%g_Earth)
368 
369  k_light = gv%nk_rho_varies + 1
370 
371  ! Set T0(k) to range from T_LIGHT to T_DENSE, and simliarly for S0(k).
372  t0(k_light) = t_light ; s0(k_light) = s_light
373  a1 = 2.0 * res_rat / (1.0 + res_rat)
374  do k=k_light+1,nz
375  k_frac = real(k-k_light)/real(nz-k_light)
376  frac_dense = a1 * k_frac + (1.0 - a1) * k_frac**2
377  t0(k) = frac_dense * (t_dense - t_light) + t_light
378  s0(k) = frac_dense * (s_dense - s_light) + s_light
379  enddo
380 
381  g_prime(1) = g_fs
382  do k=1,nz ; pref(k) = p_ref ; enddo
383  call calculate_density(t0, s0, pref, rlay, k_light,nz-k_light+1,eqn_of_state)
384  ! Extrapolate target densities for the variable density mixed and buffer layers.
385  do k=k_light-1,1,-1
386  rlay(k) = 2.0*rlay(k+1) - rlay(k+2)
387  enddo
388  do k=2,nz; g_prime(k) = (gv%g_Earth/gv%Rho0) * (rlay(k) - rlay(k-1)); enddo
389 
390  call calltree_leave(trim(mod)//'()')
391 end subroutine set_coord_from_ts_range

446 subroutine set_coord_linear(Rlay, g_prime, GV, param_file)
447  real, dimension(:), intent(out) :: Rlay, g_prime
448  type(verticalgrid_type), intent(in) :: GV
449  type(param_file_type), intent(in) :: param_file
450 ! Arguments: Rlay - the layers' target coordinate values (potential density).
451 ! (out) g_prime - the reduced gravity across the interfaces, in m s-2.
452 ! (in) GV - The ocean's vertical grid structure.
453 ! (in) param_file - A structure indicating the open file to parse for
454 ! model parameter values.
455 
456 ! This subroutine sets the layer densities (Rlay) and the interface
457 ! reduced gravities (g) according to a linear profile starting at a
458 ! reference surface layer density and spanning a range of densities
459 ! to the bottom defined by the parameter RLAY_RANGE
460 ! (defaulting to 2.0 if not defined)
461  character(len=40) :: mod = "set_coord_linear" ! This subroutine
462  real :: Rlay_ref, Rlay_range, g_fs
