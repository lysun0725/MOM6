122 !> Perform a fully implicit vertical diffusion
123 !! of momentum. Stress top and bottom boundary conditions are used.
124 !!
125 !! This is solving the tridiagonal system
126 !! \f[ \left(h_k + a_{k + 1/2} + a_{k - 1/2} + r_k\right) u_k^{n+1}
127 !! = h_k u_k^n + a_{k + 1/2} u_{k+1}^{n+1} + a_{k - 1/2} u_{k-1}^{n+1} \f]
128 !! where \f$a_{k + 1/2} = \Delta t \nu_{k + 1/2} / h_{k + 1/2}\f$
129 !! is the <em>interfacial coupling thickness per time step</em>,
130 !! encompassing background viscosity as well as contributions from
131 !! enhanced mixed and bottom layer viscosities.
132 !! $r_k$ is a Rayleight drag term due to channel drag.
133 !! There is an additional stress term on the right-hand side
134 !! if DIRECT_STRESS is true, applied to the surface layer.
135 
136 subroutine vertvisc(u, v, h, fluxes, visc, dt, OBC, ADp, CDp, G, GV, CS, &
137  taux_bot, tauy_bot)
138  type(ocean_grid_type), intent(in) :: G !< Ocean grid structure
139  type(verticalgrid_type), intent(in) :: GV !< Ocean vertical grid structure
140  real, intent(inout), &
141  dimension(SZIB_(G),SZJ_(G),SZK_(GV)) :: u !< Zonal velocity in m s-1
142  real, intent(inout), &
143  dimension(SZI_(G),SZJB_(G),SZK_(GV)) :: v !< Meridional velocity in m s-1
144  real, intent(in), &
145  dimension(SZI_(G),SZJ_(G),SZK_(GV)) :: h !< Layer thickness in H
146  type(forcing), intent(in) :: fluxes !< Structure containing forcing fields
147  type(vertvisc_type), intent(inout) :: visc !< Viscosities and bottom drag
148  real, intent(in) :: dt !< Time increment in s
149  type(ocean_obc_type), pointer :: OBC !< Open boundary condition structure
150  type(accel_diag_ptrs), intent(inout) :: ADp !< Accelerations in the momentum
151  !! equations for diagnostics
152  type(cont_diag_ptrs), intent(inout) :: CDp !< Continuity equation terms
153  type(vertvisc_cs), pointer :: CS !< Vertical viscosity control structure
154  !> Zonal bottom stress from ocean to rock in Pa
155  real, optional, intent(out), dimension(SZIB_(G),SZJ_(G)) :: taux_bot
156  !> Meridional bottom stress from ocean to rock in Pa
157  real, optional, intent(out), dimension(SZI_(G),SZJB_(G)) :: tauy_bot
158 
159  ! Fields from fluxes used in this subroutine:
160  ! taux: Zonal wind stress in Pa.
161  ! tauy: Meridional wind stress in Pa.
162 
163  ! Local variables
164 
165  real :: b1(szib_(g)) ! b1 and c1 are variables used by the
166  real :: c1(szib_(g),szk_(g)) ! tridiagonal solver. c1 is nondimensional,
167  ! while b1 has units of inverse thickness.
168  real :: d1(szib_(g)) ! d1=1-c1 is used by the tridiagonal solver, ND.
169  real :: Ray(szib_(g),szk_(g)) ! Ray is the Rayleigh-drag velocity in m s-1
170  real :: b_denom_1 ! The first term in the denominator of b1, in H.
171 
172  real :: Hmix ! The mixed layer thickness over which stress
173  ! is applied with direct_stress, translated into
174  ! thickness units - either m or kg m-2.
175  real :: I_Hmix ! The inverse of Hmix, in m-1 or m2 kg-1.
176  real :: Idt ! The inverse of the time step, in s-1.
177  real :: dt_Rho0 ! The time step divided by the mean
178  ! density, in s m3 kg-1.
179  real :: Rho0 ! A density used to convert drag laws into stress in
180  ! Pa, in kg m-3.
181  real :: dt_m_to_H ! The time step times the conversion from m to the
182  ! units of thickness - either s or s m3 kg-1.
183  real :: h_neglect ! A thickness that is so small it is usually lost
184  ! in roundoff and can be neglected, in m or kg m-2.
185 
186  real :: stress ! The surface stress times the time step, divided
187  ! by the density, in units of m2 s-1.
188  real :: zDS, hfr, h_a ! Temporary variables used with direct_stress.
189  real :: surface_stress(szib_(g))! The same as stress, unless the wind
190  ! stress is applied as a body force, in
191  ! units of m2 s-1.
192 
193  logical :: do_i(szib_(g))
194 
195  integer :: i, j, k, is, ie, Isq, Ieq, Jsq, Jeq, nz
196  is = g%isc ; ie = g%iec
197  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB ; nz = g%ke
198 
199  if (.not.associated(cs)) call mom_error(fatal,"MOM_vert_friction(visc): "// &
200  "Module must be initialized before it is used.")
201 
202  if (cs%direct_stress) then
203  hmix = cs%Hmix_stress*gv%m_to_H
204  i_hmix = 1.0 / hmix
205  endif
206  dt_rho0 = dt/gv%H_to_kg_m2
207  dt_m_to_h = dt*gv%m_to_H
208  rho0 = gv%Rho0
209  h_neglect = gv%H_subroundoff
210  idt = 1.0 / dt
211 
212  do k=1,nz ; do i=isq,ieq ; ray(i,k) = 0.0 ; enddo ; enddo
213 
214  ! Update the zonal velocity component using a modification of a standard
215  ! tridagonal solver.
216 !$OMP parallel do default(none) shared(G,Isq,Ieq,ADp,nz,u,CS,dt_Rho0,fluxes,h, &
217 !$OMP h_neglect,Hmix,I_Hmix,visc,dt_m_to_H, &
218 !$OMP Idt,taux_bot,Rho0) &
219 !$OMP firstprivate(Ray) &
220 !$OMP private(do_i,surface_stress,zDS,stress,h_a,hfr, &
221 !$OMP b_denom_1,b1,d1,c1)
222  do j=g%jsc,g%jec
223  do i=isq,ieq ; do_i(i) = (g%mask2dCu(i,j) > 0) ; enddo
224 
225  if (ASSOCIATED(adp%du_dt_visc)) then ; do k=1,nz ; do i=isq,ieq
226  adp%du_dt_visc(i,j,k) = u(i,j,k)
227  enddo ; enddo ; endif
228 
229 ! One option is to have the wind stress applied as a body force
230 ! over the topmost Hmix fluid. If DIRECT_STRESS is not defined,
231 ! the wind stress is applied as a stress boundary condition.
232  if (cs%direct_stress) then
233  do i=isq,ieq ; if (do_i(i)) then
234  surface_stress(i) = 0.0
235  zds = 0.0
236  stress = dt_rho0 * fluxes%taux(i,j)
237  do k=1,nz
238  h_a = 0.5 * (h(i,j,k) + h(i+1,j,k)) + h_neglect
239  hfr = 1.0 ; if ((zds+h_a) > hmix) hfr = (hmix - zds) / h_a
240  u(i,j,k) = u(i,j,k) + i_hmix * hfr * stress
241  zds = zds + h_a ; if (zds >= hmix) exit
242  enddo
243  endif ; enddo ! end of i loop
244  else ; do i=isq,ieq
245  surface_stress(i) = dt_rho0 * (g%mask2dCu(i,j)*fluxes%taux(i,j))
246  enddo ; endif ! direct_stress
247 
248  if (cs%Channel_drag) then ; do k=1,nz ; do i=isq,ieq
249  ray(i,k) = visc%Ray_u(i,j,k)
250  enddo ; enddo ; endif
251 
252  ! perform forward elimination on the tridiagonal system
253  !
254  ! denote the diagonal of the system as b_i, the subdiagonal as a_i
255  ! and the superdiagonal as c_i. The right-hand side terms are d_i.
256  !
257  ! ignoring the rayleigh drag contribution,
258  ! we have a_i = -dt_m_to_H * a_u(i)
259  ! b_i = h_u(i) + dt_m_to_H * (a_u(i) + a_u(i+1))
260  ! c_i = -dt_m_to_H * a_u(i+1)
261  !
262  ! for forward elimination, we want to:
263  ! calculate c'_i = - c_i / (b_i + a_i c'_(i-1))
264  ! and d'_i = (d_i - a_i d'_(i-1)) / (b_i + a_i c'_(i-1))
265  ! where c'_1 = c_1/b_1 and d'_1 = d_1/b_1
266  ! (see Thomas' tridiagonal matrix algorithm)
267  !
268  ! b1 is the denominator term 1 / (b_i + a_i c'_(i-1))
269  ! b_denom_1 is (b_i + a_i + c_i) - a_i(1 - c'_(i-1))
270  ! = (b_i + c_i + c'_(i-1))
271  ! this is done so that d1 = b1 * b_denom_1 = 1 - c'_(i-1)
272  ! c1(i) is -c'_(i - 1)
273  ! and the right-hand-side is destructively updated to be d'_i
274  !
275  do i=isq,ieq ; if (do_i(i)) then
276  b_denom_1 = cs%h_u(i,j,1) + dt_m_to_h * (ray(i,1) + cs%a_u(i,j,1))
277  b1(i) = 1.0 / (b_denom_1 + dt_m_to_h*cs%a_u(i,j,2))
278  d1(i) = b_denom_1 * b1(i)
279  u(i,j,1) = b1(i) * (cs%h_u(i,j,1) * u(i,j,1) + surface_stress(i))
280  endif ; enddo
281  do k=2,nz ; do i=isq,ieq ; if (do_i(i)) then
282  c1(i,k) = dt_m_to_h * cs%a_u(i,j,k) * b1(i)
283  b_denom_1 = cs%h_u(i,j,k) + dt_m_to_h * (ray(i,k) + cs%a_u(i,j,k)*d1(i))
284  b1(i) = 1.0 / (b_denom_1 + dt_m_to_h * cs%a_u(i,j,k+1))
285  d1(i) = b_denom_1 * b1(i)
286  u(i,j,k) = (cs%h_u(i,j,k) * u(i,j,k) + &
287  dt_m_to_h * cs%a_u(i,j,k) * u(i,j,k-1)) * b1(i)
288  endif ; enddo ; enddo
289 
290  ! back substitute to solve for the new velocities
291  ! u_i = d'_i - c'_i x_(i+1)
292  do k=nz-1,1,-1 ; do i=isq,ieq ; if (do_i(i)) then
293  u(i,j,k) = u(i,j,k) + c1(i,k+1) * u(i,j,k+1)
294  endif ; enddo ; enddo ! i and k loops
295 
296  if (ASSOCIATED(adp%du_dt_visc)) then ; do k=1,nz ; do i=isq,ieq
297  adp%du_dt_visc(i,j,k) = (u(i,j,k) - adp%du_dt_visc(i,j,k))*idt
298  enddo ; enddo ; endif
299 
300  if (ASSOCIATED(visc%taux_shelf)) then ; do i=isq,ieq
301  visc%taux_shelf(i,j) = -rho0*cs%a1_shelf_u(i,j)*u(i,j,1) ! - u_shelf?
302  enddo ; endif
303 
304  if (PRESENT(taux_bot)) then
305  do i=isq,ieq
306  taux_bot(i,j) = rho0 * (u(i,j,nz)*cs%a_u(i,j,nz+1))
307  enddo
308  if (cs%Channel_drag) then ; do k=1,nz ; do i=isq,ieq
309  taux_bot(i,j) = taux_bot(i,j) + rho0 * (ray(i,k)*u(i,j,k))
310  enddo ; enddo ; endif
311  endif
312  enddo ! end u-component j loop
313 
314  ! Now work on the meridional velocity component.
315 !$OMP parallel do default(none) shared(G,Jsq,Jeq,ADp,nz,v,CS,dt_Rho0,fluxes,h, &
316 !$OMP Hmix,I_Hmix,visc,dt_m_to_H,Idt,Rho0, &
317 !$OMP tauy_bot,is,ie) &
318 !$OMP firstprivate(Ray) &
319 !$OMP private(do_i,surface_stress,zDS,stress,h_a,hfr, &
320 !$OMP b_denom_1,b1,d1,c1)
321  do j=jsq,jeq
322  do i=is,ie ; do_i(i) = (g%mask2dCv(i,j) > 0) ; enddo
323 
324  if (ASSOCIATED(adp%dv_dt_visc)) then ; do k=1,nz ; do i=is,ie
325  adp%dv_dt_visc(i,j,k) = v(i,j,k)
326  enddo ; enddo ; endif
327 
328 ! One option is to have the wind stress applied as a body force
329 ! over the topmost Hmix fluid. If DIRECT_STRESS is not defined,
330 ! the wind stress is applied as a stress boundary condition.
331  if (cs%direct_stress) then
332  do i=is,ie ; if (do_i(i)) then
333  surface_stress(i) = 0.0
334  zds = 0.0
335  stress = dt_rho0 * fluxes%tauy(i,j)
336  do k=1,nz
337  h_a = 0.5 * (h(i,j,k) + h(i,j+1,k))
338  hfr = 1.0 ; if ((zds+h_a) > hmix) hfr = (hmix - zds) / h_a
339  v(i,j,k) = v(i,j,k) + i_hmix * hfr * stress
340  zds = zds + h_a ; if (zds >= hmix) exit
341  enddo
342  endif ; enddo ! end of i loop
343  else ; do i=is,ie
344  surface_stress(i) = dt_rho0 * (g%mask2dCv(i,j)*fluxes%tauy(i,j))
345  enddo ; endif ! direct_stress
346 
347  if (cs%Channel_drag) then ; do k=1,nz ; do i=is,ie
348  ray(i,k) = visc%Ray_v(i,j,k)
349  enddo ; enddo ; endif
350 
351  do i=is,ie ; if (do_i(i)) then
352  b_denom_1 = cs%h_v(i,j,1) + dt_m_to_h * (ray(i,1) + cs%a_v(i,j,1))
353  b1(i) = 1.0 / (b_denom_1 + dt_m_to_h*cs%a_v(i,j,2))
354  d1(i) = b_denom_1 * b1(i)
355  v(i,j,1) = b1(i) * (cs%h_v(i,j,1) * v(i,j,1) + surface_stress(i))
356  endif ; enddo
357  do k=2,nz ; do i=is,ie ; if (do_i(i)) then
358  c1(i,k) = dt_m_to_h * cs%a_v(i,j,k) * b1(i)
359  b_denom_1 = cs%h_v(i,j,k) + dt_m_to_h * (ray(i,k) + cs%a_v(i,j,k)*d1(i))
360  b1(i) = 1.0 / (b_denom_1 + dt_m_to_h * cs%a_v(i,j,k+1))
361  d1(i) = b_denom_1 * b1(i)
362  v(i,j,k) = (cs%h_v(i,j,k) * v(i,j,k) + dt_m_to_h * &
363  cs%a_v(i,j,k) * v(i,j,k-1)) * b1(i)
364  endif ; enddo ; enddo
365  do k=nz-1,1,-1 ; do i=is,ie ; if (do_i(i)) then
366  v(i,j,k) = v(i,j,k) + c1(i,k+1) * v(i,j,k+1)
367  endif ; enddo ; enddo ! i and k loops
368 
369  if (ASSOCIATED(adp%dv_dt_visc)) then ; do k=1,nz ; do i=is,ie
370  adp%dv_dt_visc(i,j,k) = (v(i,j,k) - adp%dv_dt_visc(i,j,k))*idt
371  enddo ; enddo ; endif
372 
373  if (ASSOCIATED(visc%tauy_shelf)) then ; do i=is,ie
374  visc%tauy_shelf(i,j) = -rho0*cs%a1_shelf_v(i,j)*v(i,j,1) ! - v_shelf?
375  enddo ; endif
376 
377  if (present(tauy_bot)) then
378  do i=is,ie
379  tauy_bot(i,j) = rho0 * (v(i,j,nz)*cs%a_v(i,j,nz+1))
380  enddo
381  if (cs%Channel_drag) then ; do k=1,nz ; do i=is,ie
382  tauy_bot(i,j) = tauy_bot(i,j) + rho0 * (ray(i,k)*v(i,j,k))
383  enddo ; enddo ; endif
384  endif
385  enddo ! end of v-component J loop
386 
387  call vertvisc_limit_vel(u, v, h, adp, cdp, fluxes, visc, dt, g, gv, cs)
388 
389  ! Here the velocities associated with open boundary conditions are applied.
390  if (associated(obc)) then
391  if (obc%apply_OBC_u) then
392  do k=1,nz ; do j=g%jsc,g%jec ; do i=isq,ieq
393  if (obc%OBC_mask_u(i,j) .and. (obc%OBC_kind_u(i,j) == obc_simple)) &
394  u(i,j,k) = obc%u(i,j,k)
395  enddo ; enddo ; enddo
396  endif
397  if (obc%apply_OBC_v) then
398  do k=1,nz ; do j=jsq,jeq ; do i=g%isc,g%iec
399  if (obc%OBC_mask_v(i,j) .and. (obc%OBC_kind_v(i,j) == obc_simple)) &
400  v(i,j,k) = obc%v(i,j,k)
401  enddo ; enddo ; enddo
402  endif
403  endif
404 ! Offer diagnostic fields for averaging.
405  if (cs%id_du_dt_visc > 0) &
406  call post_data(cs%id_du_dt_visc, adp%du_dt_visc, cs%diag)
407  if (cs%id_dv_dt_visc > 0) &
408  call post_data(cs%id_dv_dt_visc, adp%dv_dt_visc, cs%diag)
409  if (present(taux_bot) .and. (cs%id_taux_bot > 0)) &
410  call post_data(cs%id_taux_bot, taux_bot, cs%diag)

416 !> Calculate the fraction of momentum originally in a layer that remains
417 !! after a time-step of viscosity, and the fraction of a time-step's
418 !! worth of barotropic acceleration that a layer experiences after
419 !! viscosity is applied.
420 subroutine vertvisc_remnant(visc, visc_rem_u, visc_rem_v, dt, G, GV, CS)
421  type(ocean_grid_type), intent(in) :: G !< Ocean grid structure
422  type(verticalgrid_type), intent(in) :: GV !< Ocean vertical grid structure
423  type(vertvisc_type), intent(in) :: visc !< Viscosities and bottom drag
424  !> Fraction of a time-step's worth of a barotopic acceleration that
425  !! a layer experiences after viscosity is applied in the zonal direction
426  real, dimension(SZIB_(G),SZJ_(G),SZK_(GV)), intent(inout) :: visc_rem_u
427  !> Fraction of a time-step's worth of a barotopic acceleration that
428  !! a layer experiences after viscosity is applied in the meridional direction
429  real, dimension(SZI_(G),SZJB_(G),SZK_(GV)), intent(inout) :: visc_rem_v
430  real, intent(in) :: dt !< Time increment in s
431  type(vertvisc_cs), pointer :: CS !< Vertical viscosity control structure
432 
433  ! Local variables
434 
435  real :: b1(szib_(g)) ! b1 and c1 are variables used by the
436  real :: c1(szib_(g),szk_(g)) ! tridiagonal solver. c1 is nondimensional,
437  ! while b1 has units of inverse thickness.
438  real :: d1(szib_(g)) ! d1=1-c1 is used by the tridiagonal solver, ND.
439  real :: Ray(szib_(g),szk_(g)) ! Ray is the Rayleigh-drag velocity times the
440  ! time step, in m.
441  real :: b_denom_1 ! The first term in the denominator of b1, in m or kg m-2.
442  real :: dt_m_to_H ! The time step times the conversion from m to the
443  ! units of thickness - either s or s m3 kg-1.
444  logical :: do_i(szib_(g))
445 
446  integer :: i, j, k, is, ie, Isq, Ieq, Jsq, Jeq, nz
447  is = g%isc ; ie = g%iec
448  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB ; nz = g%ke
449 
450  if (.not.associated(cs)) call mom_error(fatal,"MOM_vert_friction(visc): "// &
451  "Module must be initialized before it is used.")
452 
453  dt_m_to_h = dt*gv%m_to_H
454 
455  do k=1,nz ; do i=isq,ieq ; ray(i,k) = 0.0 ; enddo ; enddo
456 
457  ! Find the zonal viscous using a modification of a standard tridagonal solver.
458 !$OMP parallel do default(none) shared(G,Isq,Ieq,CS,nz,visc,dt_m_to_H,visc_rem_u) &
459 !$OMP firstprivate(Ray) &
460 !$OMP private(do_i,b_denom_1,b1,d1,c1)
461  do j=g%jsc,g%jec
462  do i=isq,ieq ; do_i(i) = (g%mask2dCu(i,j) > 0) ; enddo
463 
464  if (cs%Channel_drag) then ; do k=1,nz ; do i=isq,ieq
465  ray(i,k) = visc%Ray_u(i,j,k)
466  enddo ; enddo ; endif
467 
468  do i=isq,ieq ; if (do_i(i)) then
469  b_denom_1 = cs%h_u(i,j,1) + dt_m_to_h * (ray(i,1) + cs%a_u(i,j,1))
470  b1(i) = 1.0 / (b_denom_1 + dt_m_to_h*cs%a_u(i,j,2))
471  d1(i) = b_denom_1 * b1(i)
472  visc_rem_u(i,j,1) = b1(i) * cs%h_u(i,j,1)
473  endif ; enddo
474  do k=2,nz ; do i=isq,ieq ; if (do_i(i)) then
475  c1(i,k) = dt_m_to_h * cs%a_u(i,j,k)*b1(i)
476  b_denom_1 = cs%h_u(i,j,k) + dt_m_to_h * (ray(i,k) + cs%a_u(i,j,k)*d1(i))
477  b1(i) = 1.0 / (b_denom_1 + dt_m_to_h * cs%a_u(i,j,k+1))
478  d1(i) = b_denom_1 * b1(i)
479  visc_rem_u(i,j,k) = (cs%h_u(i,j,k) + dt_m_to_h * cs%a_u(i,j,k) * visc_rem_u(i,j,k-1)) * b1(i)
480  endif ; enddo ; enddo
481  do k=nz-1,1,-1 ; do i=isq,ieq ; if (do_i(i)) then
482  visc_rem_u(i,j,k) = visc_rem_u(i,j,k) + c1(i,k+1)*visc_rem_u(i,j,k+1)
483 
484  endif ; enddo ; enddo ! i and k loops
485 
486  enddo ! end u-component j loop
487 
488  ! Now find the meridional viscous using a modification.
489 !$OMP parallel do default(none) shared(Jsq,Jeq,is,ie,G,CS,visc,dt_m_to_H,visc_rem_v,nz) &
490 !$OMP firstprivate(Ray) &
491 !$OMP private(do_i,b_denom_1,b1,d1,c1)
492  do j=jsq,jeq
493  do i=is,ie ; do_i(i) = (g%mask2dCv(i,j) > 0) ; enddo
494 
495  if (cs%Channel_drag) then ; do k=1,nz ; do i=is,ie
496  ray(i,k) = visc%Ray_v(i,j,k)
497  enddo ; enddo ; endif
498 
499  do i=is,ie ; if (do_i(i)) then
500  b_denom_1 = cs%h_v(i,j,1) + dt_m_to_h * (ray(i,1) + cs%a_v(i,j,1))
501  b1(i) = 1.0 / (b_denom_1 + dt_m_to_h*cs%a_v(i,j,2))
502  d1(i) = b_denom_1 * b1(i)
503  visc_rem_v(i,j,1) = b1(i) * cs%h_v(i,j,1)
504  endif ; enddo
505  do k=2,nz ; do i=is,ie ; if (do_i(i)) then
506  c1(i,k) = dt_m_to_h * cs%a_v(i,j,k)*b1(i)
507  b_denom_1 = cs%h_v(i,j,k) + dt_m_to_h * (ray(i,k) + cs%a_v(i,j,k)*d1(i))
508  b1(i) = 1.0 / (b_denom_1 + dt_m_to_h * cs%a_v(i,j,k+1))
509  d1(i) = b_denom_1 * b1(i)
510  visc_rem_v(i,j,k) = (cs%h_v(i,j,k) + dt_m_to_h * cs%a_v(i,j,k) * visc_rem_v(i,j,k-1)) * b1(i)
511  endif ; enddo ; enddo
512  do k=nz-1,1,-1 ; do i=is,ie ; if (do_i(i)) then
513  visc_rem_v(i,j,k) = visc_rem_v(i,j,k) + c1(i,k+1)*visc_rem_v(i,j,k+1)
514  endif ; enddo ; enddo ! i and k loops
515  enddo ! end of v-component J loop
516 
517  if (cs%debug) then
518  call uchksum(visc_rem_u,"visc_rem_u",g,haloshift=0)
519  call vchksum(visc_rem_v,"visc_rem_v",g,haloshift=0)
520  endif
521 
522 end subroutine vertvisc_remnant

525 !> Calculate the coupling coefficients (CS%a_u and CS%a_v)
526 !! and effective layer thicknesses (CS%h_u and CS%h_v) for later use in the
527 !! applying the implicit vertical viscosity via vertvisc().
528 subroutine vertvisc_coef(u, v, h, fluxes, visc, dt, G, GV, CS)
529  type(ocean_grid_type), intent(in) :: G !< Ocean grid structure
530  type(verticalgrid_type), intent(in) :: GV !< Ocean vertical grid structure
531  real, intent(in), &
532  dimension(SZIB_(G),SZJ_(G),SZK_(GV)) :: u !< Zonal velocity in m s-1
533  real, intent(in), &
534  dimension(SZI_(G),SZJB_(G),SZK_(GV)) :: v !< Meridional velocity in m s-1
535  real, intent(in), &
536  dimension(SZI_(G),SZJ_(G),SZK_(GV)) :: h !< Layer thickness in H
537  type(forcing), intent(in) :: fluxes !< Structure containing forcing fields
538  type(vertvisc_type), intent(in) :: visc !< Viscosities and bottom drag
539  real, intent(in) :: dt !< Time increment in s
540  type(vertvisc_cs), pointer :: CS !< Vertical viscosity control structure
541 
542  ! Field from fluxes used in this subroutine:
543  ! ustar: the friction velocity in m s-1, used here as the mixing
544  ! velocity in the mixed layer if NKML > 1 in a bulk mixed layer.
545 
546  ! Local variables
547 
548  real, dimension(SZIB_(G),SZK_(G)) :: &
549  h_harm, & ! Harmonic mean of the thicknesses around a velocity grid point,
550  ! given by 2*(h+ * h-)/(h+ + h-), in m or kg m-2 (H for short).
551  hvel, & ! hvel is the thickness used at a velocity grid point, in H.
552  hvel_shelf ! The equivalent of hvel under shelves, in H.
553  real, dimension(SZIB_(G),SZK_(G)+1) :: &
554  a, & ! The drag coefficients across interfaces, in m s-1. a times
555  ! the velocity difference gives the stress across an interface.
556  a_shelf, & ! The drag coefficients across interfaces in water columns under
557  ! ice shelves, in m s-1.
558  z_i ! An estimate of each interface's height above the bottom,
559  ! normalized by the bottom boundary layer thickness, nondim.
560  real, dimension(SZIB_(G)) :: &
561  kv_bbl, & ! The bottom boundary layer viscosity in m2 s-1.
562  bbl_thick, & ! The bottom boundary layer thickness in m or kg m-2.
563  I_Hbbl, & ! The inverse of the bottom boundary layer thickness, in units
564  ! of H-1 (i.e., m-1 or m2 kg-1).
565  i_htbl, & ! The inverse of the top boundary layer thickness, in units
566  ! of H-1 (i.e., m-1 or m2 kg-1).
567  zcol1, & ! The height of the interfaces to the north and south of a
568  zcol2, & ! v-point, in m or kg m-2.
569  ztop_min, & ! The deeper of the two adjacent surface heights, in H.
570  dmin, & ! The shallower of the two adjacent bottom depths converted to
571  ! thickness units, in m or kg m-2.
572  zh, & ! An estimate of the interface's distance from the bottom
573  ! based on harmonic mean thicknesses, in m or kg m-2.
574  h_ml ! The mixed layer depth, in m or kg m-2.
575  real, allocatable, dimension(:,:) :: hML_u, hML_v
576  real :: zcol(szi_(g)) ! The height of an interface at h-points, in m or kg m-2.
577  real :: botfn ! A function which goes from 1 at the bottom to 0 much more
578  ! than Hbbl into the interior.
579  real :: topfn ! A function which goes from 1 at the top to 0 much more
580  ! than Htbl into the interior.
581  real :: z2 ! The distance from the bottom, normalized by Hbbl, nondim.
582  real :: z2_wt ! A nondimensional (0-1) weight used when calculating z2.
583  real :: h_neglect ! A thickness that is so small it is usually lost
584  ! in roundoff and can be neglected, in H.
585  real :: H_to_m, m_to_H ! Unit conversion factors.
586 
587  real :: h_arith ! The arithmetic mean thickness, in m or kg m-2.
588  real :: I_valBL ! The inverse of a scaling factor determining when water is
589  ! still within the boundary layer, as determined by the sum
590  ! of the harmonic mean thicknesses.
591  logical, dimension(SZIB_(G)) :: do_i, do_i_shelf
592  logical :: do_any_shelf
593 
594  integer :: i, j, k, is, ie, js, je, Isq, Ieq, Jsq, Jeq, nz
595  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
596  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB ; nz = g%ke
597 
598  if (.not.associated(cs)) call mom_error(fatal,"MOM_vert_friction(coef): "// &
599  "Module must be initialized before it is used.")
600 
601  h_neglect = gv%H_subroundoff
602  h_to_m = gv%H_to_m ; m_to_h = gv%m_to_H
603  i_hbbl(:) = 1.0 / (cs%Hbbl * gv%m_to_H + h_neglect)
604  i_valbl = 0.0 ; if (cs%harm_BL_val > 0.0) i_valbl = 1.0 / cs%harm_BL_val
605 
606  if (cs%debug .or. (cs%id_hML_u > 0)) then
607  allocate(hml_u(g%IsdB:g%IedB,g%jsd:g%jed)) ; hml_u(:,:) = 0.0
608  endif
609  if (cs%debug .or. (cs%id_hML_v > 0)) then
610  allocate(hml_v(g%isd:g%ied,g%JsdB:g%JedB)) ; hml_v(:,:) = 0.0
611  endif
612 
613  if ((associated(visc%taux_shelf) .or. associated(fluxes%frac_shelf_u)) .and. &
614  .not.associated(cs%a1_shelf_u)) then
615  allocate(cs%a1_shelf_u(g%IsdB:g%IedB,g%jsd:g%jed)) ; cs%a1_shelf_u(:,:)=0.0
616  endif
617  if ((associated(visc%tauy_shelf) .or. associated(fluxes%frac_shelf_u)) .and. &
618  .not.associated(cs%a1_shelf_v)) then
619  allocate(cs%a1_shelf_v(g%isd:g%ied,g%JsdB:g%JedB)) ; cs%a1_shelf_v(:,:)=0.0
620  endif

622 !$OMP parallel do default(none) shared(G,GV,CS,visc,Isq,ieq,nz,u,h,fluxes,hML_u, &
623 !$OMP h_neglect,dt,m_to_H,I_valBL) &
624 !$OMP firstprivate(i_hbbl) &
625 !$OMP private(do_i,kv_bbl,bbl_thick,z_i,h_harm,h_arith,hvel,z2, &
626 !$OMP botfn,zh,Dmin,zcol,a,do_any_shelf,do_i_shelf, &
627 !$OMP a_shelf,Ztop_min,I_HTbl,hvel_shelf,topfn,h_ml,z2_wt)
628  do j=g%Jsc,g%Jec
629  do i=isq,ieq ; do_i(i) = (g%mask2dCu(i,j) > 0) ; enddo
630 
631  if (cs%bottomdraglaw) then ; do i=isq,ieq
632  kv_bbl(i) = visc%kv_bbl_u(i,j)
633  bbl_thick(i) = visc%bbl_thick_u(i,j) * m_to_h
634  if (do_i(i)) i_hbbl(i) = 1.0 / (bbl_thick(i) + h_neglect)
635  enddo ; endif
636 
637 ! The following block calculates the thicknesses at velocity
638 ! grid points for the vertical viscosity (hvel[k]). Near the
639 ! bottom an upwind biased thickness is used to control the effect
640 ! of spurious Montgomery potential gradients at the bottom where
641 ! nearly massless layers layers ride over the topography.
642  if (cs%harmonic_visc) then
643  do i=isq,ieq ; z_i(i,nz+1) = 0.0 ; enddo
644  do k=nz,1,-1 ; do i=isq,ieq ; if (do_i(i)) then
645  h_harm(i,k) = 2.0*h(i,j,k)*h(i+1,j,k) / (h(i,j,k)+h(i+1,j,k)+h_neglect)
646  h_arith = 0.5*(h(i+1,j,k)+h(i,j,k))
647 
648  hvel(i,k) = h_harm(i,k)
649  if (u(i,j,k) * (h(i+1,j,k)-h(i,j,k)) < 0) then
650  z2 = z_i(i,k+1) ; botfn = 1.0 / (1.0 + 0.09*z2*z2*z2*z2*z2*z2)
651  hvel(i,k) = (1.0-botfn)*h_harm(i,k) + botfn*h_arith
652  endif
653  z_i(i,k) = z_i(i,k+1) + h_harm(i,k)*i_hbbl(i)
654  endif ; enddo ; enddo ! i & k loops
655  else ! Not harmonic_visc
656  do i=isq,ieq
657  zh(i) = 0.0 ; z_i(i,nz+1) = 0.0
658  dmin(i) = min(g%bathyT(i,j), g%bathyT(i+1,j)) * m_to_h
659  enddo
660  do i=isq,ieq+1 ; zcol(i) = -g%bathyT(i,j) * m_to_h ; enddo
661  do k=nz,1,-1
662  do i=isq,ieq+1 ; zcol(i) = zcol(i) + h(i,j,k) ; enddo
663  do i=isq,ieq ; if (do_i(i)) then
664  h_harm(i,k) = 2.0*h(i,j,k)*h(i+1,j,k) / (h(i,j,k)+h(i+1,j,k)+h_neglect)
665  h_arith = 0.5*(h(i+1,j,k)+h(i,j,k))
666  zh(i) = zh(i) + h_harm(i,k)
667  z_i(i,k) = max(zh(i), max(zcol(i),zcol(i+1)) + dmin(i)) * i_hbbl(i)
668 
669  hvel(i,k) = h_arith
670  if (u(i,j,k) * (h(i+1,j,k)-h(i,j,k)) > 0) then
671  if (zh(i) * i_hbbl(i) < cs%harm_BL_val) then
672  hvel(i,k) = h_harm(i,k)
673  else
674  z2_wt = 1.0 ; if (zh(i) * i_hbbl(i) < 2.0*cs%harm_BL_val) &
675  z2_wt = max(0.0, min(1.0, zh(i) * i_hbbl(i) * i_valbl - 1.0))
676  z2 = z2_wt * (max(zh(i), max(zcol(i),zcol(i+1)) + dmin(i)) * i_hbbl(i))
677  botfn = 1.0 / (1.0 + 0.09*z2*z2*z2*z2*z2*z2)
678  hvel(i,k) = (1.0-botfn)*h_arith + botfn*h_harm(i,k)
679  endif
680  endif
681 
682  endif ; enddo ! i loop
683  enddo ! k loop
684  endif
685 
686  call find_coupling_coef(a, hvel, do_i, h_harm, bbl_thick, kv_bbl, z_i, h_ml, &
687  dt, j, g, gv, cs, visc, fluxes, work_on_u=.true.)
688  if (allocated(hml_u)) then
689  do i=isq,ieq ; if (do_i(i)) then ; hml_u(i,j) = h_ml(i) ; endif ; enddo
690  endif
691  do_any_shelf = .false.
692  if (associated(fluxes%frac_shelf_u)) then
693  do i=isq,ieq
694  cs%a1_shelf_u(i,j) = 0.0
695  do_i_shelf(i) = (do_i(i) .and. fluxes%frac_shelf_u(i,j) > 0.0)
696  if (do_i_shelf(i)) do_any_shelf = .true.
697  enddo
698  if (do_any_shelf) then
699  if (cs%harmonic_visc) then
700  call find_coupling_coef(a_shelf, hvel, do_i_shelf, h_harm, bbl_thick, &
701  kv_bbl, z_i, h_ml, dt, j, g, gv, cs, &
702  visc, fluxes, work_on_u=.true., shelf=.true.)
703  else ! Find upwind-biased thickness near the surface.
704  ! Perhaps this needs to be done more carefully, via find_eta.
705  do i=isq,ieq ; if (do_i_shelf(i)) then
706  zh(i) = 0.0 ; ztop_min(i) = min(zcol(i), zcol(i+1))
707  i_htbl(i) = 1.0 / (visc%tbl_thick_shelf_u(i,j)*m_to_h + h_neglect)
708  endif ; enddo
709  do k=1,nz
710  do i=isq,ieq+1 ; zcol(i) = zcol(i) - h(i,j,k) ; enddo
711  do i=isq,ieq ; if (do_i_shelf(i)) then
712  h_arith = 0.5*(h(i+1,j,k)+h(i,j,k))
713  zh(i) = zh(i) + h_harm(i,k)
714 
715  hvel_shelf(i,k) = hvel(i,k)
716  if (u(i,j,k) * (h(i+1,j,k)-h(i,j,k)) > 0) then
717  if (zh(i) * i_htbl(i) < cs%harm_BL_val) then
718  hvel_shelf(i,k) = min(hvel(i,k), h_harm(i,k))
719  else
720  z2_wt = 1.0 ; if (zh(i) * i_htbl(i) < 2.0*cs%harm_BL_val) &
721  z2_wt = max(0.0, min(1.0, zh(i) * i_htbl(i) * i_valbl - 1.0))
722  z2 = z2_wt * (max(zh(i), ztop_min(i) - min(zcol(i),zcol(i+1))) * i_htbl(i))
723  topfn = 1.0 / (1.0 + 0.09*z2**6)
724  hvel_shelf(i,k) = min(hvel(i,k), (1.0-topfn)*h_arith + topfn*h_harm(i,k))
725  endif
726  endif
727  endif ; enddo
728  enddo
729  call find_coupling_coef(a_shelf, hvel_shelf, do_i_shelf, h_harm, &
730  bbl_thick, kv_bbl, z_i, h_ml, dt, j, g, gv, cs, &
731  visc, fluxes, work_on_u=.true., shelf=.true.)
732  endif
733  do i=isq,ieq ; if (do_i_shelf(i)) cs%a1_shelf_u(i,j) = a_shelf(i,1) ; enddo
734  endif
735  endif
736 
737  if (do_any_shelf) then
738  do k=1,nz+1 ; do i=isq,ieq ; if (do_i_shelf(i)) then
739  cs%a_u(i,j,k) = fluxes%frac_shelf_u(i,j) * a_shelf(i,k) + &
740  (1.0-fluxes%frac_shelf_u(i,j)) * a(i,k)
741 ! This is Alistair's suggestion, but it destabilizes the model. I do not know why. RWH
742 ! CS%a_u(I,j,K) = fluxes%frac_shelf_u(I,j) * max(a_shelf(I,K), a(I,K)) + &
743 ! (1.0-fluxes%frac_shelf_u(I,j)) * a(I,K)
744  elseif (do_i(i)) then
745  cs%a_u(i,j,k) = a(i,k)
746  endif ; enddo ; enddo
747  do k=1,nz ; do i=isq,ieq ; if (do_i_shelf(i)) then
748  ! Should we instead take the inverse of the average of the inverses?
749  cs%h_u(i,j,k) = fluxes%frac_shelf_u(i,j) * hvel_shelf(i,k) + &
750  (1.0-fluxes%frac_shelf_u(i,j)) * hvel(i,k)
751  elseif (do_i(i)) then
752  cs%h_u(i,j,k) = hvel(i,k)
753  endif ; enddo ; enddo
754  else
755  do k=1,nz+1 ; do i=isq,ieq ; if (do_i(i)) cs%a_u(i,j,k) = a(i,k) ; enddo ; enddo
756  do k=1,nz ; do i=isq,ieq ; if (do_i(i)) cs%h_u(i,j,k) = hvel(i,k) ; enddo ; enddo
757  endif
758 
759  enddo

762  ! Now work on v-points.
763 !$OMP parallel do default(none) shared(G,GV,CS,visc,is,ie,Jsq,Jeq,nz,v,h,fluxes,hML_v, &
764 !$OMP h_neglect,dt,m_to_H,I_valBL) &
765 !$OMP firstprivate(i_hbbl) &
766 !$OMP private(do_i,kv_bbl,bbl_thick,z_i,h_harm,h_arith,hvel,z2, &

770  do i=is,ie ; do_i(i) = (g%mask2dCv(i,j) > 0) ; enddo
771 
772  if (cs%bottomdraglaw) then ; do i=is,ie
773  kv_bbl(i) = visc%kv_bbl_v(i,j)
774  bbl_thick(i) = visc%bbl_thick_v(i,j) * m_to_h
775  if (do_i(i)) i_hbbl(i) = 1.0 / bbl_thick(i)
776  enddo ; endif

778 ! The following block calculates the thicknesses at velocity
779 ! grid points for the vertical viscosity (hvel[k]). Near the
780 ! bottom an upwind biased thickness is used to control the effect
781 ! of spurious Montgomery potential gradients at the bottom where
782 ! nearly massless layers layers ride over the topography.
783  if (cs%harmonic_visc) then
784  do i=is,ie ; z_i(i,nz+1) = 0.0 ; enddo
785

787  h_harm(i,k) = 2.0*h(i,j,k)*h(i,j+1,k) / (h(i,j,k)+h(i,j+1,k)+h_neglect)
788  h_arith = 0.5*(h(i,j+1,k)+h(i,j,k))

793  hvel(i,k) = (1.0-botfn)*h_harm(i,k) + botfn*h_arith
794  endif
795  z_i(i,k) = z_i(i,k+1) + h_harm(i,k)*i_hbbl(i)
796  endif ; enddo ; enddo ! i & k loops
797  else ! Not harmonic_visc
798  do i=is,ie
799  zh(i) = 0.0 ; z_i(i,nz+1) = 0.0
800  dmin(i) = min(g%bathyT(i,j), g%bathyT(i,j+1)) * m_to_h
801  zcol1(i) = -g%bathyT(i,j) * m_to_h
802  zcol2(i) = -g%bathyT(i,j+1) * m_to_h
803  enddo
804  do k=nz,1,-1 ; do i=is,ie ; if (do_i(i)) then
805  h_harm(i,k) = 2.0*h(i,j,k)*h(i,j+1,k) / (h(i,j,k)+h(i,j+1,k)+h_neglect)
806  h_arith = 0.5*(h(i,j+1,k)+h(i,j,k))
807  zh(i) = zh(i) + h_harm(i,k)
808  zcol1(i) = zcol1(i) + h(i,j,k) ; zcol2(i) = zcol2(i) + h(i,j+1,k)
809  z_i(i,k) = max(zh(i), max(zcol1(i),zcol2(i)) + dmin(i)) * i_hbbl(i)
810 
811  hvel(i,k) = h_arith
812  if (v(i,j,k) * (h(i,j+1,k)-h(i,j,k)) > 0) then
813  if (zh(i) * i_hbbl(i) < cs%harm_BL_val) then
814  hvel(i,k) = h_harm(i,k)
815  else
816  z2_wt = 1.0 ; if (zh(i) * i_hbbl(i) < 2.0*cs%harm_BL_val) &
817  z2_wt = max(0.0, min(1.0, zh(i) * i_hbbl(i) * i_valbl - 1.0))
818  z2 = z2_wt * (max(zh(i), max(zcol1(i),zcol2(i)) + dmin(i)) * i_hbbl(i))
819  botfn = 1.0 / (1.0 + 0.09*z2*z2*z2*z2*z2*z2)
820  hvel(i,k) = (1.0-botfn)*h_arith + botfn*h_harm(i,k)
821  endif
822  endif
823 
824  endif ; enddo ; enddo ! i & k loops
825  endif
826 
827  call find_coupling_coef(a, hvel, do_i, h_harm, bbl_thick, kv_bbl, z_i, h_ml, &
828  dt, j, g, gv, cs, visc, fluxes, work_on_u=.false.)
829  if ( allocated(hml_v)) then
830  do i=is,ie ; if (do_i(i)) then ; hml_v(i,j) = h_ml(i) ; endif ; enddo
831  endif
832  do_any_shelf = .false.
833  if (associated(fluxes%frac_shelf_v)) then
834  do i=is,ie
835  cs%a1_shelf_v(i,j) = 0.0
836  do_i_shelf(i) = (do_i(i) .and. fluxes%frac_shelf_v(i,j) > 0.0)
837  if (do_i_shelf(i)) do_any_shelf = .true.
838  enddo
839  if (do_any_shelf) then
840  if (cs%harmonic_visc) then
841  call find_coupling_coef(a_shelf, hvel, do_i_shelf, h_harm, bbl_thick, &
842  kv_bbl, z_i, h_ml, dt, j, g, gv, cs, visc, &
843  fluxes, work_on_u=.false., shelf=.true.)
844  else ! Find upwind-biased thickness near the surface.
845  ! Perhaps this needs to be done more carefully, via find_eta.
846  do i=is,ie ; if (do_i_shelf(i)) then
847  zh(i) = 0.0 ; ztop_min(i) = min(zcol1(i), zcol2(i))
848  i_htbl(i) = 1.0 / (visc%tbl_thick_shelf_v(i,j)*m_to_h + h_neglect)
849  endif ; enddo
850  do k=1,nz
851  do i=is,ie ; if (do_i_shelf(i)) then
852  zcol1(i) = zcol1(i) - h(i,j,k) ; zcol2(i) = zcol2(i) - h(i,j+1,k)
853  h_arith = 0.5*(h(i,j+1,k)+h(i,j,k))
854  zh(i) = zh(i) + h_harm(i,k)
855 
856  hvel_shelf(i,k) = hvel(i,k)
857  if (v(i,j,k) * (h(i,j+1,k)-h(i,j,k)) > 0) then
858  if (zh(i) * i_htbl(i) < cs%harm_BL_val) then
859  hvel_shelf(i,k) = min(hvel(i,k), h_harm(i,k))
860  else
861  z2_wt = 1.0 ; if (zh(i) * i_htbl(i) < 2.0*cs%harm_BL_val) &
862  z2_wt = max(0.0, min(1.0, zh(i) * i_htbl(i) * i_valbl - 1.0))
863  z2 = z2_wt * (max(zh(i), ztop_min(i) - min(zcol1(i),zcol2(i))) * i_htbl(i))
864  topfn = 1.0 / (1.0 + 0.09*z2**6)
865  hvel_shelf(i,k) = min(hvel(i,k), (1.0-topfn)*h_arith + topfn*h_harm(i,k))
866  endif
867  endif
868  endif ; enddo
869  enddo
870  call find_coupling_coef(a_shelf, hvel_shelf, do_i_shelf, h_harm, &
871  bbl_thick, kv_bbl, z_i, h_ml, dt, j, g, gv, cs, &
872  visc, fluxes, work_on_u=.false., shelf=.true.)
873  endif
874  do i=is,ie ; if (do_i_shelf(i)) cs%a1_shelf_v(i,j) = a_shelf(i,1) ; enddo
875  endif
876  endif
877 
878  if (do_any_shelf) then
879  do k=1,nz+1 ; do i=is,ie ; if (do_i_shelf(i)) then
880  cs%a_v(i,j,k) = fluxes%frac_shelf_v(i,j) * a_shelf(i,k) + &
881  (1.0-fluxes%frac_shelf_v(i,j)) * a(i,k)
882 ! This is Alistair's suggestion, but it destabilizes the model. I do not know why. RWH
883 ! CS%a_v(i,J,K) = fluxes%frac_shelf_v(i,J) * max(a_shelf(i,K), a(i,K)) + &
884 ! (1.0-fluxes%frac_shelf_v(i,J)) * a(i,K)
885  elseif (do_i(i)) then
886  cs%a_v(i,j,k) = a(i,k)
887  endif ; enddo ; enddo
888  do k=1,nz ; do i=is,ie ; if (do_i_shelf(i)) then
889  ! Should we instead take the inverse of the average of the inverses?
890  cs%h_v(i,j,k) = fluxes%frac_shelf_v(i,j) * hvel_shelf(i,k) + &
891  (1.0-fluxes%frac_shelf_v(i,j)) * hvel(i,k)
892  elseif (do_i(i)) then
893  cs%h_v(i,j,k) = hvel(i,k)
894  endif ; enddo ; enddo
895  else
896  do k=1,nz+1 ; do i=is,ie ; if (do_i(i)) cs%a_v(i,j,k) = a(i,k) ; enddo ; enddo
897  do k=1,nz ; do i=is,ie ; if (do_i(i)) cs%h_v(i,j,k) = hvel(i,k) ; enddo ; enddo
898  endif
899  enddo ! end of v-point j loop
900 
901 
902  if (cs%debug) then
903  call uchksum(cs%h_u*h_to_m,"vertvisc_coef h_u",g,haloshift=0)
904  call vchksum(cs%h_v*h_to_m,"vertvisc_coef h_v",g,haloshift=0)
905  call uchksum(cs%a_u,"vertvisc_coef a_u",g,haloshift=0)
906  call vchksum(cs%a_v,"vertvisc_coef a_v",g,haloshift=0)
907  if (allocated(hml_u)) call uchksum(hml_u*h_to_m,"vertvisc_coef hML_u",g,haloshift=0)
908  if (allocated(hml_v)) call vchksum(hml_v*h_to_m,"vertvisc_coef hML_v",g,haloshift=0)
909  endif
910 
911 ! Offer diagnostic fields for averaging.
912  if (cs%id_au_vv > 0) call post_data(cs%id_au_vv, cs%a_u, cs%diag)
913  if (cs%id_av_vv > 0) call post_data(cs%id_av_vv, cs%a_v, cs%diag)
914  if (cs%id_h_u > 0) call post_data(cs%id_h_u, cs%h_u, cs%diag)
915  if (cs%id_h_v > 0) call post_data(cs%id_h_v, cs%h_v, cs%diag)
916  if (cs%id_hML_u > 0) call post_data(cs%id_hML_u, hml_u, cs%diag)
917  if (cs%id_hML_v > 0) call post_data(cs%id_hML_v, hml_v, cs%diag)
918 
919  if (allocated(hml_u)) deallocate(hml_u)
920  if (allocated(hml_v)) deallocate(hml_v)

924 !> Calculate the 'coupling coefficient' (a[k]) at the
925 !! interfaces. If BOTTOMDRAGLAW is defined, the minimum of Hbbl and half the
926 !! adjacent layer thicknesses are used to calculate a[k] near the bottom.
927 subroutine find_coupling_coef(a, hvel, do_i, h_harm, bbl_thick, kv_bbl, z_i, h_ml, &
928  dt, j, g, gv, cs, visc, fluxes, work_on_u, shelf)
929  type(ocean_grid_type), intent(in) :: G !< Ocean grid structure
930  type(verticalgrid_type), intent(in) :: GV !< Ocean vertical grid structure
931  !> Coupling coefficient across interfaces, in m s-1
932  real, dimension(SZIB_(G),SZK_(GV)+1), intent(out) :: a
933  !> Thickness at velocity points, in H
934  real, dimension(SZIB_(G),SZK_(GV)), intent(in) :: hvel
935  !> If true, determine coupling coefficient for a column
936  logical, dimension(SZIB_(G)), intent(in) :: do_i
937  !> Harmonic mean of thicknesses around a velocity grid point, in H
938  real, dimension(SZIB_(G),SZK_(GV)), intent(in) :: h_harm
939  !> Bottom boundary layer thickness, in H
940  real, dimension(SZIB_(G)), intent(in) :: bbl_thick
941  !> Bottom boundary layer viscosity, in m2 s-1
942  real, dimension(SZIB_(G)), intent(in) :: kv_bbl
943  !> Estimate of interface heights above the bottom,
944  !! normalised by the bottom boundary layer thickness
945  real, dimension(SZIB_(G),SZK_(GV)+1), intent(in) :: z_i
946  !> Mixed layer depth, in H
947  real, dimension(SZIB_(G)), intent(out) :: h_ml
948  !> j-index to find coupling coefficient for
949  integer, intent(in) :: j
950  !> Time increment, in s
951  real, intent(in) :: dt
952  !> Vertical viscosity control structure
953  type(vertvisc_cs), pointer :: CS
954  !> Structure containing viscosities and bottom drag
955  type(vertvisc_type), intent(in) :: visc
956  !> Structure containing forcing fields
957  type(forcing), intent(in) :: fluxes
958  !> If true, u-points are being calculated, otherwise v-points
959  logical, intent(in) :: work_on_u
960  !> If present and true, use a surface boundary condition
961  !! appropriate for an ice shelf.
962  logical, optional, intent(in) :: shelf
963 
964  ! Local variables
965 
966  real, dimension(SZIB_(G)) :: &
967  u_star, & ! ustar at a velocity point, in m s-1.
968  absf, & ! The average of the neighboring absolute values of f, in s-1.
969 ! h_ml, & ! The mixed layer depth, in m or kg m-2.
970  nk_visc, & ! The (real) interface index of the base of mixed layer.
971  z_t, & ! The distance from the top, sometimes normalized
972  ! by Hmix, in m or nondimensional.
973  kv_tbl, &
974  tbl_thick
975  real :: h_shear ! The distance over which shears occur, m or kg m-2.
976  real :: r ! A thickness to compare with Hbbl, in m or kg m-2.
977  real :: visc_ml ! The mixed layer viscosity, in m2 s-1.
978  real :: I_Hmix ! The inverse of the mixed layer thickness, in m-1 or m2 kg-1.
979  real :: a_ml ! The layer coupling coefficient across an interface in
980  ! the mixed layer, in m s-1.
981  real :: temp1 ! A temporary variable in m2 s-1.
982  real :: h_neglect ! A thickness that is so small it is usually lost
983  ! in roundoff and can be neglected, in H.
984  real :: dz_neglect ! A thickness in m that is so small it is usually lost
985  ! in roundoff and can be neglected, in m.
986  real :: z2 ! A copy of z_i, nondim.
987  real :: H_to_m, m_to_H ! Unit conversion factors.
988  real :: topfn
989  real :: a_top
990  logical :: do_shelf
991  integer :: i, k, is, ie, max_nk
992  integer :: nz
993  real :: botfn
994 
995  a(:,:) = 0.0
996 
997  if (work_on_u) then ; is = g%IscB ; ie = g%IecB
998  else ; is = g%isc ; ie = g%iec ; endif
999  nz = g%ke
1000  h_neglect = gv%H_subroundoff
1001  h_to_m = gv%H_to_m ; m_to_h = gv%m_to_H
1002  dz_neglect = gv%H_subroundoff*gv%H_to_m
1003 
1004  do_shelf = .false. ; if (present(shelf)) do_shelf = shelf
1005  h_ml(:) = 0.0
1006 
1007 ! The following loop calculates the vertical average velocity and
1008 ! surface mixed layer contributions to the vertical viscosity.
1009  do i=is,ie ; a(i,1) = 0.0 ; enddo
1010  if ((gv%nkml>0) .or. do_shelf) then ; do k=2,nz ; do i=is,ie
1011  if (do_i(i)) a(i,k) = 2.0*cs%Kv
1012  enddo ; enddo ; else
1013  i_hmix = 1.0 / (cs%Hmix * m_to_h + h_neglect)
1014  do i=is,ie ; z_t(i) = h_neglect*i_hmix ; enddo
1015  do k=2,nz ; do i=is,ie ; if (do_i(i)) then
1016  z_t(i) = z_t(i) + h_harm(i,k-1)*i_hmix
1017  a(i,k) = 2.0*cs%Kv + 2.0*cs%Kvml / ((z_t(i)*z_t(i)) * &
1018  (1.0 + 0.09*z_t(i)*z_t(i)*z_t(i)*z_t(i)*z_t(i)*z_t(i)))
1019  endif ; enddo ; enddo
1020  endif
1021 
1022  do i=is,ie ; if (do_i(i)) then
1023  if (cs%bottomdraglaw) then
1024  r = hvel(i,nz)*0.5
1025  if (r < bbl_thick(i)) then
1026  a(i,nz+1) = 1.0*kv_bbl(i) / (1e-10*dt*kv_bbl(i) + r*h_to_m)
1027  else
1028  a(i,nz+1) = 1.0*kv_bbl(i) / (1e-10*dt*kv_bbl(i) + bbl_thick(i)*h_to_m)
1029  endif
1030  else
1031  a(i,nz+1) = 2.0*cs%Kvbbl / (hvel(i,nz)*h_to_m + 2.0e-10*dt*cs%Kvbbl)
1032  endif
1033  endif ; enddo
1034 
1035  if (associated(visc%Kv_turb)) then
1036  ! BGR/ Add factor of 2. * the averaged Kv_turb.
1037  ! this is needed to reproduce the analytical solution to
1038  ! a simple diffusion problem, likely due to h_shear being
1039  ! equal to 2 x \delta z
1040  if (work_on_u) then
1041  do k=nz,2,-1 ; do i=is,ie ; if (do_i(i)) then
1042  a(i,k) = a(i,k) + (2.*0.5)*(visc%Kv_turb(i,j,k) + visc%Kv_turb(i+1,j,k))
1043  endif ; enddo ; enddo
1044  else
1045  do k=nz,2,-1 ; do i=is,ie ; if (do_i(i)) then
1046  a(i,k) = a(i,k) + (2.*0.5)*(visc%Kv_turb(i,j,k) + visc%Kv_turb(i,j+1,k))
1047  endif ; enddo ; enddo
1048  endif
1049  endif
1050 
1051  do k=nz,2,-1 ; do i=is,ie ; if (do_i(i)) then
1052  ! botfn determines when a point is within the influence of the bottom
1053  ! boundary layer, going from 1 at the bottom to 0 in the interior.
1054  z2 = z_i(i,k)
1055  botfn = 1.0 / (1.0 + 0.09*z2*z2*z2*z2*z2*z2)
1056 
1057  if (cs%bottomdraglaw) then
1058  a(i,k) = a(i,k) + 2.0*(kv_bbl(i)-cs%Kv)*botfn
1059  r = (hvel(i,k)+hvel(i,k-1))
1060  if (r > 2.0*bbl_thick(i)) then
1061  h_shear = ((1.0 - botfn) * r + botfn*2.0*bbl_thick(i))
1062  else
1063  h_shear = r
1064  endif
1065  else
1066  a(i,k) = a(i,k) + 2.0*(cs%Kvbbl-cs%Kv)*botfn
1067  h_shear = hvel(i,k) + hvel(i,k-1) + h_neglect
1068  endif
1069 
1070  ! Up to this point a has units of m2 s-1, but now is converted to m s-1.
1071  ! The term including 1e-10 in the denominators is here to avoid
1072  ! truncation error problems in the tridiagonal solver. Effectively, this
1073  ! sets the maximum coupling coefficient at 1e10 m.
1074  a(i,k) = a(i,k) / (h_shear*h_to_m + 1.0e-10*dt*a(i,k))
1075  endif ; enddo ; enddo ! i & k loops
1076 
1077  if (do_shelf) then
1078  ! Set the coefficients to include the no-slip surface stress.
1079  do i=is,ie ; if (do_i(i)) then
1080  if (work_on_u) then
1081  kv_tbl(i) = visc%kv_tbl_shelf_u(i,j)
1082  tbl_thick(i) = visc%tbl_thick_shelf_u(i,j) * m_to_h
1083  else
1084  kv_tbl(i) = visc%kv_tbl_shelf_v(i,j)
1085  tbl_thick(i) = visc%tbl_thick_shelf_v(i,j) * m_to_h
1086  endif
1087  z_t(i) = 0.0
1088 
1089  ! If a(i,1) were not already 0, it would be added here.
1090  if (0.5*hvel(i,1) > tbl_thick(i)) then
1091  a(i,1) = kv_tbl(i) / (tbl_thick(i) *h_to_m + (1.0e-10*dt)*kv_tbl(i))
1092  else
1093  a(i,1) = kv_tbl(i) / (0.5*hvel(i,1)*h_to_m + (1.0e-10*dt)*kv_tbl(i))
1094  endif
1095  endif ; enddo
1096 
1097  do k=2,nz ; do i=is,ie ; if (do_i(i)) then
1098  z_t(i) = z_t(i) + hvel(i,k-1) / tbl_thick(i)
1099  topfn = 1.0 / (1.0 + 0.09 * z_t(i)**6)
1100 
1101  r = (hvel(i,k)+hvel(i,k-1))
1102  if (r > 2.0*tbl_thick(i)) then
1103  h_shear = ((1.0 - topfn) * r + topfn*2.0*tbl_thick(i))
1104  else
1105  h_shear = r
1106  endif
1107  ! The term including 1e-10 in the denominators is here to avoid
1108  ! truncation error problems in the tridiagonal solver. Effectively, this
1109  ! sets the maximum coupling coefficient increment to 1e10 m.
1110  a_top = 2.0 * topfn * kv_tbl(i)
1111  a(i,k) = a(i,k) + a_top / (h_shear*h_to_m + 1.0e-10*dt*a_top)
1112  endif ; enddo ; enddo
1113  elseif (cs%dynamic_viscous_ML .or. (gv%nkml>0)) then
1114  max_nk = 0
1115  do i=is,ie ; if (do_i(i)) then
1116  if (gv%nkml>0) nk_visc(i) = real(gv%nkml+1)
1117  if (work_on_u) then
1118  u_star(i) = 0.5*(fluxes%ustar(i,j) + fluxes%ustar(i+1,j))
1119  absf(i) = 0.5*(abs(g%CoriolisBu(i,j-1)) + abs(g%CoriolisBu(i,j)))
1120  if (cs%dynamic_viscous_ML) nk_visc(i) = visc%nkml_visc_u(i,j) + 1
1121  else
1122  u_star(i) = 0.5*(fluxes%ustar(i,j) + fluxes%ustar(i,j+1))
1123  absf(i) = 0.5*(abs(g%CoriolisBu(i-1,j)) + abs(g%CoriolisBu(i,j)))
1124  if (cs%dynamic_viscous_ML) nk_visc(i) = visc%nkml_visc_v(i,j) + 1
1125  endif
1126  h_ml(i) = h_neglect ; z_t(i) = 0.0
1127  max_nk = max(max_nk,ceiling(nk_visc(i) - 1.0))
1128  endif ; enddo
1129 
1130  do k=1,max_nk ; do i=is,ie ; if (do_i(i)) then
1131  if (k+1 <= nk_visc(i)) then ! This layer is all in the ML.
1132  h_ml(i) = h_ml(i) + hvel(i,k)
1133  elseif (k < nk_visc(i)) then ! Part of this layer is in the ML.

1586 !> \namespace mom_vert_friction
1587 !! \author Robert Hallberg
1588 !! \date April 1994 - October 2006
1589 !!
1590 !! The vertical diffusion of momentum is fully implicit. This is
1591 !! necessary to allow for vanishingly small layers. The coupling
1592 !! is based on the distance between the centers of adjacent layers,
1593 !! except where a layer is close to the bottom compared with a
1594 !! bottom boundary layer thickness when a bottom drag law is used.
1595 !! A stress top b.c. and a no slip bottom b.c. are used. There
1596 !! is no limit on the time step for vertvisc.
1597 !!
1598 !! Near the bottom, the horizontal thickness interpolation scheme
1599 !! changes to an upwind biased estimate to control the effect of
1600 !! spurious Montgomery potential gradients at the bottom where
1601 !! nearly massless layers layers ride over the topography. Within a
1602 !! few boundary layer depths of the bottom, the harmonic mean
1603 !! thickness (i.e. (2 h+ h-) / (h+ + h-) ) is used if the velocity
1604 !! is from the thinner side and the arithmetic mean thickness
1605 !! (i.e. (h+ + h-)/2) is used if the velocity is from the thicker
1606 !! side. Both of these thickness estimates are second order
1607 !! accurate. Above this the arithmetic mean thickness is used.
1608 !!
1609 !! In addition, vertvisc truncates any velocity component that
1610 !! exceeds maxvel to truncvel. This basically keeps instabilities
1611 !! spatially localized. The number of times the velocity is
1612 !! truncated is reported each time the energies are saved, and if
1613 !! exceeds CS%Maxtrunc the model will stop itself and change the time
1614 !! to a large value. This has proven very useful in (1) diagnosing
1615 !! model failures and (2) letting the model settle down to a
1616 !! meaningful integration from a poorly specified initial condition.
1617 !!
1618 !! The same code is used for the two velocity components, by
1619 !! indirectly referencing the velocities and defining a handful of
1620 !! direction-specific defined variables.
1621 !!
1622 !! Macros written all in capital letters are defined in MOM_memory.h.
1623 !!
1624 !! A small fragment of the grid is shown below:
1625 !! \verbatim
1626 !! j+1 x ^ x ^ x At x: q
1627 !! j+1 > o > o > At ^: v, frhatv, tauy
1628 !! j x ^ x ^ x At >: u, frhatu, taux
1629 !! j > o > o > At o: h
1630 !! j-1 x ^ x ^ x
1631 !! i-1 i i+1 At x & ^:
1632 !! i i+1 At > & o:
1633 !! \endverbatim
1634 !!
1635 !! The boundaries always run through q grid points (x).
1636 end module mom_vert_friction
