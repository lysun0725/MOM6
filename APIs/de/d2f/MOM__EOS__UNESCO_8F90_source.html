61 subroutine calculate_density_scalar_unesco(T, S, pressure, rho)
62 real, intent(in) :: T, S, pressure
63 real, intent(out) :: rho
64 ! * Arguments: T - potential temperature relative to the surface in C. *
65 ! * (in) S - salinity in PSU. *
66 ! * (in) pressure - pressure in Pa. *
67 ! * (out) rho - in situ density in kg m-3. *
68 ! * (in) start - the starting point in the arrays. *
69 ! * (in) npts - the number of values to calculate. *
70 
71 ! *====================================================================*
72 ! * This subroutine computes the in situ density of sea water (rho in *
73 ! * units of kg/m^3) from salinity (S in psu), potential temperature *
74 ! * (T in deg C), and pressure in Pa. It uses the expression from *
75 ! * Wright, 1997, J. Atmos. Ocean. Tech., 14, 735-740. *
76 ! * Coded by R. Hallberg, 7/00 *
77 ! *====================================================================*
78 
79  real, dimension(1) :: T0, S0, pressure0
80  real, dimension(1) :: rho0
81 
82  t0(1) = t
83  s0(1) = s
84  pressure0(1) = pressure
85 
86  call calculate_density_array_unesco(t0, s0, pressure0, rho0, 1, 1)
87  rho = rho0(1)
88 
89 end subroutine calculate_density_scalar_unesco

91 subroutine calculate_density_array_unesco(T, S, pressure, rho, start, npts)
92  real, intent(in), dimension(:) :: T, S, pressure
93  real, intent(out), dimension(:) :: rho
94  integer, intent(in) :: start, npts
95 
96 ! * This subroutine computes the in situ density of sea water (rho in *
97 ! * units of kg/m^3) from salinity (S in psu), potential temperature *
98 ! * (T in deg C), and pressure in Pa. *
99 
100 ! * Arguments: T - potential temperature relative to the surface in C. *
101 ! * (in) S - salinity in PSU. *
102 ! * (in) pressure - pressure in Pa. *
103 ! * (out) rho - in situ density in kg m-3. *
104 ! * (in) start - the starting point in the arrays. *
105 ! * (in) npts - the number of values to calculate. *
106 
107  real :: t_local, t2, t3, t4, t5; ! Temperature to the 1st - 5th power.
108  real :: s_local, s32, s2; ! Salinity to the 1st, 3/2, & 2nd power.
109  real :: p1, p2; ! Pressure (in bars) to the 1st and 2nd power.
110  real :: rho0; ! Density at 1 bar pressure, in kg m-3.
111  real :: ks; ! The secant bulk modulus in bar.
112  integer :: j
113 
114  do j=start,start+npts-1
115  p1 = pressure(j)*1.0e-5; p2 = p1*p1;
116  t_local = t(j); t2 = t_local*t_local; t3 = t_local*t2; t4 = t2*t2; t5 = t3*t2;
117  s_local = s(j); s2 = s_local*s_local; s32 = s_local*sqrt(s_local);
118 
119 ! Compute rho(s,theta,p=0) - (same as rho(s,t_insitu,p=0) ).
120 
121  rho0 = r00 + r10*t_local + r20*t2 + r30*t3 + r40*t4 + r50*t5 + &
122  s_local*(r01 + r11*t_local + r21*t2 + r31*t3 + r41*t4) + &
123  s32*(r032 + r132*t_local + r232*t2) + r02*s2;
124 
125 ! Compute rho(s,theta,p), first calculating the secant bulk modulus.
126 
127  ks = s00 + s10*t_local + s20*t2 + s30*t3 + s40*t4 + s_local*(s01 + s11*t_local + s21*t2 + s31*t3) + &
128  s32*(s032 + s132*t_local + s232*t2) + &
129  p1*(sp00 + sp10*t_local + sp20*t2 + sp30*t3 + &
130  s_local*(sp01 + sp11*t_local + sp21*t2) + sp032*s32) + &
131  p2*(sp000 + sp010*t_local + sp020*t2 + s_local*(sp001 + sp011*t_local + sp021*t2));
132 
133  rho(j) = rho0*ks / (ks - p1);
134  enddo
135 end subroutine calculate_density_array_unesco

137 subroutine calculate_density_derivs_unesco(T, S, pressure, drho_dT, drho_dS, start, npts)
138  real, intent(in), dimension(:) :: T, S, pressure
139  real, intent(out), dimension(:) :: drho_dT, drho_dS
140  integer, intent(in) :: start, npts
141 ! * This subroutine calculates the partial derivatives of density *
142 ! * with potential temperature and salinity. *
143 ! * *
144 ! * Arguments: T - potential temperature relative to the surface in C. *
145 ! * (in) S - salinity in PSU. *
146 ! * (in) pressure - pressure in Pa. *
147 ! * (out) drho_dT - the partial derivative of density with *
148 ! * potential temperature, in kg m-3 K-1. *
149 ! * (out) drho_dS - the partial derivative of density with *
150 ! * salinity, in kg m-3 psu-1. *
151 ! * (in) start - the starting point in the arrays. *
152 ! * (in) npts - the number of values to calculate. *
153  real :: t_local, t2, t3, t4, t5; ! Temperature to the 1st - 5th power.
154  real :: s12, s_local, s32, s2; ! Salinity to the 1/2 - 2nd powers.
155  real :: p1, p2; ! Pressure (in bars) to the 1st & 2nd power.
156  real :: rho0; ! Density at 1 bar pressure, in kg m-3.
157  real :: ks; ! The secant bulk modulus, in bar.
158  real :: drho0_dT; ! Derivative of rho0 with T, in kg m-3 K-1.
159  real :: drho0_dS; ! Derivative of rho0 with S, kg m-3 psu-1.
160  real :: dks_dT; ! Derivative of ks with T, in bar K-1.
161  real :: dks_dS; ! Derivative of ks with S, in bar psu-1.
162  real :: denom; ! 1.0 / (ks - p1) in bar-1.
163  integer :: j
164 
165  do j=start,start+npts-1
166  p1 = pressure(j)*1.0e-5; p2 = p1*p1;
167  t_local = t(j); t2 = t_local*t_local; t3 = t_local*t2; t4 = t2*t2; t5 = t3*t2;
168  s_local = s(j); s2 = s_local*s_local; s12 = sqrt(s_local); s32 = s_local*s12;
169 
170 ! compute rho(s,theta,p=0) - (same as rho(s,t_insitu,p=0) )
171 
172  rho0 = r00 + r10*t_local + r20*t2 + r30*t3 + r40*t4 + r50*t5 + &
173  s_local*(r01 + r11*t_local + r21*t2 + r31*t3 + r41*t4) + &
174  s32*(r032 + r132*t_local + r232*t2) + r02*s2;
175  drho0_dt = r10 + 2.0*r20*t_local + 3.0*r30*t2 + 4.0*r40*t3 + 5.0*r50*t4 + &
176  s_local*(r11 + 2.0*r21*t_local + 3.0*r31*t2 + 4.0*r41*t3) + &
177  s32*(r132 + 2.0*r232*t_local);
178  drho0_ds = (r01 + r11*t_local + r21*t2 + r31*t3 + r41*t4) + &
179  1.5*s12*(r032 + r132*t_local + r232*t2) + 2.0*r02*s_local;
180 
181 ! compute rho(s,theta,p)
182 
183  ks = s00 + s10*t_local + s20*t2 + s30*t3 + s40*t4 + s_local*(s01 + s11*t_local + s21*t2 + s31*t3) + &
184  s32*(s032 + s132*t_local + s232*t2) + &
185  p1*(sp00 + sp10*t_local + sp20*t2 + sp30*t3 + &
186  s_local*(sp01 + sp11*t_local + sp21*t2) + sp032*s32) + &
187  p2*(sp000 + sp010*t_local + sp020*t2 + s_local*(sp001 + sp011*t_local + sp021*t2));
188  dks_dt = s10 + 2.0*s20*t_local + 3.0*s30*t2 + 4.0*s40*t3 + &
189  s_local*(s11 + 2.0*s21*t_local + 3.0*s31*t2) + s32*(s132 + 2.0*s232*t_local) + &
190  p1*(sp10 + 2.0*sp20*t_local + 3.0*sp30*t2 + s_local*(sp11 + 2.0*sp21*t_local)) + &
191  p2*(sp010 + 2.0*sp020*t_local + s_local*(sp011 + 2.0*sp021*t_local));
192  dks_ds = (s01 + s11*t_local + s21*t2 + s31*t3) + 1.5*s12*(s032 + s132*t_local + s232*t2) + &
193  p1*(sp01 + sp11*t_local + sp21*t2 + 1.5*sp032*s12) + &
194  p2*(sp001 + sp011*t_local + sp021*t2);
195 
196  denom = 1.0 / (ks - p1);
197  drho_dt(j) = denom*(ks*drho0_dt - rho0*p1*denom*dks_dt);
198  drho_ds(j) = denom*(ks*drho0_ds - rho0*p1*denom*dks_ds);
199  enddo
200 
201 end subroutine calculate_density_derivs_unesco

203 subroutine calculate_compress_unesco(T, S, pressure, rho, drho_dp, start, npts)
204  real, intent(in), dimension(:) :: T, S, pressure
205  real, intent(out), dimension(:) :: rho, drho_dp
206  integer, intent(in) :: start, npts
207 ! * This subroutine computes the in situ density of sea water (rho) *
208 ! * and the compressibility (drho/dp == C_sound^-2) at the given *
209 ! * salinity, potential temperature, and pressure. *
210 ! * *
211 ! * Arguments: T - potential temperature relative to the surface in C. *
212 ! * (in) S - salinity in PSU. *
213 ! * (in) pressure - pressure in Pa. *
214 ! * (out) rho - in situ density in kg m-3. *
215 ! * (out) drho_dp - the partial derivative of density with *
216 ! * pressure (also the inverse of the square of *
217 ! * sound speed) in s2 m-2. *
218 ! * (in) start - the starting point in the arrays. *
219 ! * (in) npts - the number of values to calculate. *
220 
221  real :: t_local, t2, t3, t4, t5; ! Temperature to the 1st - 5th power.
222  real :: s_local, s32, s2; ! Salinity to the 1st, 3/2, & 2nd power.
223  real :: p1, p2; ! Pressure (in bars) to the 1st and 2nd power.
224  real :: rho0; ! Density at 1 bar pressure, in kg m-3.
225  real :: ks; ! The secant bulk modulus in bar.
226  real :: ks_0, ks_1, ks_2;
227  real :: dks_dp; ! The derivative of the secant bulk modulus
228  ! with pressure, nondimensional.
229  integer :: j
230 
231  do j=start,start+npts-1
232  p1 = pressure(j)*1.0e-5; p2 = p1*p1;
233  t_local = t(j); t2 = t_local*t_local; t3 = t_local*t2; t4 = t2*t2; t5 = t3*t2;
234  s_local = s(j); s2 = s_local*s_local; s32 = s_local*sqrt(s_local);
235 
236 ! Compute rho(s,theta,p=0) - (same as rho(s,t_insitu,p=0) ).
237 
238  rho0 = r00 + r10*t_local + r20*t2 + r30*t3 + r40*t4 + r50*t5 + &
239  s_local*(r01 + r11*t_local + r21*t2 + r31*t3 + r41*t4) + &
240  s32*(r032 + r132*t_local + r232*t2) + r02*s2;
241 
242 ! Compute rho(s,theta,p), first calculating the secant bulk modulus.
243  ks_0 = s00 + s10*t_local + s20*t2 + s30*t3 + s40*t4 + &
244  s_local*(s01 + s11*t_local + s21*t2 + s31*t3) + s32*(s032 + s132*t_local + s232*t2);
245  ks_1 = sp00 + sp10*t_local + sp20*t2 + sp30*t3 + &
246  s_local*(sp01 + sp11*t_local + sp21*t2) + sp032*s32;
247  ks_2 = sp000 + sp010*t_local + sp020*t2 + s_local*(sp001 + sp011*t_local + sp021*t2);
248 
249  ks = ks_0 + p1*ks_1 + p2*ks_2;
250  dks_dp = ks_1 + 2.0*p1*ks_2;
251 
252  rho(j) = rho0*ks / (ks - p1);
253 ! The factor of 1.0e-5 is because pressure here is in bars, not Pa.
254  drho_dp(j) = 1.0e-5 * (rho(j) / (ks - p1)) * (1.0 - dks_dp*p1/ks);
255  enddo
256 end subroutine calculate_compress_unesco

258 subroutine calculate_2_densities_unesco( T, S, pressure1, pressure2, rho1, rho2, start, npts)
259  real, intent(in), dimension(:) :: T, S
260  real, intent(in) :: pressure1, pressure2
261  real, intent(out), dimension(:) :: rho1, rho2
262  integer, intent(in) :: start, npts
263 ! * This subroutine computes the densities of sea water (rho1 and *
264 ! * rho2) at two reference pressures (pressure1 and pressure2) from *
265 ! * salinity and potential temperature. *
266 ! * *
267 ! * Arguments: T - potential temperature relative to the surface in C. *
268 ! * (in) S - salinity in PSU. *
269 ! * (in) pressure1 - the first pressure in Pa. *
270 ! * (in) pressure2 - the second pressure in Pa. *
271 ! * (out) rho1 - density at pressure1 in kg m-3. *
272 ! * (out) rho2 - density at pressure2 in kg m-3. *
273 ! * (in) start - the starting point in the arrays. *
274 ! * (in) npts - the number of values to calculate. *
275  real :: t_local, t2, t3, t4, t5; ! Temperature to the 1st - 5th power.
276  real :: s_local, s32, s2; ! Salinity to the 1st, 3/2, & 2nd power.
277  real :: p1a, p2a; ! Pressure1 (in bars) to the 1st and 2nd power.
278  real :: p1b, p2b; ! Pressure2 (in bars) to the 1st and 2nd power.
279  real :: rho0; ! Density at 1 bar pressure, in kg m-3.
280  real :: ksa, ksb; ! The secant bulk modulus in bar.
281  real :: ks_0, ks_1, ks_2;
282  integer :: j
283 
284  p1a = pressure1*1.0e-5; p2a = p1a*p1a;
285  p1b = pressure2*1.0e-5; p2b = p1b*p1b;
286 
287  do j=start, start+npts-1
288  t_local = t(j); t2 = t_local*t_local; t3 = t_local*t2; t4 = t2*t2; t5 = t3*t2;
289  s_local = s(j); s2 = s_local*s_local; s32 = s_local*sqrt(s_local);
290 
291 ! Compute rho(s,theta,p=0) - (same as rho(s,t_insitu,p=0) ).
292 
293  rho0 = r00 + r10*t_local + r20*t2 + r30*t3 + r40*t4 + r50*t5 + &
294  s_local*(r01 + r11*t_local + r21*t2 + r31*t3 + r41*t4) + &
295  s32*(r032 + r132*t_local + r232*t2) + r02*s2;
296 
297 ! Compute rho(s,theta,p), first calculating the secant bulk modulus.
298 
299  ks_0 = s00 + s10*t_local + s20*t2 + s30*t3 + s40*t4 + &
300  s_local*(s01 + s11*t_local + s21*t2 + s31*t3) + s32*(s032 + s132*t_local + s232*t2);
301  ks_1 = sp00 + sp10*t_local + sp20*t2 + sp30*t3 + &
302  s_local*(sp01 + sp11*t_local + sp21*t2) + sp032*s32;
303  ks_2 = sp000 + sp010*t_local + sp020*t2 + s_local*(sp001 + sp011*t_local + sp021*t2);
304 
305  ksa = ks_0 + p1a*ks_1 + p2a*ks_2;
306  ksb = ks_0 + p1b*ks_1 + p2b*ks_2;
307 
308  rho1(j) = rho0*ksa / (ksa - p1a);
309  rho2(j) = rho0*ksb / (ksb - p1b);
310  enddo
311 end subroutine calculate_2_densities_unesco
