25 !* Alistair Adcroft, and Stephen Griffies *
26 !* *
27 !* This program contains the subroutine that, along with the *
28 !* subroutines that it calls, implements diapycnal mass and momentum *
29 !* fluxes and a bulk mixed layer. The diapycnal diffusion can be *
30 !* used without the bulk mixed layer. *
31 !* *
32 !* diabatic first determines the (diffusive) diapycnal mass fluxes *
33 !* based on the convergence of the buoyancy fluxes within each layer. *
34 !* The dual-stream entrainment scheme of MacDougall and Dewar (JPO, *
35 !* 1997) is used for combined diapycnal advection and diffusion, *
36 !* calculated implicitly and potentially with the Richardson number *
37 !* dependent mixing, as described by Hallberg (MWR, 2000). Diapycnal *
38 !* advection is fundamentally the residual of diapycnal diffusion, *
39 !* so the fully implicit upwind differencing scheme that is used is *
40 !* entirely appropriate. The downward buoyancy flux in each layer *
41 !* is determined from an implicit calculation based on the previously *
42 !* calculated flux of the layer above and an estimated flux in the *
43 !* layer below. This flux is subject to the following conditions: *
44 !* (1) the flux in the top and bottom layers are set by the boundary *
45 !* conditions, and (2) no layer may be driven below an Angstrom thick-*
46 !* ness. If there is a bulk mixed layer, the buffer layer is treat- *
47 !* ed as a fixed density layer with vanishingly small diffusivity. *
48 !* *
49 !* diabatic takes 5 arguments: the two velocities (u and v), the *
50 !* thicknesses (h), a structure containing the forcing fields, and *
51 !* the length of time over which to act (dt). The velocities and *
52 !* thickness are taken as inputs and modified within the subroutine. *
53 !* There is no limit on the time step. *
54 !* *
55 !* A small fragment of the grid is shown below: *
56 !* *
57 !* j+1 x ^ x ^ x At x: q *
58 !* j+1 > o > o > At ^: v *
59 !* j x ^ x ^ x At >: u *
60 !* j > o > o > At o: h, T, S, buoy, ustar, ea, eb, etc. *
61 !* j-1 x ^ x ^ x *
62 !* i-1 i i+1 At x & ^: *
63 !* i i+1 At > & o: *
64 !* *
65 !* The boundaries always run through q grid points (x). *
66 !* *

143 subroutine make_frazil(h, tv, G, GV, CS, p_surf)
144  type(ocean_grid_type), intent(in) :: G
145  type(verticalgrid_type), intent(in) :: GV
146  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
147  type(thermo_var_ptrs), intent(inout) :: tv
148  type(diabatic_aux_cs), intent(in) :: CS
149  real, dimension(SZI_(G),SZJ_(G)), intent(in), optional :: p_surf
150 
151 ! Frazil formation keeps the temperature above the freezing point.
152 ! This subroutine warms any water that is colder than the (currently
153 ! surface) freezing point up to the freezing point and accumulates
154 ! the required heat (in J m-2) in tv%frazil.
155 ! The expression, below, for the freezing point of sea water comes
156 ! from Millero (1978) via Appendix A of Gill, 1982.
157 
158 ! Arguments: h - Layer thickness, in m or kg m-2.
159 ! (in/out) tv - A structure containing pointers to any available
160 ! thermodynamic fields. Absent fields have NULL ptrs.
161 ! (in) G - The ocean's grid structure.
162 ! (in) GV - The ocean's vertical grid structure.
163 ! (in) CS - The control structure returned by a previous call to
164 ! diabatic_driver_init.
165  real, dimension(SZI_(G)) :: &
166  fraz_col, & ! The accumulated heat requirement due to frazil, in J.
167  T_freeze, & ! The freezing potential temperature at the current salinity, C.
168  ps ! pressure
169  real, dimension(SZI_(G),SZK_(G)) :: &
170  pressure ! The pressure at the middle of each layer in Pa.
171  real :: hc ! A layer's heat capacity in J m-2 K-1.
172  logical :: T_fr_set ! True if the freezing point has been calculated for a
173  ! row of points.
174  integer :: i, j, k, is, ie, js, je, nz
175  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
176 
177  call cpu_clock_begin(id_clock_frazil)
178 
179  if (.not.cs%pressure_dependent_frazil) then
180  do k=1,nz ; do i=is,ie ; pressure(i,k) = 0.0 ; enddo ; enddo
181  endif
182 !$OMP parallel do default(none) shared(is,ie,js,je,CS,G,GV,h,nz,tv,p_surf) &
183 !$OMP private(fraz_col,T_fr_set,T_freeze,hc,ps) &
184 !$OMP firstprivate(pressure)
185  do j=js,je
186  ps(:) = 0.0
187  if (PRESENT(p_surf)) then
188  ps(:) = p_surf(:,j)
189  endif
190 
191  do i=is,ie ; fraz_col(:) = 0.0 ; enddo
192 
193  if (cs%pressure_dependent_frazil) then
194  do i=is,ie
195  pressure(i,1) = ps(i) + (0.5*gv%H_to_Pa)*h(i,j,1)
196  enddo
197  do k=2,nz ; do i=is,ie
198  pressure(i,k) = pressure(i,k-1) + &
199  (0.5*gv%H_to_Pa) * (h(i,j,k) + h(i,j,k-1))
200  enddo ; enddo
201  endif
202 
203  if (cs%reclaim_frazil) then
204  t_fr_set = .false.
205  do i=is,ie ; if (tv%frazil(i,j) > 0.0) then
206  if (.not.t_fr_set) then
207  call calculate_tfreeze(tv%S(i:,j,1), pressure(i:,1), t_freeze(i:), &
208  1, ie-i+1, tv%eqn_of_state)
209  t_fr_set = .true.
210  endif
211 
212  if (tv%T(i,j,1) > t_freeze(i)) then
213  ! If frazil had previously been formed, but the surface temperature is now
214  ! above freezing, cool the surface layer with the frazil heat deficit.
215  hc = (tv%C_p*gv%H_to_kg_m2) * h(i,j,1)
216  if (tv%frazil(i,j) - hc * (tv%T(i,j,1) - t_freeze(i)) <= 0.0) then
217  tv%T(i,j,1) = tv%T(i,j,1) - tv%frazil(i,j)/hc
218  tv%frazil(i,j) = 0.0
219  else
220  tv%frazil(i,j) = tv%frazil(i,j) - hc * (tv%T(i,j,1) - t_freeze(i))
221  tv%T(i,j,1) = t_freeze(i)
222  endif
223  endif
224  endif ; enddo
225  endif
226 
227  do k=nz,1,-1
228  t_fr_set = .false.
229  do i=is,ie
230  if ((g%mask2dT(i,j) > 0.0) .and. &
231  ((tv%T(i,j,k) < 0.0) .or. (fraz_col(i) > 0.0))) then
232  if (.not.t_fr_set) then
233  call calculate_tfreeze(tv%S(i:,j,k), pressure(i:,k), t_freeze(i:), &
234  1, ie-i+1, tv%eqn_of_state)
235  t_fr_set = .true.
236  endif
237 
238  hc = (tv%C_p*gv%H_to_kg_m2) * h(i,j,k)
239  if (h(i,j,k) <= 10.0*gv%Angstrom) then
240  ! Very thin layers should not be cooled by the frazil flux.
241  if (tv%T(i,j,k) < t_freeze(i)) then
242  fraz_col(i) = fraz_col(i) + hc * (t_freeze(i) - tv%T(i,j,k))
243  tv%T(i,j,k) = t_freeze(i)
244  endif
245  else
246  if (fraz_col(i) + hc * (t_freeze(i) - tv%T(i,j,k)) <= 0.0) then
247  tv%T(i,j,k) = tv%T(i,j,k) - fraz_col(i)/hc
248  fraz_col(i) = 0.0
249  else
250  fraz_col(i) = fraz_col(i) + hc * (t_freeze(i) - tv%T(i,j,k))
251  tv%T(i,j,k) = t_freeze(i)
252  endif
253  endif
254  endif
255  enddo
256  enddo
257  do i=is,ie
258  tv%frazil(i,j) = tv%frazil(i,j) + fraz_col(i)
259  enddo
260  enddo
261  call cpu_clock_end(id_clock_frazil)
262 
263 end subroutine make_frazil

265 subroutine differential_diffuse_t_s(h, tv, visc, dt, G, GV)
266  type(ocean_grid_type), intent(in) :: G
267  type(verticalgrid_type), intent(in) :: GV
268  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
269  type(thermo_var_ptrs), intent(inout) :: tv
270  type(vertvisc_type), intent(in) :: visc
271  real, intent(in) :: dt
272 
273 ! This subroutine applies double diffusion to T & S, assuming no diapycal mass
274 ! fluxes, using a simple triadiagonal solver.
275 
276 ! Arguments: h - Layer thickness, in m or kg m-2.
277 ! (in) tv - A structure containing pointers to any available
278 ! thermodynamic fields. Absent fields have NULL ptrs.
279 ! (in) visc - A structure containing vertical viscosities, bottom boundary
280 ! layer properies, and related fields.
281 ! (in) dt - Time increment, in s.
282 ! (in) G - The ocean's grid structure.
283 ! (in) GV - The ocean's vertical grid structure.
284 
285  real, dimension(SZI_(G)) :: &
286  b1_T, b1_S, & ! Variables used by the tridiagonal solvers of T & S, in H.
287  d1_T, d1_S ! Variables used by the tridiagonal solvers, nondim.
288  real, dimension(SZI_(G),SZK_(G)) :: &
289  c1_T, c1_S ! Variables used by the tridiagonal solvers, in m or kg m-2.
290  real, dimension(SZI_(G),SZK_(G)+1) :: &
291  mix_T, mix_S ! Mixing distances in both directions across each
292  ! interface, in m or kg m-2.
293  real :: h_tr ! h_tr is h at tracer points with a tiny thickness
294  ! added to ensure positive definiteness, in m or kg m-2.
295  real :: h_neglect ! A thickness that is so small it is usually lost
296  ! in roundoff and can be neglected, in m or kg m-2.
297  real :: I_h_int ! The inverse of the thickness associated with an
298  ! interface, in m-1 or m2 kg-1.
299  real :: b_denom_T ! The first term in the denominators for the expressions
300  real :: b_denom_S ! for b1_T and b1_S, both in m or kg m-2.
301 
302  integer :: i, j, k, is, ie, js, je, nz
303  real, pointer :: T(:,:,:), S(:,:,:), Kd_T(:,:,:), Kd_S(:,:,:)
304  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
305  h_neglect = gv%H_subroundoff
306 
307  if (.not.associated(tv%T)) call mom_error(fatal, &
308  "differential_diffuse_T_S: Called with an unassociated tv%T")
309  if (.not.associated(tv%S)) call mom_error(fatal, &
310  "differential_diffuse_T_S: Called with an unassociated tv%S")
311  if (.not.associated(visc%Kd_extra_T)) call mom_error(fatal, &
312  "differential_diffuse_T_S: Called with an unassociated visc%Kd_extra_T")
313  if (.not.associated(visc%Kd_extra_S)) call mom_error(fatal, &
314  "differential_diffuse_T_S: Called with an unassociated visc%Kd_extra_S")
315 
316  t => tv%T ; s => tv%S
317  kd_t => visc%Kd_extra_T ; kd_s => visc%Kd_extra_S
318 !$OMP parallel do default(none) shared(is,ie,js,je,h,h_neglect,dt,Kd_T,Kd_S,G,GV,T,S,nz) &
319 !$OMP private(I_h_int,mix_T,mix_S,h_tr,b1_T,b1_S, &
320 !$OMP d1_T,d1_S,c1_T,c1_S,b_denom_T,b_denom_S)
321  do j=js,je
322  do i=is,ie
323  i_h_int = 1.0 / (0.5 * (h(i,j,1) + h(i,j,2)) + h_neglect)
324  mix_t(i,2) = ((dt * kd_t(i,j,2)) * gv%m_to_H**2) * i_h_int
325  mix_s(i,2) = ((dt * kd_s(i,j,2)) * gv%m_to_H**2) * i_h_int
326 
327  h_tr = h(i,j,1) + h_neglect
328  b1_t(i) = 1.0 / (h_tr + mix_t(i,2))
329  b1_s(i) = 1.0 / (h_tr + mix_s(i,2))
330  d1_t(i) = h_tr * b1_t(i)
331  d1_s(i) = h_tr * b1_s(i)
332  t(i,j,1) = (b1_t(i)*h_tr)*t(i,j,1)
333  s(i,j,1) = (b1_s(i)*h_tr)*s(i,j,1)
334  enddo
335  do k=2,nz-1 ; do i=is,ie
336  ! Calculate the mixing across the interface below this layer.
337  i_h_int = 1.0 / (0.5 * (h(i,j,k) + h(i,j,k+1)) + h_neglect)
338  mix_t(i,k+1) = ((dt * kd_t(i,j,k+1)) * gv%m_to_H**2) * i_h_int
339  mix_s(i,k+1) = ((dt * kd_s(i,j,k+1)) * gv%m_to_H**2) * i_h_int
340 
341  c1_t(i,k) = mix_t(i,k) * b1_t(i)
342  c1_s(i,k) = mix_s(i,k) * b1_s(i)
343 
344  h_tr = h(i,j,k) + h_neglect
345  b_denom_t = h_tr + d1_t(i)*mix_t(i,k)
346  b_denom_s = h_tr + d1_s(i)*mix_s(i,k)
347  b1_t(i) = 1.0 / (b_denom_t + mix_t(i,k+1))
348  b1_s(i) = 1.0 / (b_denom_s + mix_s(i,k+1))
349  d1_t(i) = b_denom_t * b1_t(i)
350  d1_s(i) = b_denom_s * b1_s(i)
351 
352  t(i,j,k) = b1_t(i) * (h_tr*t(i,j,k) + mix_t(i,k)*t(i,j,k-1))
353  s(i,j,k) = b1_s(i) * (h_tr*s(i,j,k) + mix_s(i,k)*s(i,j,k-1))
354  enddo ; enddo
355  do i=is,ie
356  c1_t(i,nz) = mix_t(i,nz) * b1_t(i)
357  c1_s(i,nz) = mix_s(i,nz) * b1_s(i)
358 
359  h_tr = h(i,j,nz) + h_neglect
360  b1_t(i) = 1.0 / (h_tr + d1_t(i)*mix_t(i,nz))
361  b1_s(i) = 1.0 / (h_tr + d1_s(i)*mix_s(i,nz))
362 
363  t(i,j,nz) = b1_t(i) * (h_tr*t(i,j,nz) + mix_t(i,nz)*t(i,j,nz-1))
364  s(i,j,nz) = b1_s(i) * (h_tr*s(i,j,nz) + mix_s(i,nz)*s(i,j,nz-1))
365  enddo
366  do k=nz-1,1,-1 ; do i=is,ie
367  t(i,j,k) = t(i,j,k) + c1_t(i,k+1)*t(i,j,k+1)
368  s(i,j,k) = s(i,j,k) + c1_s(i,k+1)*s(i,j,k+1)
369  enddo ; enddo
370  enddo
371 
372 end subroutine differential_diffuse_t_s

374 subroutine adjust_salt(h, tv, G, GV, CS)
375  type(ocean_grid_type), intent(in) :: G
376  type(verticalgrid_type), intent(in) :: GV
377  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
378  type(thermo_var_ptrs), intent(inout) :: tv
379  type(diabatic_aux_cs), intent(in) :: CS
380 
381 ! Keep salinity from falling below a small but positive threshold
382 ! This occurs when the ice model attempts to extract more salt then
383 ! is actually available to it from the ocean.
384 
385 ! Arguments: h - Layer thickness, in m.
386 ! (in/out) tv - A structure containing pointers to any available
387 ! thermodynamic fields. Absent fields have NULL ptrs.
388 ! (in) G - The ocean's grid structure.
389 ! (in) GV - The ocean's vertical grid structure.
390 ! (in) CS - The control structure returned by a previous call to
391 ! diabatic_driver_init.
392  real :: salt_add_col(szi_(g),szj_(g)) ! The accumulated salt requirement
393  real :: S_min ! The minimum salinity
394  real :: mc ! A layer's mass kg m-2 .
395  integer :: i, j, k, is, ie, js, je, nz
396  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
397 
398 ! call cpu_clock_begin(id_clock_adjust_salt)
399 
400  s_min = 0.01
401 
402  salt_add_col(:,:) = 0.0
403 
404 !$OMP parallel do default(none) shared(is,ie,js,je,nz,G,GV,tv,h,salt_add_col, S_min) &
405 !$OMP private(mc)
406  do j=js,je
407  do k=nz,1,-1 ; do i=is,ie
408  if ((g%mask2dT(i,j) > 0.0) .and. &
409  ((tv%S(i,j,k) < s_min) .or. (salt_add_col(i,j) > 0.0))) then
410  mc = gv%H_to_kg_m2 * h(i,j,k)
411  if (h(i,j,k) <= 10.0*gv%Angstrom) then
412  ! Very thin layers should not be adjusted by the salt flux
413  if (tv%S(i,j,k) < s_min) then
414  salt_add_col(i,j) = salt_add_col(i,j) + mc * (s_min - tv%S(i,j,k))
415  tv%S(i,j,k) = s_min
416  endif
417  else
418  if (salt_add_col(i,j) + mc * (s_min - tv%S(i,j,k)) <= 0.0) then
419  tv%S(i,j,k) = tv%S(i,j,k) - salt_add_col(i,j)/mc
420  salt_add_col(i,j) = 0.0
421  else
422  salt_add_col(i,j) = salt_add_col(i,j) + mc * (s_min - tv%S(i,j,k))
423  tv%S(i,j,k) = s_min
424  endif
425  endif
426  endif
427  enddo ; enddo

563 subroutine tridiagts(G, GV, is, ie, js, je, hold, ea, eb, T, S)
564 ! Simple tri-diagnonal solver for T and S
565 ! "Simple" means it only uses arrays hold, ea and eb
566  ! Arguments
567  type(ocean_grid_type), intent(in) :: G
568  type(verticalgrid_type), intent(in) :: GV
569  integer, intent(in) :: is, ie, js, je
570  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: hold, ea, eb
571  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(inout) :: T, S
572  ! Local variables
573  real :: b1(szib_(g)), d1(szib_(g)) ! b1, c1, and d1 are variables used by the
574  real :: c1(szib_(g),szk_(g)) ! tridiagonal solver.
575  real :: h_tr, b_denom_1

591  b_denom_1 = h_tr + d1(i)*ea(i,j,k)
592  b1(i) = 1.0 / (b_denom_1 + eb(i,j,k))
593  d1(i) = b_denom_1 * b1(i)
594  t(i,j,k) = b1(i) * (h_tr*t(i,j,k) + ea(i,j,k)*t(i,j,k-1))
595  s(i,j,k) = b1(i) * (h_tr*s(i,j,k) + ea(i,j,k)*s(i,j,k-1))
596  enddo ; enddo
597  do k=g%ke-1,1,-1 ; do i=is,ie
598  t(i,j,k) = t(i,j,k) + c1(i,k+1)*t(i,j,k+1)
599  s(i,j,k) = s(i,j,k) + c1(i,k+1)*s(i,j,k+1)
600  enddo ; enddo

638  real :: a_e(szi_(g)), a_w(szi_(g)) ! velocity points, ~1/2 in the open
639  ! ocean, nondimensional.
640  real :: s, Idenom
641  logical :: mix_vertically
642  integer :: i, j, k, is, ie, js, je, nz
643  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
644  call cpu_clock_begin(id_clock_uv_at_h)
645  h_neglect = gv%H_subroundoff

649  "find_uv_at_h: Either both ea and eb or neither one must be present "// &
650  "in call to find_uv_at_h.")

652 !$OMP eb,u_h,u,v_h,v,nz,ea) &
653 !$OMP private(s,Idenom,a_w,a_e,a_s,a_n,b_denom_1,b1,d1,c1)

655  do i=is,ie
656  s = g%areaCu(i-1,j)+g%areaCu(i,j)

669  a_n(i) = g%areaCv(i,j)*idenom
670  else
671  a_s(i) = 0.0 ; a_n(i) = 0.0
672  endif
673  enddo
674 
675  if (mix_vertically) then
676  do i=is,ie
677  b_denom_1 = h(i,j,1) + h_neglect
678  b1(i) = 1.0 / (b_denom_1 + eb(i,j,1))
679  d1(i) = b_denom_1 * b1(i)
680  u_h(i,j,1) = (h(i,j,1)*b1(i)) * (a_e(i)*u(i,j,1) + a_w(i)*u(i-1,j,1))

687  d1(i) = b_denom_1 * b1(i)

689  ea(i,j,k)*u_h(i,j,k-1))*b1(i)

691  ea(i,j,k)*v_h(i,j,k-1))*b1(i)

694  u_h(i,j,k) = u_h(i,j,k) + c1(i,k+1)*u_h(i,j,k+1)
695  v_h(i,j,k) = v_h(i,j,k) + c1(i,k+1)*v_h(i,j,k+1)
696  enddo ; enddo
697  else
698  do k=1,nz ; do i=is,ie
699  u_h(i,j,k) = a_e(i)*u(i,j,k) + a_w(i)*u(i-1,j,k)
700  v_h(i,j,k) = a_n(i)*v(i,j,k) + a_s(i)*v(i,j-1,k)

702  endif
703  enddo

971  dsalt = 0.

974  ! If fractionOfForcing = 1, then updated netMassIn, netHeat, and netSalt vanish.

979  salin_in = 0.0
980  dtemp = dtemp + dthickness*temp_in

984  fluxes%heat_content_massin(i,j) = fluxes%heat_content_massin(i,j) + &
985  t2d(i,k) * max(0.,dthickness) * gv%H_to_kg_m2 * fluxes%C_p * idt
986  if (ASSOCIATED(fluxes%heat_content_massout)) &
987  fluxes%heat_content_massout(i,j) = fluxes%heat_content_massout(i,j) + &
988  t2d(i,k) * min(0.,dthickness) * gv%H_to_kg_m2 * fluxes%C_p * idt

995  ! is present to simulate unresolved estuaries. The TKE input is diagnosed

997  ! TKE_river[m3 s-3] = 0.5*rivermix_depth*g*(1/rho)*drho_ds*
998  ! River*(Samb - Sriver) = CS%mstar*U_star^3
999  ! where River is in units of m s-1.

1002  ! drho_ds = The gradient of density wrt salt at the ambient surface salinity.

1008  endif

1011  hold = h2d(i,k) ! Keep original thickness in hand
1012  h2d(i,k) = h2d(i,k) + dthickness ! New thickness
1013  if (h2d(i,k) > 0.0) then
1014  if (calculate_energetics .and. (dthickness > 0.)) then
1015  ! Calculate the energy required to mix the newly added water over
1016  ! the topmost grid cell. ###CHECK THE SIGNS!!!
1017  ctke(i,j,k) = ctke(i,j,k) + 0.5*g_hconv2*(hold*dthickness) * &
1018  ((t2d(i,k) - temp_in) * dsv_dt(i,j,k) + (tv%S(i,j,k) - salin_in) * dsv_ds(i,j,k))
1019  endif
1020  ithickness = 1.0/h2d(i,k) ! Inverse new thickness
1021  ! The "if"s below avoid changing T/S by roundoff unnecessarily
1022  if (dthickness /= 0. .or. dtemp /= 0.) t2d(i,k) = (hold*t2d(i,k) + dtemp)*ithickness
1023  if (dthickness /= 0. .or. dsalt /= 0.) tv%S(i,j,k) = (hold*tv%S(i,j,k) + dsalt)*ithickness
1024 
1025  endif
1026 
1027  enddo ! k=1,1
1028

1030  do k=1,nz
1031 
1032  ! Place forcing into this layer if this layer has nontrivial thickness.
1033  ! For layers thin relative to 1/IforcingDepthScale, then distribute
1034  ! forcing into deeper layers.
1035  iforcingdepthscale = 1. / max(gv%H_subroundoff, cs%minimum_forcing_depth*gv%m_to_H - netmassout(i) )
1036  ! fractionOfForcing = 1.0, unless h2d is less than IforcingDepthScale.
1037  fractionofforcing = min(1.0, h2d(i,k)*iforcingdepthscale)
1038 
1039  ! In the case with (-1)*netMassOut*fractionOfForcing greater than cfl*h, we
1040  ! limit the forcing applied to this cell, leaving the remaining forcing to
1041  ! be distributed downwards.
1042  if (-fractionofforcing*netmassout(i) > cs%evap_CFL_limit*h2d(i,k)) then
1043  fractionofforcing = -cs%evap_CFL_limit*h2d(i,k)/netmassout(i)
1044  endif
1045

1047  dthickness = max( fractionofforcing*netmassout(i), -h2d(i,k) )
1048  dtemp = fractionofforcing*netheat(i)
1049  ! ### The 0.9999 here should become a run-time parameter?
1050  dsalt = max( fractionofforcing*netsalt(i), -0.9999*h2d(i,k)*tv%S(i,j,k))
1051 
1052  ! Update the forcing by the part to be consumed within the present k-layer.
1053  ! If fractionOfForcing = 1, then new netMassOut vanishes.
1054  netmassout(i) = netmassout(i) - dthickness
1055  netheat(i) = netheat(i) - dtemp
1056  netsalt(i) = netsalt(i) - dsalt
1057 
1058  ! This line accounts for the temperature of the mass exchange
1059  dtemp = dtemp + dthickness*t2d(i,k)
1060 
1061  ! Diagnostics of heat content associated with mass fluxes
1062  if (ASSOCIATED(fluxes%heat_content_massin)) &
1063  fluxes%heat_content_massin(i,j) = fluxes%heat_content_massin(i,j) + &
1064  tv%T(i,j,k) * max(0.,dthickness) * gv%H_to_kg_m2 * fluxes%C_p * idt
1065  if (ASSOCIATED(fluxes%heat_content_massout)) &
1066  fluxes%heat_content_massout(i,j) = fluxes%heat_content_massout(i,j) + &
1067  tv%T(i,j,k) * min(0.,dthickness) * gv%H_to_kg_m2 * fluxes%C_p * idt
1068  if (ASSOCIATED(tv%TempxPmE)) tv%TempxPmE(i,j) = tv%TempxPmE(i,j) + &
1069  tv%T(i,j,k) * dthickness * gv%H_to_kg_m2
1070 !NOTE tv%T should be T2d
1071 
1072  ! Update state by the appropriate increment.
1073  hold = h2d(i,k) ! Keep original thickness in hand
1074  h2d(i,k) = h2d(i,k) + dthickness ! New thickness
1075  if (h2d(i,k) > 0.) then
1076  if (calculate_energetics) then
1077  ! Calculate the energy required to mix the newly added water over
1078  ! the topmost grid cell, assuming that the fluxes of heat and salt
1079  ! and rejected brine are initially applied in vanishingly thin
1080  ! layers at the top of the layer before being mixed throughout
1081  ! the layer. Note that dThickness is always <= 0. ###CHECK THE SIGNS!!!
1082  ctke(i,j,k) = ctke(i,j,k) - (0.5*h2d(i,k)*g_hconv2) * &
1083  ((dtemp - dthickness*t2d(i,k)) * dsv_dt(i,j,k) + &
1084  (dsalt - dthickness*tv%S(i,j,k)) * dsv_ds(i,j,k))
1085  endif

1087  t2d(i,k) = (hold*t2d(i,k) + dtemp)*ithickness
1088  tv%S(i,j,k) = (hold*tv%S(i,j,k) + dsalt)*ithickness
1089  elseif (h2d(i,k) < 0.0) then ! h2d==0 is a special limit that needs no extra handling
1090  call forcing_singlepointprint(fluxes,g,i,j,'applyBoundaryFluxesInOut (h<0)')
1091  write(0,*) 'applyBoundaryFluxesInOut(): lon,lat=',g%geoLonT(i,j),g%geoLatT(i,j)
1092  write(0,*) 'applyBoundaryFluxesInOut(): netT,netS,netH=',netheat(i),netsalt(i),netmassinout(i)
1093  write(0,*) 'applyBoundaryFluxesInOut(): dT,dS,dH=',dtemp,dsalt,dthickness
1094  write(0,*) 'applyBoundaryFluxesInOut(): h(n),h(n+1),k=',hold,h2d(i,k),k
1095  call mom_error(fatal, "MOM_diabatic_driver.F90, applyBoundaryFluxesInOut(): "//&
1096  "Complete mass loss in column!")
1097  endif
1098 
1099  enddo ! k
1100 
1101  ! Check if trying to apply fluxes over land points
1102  elseif((abs(netheat(i))+abs(netsalt(i))+abs(netmassin(i))+abs(netmassout(i)))>0.) then
1103  call forcing_singlepointprint(fluxes,g,i,j,'applyBoundaryFluxesInOut (land)')
1104  write(0,*) 'applyBoundaryFluxesInOut(): lon,lat=',g%geoLonT(i,j),g%geoLatT(i,j)
1105  write(0,*) 'applyBoundaryFluxesInOut(): netHeat,netSalt,netMassIn,netMassOut=',&
1106  netheat(i),netsalt(i),netmassin(i),netmassout(i)
1107  call mom_error(fatal, "MOM_diabatic_driver.F90, applyBoundaryFluxesInOut(): "//&
1108  "Mass loss over land?")
1109  endif

1112  if (netmassin(i)+netmassout(i) /= 0.0) then
1113 !$OMP critical
1114  numberofgroundings = numberofgroundings +1
1115  if (numberofgroundings<=maxgroundings) then
1116  iground(numberofgroundings) = i ! Record i,j location of event for
1117  jground(numberofgroundings) = j ! warning message
1118  hgrounding(numberofgroundings) = netmassin(i)+netmassout(i)
1119  endif
1120 !$OMP end critical
1121  if (cs%id_createdH>0) cs%createdH(i,j) = cs%createdH(i,j) - (netmassin(i)+netmassout(i))/dt
1122  endif
1123 
1124  enddo ! i
1125 
1126  ! Step C/ in the application of fluxes
1127  ! Heat by the convergence of penetrating SW.
1128  ! SW penetrative heating uses the updated thickness from above.
1129 
1130  ! Save temperature before increment with SW heating
1131  ! and initialize CS%penSWflux_diag to zero.
1132  if(cs%id_penSW_diag > 0 .or. cs%id_penSWflux_diag > 0) then
1133  do k=1,nz ; do i=is,ie
1134  cs%penSW_diag(i,j,k) = t2d(i,k)
1135  cs%penSWflux_diag(i,j,k) = 0.0
1136  enddo ; enddo
1137  k=nz+1 ; do i=is,ie
1138  cs%penSWflux_diag(i,j,k) = 0.0
1139  enddo
1140  endif
1141 
1142  if (calculate_energetics) then
1143  call absorbremainingsw(g, gv, h2d, opacityband, nsw, j, dt, h_limit_fluxes, &
1144  .false., .true., t2d, pen_sw_bnd, tke=pen_tke_2d, dsv_dt=dsv_dt_2d)

1146  do k=1,nz ; do i=is,ie
1147  ctke(i,j,k) = ctke(i,j,k) + pen_tke_2d(i,k)
1148  enddo ; enddo
1149  else
1150  call absorbremainingsw(g, gv, h2d, opacityband, nsw, j, dt, h_limit_fluxes, &
1151  .false., .true., t2d, pen_sw_bnd)
1152  endif
1153 
1154 
1155  ! Step D/ copy updated thickness and temperature
1156  ! 2d slice now back into model state.
1157  do k=1,nz ; do i=is,ie
1158  h(i,j,k) = h2d(i,k)
1159  tv%T(i,j,k) = t2d(i,k)
1160  enddo ; enddo
1161 
1162  ! Diagnose heating (W/m2) applied to a grid cell from SW penetration
1163  ! Also diagnose the penetrative SW heat flux at base of layer.
1164  if(cs%id_penSW_diag > 0 .or. cs%id_penSWflux_diag > 0) then
1165 
1166  ! convergence of SW into a layer
1167  do k=1,nz ; do i=is,ie
1168  cs%penSW_diag(i,j,k) = (t2d(i,k)-cs%penSW_diag(i,j,k))*h(i,j,k) * idt * tv%C_p * gv%H_to_kg_m2
1169  enddo ; enddo
1170 
1171  ! Perform a cumulative sum upwards from bottom to
1172  ! diagnose penetrative SW flux at base of tracer cell.
1173  ! CS%penSWflux_diag(i,j,k=1) is penetrative shortwave at top of ocean.
1174  ! CS%penSWflux_diag(i,j,k=kbot+1) is zero, since assume no SW penetrates rock.
1175  ! CS%penSWflux_diag = rsdo and CS%penSW_diag = rsdoabsorb
1176  ! rsdoabsorb(k) = rsdo(k) - rsdo(k+1), so that rsdo(k) = rsdo(k+1) + rsdoabsorb(k)
1177  if(cs%id_penSWflux_diag > 0) then
1178  do k=nz,1,-1 ; do i=is,ie
1179  cs%penSWflux_diag(i,j,k) = cs%penSW_diag(i,j,k) + cs%penSWflux_diag(i,j,k+1)
1180  enddo ; enddo
1181  endif
1182 
1183  endif
1184 
1185  ! Fill CS%nonpenSW_diag
1186  if(cs%id_nonpenSW_diag > 0) then
1187  do i=is,ie
1188  cs%nonpenSW_diag(i,j) = nonpensw(i)
1189  enddo
1190  endif
1191 
1192  enddo ! j-loop finish
1193 
1194  ! Post the diagnostics
1195  if (cs%id_createdH > 0) call post_data(cs%id_createdH , cs%createdH , cs%diag)
1196  if (cs%id_penSW_diag > 0) call post_data(cs%id_penSW_diag , cs%penSW_diag , cs%diag)
1197  if (cs%id_penSWflux_diag > 0) call post_data(cs%id_penSWflux_diag, cs%penSWflux_diag, cs%diag)
1198  if (cs%id_nonpenSW_diag > 0) call post_data(cs%id_nonpenSW_diag , cs%nonpenSW_diag , cs%diag)
1199 
1200  if (numberofgroundings>0) then
1201  do i = 1, min(numberofgroundings, maxgroundings)
1202  call forcing_singlepointprint(fluxes,g,iground(i),jground(i),'applyBoundaryFluxesInOut (grounding)')
1203  write(mesg(1:45),'(3es15.3)') g%geoLonT( iground(i), jground(i) ), &
1204  g%geoLatT( iground(i), jground(i)) , hgrounding(i)
1205  call mom_error(warning, "MOM_diabatic_driver.F90, applyBoundaryFluxesInOut(): "//&
1206  "Mass created. x,y,dh= "//trim(mesg), all_print=.true.)
1207  enddo
1208 
1209  if (numberofgroundings - maxgroundings > 0) then
1210  write(mesg, '(i4)') numberofgroundings - maxgroundings
1211  call mom_error(warning, "MOM_diabatic_driver:F90, applyBoundaryFluxesInOut(): "//&
1212  trim(mesg) // " groundings remaining")
1213  endif
1214  endif
