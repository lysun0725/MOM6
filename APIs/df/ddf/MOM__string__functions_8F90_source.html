108 function left_real(val)
109  real, intent(in) :: val
110  character(len=32) :: left_real
111 ! Returns a left-justified string with a real formatted like '(G)'
112  integer :: l, ind
113 
114  if ((abs(val) < 1.0e4) .and. (abs(val) >= 1.0e-3)) then
115  write(left_real, '(F30.11)') val
116  if (.not.isformattedfloatequalto(left_real,val)) then
117  write(left_real, '(F30.12)') val
118  if (.not.isformattedfloatequalto(left_real,val)) then
119  write(left_real, '(F30.13)') val
120  if (.not.isformattedfloatequalto(left_real,val)) then

124  if (.not.isformattedfloatequalto(left_real,val)) then
125  write(left_real, '(F30.16)') val
126  endif
127  endif
128  endif
129  endif
130  endif
131  do
132  l = len_trim(left_real)
133  if ((l<2) .or. (left_real(l-1:l) == ".0") .or. &
134  (left_real(l:l) /= "0")) exit

136  enddo
137  elseif (val == 0.) then
138  left_real = "0.0"

154 function left_reals(r,sep)
155 ! Returns a character string of a comma-separated, compact formatted, reals
156 ! e.g. "1., 2., 5*3., 5.E2"
157  ! Arguments
158  character(len=1320) :: left_reals
159  real, intent(in) :: r(:)
160  character(len=*), optional :: sep
161  ! Local variables
162  integer :: j, n, b, ns
163  logical :: doWrite
164  character(len=10) :: separator
165  n=1 ; dowrite=.true. ; left_reals='' ; b=1
166  if (present(sep)) then
167  separator=sep ; ns=len(sep)
168  else
169  separator=', ' ; ns=2
170  endif
171  do j=1,size(r)
172  dowrite=.true.
173  if (j<size(r)) then
174  if (r(j)==r(j+1)) then
175  n=n+1
176  dowrite=.false.
177  endif
178  endif
179  if (dowrite) then
180  if (b>1) then ! Write separator if a number has already been written
181  write(left_reals(b:),'(A)') separator
182  b=b+ns
183  endif
184  if (n>1) then
185  write(left_reals(b:),'(A,"*",A)') trim(left_int(n)),trim(left_real(r(j)))
186  else
187  write(left_reals(b:),'(A)') trim(left_real(r(j)))
188  endif
189  n=1 ; b=len_trim(left_reals)+1
190  endif
191  enddo

209 function extractword(string,n)
210 ! Returns string corresponding to the nth word in the argument
211 ! or "" if the string is not long enough. Both spaces and commas
212 ! are interpretted as separators.
213  character(len=*), intent(in) :: string
214  integer, intent(in) :: n
215  character(len=120) :: extractWord
216  ! Local variables
217  integer :: ns, i, b, e, nw
218  logical :: lastCharIsSeperator
219  extractword = ''
220  lastcharisseperator = .true.
221  ns = len_trim(string)
222  i = 0; b=0; e=0; nw=0;
223  do while (i<ns)
224  i = i+1
225  if (lastcharisseperator) then ! search for end of word
226  if (string(i:i)==' ' .or. string(i:i)==',') then
227  continue ! Multiple separators, .e.g ' ' or ', '
228  else
229  lastcharisseperator = .false. ! character is beginning of word
230  b = i
231  continue
232  endif
233  else ! continue search for end of word
234  if (string(i:i)==' ' .or. string(i:i)==',') then
235  lastcharisseperator = .true.
236  e = i-1 ! Previous character is end of word
237  nw = nw+1
238  if (nw==n) then
239  extractword = trim(string(b:e))
240  return
241  endif
242  endif
243  endif
244  enddo
245  if (b<=ns) extractword = trim(string(b:ns))
246 end function extractword
