47 subroutine scm_idealized_hurricane_ts_init(T, S, h, G, GV, param_file)

51  real, dimension(SZI_(G),SZJ_(G), SZK_(G)), intent(out) :: S !< Salinity (psu)

53  type(param_file_type), intent(in) :: param_file !< Input parameter structure

65  'Reference salinity', units='1e-3',default=35.0)
66  call get_param(param_file,mod,"SCM_SST_REF",sst_ref, &
67  'Reference surface temperature', units='C', fail_if_missing=.true.)

149 subroutine scm_idealized_hurricane_wind_forcing(state, fluxes, day, G, CS)
150  type(surface), intent(in) :: state !< Surface state structure
151  type(forcing), intent(inout) :: fluxes !< Surface fluxes structure
152  type(time_type), intent(in) :: day !< Time in days
153  type(ocean_grid_type), intent(inout) :: G !< Grid structure
154  type(scm_idealized_hurricane_cs), pointer :: CS !< Container for SCM parameters
155  ! Local variables
156  integer :: i, j, is, ie, js, je, Isq, Ieq, Jsq, Jeq
157  integer :: isd, ied, jsd, jed, IsdB, IedB, JsdB, JedB
158  real :: pie, Deg2Rad
159  real :: U10, A, B, C, r, f,du10,rkm ! For wind profile expression
160  real :: xx, t0 !for location
161  real :: dp, rB
162  real :: Cd ! Air-sea drag coefficient
163  real :: Uocn, Vocn ! Surface ocean velocity components
164  real :: dU, dV ! Air-sea differential motion
165  !Wind angle variables
166  real :: Alph,Rstr, A0, A1, P1, Adir, transdir, V_TS, U_TS
167  logical :: BR_Bench
168  ! Bounds for loops and memory allocation
169  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
170  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
171  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
172  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
173 
174  ! Allocate the forcing arrays, if necessary.
175  call allocate_forcing_type(g, fluxes, stress=.true., ustar=.true.)
176 
177  pie = 4.0*atan(1.0) ; deg2rad = pie/180.
178 
179  !/ BR
180  ! Implementing Holland (1980) parameteric wind profile
181  !------------------------------------------------------|
182  br_bench = .true. !true if comparing to LES runs |
183  t0 = 129600. !TC 'eye' crosses (0,0) at 36 hours|
184  transdir = pie !translation direction (-x) |
185  !------------------------------------------------------|
186  dp = cs%p_n - cs%p_c
187  c = cs%U_max / sqrt( dp )
188  b = c**2 * cs%rho_a * exp(1.0)
189  if (br_bench) then
190  ! rho_a reset to value used in generated wind for benchmark test
191  b = c**2 * 1.2 * exp(1.0)
192  endif
193  a = (cs%r_max/1000.)**b
194  f =g%CoriolisBu(is,js) ! f=f(x,y) but in the SCM is constant
195  if (br_bench) then
196  ! f reset to value used in generated wind for benchmark test
197  f = 5.5659e-05
198  endif
199  !/ BR
200  ! Calculate x position as a function of time.
201  xx = ( t0 - time_type_to_real(day)) * cs%tran_speed * cos(transdir)
202  r = sqrt(xx**2.+cs%YY**2.)
203  !/ BR
204  ! rkm - r converted to km for Holland prof.
205  ! used in km due to error, correct implementation should
206  ! not need rkm, but to match winds w/ experiment this must
207  ! be maintained. Causes winds far from storm center to be a
208  ! couple of m/s higher than the correct Holland prof.
209  if (br_bench) then
210  rkm = r/1000.
211  rb = (rkm)**b
212  else
213  ! if not comparing to benchmark, then use correct Holland prof.
214  rkm = r
215  rb = r**b
216  endif
217  !/ BR
218  ! Calculate U10 in the interior (inside of 10x radius of maximum wind),
219  ! while adjusting U10 to 0 outside of 12x radius of maximum wind.
220  ! Note that rho_a is set to 1.2 following generated wind for experiment
221  if (r/cs%r_max.gt.0.001 .AND. r/cs%r_max.lt.10.) then
222  u10 = sqrt( a*b*dp*exp(-a/rb)/(1.2*rb) + 0.25*(rkm*f)**2 ) - 0.5*rkm*f
223  elseif (r/cs%r_max.gt.10. .AND. r/cs%r_max.lt.12.) then
224  r=cs%r_max*10.
225  if (br_bench) then
226  rkm = r/1000.
227  rb=rkm**b
228  else
229  rkm = r
230  rb = r**b
231  endif
232  u10 = ( sqrt( a*b*dp*exp(-a/rb)/(1.2*rb) + 0.25*(rkm*f)**2 ) - 0.5*rkm*f) &
233  * (12. - r/cs%r_max)/2.
234  else
235  u10 = 0.
236  end if
237  adir = atan2(cs%YY,xx)
238 
239  !/ BR
240  ! Wind angle model following Zhang and Ulhorn (2012)
241  ! ALPH is inflow angle positive outward.
242  rstr = min(10.,r / cs%r_max)
243  a0 = -0.9*rstr -0.09*cs%U_max -14.33
244  a1 = -a0 *(0.04*rstr +0.05*cs%tran_speed+0.14)
245  p1 = (6.88*rstr -9.60*cs%tran_speed+85.31)*pie/180.
246  alph = a0 - a1*cos( (transdir - adir ) - p1)
247  if (r/cs%r_max.gt.10. .AND. r/cs%r_max.lt.12.) then
248  alph = alph* (12. - r/cs%r_max)/2.
249  elseif (r/cs%r_max.gt.12.) then
250  alph = 0.0
251  endif
252  alph = alph * deg2rad
253 
254  !/BR
255  ! Prepare for wind calculation
256  ! X_TS is component of translation speed added to wind vector
257  ! due to background steering wind.
258  u_ts = cs%tran_speed/2.*cos(transdir)
259  v_ts = cs%tran_speed/2.*sin(transdir)
260 
261  ! Set the surface wind stresses, in units of Pa. A positive taux
262  ! accelerates the ocean to the (pseudo-)east.
263  ! The i-loop extends to is-1 so that taux can be used later in the
264  ! calculation of ustar - otherwise the lower bound would be Isq.
265  do j=js,je ; do i=is-1,ieq
266  !/BR
267  ! Turn off surface current for stress calculation to be
268  ! consistent with test case.
269  uocn = 0.!state%u(I,j)
270  vocn = 0.!0.25*( (state%v(i,J) + state%v(i+1,J-1)) &
271  ! +(state%v(i+1,J) + state%v(i,J-1)) )
272  !/BR
273  ! Wind vector calculated from location/direction (sin/cos flipped b/c
274  ! cyclonic wind is 90 deg. phase shifted from position angle).
275  du = u10*sin(adir-pie-alph) - uocn + u_ts
276  dv = u10*cos(adir-alph) - vocn + v_ts
277  !/----------------------------------------------------|
278  !BR
279  ! Add a simple drag coefficient as a function of U10 |
280  !/----------------------------------------------------|
281  du10=sqrt(du**2+dv**2)
282  if (du10.LT.11.) then
283  cd = 1.2e-3
284  elseif (du10.LT.20.) then
285  cd = (0.49 + 0.065 * u10 )*0.001
286  else
287  cd = 0.0018
288  endif
289  fluxes%taux(i,j) = cs%rho_a * g%mask2dCu(i,j) * cd*sqrt(du**2+dv**2)*du
290  enddo ; enddo
291  !/BR
292  ! See notes above
293  do j=js-1,jeq ; do i=is,ie
294  uocn = 0.!0.25*( (state%u(I,j) + state%u(I-1,j+1)) &
295  ! +(state%u(I-1,j) + state%u(I,j+1)) )
296  vocn = 0.!state%v(i,J)
297  du = u10*sin(adir-pie-alph) - uocn + u_ts
298  dv = u10*cos(adir-alph) - vocn + v_ts
299  du10=sqrt(du**2+dv**2)
300  if (du10.LT.11.) then
301  cd = 1.2e-3
302  elseif (du10.LT.20.) then
303  cd = (0.49 + 0.065 * u10 )*0.001
304  else
305  cd = 0.0018
306  endif
307  fluxes%tauy(i,j) = cs%rho_a * g%mask2dCv(i,j) * cd*du10*dv
308  enddo ; enddo
309  ! Set the surface friction velocity, in units of m s-1. ustar is always positive.
310  do j=js,je ; do i=is,ie
311  ! This expression can be changed if desired, but need not be.
312  fluxes%ustar(i,j) = g%mask2dT(i,j) * sqrt(cs%gust_const/cs%Rho0 + &
313  sqrt(0.5*(fluxes%taux(i-1,j)**2 + fluxes%taux(i,j)**2) + &
314  0.5*(fluxes%tauy(i,j-1)**2 + fluxes%tauy(i,j)**2))/cs%Rho0)
315  enddo ; enddo
316 
317 end subroutine scm_idealized_hurricane_wind_forcing
