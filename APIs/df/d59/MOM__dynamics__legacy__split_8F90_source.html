25 !* By Robert Hallberg and Alistair Adcroft, 1996-2012 *
26 !* *
27 !* This file contains code that does the time-stepping of the *
28 !* adiabatic dynamic core, in this case with mode-splitting between *
29 !* the baroclinic and barotropic modes and a pseudo-second order *
30 !* Runge-Kutta time stepping scheme for the baroclinic momentum *
31 !* equation and a forward-backward coupling between the baroclinic *
32 !* momentum and continuity equations. This split time-stepping *
33 !* scheme is described in detail in Hallberg (JCP, 1997), with the *
34 !* primary issues related to exact tracer conservation and how to *
35 !* ensure consistency between the barotropic and layered estimates *
36 !* of the free surface height described carefully in Hallberg and *
37 !* Adcroft (Ocean Modelling, 2009). This was the time stepping code *
38 !* that is used for most GOLD applications, including GFDL's ESM2G *
39 !* Earth system model, and all of the examples provided with the *
40 !* MOM code (although several of these solutions are routinely *
41 !* verified by comparison with the slower unsplit schemes). *
42 !* *
43 !* The subroutine step_MOM_dyn_legacy_split actually does the time *
44 !* stepping, while register_restarts_dyn_legacy_split sets the fields *
45 !* that are found in a full restart file with this scheme, and *
46 !* initialize_dyn_legacy_split initializes the cpu clocks that are * *
47 !* used in this module. For largely historical reasons, this module *
48 !* does not have its own control structure, but shares the same *
49 !* control structure with MOM.F90 and the other MOM_dynamics_... *
50 !* modules. *
51 !* *
52 !* Macros written all in capital letters are defined in MOM_memory.h. *
53 !* *
54 !* A small fragment of the grid is shown below: *
55 !* *
56 !* j+1 x ^ x ^ x At x: q, CoriolisBu *
57 !* j+1 > o > o > At ^: v, PFv, CAv, vh, diffv, tauy, vbt, vhtr *
58 !* j x ^ x ^ x At >: u, PFu, CAu, uh, diffu, taux, ubt, uhtr *
59 !* j > o > o > At o: h, bathyT, eta, T, S, tr *
60 !* j-1 x ^ x ^ x *
61 !* i-1 i i+1 *
62 !* i i+1 *
63 !* *
64 !* The boundaries always run through q grid points (x). *
65 !* *

798  call uchksum(u_av,"Predictor avg u",g,haloshift=1)
799  call vchksum(v_av,"Predictor avg v",g,haloshift=1)
800  call hchksum(gv%H_to_kg_m2*h_av,"Predictor avg h",g,haloshift=0)
801  ! call MOM_state_chksum("Predictor avg ", u_av, v_av, h_av,uh, vh, G, GV)
802  call check_redundant("Predictor up ", up, vp, g)
803  call check_redundant("Predictor uh ", uh, vh, g)
804  endif
805 
806 ! diffu = horizontal viscosity terms (u_av)
807  call cpu_clock_begin(id_clock_horvisc)
808  call horizontal_viscosity(u_av, v_av, h_av, cs%diffu, cs%diffv, &
809  meke, varmix, g, gv, cs%hor_visc_CSp, obc=cs%OBC)
810  call cpu_clock_end(id_clock_horvisc)
811 
812 ! CAu = -(f+zeta_av)/h_av vh + d/dx KE_av
813  call cpu_clock_begin(id_clock_cor)
814  call coradcalc(u_av, v_av, h_av, uh, vh, cs%CAu, cs%CAv, cs%ADp, g, gv, &
815  cs%CoriolisAdv_CSp)
816  call cpu_clock_end(id_clock_cor)
817 
818 ! Calculate the momentum forcing terms for the barotropic equations.
819 
820 ! u_bc_accel = CAu + PFu + diffu(u[n-1])
821  call cpu_clock_begin(id_clock_btforce)
822  do k=1,nz ; do j=js,je ; do i=isq,ieq
823  u_bc_accel(i,j,k) = (cs%Cau(i,j,k) + cs%PFu(i,j,k)) + cs%diffu(i,j,k)
824  enddo ; enddo ; enddo
825  do k=1,nz ; do j=jsq,jeq ; do i=is,ie
826  v_bc_accel(i,j,k) = (cs%Cav(i,j,k) + cs%PFv(i,j,k)) + cs%diffv(i,j,k)
827  enddo ; enddo ; enddo
828  call cpu_clock_end(id_clock_btforce)
829 
830  if (cs%debug) then
831  call check_redundant("corr pre-btstep CS%Ca ", cs%Cau, cs%Cav, g)
832  call check_redundant("corr pre-btstep CS%PF ", cs%PFu, cs%PFv, g)
833  call check_redundant("corr pre-btstep CS%diff ", cs%diffu, cs%diffv, g)
834  call check_redundant("corr pre-btstep u_bc_accel ", u_bc_accel, v_bc_accel, g)
835  endif
836 
837 ! u_accel_bt = layer accelerations due to barotropic solver
838 ! pbce = dM/deta
839  call cpu_clock_begin(id_clock_btstep)
840  if (cs%flux_BT_coupling) then
841  call legacy_btstep(.true., uh_in, vh_in, eta, dt, u_bc_accel, v_bc_accel, &
842  fluxes, cs%pbce, cs%eta_PF, uh, vh, cs%u_accel_bt, &
843  cs%v_accel_bt, eta, cs%uhbt, cs%vhbt, g, gv, &
844  cs%barotropic_CSp, cs%visc_rem_u, cs%visc_rem_v, etaav=eta_av, &
845  uhbt_out = uhbt_out, vhbt_out = vhbt_out, obc=cs%OBC, &
846  bt_cont = cs%BT_cont, eta_pf_start = eta_pf_start, &
847  taux_bot=taux_bot, tauy_bot=tauy_bot)
848  else
849  if (cs%BT_use_layer_fluxes) then
850  uh_ptr => uh ; vh_ptr => vh ; u_ptr => u_av ; v_ptr => v_av
851  endif
852 
853  call legacy_btstep(.false., u, v, eta, dt, u_bc_accel, v_bc_accel, &
854  fluxes, cs%pbce, cs%eta_PF, u_av, v_av, cs%u_accel_bt, &
855  cs%v_accel_bt, eta, cs%uhbt, cs%vhbt, g, gv, &
856  cs%barotropic_CSp, cs%visc_rem_u, cs%visc_rem_v, &
857  etaav=eta_av, obc=cs%OBC, &
858  bt_cont = cs%BT_cont, eta_pf_start=eta_pf_start, &
859  taux_bot=taux_bot, tauy_bot=tauy_bot, &
860  uh0=uh_ptr, vh0=vh_ptr, u_uh0=u_ptr, v_vh0=v_ptr)
861  endif
862  call cpu_clock_end(id_clock_btstep)
863 
864  if (cs%debug) then
865  call check_redundant("u_accel_bt ", cs%u_accel_bt, cs%v_accel_bt, g)
866  endif
867 
868 ! u = u + dt*( u_bc_accel + u_accel_bt )
869  call cpu_clock_begin(id_clock_mom_update)
870  do k=1,nz ; do j=js,je ; do i=isq,ieq
871  u(i,j,k) = g%mask2dCu(i,j) * (u_init(i,j,k) + dt * &
872  (u_bc_accel(i,j,k) + cs%u_accel_bt(i,j,k)))
873  enddo ; enddo ; enddo
874 
875  do k=1,nz ; do j=jsq,jeq ; do i=is,ie
876  v(i,j,k) = g%mask2dCv(i,j) * (v_init(i,j,k) + dt * &
877  (v_bc_accel(i,j,k) + cs%v_accel_bt(i,j,k)))
878  enddo ; enddo ; enddo
879  call cpu_clock_end(id_clock_mom_update)
880 
881  if (cs%debug) then
882  call uchksum(u,"Corrector 1 u",g,haloshift=0)
883  call vchksum(v,"Corrector 1 v",g,haloshift=0)
884  call hchksum(gv%H_to_kg_m2*h,"Corrector 1 h",g,haloshift=2)
885  call uchksum(gv%H_to_kg_m2*uh,"Corrector 1 uh",g,haloshift=2)
886  call vchksum(gv%H_to_kg_m2*vh,"Corrector 1 vh",g,haloshift=2)
887  ! call MOM_state_chksum("Corrector 1", u, v, h, uh, vh, G, GV, haloshift=1)
888  call mom_accel_chksum("Corrector accel", cs%CAu, cs%CAv, cs%PFu, cs%PFv, &
889  cs%diffu, cs%diffv, g, gv, cs%pbce, cs%u_accel_bt, cs%v_accel_bt)
890  endif
891 
892 ! u <- u + dt d/dz visc d/dz u
893 ! u_av <- u_av + dt d/dz visc d/dz u_av
894  call cpu_clock_begin(id_clock_vertvisc)
895  call vertvisc_coef(u, v, h, fluxes, visc, dt, g, gv, cs%vertvisc_CSp)
896  call vertvisc(u, v, h, fluxes, visc, dt, cs%OBC, cs%ADp, cs%CDp, g, gv, &
897  cs%vertvisc_CSp, cs%taux_bot, cs%tauy_bot)
898  if (g%nonblocking_updates) then
899  call cpu_clock_end(id_clock_vertvisc) ; call cpu_clock_begin(id_clock_pass)
900  pid_u = pass_vector_start(u, v, g%Domain)
901  call cpu_clock_end(id_clock_pass) ; call cpu_clock_begin(id_clock_vertvisc)
902  endif
903  call vertvisc_remnant(visc, cs%visc_rem_u, cs%visc_rem_v, dt, g, gv, cs%vertvisc_CSp)
904  call cpu_clock_end(id_clock_vertvisc)
905 
906 ! Later, h_av = (h_in + h_out)/2, but for now use h_av to store h_in.
907  do k=1,nz ; do j=js-2,je+2 ; do i=is-2,ie+2
908  h_av(i,j,k) = h(i,j,k)
909  enddo ; enddo ; enddo
910 
911  call cpu_clock_begin(id_clock_pass)
912  call pass_vector(cs%visc_rem_u, cs%visc_rem_v, g%Domain, &
913  to_all+scalar_pair, cgrid_ne)
914  if (g%nonblocking_updates) then
915  call pass_vector_complete(pid_u, u, v, g%Domain)
916  else
917  call pass_vector(u, v, g%Domain)
918  endif
919  call cpu_clock_end(id_clock_pass)
920 
921 ! uh = u_av * h
922 ! h = h + dt * div . uh
923  if (cs%flux_BT_coupling) then
924  ! u_av and v_av adjusted so their mass transports match uhbt and vhbt.
925  ! Also, determine the values of u and v so that their transports
926  ! that agree with uhbt_out and vhbt_out.
927  if (ASSOCIATED(cs%ADp%du_other)) then ; do k=1,nz ; do j=js,je ; do i=isq,ieq
928  u_tmp(i,j,k) = u(i,j,k)
929  enddo ; enddo ; enddo ; endif
930  if (ASSOCIATED(cs%ADp%dv_other)) then ; do k=1,nz ; do j=jsq,jeq ; do i=is,ie
931  v_tmp(i,j,k) = v(i,j,k)
932  enddo ; enddo ; enddo ; endif
933  call cpu_clock_begin(id_clock_continuity)
934  call continuity(u, v, h, h, uh, vh, dt, g, gv, &
935  cs%continuity_CSp, cs%uhbt, cs%vhbt, cs%OBC, &
936  cs%visc_rem_u, cs%visc_rem_v, u_av, v_av, &
937  uhbt_out, vhbt_out, u, v)
938  call cpu_clock_end(id_clock_continuity)
939  ! Whenever thickness changes let the diag manager know, target grids
940  ! for vertical remapping may need to be regenerated.
941  call diag_update_target_grids(cs%diag)
942  if (g%nonblocking_updates) then
943  call cpu_clock_begin(id_clock_pass)
944  pid_h = pass_var_start(h, g%Domain)
945  call cpu_clock_end(id_clock_pass)
946  endif
947  if (ASSOCIATED(cs%ADp%du_other)) then ; do k=1,nz ; do j=js,je ; do i=isq,ieq
948  cs%ADp%du_other(i,j,k) = cs%ADp%du_other(i,j,k) + (u(i,j,k) - u_tmp(i,j,k))
949  enddo ; enddo ; enddo ; endif
950  if (ASSOCIATED(cs%ADp%dv_other)) then ; do k=1,nz ; do j=jsq,jeq ; do i=is,ie
951  cs%ADp%dv_other(i,j,k) = cs%ADp%dv_other(i,j,k) + (v(i,j,k) - v_tmp(i,j,k))
952  enddo ; enddo ; enddo ; endif
953 
954  call cpu_clock_begin(id_clock_vertvisc)
955  call vertvisc_limit_vel(u, v, h_av, cs%ADp, cs%CDp, fluxes, visc, dt, g, gv, cs%vertvisc_CSp)
956  if (g%nonblocking_updates) then
957  call cpu_clock_end(id_clock_vertvisc) ; call cpu_clock_begin(id_clock_pass)
958  pid_u = pass_vector_start(u, v, g%Domain)
959  call cpu_clock_end(id_clock_pass) ; call cpu_clock_begin(id_clock_vertvisc)
960  endif
961  call vertvisc_limit_vel(u_av, v_av, h_av, cs%ADp, cs%CDp, fluxes, visc, dt, g, gv, cs%vertvisc_CSp)
962  call cpu_clock_end(id_clock_vertvisc)
963 
964  call cpu_clock_begin(id_clock_pass)
965  if (g%nonblocking_updates) then
966  call pass_var_complete(pid_h, h, g%Domain)
967  call pass_vector_complete(pid_u, u, v, g%Domain)
968  else
969  call pass_var(h, g%Domain)
970  call pass_vector(u, v, g%Domain, complete=.false.)
971  endif
972  call cpu_clock_end(id_clock_pass)
973  else
974  ! u_av and v_av adjusted so their mass transports match uhbt and vhbt.
975  call cpu_clock_begin(id_clock_continuity)
976  call continuity(u, v, h, h, uh, vh, dt, g, gv, &
977  cs%continuity_CSp, cs%uhbt, cs%vhbt, cs%OBC, &
978  cs%visc_rem_u, cs%visc_rem_v, u_av, v_av)
979  call cpu_clock_end(id_clock_continuity)
980  ! Whenever thickness changes let the diag manager know, target grids
981  ! for vertical remapping may need to be regenerated.
982  call diag_update_target_grids(cs%diag)
983  call cpu_clock_begin(id_clock_pass)
984  call pass_var(h, g%Domain)
985  call cpu_clock_end(id_clock_pass)
986  endif
987 
988  call cpu_clock_begin(id_clock_pass)
989  if (g%nonblocking_updates) then
990  pid_uh = pass_vector_start(uh(:,:,:), vh(:,:,:), g%Domain)
991  pid_u_av = pass_vector_start(u_av, v_av, g%Domain)
992  else
993  call pass_vector(u_av, v_av, g%Domain, complete=.false.)
994  call pass_vector(uh(:,:,:), vh(:,:,:), g%Domain)
995  endif
996  call cpu_clock_end(id_clock_pass)
997 
998  if (associated(cs%OBC)) then
999  call radiation_open_bdry_conds(cs%OBC, u, u_old_rad_obc, v, &
1000  v_old_rad_obc, h, h_old_rad_obc, g, cs%open_boundary_CSp)
1001  endif
1002 
1003 ! h_av = (h_in + h_out)/2 . Going in to this line, h_av = h_in.
1004  do k=1,nz ; do j=js-2,je+2 ; do i=is-2,ie+2
1005  h_av(i,j,k) = 0.5*(h_av(i,j,k) + h(i,j,k))
1006  enddo ; enddo ; enddo
1007 
1008  if (g%nonblocking_updates) then
1009  call cpu_clock_begin(id_clock_pass)
1010  call pass_vector_complete(pid_uh, uh(:,:,:), vh(:,:,:), g%Domain)
1011  call pass_vector_complete(pid_u_av, u_av, v_av, g%Domain)
1012  call cpu_clock_end(id_clock_pass)
1013  endif
1014 
1015  do k=1,nz ; do j=js-2,je+2 ; do i=isq-2,ieq+2
1016  uhtr(i,j,k) = uhtr(i,j,k) + uh(i,j,k)*dt
1017  enddo ; enddo ; enddo
1018  do k=1,nz ; do j=jsq-2,jeq+2 ; do i=is-2,ie+2
1019  vhtr(i,j,k) = vhtr(i,j,k) + vh(i,j,k)*dt
1020  enddo ; enddo ; enddo
1021 
1022 ! The time-averaged free surface height has already been set by the last
1023 ! call to btstep.
1024 
1025 ! Here various terms used in to update the momentum equations are
1026 ! offered for averaging.
1027  if (cs%id_PFu > 0) call post_data(cs%id_PFu, cs%PFu, cs%diag)
1028  if (cs%id_PFv > 0) call post_data(cs%id_PFv, cs%PFv, cs%diag)
1029  if (cs%id_CAu > 0) call post_data(cs%id_CAu, cs%CAu, cs%diag)
1030  if (cs%id_CAv > 0) call post_data(cs%id_CAv, cs%CAv, cs%diag)
1031 
1032 ! Here the thickness fluxes are offered for averaging.
1033  if (cs%id_uh > 0) call post_data(cs%id_uh, uh, cs%diag)
1034  if (cs%id_vh > 0) call post_data(cs%id_vh, vh, cs%diag)
1035  if (cs%id_uav > 0) call post_data(cs%id_uav, u_av, cs%diag)
1036  if (cs%id_vav > 0) call post_data(cs%id_vav, v_av, cs%diag)
1037  if (cs%id_u_BT_accel > 0) call post_data(cs%id_u_BT_accel, cs%u_accel_bt, cs%diag)
1038  if (cs%id_v_BT_accel > 0) call post_data(cs%id_v_BT_accel, cs%v_accel_bt, cs%diag)
1039  if (cs%id_du_adj > 0) call post_data(cs%id_du_adj, cs%ADp%du_other, cs%diag)
1040  if (cs%id_dv_adj > 0) call post_data(cs%id_dv_adj, cs%ADp%dv_other, cs%diag)
1041  if (cs%debug) then
1042  call mom_state_chksum("Corrector ", u, v, h, uh, vh, g, gv)
1043  call uchksum(u_av,"Corrector avg u",g,haloshift=1)
1044  call vchksum(v_av,"Corrector avg v",g,haloshift=1)
1045  call hchksum(gv%H_to_kg_m2*h_av,"Corrector avg h",g,haloshift=1)
1046  ! call MOM_state_chksum("Corrector avg ", u_av, v_av, h_av, uh, vh, G, GV)
1047  endif
1048 
1049 end subroutine step_mom_dyn_legacy_split
