24 !* By Robert Hallberg, April 1994 - June 2002. *
25 !* *
26 !* This program contains the subroutine that calculates the *
27 !* effects of horizontal viscosity, including parameterizations of *
28 !* the value of the viscosity itself. horizontal_viscosity calc- *
29 !* ulates the acceleration due to some combination of a biharmonic *
30 !* viscosity and a Laplacian viscosity. Either or both may use a *
31 !* coefficient that depends on the shear and strain of the flow. *
32 !* All metric terms are retained. The Laplacian is calculated as *
33 !* the divergence of a stress tensor, using the form suggested by *
34 !* Smagorinsky (1993). The biharmonic is calculated by twice *
35 !* applying the divergence of the stress tensor that is used to *
36 !* calculate the Laplacian, but without the dependence on thickness *
37 !* in the first pass. This form permits a variable viscosity, and *
38 !* indicates no acceleration for either resting fluid or solid body *
39 !* rotation. *
40 !* *
41 !* set_up_hor_visc calculates and stores the values of a number of *
42 !* metric functions that are used in horizontal_viscosity. It is *
43 !* called by horizontal_viscosity the first time that the latter is *
44 !* called. *
45 !* *
46 !* The form of the Laplacian viscosity is: *
47 !* *
48 !* diffu = 1/h * {d/dx[KH*h*sh_xx] + d/dy[KH*h*sh_xy]} *
49 !* diffv = 1/h * {d/dx[KH*h*sh_xy] - d/dy[KH*h*sh_xx]} *
50 !* *
51 !* sh_xx = du/dx - dv/dy sh_xy = du/dy + dv/dx *
52 !* *
53 !* with appropriate metric terms thrown in. KH may either be a *
54 !* constant or may vary with the shear, as proposed by Smagorinsky. *
55 !* The form of this term is discussed extensively in Griffies and *
56 !* Hallberg (MWR, 2000), and the implementation here follows that *
57 !* discussion closely. *
58 !* *
59 !* Only free slip boundary conditions have been coded, although *
60 !* no slip boundary conditions could be used with the Laplacian *
61 !* viscosity. For a western boundary, for example, the boundary *
62 !* conditions with the biharmonic operator would be written as: *
63 !* dv/dx = 0, d^3v/dx^3 = 0, u = 0, d^2u/dx^2 = 0 , *
64 !* while for a Laplacian operator, they are simply: *
65 !* dv/dx = 0, u = 0 . *
66 !* These boundary conditions are largely dictated by the use of *
67 !* a an Arakawa C-grid and by the varying layer thickness. *
68 !* *
69 !* *
70 !* *
71 !* *
72 !* *
73 !* Macros written all in capital letters are defined in MOM_memory.h. *
74 !* *
75 !* A small fragment of the C-grid is shown below: *
76 !* *
77 !* j+1 x ^ x ^ x At x: q, CoriolisBu, hq, str_xy, sh_xy *
78 !* j+1 > o > o > At ^: v, diffv, v0 *
79 !* j x ^ x ^ x At >: u, diffu, u0 *
80 !* j > o > o > At o: h, str_xx, sh_xx *
81 !* j-1 x ^ x ^ x *
82 !* i-1 i i+1 At x & ^: *
83 !* i i+1 At > & o: *
84 !* *
85 !* The boundaries always run through q grid points (x). *
86 !* *

202 subroutine horizontal_viscosity(u, v, h, diffu, diffv, MEKE, VarMix, G, GV, CS, OBC)
203  type(ocean_grid_type), intent(in) :: G
204  type(verticalgrid_type), intent(in) :: GV
205  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in) :: u
206  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(in) :: v
207  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h
208  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out) :: diffu
209  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(out) :: diffv
210  type(meke_type), pointer :: MEKE
211  type(varmix_cs), pointer :: VarMix
212  type(hor_visc_cs), pointer :: CS
213  type(ocean_obc_type), pointer, optional :: OBC
214 
215 ! Arguments:
216 ! (in) u - zonal velocity (m/s)
217 ! (in) v - meridional velocity (m/s)
218 ! (in) h - layer thickness (m or kg m-2); h units are referred to as H.
219 ! (out) diffu - zonal acceleration due to convergence of
220 ! along-coordinate stress tensor (m/s2)
221 ! (out) diffv - meridional acceleration due to convergence of
222 ! along-coordinate stress tensor (m/s2)
223 ! (inout) MEKE - pointer to a structure containing fields related to
224 ! Mesoscale Eddy Kinetic Energy
225 ! (in) VarMix - pointer to a structure with fields that specify the
226 ! spatially variable viscosities
227 ! (in) G - ocean grid structure
228 ! (in) GV - The ocean's vertical grid structure.
229 ! (in) CS - control structure returned by a previous call to
230 ! hor_visc_init
231 ! (in) OBC - pointer to an open boundary condition type
232 
233 ! By R. Hallberg, August 1998 - November 1998.
234 ! This subroutine determines the acceleration due to the
235 ! horizontal viscosity. A combination of biharmonic and Laplacian
236 ! forms can be used. The coefficient may either be a constant or
237 ! a shear-dependent form. The biharmonic is determined by twice
238 ! taking the divergence of an appropriately defined stress tensor.
239 ! The Laplacian is determined by doing so once.
240 ! To work, the following fields must be set outside of the usual
241 ! is to ie range before this subroutine is called:
242 ! v[is-2,is-1,ie+1,ie+2], u[is-2,is-1,ie+1,ie+2], and h[is-1,ie+1],
243 ! with a similarly sized halo in the y-direction.
244 
245  real, dimension(SZIB_(G),SZJ_(G)) :: u0 ! Laplacian of u (m-1 s-1)
246  real, dimension(SZI_(G),SZJB_(G)) :: v0 ! Laplacian of v (m-1 s-1)
247 
248  real, dimension(SZI_(G),SZJ_(G)) :: &
249  sh_xx, & ! horizontal tension (du/dx - dv/dy) (1/sec) including metric terms
250  str_xx,& ! str_xx is the diagonal term in the stress tensor (H m2 s-2)
251  bhstr_xx,& ! A copy of str_xx that only contains the biharmonic contribution (H m2 s-2)
252  FrictWorkIntz ! depth integrated energy dissipated by lateral friction (W/m2)
253 
254  real, dimension(SZIB_(G),SZJB_(G)) :: &
255  sh_xy, & ! horizontal shearing strain (du/dy + dv/dx) (1/sec) including metric terms
256  str_xy, & ! str_xy is the cross term in the stress tensor (H m2 s-2)
257  bhstr_xy ! A copy of str_xy that only contains the biharmonic contribution (H m2 s-2)
258 
259  real, dimension(SZIB_(G),SZJB_(G),SZK_(G)) :: &
260  Ah_q, & ! biharmonic viscosity at corner points (m4/s)
261  Kh_q ! Laplacian viscosity at corner points (m2/s)
262 
263  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: &
264  Ah_h, & ! biharmonic viscosity at thickness points (m4/s)
265  Kh_h, & ! Laplacian viscosity at thickness points (m2/s)
266  FrictWork ! energy dissipated by lateral friction (W/m2)
267 
268  real :: Ah ! biharmonic viscosity (m4/s)
269  real :: Kh ! Laplacian viscosity (m2/s)
270  real :: AhSm ! Smagorinsky biharmonic viscosity (m4/s)
271  real :: KhSm ! Smagorinsky Laplacian viscosity (m2/s)
272  real :: Shear_mag ! magnitude of the shear (1/s)
273  real :: huq, hvq ! temporary variables in units of H^2 (i.e. m2 or kg2 m-4).
274  real :: hu, hv ! thicknesses at velocity points in units of H (i.e. m or kg m-2).
275  real :: hq ! harmonic mean of the harmonic means of the u- & v-
276  ! point thicknesses, in H; guarantees that hq/hu < 4.
277  real :: h_neglect ! thickness so small it can be lost in roundoff and so neglected (H)
278  real :: h_neglect3 ! h_neglect^3, in H3
279  real :: hrat_min ! minimum thicknesses at the 4 neighboring
280  ! velocity points divided by the thickness at the stress
281  ! point (h or q point) (nondimensional)
282  real :: visc_bound_rem ! fraction of overall viscous bounds that
283  ! remain to be applied (nondim)
284  real :: Kh_scale ! A factor between 0 and 1 by which the horizontal
285  ! Laplacian viscosity is rescaled
286  real :: RoScl ! The scaling function for MEKE source term
287  real :: FatH ! abs(f) at h-point for MEKE source term (s-1)
288 
289  logical :: rescale_Kh
290  logical :: find_FrictWork
291  logical :: apply_OBC = .false.
292  logical :: use_MEKE_Ku
293  integer :: is, ie, js, je, Isq, Ieq, Jsq, Jeq, nz
294  integer :: i, j, k
295 
296  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
297  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
298 
299  h_neglect = gv%H_subroundoff
300  h_neglect3 = h_neglect**3
301 
302  if (present(obc)) then ; if (associated(obc)) then
303  apply_obc = obc%apply_OBC_u_flather_east .or. obc%apply_OBC_u_flather_west .or. &
304  obc%apply_OBC_v_flather_north .or. obc%apply_OBC_v_flather_south
305  endif ; endif
306 
307  if (.not.associated(cs)) call mom_error(fatal, &
308  "MOM_hor_visc: Module must be initialized before it is used.")
309 
310  find_frictwork = (cs%id_FrictWork > 0)
311  if (cs%id_FrictWorkIntz > 0) find_frictwork = .true.
312  if (associated(meke)) then
313  if (associated(meke%mom_src)) find_frictwork = .true.
314  endif
315 
316  rescale_kh = .false.
317  if (associated(varmix)) then
318  rescale_kh = varmix%Resoln_scaled_Kh
319  if (rescale_kh .and. &
320  (.not.associated(varmix%Res_fn_h) .or. .not.associated(varmix%Res_fn_q))) &
321  call mom_error(fatal, "MOM_hor_visc: VarMix%Res_fn_h and " //&
322  "VarMix%Res_fn_q both need to be associated with Resoln_scaled_Kh.")
323  endif
324 
325  ! Toggle whether to use a Laplacian viscosity derived from MEKE
326  use_meke_ku = associated(meke%Ku)
327 
328 !$OMP parallel do default(none) shared(Isq,Ieq,Jsq,Jeq,nz,CS,G,GV,u,v,is,js,ie,je,h, &
329 !$OMP rescale_Kh,VarMix,h_neglect,h_neglect3, &
330 !$OMP Kh_h,Ah_h,Kh_q,Ah_q,diffu,apply_OBC,OBC,diffv, &
331 !$OMP find_FrictWork,FrictWork,use_MEKE_Ku,MEKE) &
332 !$OMP private(u0, v0, sh_xx, str_xx, visc_bound_rem, &
333 !$OMP sh_xy, str_xy, Ah, Kh, AhSm, KhSm, &
334 !$OMP bhstr_xx, bhstr_xy,FatH,RoScl, hu, hv, &
335 !$OMP Shear_mag, huq, hvq, hq, Kh_scale, hrat_min)
336  do k=1,nz
337 
338 ! This code uses boundary conditions that are consistent with
339 ! free slip and no normal flow boundary conditions. The boundary
340 ! conditions for the western boundary, for example, are:
341 ! dv/dx = 0, d^3v/dx^3 = 0, u = 0, d^2u/dx^2 = 0 .
342 ! The overall scheme is second order accurate.
343 ! All of the metric terms are retained, and the repeated use of
344 ! the symmetric stress tensor insures that no stress is applied with
345 ! no flow or solid-body rotation, even with non-constant values of
346 ! of the biharmonic viscosity.
347 
348 ! The following are the forms of the horizontal tension and hori-
349 ! shearing strain advocated by Smagorinsky (1993) and discussed in
350 ! Griffies and Hallberg (MWR, 2000).
351  do j=jsq-1,jeq+2 ; do i=isq-1,ieq+2
352  sh_xx(i,j) = (cs%DY_dxT(i,j)*(g%IdyCu(i,j) * u(i,j,k) - &
353  g%IdyCu(i-1,j) * u(i-1,j,k)) - &
354  cs%DX_dyT(i,j)*(g%IdxCv(i,j) * v(i,j,k) - &
355  g%IdxCv(i,j-1)*v(i,j-1,k)))
356  enddo ; enddo
357 
358  if (cs%no_slip) then
359  do j=js-2,jeq+1 ; do i=is-2,ieq+1
360  sh_xy(i,j) = (2.0-g%mask2dBu(i,j)) * &
361  (cs%DX_dyBu(i,j)*(u(i,j+1,k)*g%IdxCu(i,j+1) - u(i,j,k)*g%IdxCu(i,j)) + &
362  cs%DY_dxBu(i,j)*(v(i+1,j,k)*g%IdyCv(i+1,j) - v(i,j,k)*g%IdyCv(i,j)))
363  enddo ; enddo
364  else
365  do j=js-2,jeq+1 ; do i=is-2,ieq+1
366  sh_xy(i,j) = g%mask2dBu(i,j) * &
367  (cs%DX_dyBu(i,j)*(u(i,j+1,k)*g%IdxCu(i,j+1) - u(i,j,k)*g%IdxCu(i,j)) + &
368  cs%DY_dxBu(i,j)*(v(i+1,j,k)*g%IdyCv(i+1,j) - v(i,j,k)*g%IdyCv(i,j)))
369  enddo ; enddo
370  endif
371 
372 ! Evaluate u0 = x.Div(Grad u) and v0 = y.Div( Grad u)
373  if (cs%biharmonic) then
374  do j=js-1,jeq+1 ; do i=isq-1,ieq+1
375  u0(i,j) = cs%IDXDY2u(i,j)*(cs%DY2h(i+1,j)*sh_xx(i+1,j) - cs%DY2h(i,j)*sh_xx(i,j)) + &
376  cs%IDX2dyCu(i,j)*(cs%DX2q(i,j)*sh_xy(i,j) - cs%DX2q(i,j-1)*sh_xy(i,j-1))
377  enddo ; enddo
378  do j=jsq-1,jeq+1 ; do i=is-1,ieq+1
379  v0(i,j) = cs%IDXDY2v(i,j)*(cs%DY2q(i,j)*sh_xy(i,j) - cs%DY2q(i-1,j)*sh_xy(i-1,j)) - &
380  cs%IDX2dyCv(i,j)*(cs%DX2h(i,j+1)*sh_xx(i,j+1) - cs%DX2h(i,j)*sh_xx(i,j))
381  enddo ; enddo
382  endif
383 
384  do j=jsq,jeq+1 ; do i=isq,ieq+1
385  if ((cs%Smagorinsky_Kh) .or. (cs%Smagorinsky_Ah)) &
386  shear_mag = sqrt(sh_xx(i,j)*sh_xx(i,j) + &
387  0.25*((sh_xy(i-1,j-1)*sh_xy(i-1,j-1) + sh_xy(i,j)*sh_xy(i,j)) + &
388  (sh_xy(i-1,j)*sh_xy(i-1,j) + sh_xy(i,j-1)*sh_xy(i,j-1))))
389  if (cs%better_bound_Ah .or. cs%better_bound_Kh) then
390  hrat_min = min(1.0, &
391  0.5*min((h(i+1,j,k) + h(i,j,k)), (h(i,j,k) + h(i-1,j,k)), &
392  (h(i,j+1,k) + h(i,j,k)), (h(i,j-1,k) + h(i,j,k))) / &
393  (h(i,j,k) + h_neglect) )
394  visc_bound_rem = 1.0
395  endif
396 
397  if (cs%Laplacian) then
398  ! Determine the Laplacian viscosity at h points, using the
399  ! largest value from several parameterizations.
400  kh_scale = 1.0 ; if (rescale_kh) kh_scale = varmix%Res_fn_h(i,j)
401  if (cs%Smagorinsky_Kh) then
402  khsm = cs%LAPLAC_CONST_xx(i,j) * shear_mag
403  kh = kh_scale * max(cs%Kh_bg_xx(i,j), khsm)
404  if (cs%bound_Kh .and. .not.cs%better_bound_Kh) &
405  kh = min(kh, cs%Kh_Max_xx(i,j))
406  else
407  kh = kh_scale * cs%Kh_bg_xx(i,j)
408  endif
409  if (use_meke_ku) then
410  kh = kh + meke%Ku(i,j)
411  endif
412  if (cs%better_bound_Kh) then
413  if (kh >= hrat_min*cs%Kh_Max_xx(i,j)) then
414  visc_bound_rem = 0.0
415  kh = hrat_min*cs%Kh_Max_xx(i,j)
416  else
417  !visc_bound_rem = 1.0 - abs(Kh) / (hrat_min*CS%Kh_Max_xx(i,j))
418  visc_bound_rem = 1.0 - kh / (hrat_min*cs%Kh_Max_xx(i,j))
419  endif
420  endif
421 
422  if (cs%id_Kh_h>0) kh_h(i,j,k) = kh
423 
424  str_xx(i,j) = -kh * sh_xx(i,j)
425  else ! not Laplacian
426  str_xx(i,j) = 0.0
427  endif ! Laplacian
428 
429  if (cs%biharmonic) then
430 ! Determine the biharmonic viscosity at h points, using the
431 ! largest value from several parameterizations.
432  if (cs%Smagorinsky_Ah) then
433  if (cs%bound_Coriolis) then
434  ahsm = shear_mag * (cs%BIHARM_CONST_xx(i,j) + &
435  cs%Biharm_Const2_xx(i,j)*shear_mag)
436  else
437  ahsm = cs%BIHARM_CONST_xx(i,j) * shear_mag
438  endif
439  ah = max(cs%Ah_bg_xx(i,j), ahsm)
440  if (cs%bound_Ah .and. .not.cs%better_bound_Ah) &
441  ah = min(ah, cs%Ah_Max_xx(i,j))
442  else
443  ah = cs%Ah_bg_xx(i,j)
444  endif ! Smagorinsky_Ah
445  if (cs%better_bound_Ah) then
446  ah = min(ah, visc_bound_rem*hrat_min*cs%Ah_Max_xx(i,j))
447  endif
448 
449  if (cs%id_Ah_h>0) ah_h(i,j,k) = ah
450 
451  str_xx(i,j) = str_xx(i,j) + ah * &
452  (cs%DY_dxT(i,j)*(g%IdyCu(i,j)*u0(i,j) - g%IdyCu(i-1,j)*u0(i-1,j)) - &
453  cs%DX_dyT(i,j) *(g%IdxCv(i,j)*v0(i,j) - g%IdxCv(i,j-1)*v0(i,j-1)))
454 
455  ! Keep a copy of the biharmonic contribution for backscatter parameterization
456  bhstr_xx(i,j) = ah * &
457  (cs%DY_dxT(i,j)*(g%IdyCu(i,j)*u0(i,j) - g%IdyCu(i-1,j)*u0(i-1,j)) - &
458  cs%DX_dyT(i,j) *(g%IdxCv(i,j)*v0(i,j) - g%IdxCv(i,j-1)*v0(i,j-1)))
459  bhstr_xx(i,j) = bhstr_xx(i,j) * (h(i,j,k) * cs%reduction_xx(i,j))
460 
461  endif ! biharmonic
462 
463  str_xx(i,j) = str_xx(i,j) * (h(i,j,k) * cs%reduction_xx(i,j))
464  enddo ; enddo
465 
466  do j=js-1,jeq ; do i=is-1,ieq
467  if ((cs%Smagorinsky_Kh) .or. (cs%Smagorinsky_Ah)) &
468  shear_mag = sqrt(sh_xy(i,j)*sh_xy(i,j) + &
469  0.25*((sh_xx(i,j)*sh_xx(i,j) + sh_xx(i+1,j+1)*sh_xx(i+1,j+1)) + &
470  (sh_xx(i,j+1)*sh_xx(i,j+1) + sh_xx(i+1,j)*sh_xx(i+1,j))))
471 
472  huq = (h(i,j,k) + h(i+1,j,k)) * (h(i,j+1,k) + h(i+1,j+1,k))
473  hvq = (h(i,j,k) + h(i,j+1,k)) * (h(i+1,j,k) + h(i+1,j+1,k))
474  hq = 2.0 * huq * hvq / (h_neglect3 + (huq + hvq) * &
475  ((h(i,j,k) + h(i+1,j+1,k)) + (h(i,j+1,k) + h(i+1,j,k))))
476 
477  if (cs%better_bound_Ah .or. cs%better_bound_Kh) then
478  hrat_min = min(1.0, &
479  0.5*min((h(i,j,k) + h(i+1,j,k)), (h(i,j+1,k) + h(i+1,j+1,k)), &
480  (h(i,j,k) + h(i,j+1,k)), (h(i+1,j,k) + h(i+1,j+1,k))) / &
481  (hq + h_neglect) )
482  visc_bound_rem = 1.0
483  endif
484 
485  if (cs%no_slip .and. (g%mask2dBu(i,j) < 0.5)) then
486  if ((g%mask2dCu(i,j) + g%mask2dCu(i,j+1)) + &
487  (g%mask2dCv(i,j) + g%mask2dCv(i+1,j)) > 0.0) then
488  ! This is a coastal vorticity point, so modify hq and hrat_min.
489 
490  hu = 0.5 * (g%mask2dCu(i,j) * (h(i,j,k) + h(i+1,j,k)) + &
491  g%mask2dCu(i,j+1) * (h(i,j+1,k) + h(i+1,j+1,k)))
492  hv = 0.5 * (g%mask2dCv(i,j) * (h(i,j,k) + h(i,j+1,k)) + &
493  g%mask2dCv(i+1,j) * (h(i+1,j,k) + h(i+1,j+1,k)))
494  if ((g%mask2dCu(i,j) + g%mask2dCu(i,j+1)) * &
495  (g%mask2dCv(i,j) + g%mask2dCv(i+1,j)) == 0.0) then
496  ! Only one of hu and hv is nonzero, so just add them.
497  hq = hu + hv
498  hrat_min = 1.0
499  else
500  ! Both hu and hv are nonzero, so take the harmonic mean.
501  hq = 2.0 * (hu * hv) / ((hu + hv) + h_neglect)
502  hrat_min = min(1.0, min(hu, hv) / (hq + h_neglect) )
503  endif
504  endif
505  endif
506 
507  if (cs%Laplacian) then
508  ! Determine the Laplacian viscosity at q points, using the
509  ! largest value from several parameterizations.
510  kh_scale = 1.0 ; if (rescale_kh) kh_scale = varmix%Res_fn_q(i,j)
511  if (cs%Smagorinsky_Kh) then
512  khsm = cs%LAPLAC_CONST_xy(i,j) * shear_mag
513  kh = kh_scale * max(cs%Kh_bg_xy(i,j), khsm)
514  if (cs%bound_Kh .and. .not.cs%better_bound_Kh) &
515  kh = min(kh, cs%Kh_Max_xy(i,j))
516  else
517  kh = kh_scale * cs%Kh_bg_xy(i,j)
518  endif
519  if (use_meke_ku) then
520  kh = kh + 0.25*( (meke%Ku(i,j)+meke%Ku(i+1,j+1)) &
521  +(meke%Ku(i+1,j)+meke%Ku(i,j+1)) )
522  endif
523  if (cs%better_bound_Kh) then
524  if (kh >= hrat_min*cs%Kh_Max_xy(i,j)) then
525  visc_bound_rem = 0.0
526  kh = hrat_min*cs%Kh_Max_xy(i,j)
527  elseif (cs%Kh_Max_xy(i,j)>0.) then
528  !visc_bound_rem = 1.0 - abs(Kh) / (hrat_min*CS%Kh_Max_xy(I,J))
529  visc_bound_rem = 1.0 - kh / (hrat_min*cs%Kh_Max_xy(i,j))
530  endif
531  endif
532 
533  if (cs%id_Kh_q>0) kh_q(i,j,k) = kh
534 
535  str_xy(i,j) = -kh * sh_xy(i,j)
536  else ! not Laplacian
537  str_xy(i,j) = 0.0
538  endif ! Laplacian
539 
540  if (cs%biharmonic) then
541  ! Determine the biharmonic viscosity at q points, using the
542  ! largest value from several parameterizations.
543  if (cs%Smagorinsky_Ah) then
544  if (cs%bound_Coriolis) then
545  ahsm = shear_mag * (cs%BIHARM_CONST_xy(i,j) + &
546  cs%Biharm_Const2_xy(i,j)*shear_mag)
547  else
548  ahsm = cs%BIHARM_CONST_xy(i,j) * shear_mag
549  endif
550  ah = max(cs%Ah_bg_xy(i,j), ahsm)
551  if (cs%bound_Ah .and. .not.cs%better_bound_Ah) &
552  ah = min(ah, cs%Ah_Max_xy(i,j))
553  else
554  ah = cs%Ah_bg_xy(i,j)
555  endif ! Smagorinsky_Ah
556  if (cs%better_bound_Ah) then
557  ah = min(ah, visc_bound_rem*hrat_min*cs%Ah_Max_xy(i,j))
558  endif
559 
560  if (cs%id_Ah_q>0) ah_q(i,j,k) = ah
561 
562  str_xy(i,j) = str_xy(i,j) + ah * &
563  (cs%DX_dyBu(i,j)*(u0(i,j+1)*g%IdxCu(i,j+1) - u0(i,j)*g%IdxCu(i,j)) + &
564  cs%DY_dxBu(i,j)*(v0(i+1,j)*g%IdyCv(i+1,j) - v0(i,j)*g%IdyCv(i,j)))
565 
566  ! Keep a copy of the biharmonic contribution for backscatter parameterization
567  bhstr_xy(i,j) = ah * &
568  (cs%DX_dyBu(i,j)*(u0(i,j+1)*g%IdxCu(i,j+1) - u0(i,j)*g%IdxCu(i,j)) + &
569  cs%DY_dxBu(i,j)*(v0(i+1,j)*g%IdyCv(i+1,j) - v0(i,j)*g%IdyCv(i,j)))
570  bhstr_xy(i,j) = bhstr_xy(i,j) * (hq * g%mask2dBu(i,j) * cs%reduction_xy(i,j))
571 
572  endif ! biharmonic
573 
574  if (cs%no_slip) then
575  str_xy(i,j) = str_xy(i,j) * (hq * cs%reduction_xy(i,j))
576  else
577  str_xy(i,j) = str_xy(i,j) * (hq * g%mask2dBu(i,j) * cs%reduction_xy(i,j))
578  endif
579  enddo ; enddo
580 
581  do j=js,je ; do i=isq,ieq
582 ! Evaluate 1/h x.Div(h Grad u) or the biharmonic equivalent.
583  diffu(i,j,k) = ((g%IdyCu(i,j)*(cs%DY2h(i,j) *str_xx(i,j) - &
584  cs%DY2h(i+1,j)*str_xx(i+1,j)) + &
585  g%IdxCu(i,j)*(cs%DX2q(i,j-1)*str_xy(i,j-1) - &
586  cs%DX2q(i,j) *str_xy(i,j))) * &
587  g%IareaCu(i,j)) / (0.5*(h(i+1,j,k) + h(i,j,k)) + h_neglect)
588 
589  if (apply_obc) then ; if (obc%OBC_mask_u(i,j)) then
590  if ((obc%OBC_kind_u(i,j) == obc_flather_e) .or. &
591  (obc%OBC_kind_u(i,j) == obc_flather_w)) diffu(i,j,k) = 0.0
592  endif ; endif
593  enddo ; enddo
594 
595 ! Evaluate 1/h y.Div(h Grad u) or the biharmonic equivalent.
596  do j=jsq,jeq ; do i=is,ie
597  diffv(i,j,k) = ((g%IdyCv(i,j)*(cs%DY2q(i-1,j)*str_xy(i-1,j) - &
598  cs%DY2q(i,j) *str_xy(i,j)) - &
599  g%IdxCv(i,j)*(cs%DX2h(i,j) *str_xx(i,j) - &
600  cs%DX2h(i,j+1)*str_xx(i,j+1))) * &
601  g%IareaCv(i,j)) / (0.5*(h(i,j+1,k) + h(i,j,k)) + h_neglect)
602  if (apply_obc) then ; if (obc%OBC_mask_v(i,j)) then
603  if ((obc%OBC_kind_v(i,j) == obc_flather_n) .or. &
604  (obc%OBC_kind_v(i,j) == obc_flather_s)) diffv(i,j,k) = 0.0
605  endif ; endif
606  enddo ; enddo
607 
608  if (find_frictwork) then ; do j=js,je ; do i=is,ie
609  ! Diagnose str_xx*d_x u - str_yy*d_y v + str_xy*(d_y u + d_x v)
610  frictwork(i,j,k) = gv%H_to_kg_m2 * ( &
611  (str_xx(i,j)*(u(i,j,k)-u(i-1,j,k))*g%IdxT(i,j) &
612  -str_xx(i,j)*(v(i,j,k)-v(i,j-1,k))*g%IdyT(i,j)) &
613  +0.25*((str_xy(i,j)*( &
614  (u(i,j+1,k)-u(i,j,k))*g%IdyBu(i,j) &
615  +(v(i+1,j,k)-v(i,j,k))*g%IdxBu(i,j) ) &
616  +str_xy(i-1,j-1)*( &
617  (u(i-1,j,k)-u(i-1,j-1,k))*g%IdyBu(i-1,j-1) &
618  +(v(i,j-1,k)-v(i-1,j-1,k))*g%IdxBu(i-1,j-1) )) &
619  +(str_xy(i-1,j)*( &
620  (u(i-1,j+1,k)-u(i-1,j,k))*g%IdyBu(i-1,j) &
621  +(v(i,j,k)-v(i-1,j,k))*g%IdxBu(i-1,j) ) &
622  +str_xy(i,j-1)*( &
623  (u(i,j,k)-u(i,j-1,k))*g%IdyBu(i,j-1) &
624  +(v(i+1,j-1,k)-v(i,j-1,k))*g%IdxBu(i,j-1) )) ) )
625  enddo ; enddo ; endif
626 
627  ! Make a similar calculation as for FrictWork above but accumulating into
628  ! the vertically integrated MEKE source term, and adjusting for any
629  ! energy loss seen as a reduction in the [biharmonic] frictional source term.
630  if (find_frictwork .and. associated(meke)) then ; if (associated(meke%mom_src)) then
631  if (k==1) then
632  do j=js,je ; do i=is,ie
633  meke%mom_src(i,j) = 0.
634  enddo ; enddo
635  endif
636  if (meke%backscatter_Ro_c /= 0.) then
637  do j=js,je ; do i=is,ie
638  fath = 0.25*( (abs(g%CoriolisBu(i-1,j-1)) + abs(g%CoriolisBu(i,j))) &
639  +(abs(g%CoriolisBu(i-1,j)) + abs(g%CoriolisBu(i,j-1))) )
640  shear_mag = sqrt(sh_xx(i,j)*sh_xx(i,j) + &
641  0.25*((sh_xy(i-1,j-1)*sh_xy(i-1,j-1) + sh_xy(i,j)*sh_xy(i,j)) + &
642  (sh_xy(i-1,j)*sh_xy(i-1,j) + sh_xy(i,j-1)*sh_xy(i,j-1))))
643  fath = fath ** meke%backscatter_Ro_pow ! f^n
644  shear_mag = ( ( shear_mag ** meke%backscatter_Ro_pow ) + 1.e-30 ) &
645  * meke%backscatter_Ro_c ! c * D^n
646  ! The Rossby number function is g(Ro) = 1/(1+c.Ro^n)
647  ! RoScl = 1 - g(Ro)
648  roscl = shear_mag / ( fath + shear_mag ) ! = 1 - f^n/(f^n+c*D^n)
649  meke%mom_src(i,j) = meke%mom_src(i,j) + gv%H_to_kg_m2 * ( &
650  ((str_xx(i,j)-roscl*bhstr_xx(i,j))*(u(i,j,k)-u(i-1,j,k))*g%IdxT(i,j) &
651  -(str_xx(i,j)-roscl*bhstr_xx(i,j))*(v(i,j,k)-v(i,j-1,k))*g%IdyT(i,j)) &
652  +0.25*(((str_xy(i,j)-roscl*bhstr_xy(i,j))*( &
653  (u(i,j+1,k)-u(i,j,k))*g%IdyBu(i,j) &
654  +(v(i+1,j,k)-v(i,j,k))*g%IdxBu(i,j) ) &
655  +(str_xy(i-1,j-1)-roscl*bhstr_xy(i-1,j-1))*( &
656  (u(i-1,j,k)-u(i-1,j-1,k))*g%IdyBu(i-1,j-1) &
657  +(v(i,j-1,k)-v(i-1,j-1,k))*g%IdxBu(i-1,j-1) )) &
658  +((str_xy(i-1,j)-roscl*bhstr_xy(i-1,j))*( &
659  (u(i-1,j+1,k)-u(i-1,j,k))*g%IdyBu(i-1,j) &
660  +(v(i,j,k)-v(i-1,j,k))*g%IdxBu(i-1,j) ) &
661  +(str_xy(i,j-1)-roscl*bhstr_xy(i,j-1))*( &
662  (u(i,j,k)-u(i,j-1,k))*g%IdyBu(i,j-1) &
663  +(v(i+1,j-1,k)-v(i,j-1,k))*g%IdxBu(i,j-1) )) ) )
664  enddo ; enddo
665  else
666  do j=js,je ; do i=is,ie
667  meke%mom_src(i,j) = meke%mom_src(i,j) + frictwork(i,j,k)
668  enddo ; enddo
669  endif
670  endif ; endif
671 
672  enddo ! end of k loop
673 
674 ! Offer fields for diagnostic averaging.
675  if (cs%id_diffu>0) call post_data(cs%id_diffu, diffu, cs%diag)
676  if (cs%id_diffv>0) call post_data(cs%id_diffv, diffv, cs%diag)
677  if (cs%id_FrictWork>0) call post_data(cs%id_FrictWork, frictwork, cs%diag)
678  if (cs%id_Ah_h>0) call post_data(cs%id_Ah_h, ah_h, cs%diag)
679  if (cs%id_Ah_q>0) call post_data(cs%id_Ah_q, ah_q, cs%diag)
680  if (cs%id_Kh_h>0) call post_data(cs%id_Kh_h, kh_h, cs%diag)
681  if (cs%id_Kh_q>0) call post_data(cs%id_Kh_q, kh_q, cs%diag)
682 
683  if (cs%id_FrictWorkIntz > 0) then
684  do j=js,je
685  do i=is,ie ; frictworkintz(i,j) = frictwork(i,j,1) ; enddo
686  do k=2,nz ; do i=is,ie
687  frictworkintz(i,j) = frictworkintz(i,j) + frictwork(i,j,k)
688  enddo ; enddo
689  enddo
690  call post_data(cs%id_FrictWorkIntz, frictworkintz, cs%diag)
691  endif
692 
693 
694 end subroutine horizontal_viscosity

800  "If true, use a Smagorinsky nonlinear eddy viscosity.", &
801  default=.false.)
