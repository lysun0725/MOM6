26 !* This file contains the subroutines that a user should modify to *
27 !* to set the surface wind stresses and fluxes of buoyancy or *
28 !* temperature and fresh water. They are called when the run-time *
29 !* parameters WIND_CONFIG or BUOY_CONFIG are set to "USER". The *
30 !* standard version has simple examples, along with run-time error *
31 !* messages that will cause the model to abort if this code has not *
32 !* been modified. This code is intended for use with relatively *
33 !* simple specifications of the forcing. For more complicated forms, *
34 !* it is probably a good idea to read the forcing from input files *
35 !* using "file" for WIND_CONFIG and BUOY_CONFIG. *
36 !* *
37 !* USER_wind_forcing should set the surface wind stresses (taux and *
38 !* tauy) perhaps along with the surface friction velocity (ustar). *
39 !* *
40 !* USER_buoyancy forcing is used to set the surface buoyancy *
41 !* forcing, which may include a number of fresh water flux fields *
42 !* (evap, liq_precip, froz_precip, liq_runoff, froz_runoff, and *
43 !* virt_precip) and the surface heat fluxes (sw, lw, latent and sens) *
44 !* if temperature and salinity are state variables, or it may simply *
45 !* be the buoyancy flux if it is not. This routine also has coded a *
46 !* restoring to surface values of temperature and salinity. *
47 !* *
48 !* Macros written all in capital letters are defined in MOM_memory.h. *
49 !* *
50 !* A small fragment of the grid is shown below: *
51 !* *
52 !* j+1 x ^ x ^ x At x: q *
53 !* j+1 > o > o > At ^: v, tauy *
54 !* j x ^ x ^ x At >: u, taux *
55 !* j > o > o > At o: h, fluxes. *
56 !* j-1 x ^ x ^ x *
57 !* i-1 i i+1 At x & ^: *
58 !* i i+1 At > & o: *
59 !* *
60 !* The boundaries always run through q grid points (x). *
61 !* *
62 !********+*********+*********+*********+*********+*********+*********+**

104 subroutine user_wind_forcing(state, fluxes, day, G, CS)
105  type(surface), intent(inout) :: state
106  type(forcing), intent(inout) :: fluxes
107  type(time_type), intent(in) :: day
108  type(ocean_grid_type), intent(inout) :: G
109  type(user_surface_forcing_cs), pointer :: CS
110 
111 ! This subroutine sets the surface wind stresses, fluxes%taux and fluxes%tauy.
112 ! These are the stresses in the direction of the model grid (i.e. the same
113 ! direction as the u- and v- velocities.) They are both in Pa.
114 ! In addition, this subroutine can be used to set the surface friction
115 ! velocity, fluxes%ustar, in m s-1. This is needed with a bulk mixed layer.
116 !
117 ! Arguments: state - A structure containing fields that describe the
118 ! surface state of the ocean.
119 ! (out) fluxes - A structure containing pointers to any possible
120 ! forcing fields. Unused fields have NULL ptrs.
121 ! (in) day - Time of the fluxes.
122 ! (in) G - The ocean's grid structure.
123 ! (in) CS - A pointer to the control structure returned by a previous
124 ! call to user_surface_forcing_init
125 
126  integer :: i, j, is, ie, js, je, Isq, Ieq, Jsq, Jeq
127  integer :: isd, ied, jsd, jed, IsdB, IedB, JsdB, JedB
128 
129  ! When modifying the code, comment out this error message. It is here
130  ! so that the original (unmodified) version is not accidentally used.
131  call mom_error(fatal, "User_wind_surface_forcing: " // &
132  "User forcing routine called without modification." )
133 
134  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
135  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
136  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
137  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
138 
139  ! Allocate the forcing arrays, if necessary.
140  call allocate_forcing_type(g, fluxes, stress=.true., ustar=.true.)
141 
142  ! Set the surface wind stresses, in units of Pa. A positive taux
143  ! accelerates the ocean to the (pseudo-)east.
144 
145  ! The i-loop extends to is-1 so that taux can be used later in the
146  ! calculation of ustar - otherwise the lower bound would be Isq.
147  do j=js,je ; do i=is-1,ieq
148  fluxes%taux(i,j) = g%mask2dCu(i,j) * 0.0 ! Change this to the desired expression.
149  enddo ; enddo
150  do j=js-1,jeq ; do i=is,ie
151  fluxes%tauy(i,j) = g%mask2dCv(i,j) * 0.0 ! Change this to the desired expression.
152  enddo ; enddo
153 
154  ! Set the surface friction velocity, in units of m s-1. ustar
155  ! is always positive.
156  if (associated(fluxes%ustar)) then ; do j=js,je ; do i=is,ie
157  ! This expression can be changed if desired, but need not be.
158  fluxes%ustar(i,j) = g%mask2dT(i,j) * sqrt(cs%gust_const/cs%Rho0 + &
159  sqrt(0.5*(fluxes%taux(i-1,j)**2 + fluxes%taux(i,j)**2) + &
160  0.5*(fluxes%tauy(i,j-1)**2 + fluxes%tauy(i,j)**2))/cs%Rho0)
161  enddo ; enddo ; endif
162 
163 end subroutine user_wind_forcing

165 subroutine user_buoyancy_forcing(state, fluxes, day, dt, G, CS)
166  type(surface), intent(inout) :: state
167  type(forcing), intent(inout) :: fluxes
168  type(time_type), intent(in) :: day
169  real, intent(in) :: dt
170  type(ocean_grid_type), intent(in) :: G
171  type(user_surface_forcing_cs), pointer :: CS
172 
173 ! This subroutine specifies the current surface fluxes of buoyancy or
174 ! temperature and fresh water. It may also be modified to add
175 ! surface fluxes of user provided tracers.
176 
177 ! When temperature is used, there are long list of fluxes that need to be
178 ! set - essentially the same as for a full coupled model, but most of these
179 ! can be simply set to zero. The net fresh water flux should probably be
180 ! set in fluxes%evap and fluxes%liq_precip, with any salinity restoring
181 ! appearing in fluxes%virt_precip, and the other water flux components
182 ! (froz_precip, liq_runoff and froz_runoff) left as arrays full of zeros.
183 ! Evap is usually negative and precip is usually positive. All heat fluxes
184 ! are in W m-2 and positive for heat going into the ocean. All fresh water
185 ! fluxes are in kg m-2 s-1 and positive for water moving into the ocean.
186 
187 ! Arguments: state - A structure containing fields that describe the
188 ! surface state of the ocean.
189 ! (out) fluxes - A structure containing pointers to any possible
190 ! forcing fields. Unused fields have NULL ptrs.
191 ! (in) day_start - Start time of the fluxes.
192 ! (in) day_interval - Length of time over which these fluxes
193 ! will be applied.
194 ! (in) G - The ocean's grid structure.
195 ! (in) CS - A pointer to the control structure returned by a previous
196 ! call to user_surface_forcing_init
197 
198  real :: Temp_restore ! The temperature that is being restored toward, in C.
199  real :: Salin_restore ! The salinity that is being restored toward, in PSU.
200  real :: density_restore ! The potential density that is being restored
201  ! toward, in kg m-3.
202  real :: rhoXcp ! The mean density times the heat capacity, in J m-3 K-1.
203  real :: buoy_rest_const ! A constant relating density anomalies to the
204  ! restoring buoyancy flux, in m5 s-3 kg-1.
205 
206  integer :: i, j, is, ie, js, je
207  integer :: isd, ied, jsd, jed
208 
209  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
210  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
211 
212  ! When modifying the code, comment out this error message. It is here
213  ! so that the original (unmodified) version is not accidentally used.
214  call mom_error(fatal, "User_buoyancy_surface_forcing: " // &
215  "User forcing routine called without modification." )
216 
217  ! Allocate and zero out the forcing arrays, as necessary. This portion is
218  ! usually not changed.
219  if (cs%use_temperature) then
220  call alloc_if_needed(fluxes%evap, isd, ied, jsd, jed)
221  call alloc_if_needed(fluxes%liq_precip, isd, ied, jsd, jed)
222  call alloc_if_needed(fluxes%froz_precip, isd, ied, jsd, jed)
223  call alloc_if_needed(fluxes%liq_runoff, isd, ied, jsd, jed)
224  call alloc_if_needed(fluxes%froz_runoff, isd, ied, jsd, jed)
225  call alloc_if_needed(fluxes%virt_precip, isd, ied, jsd, jed)
226 
227  call alloc_if_needed(fluxes%sw, isd, ied, jsd, jed)
228  call alloc_if_needed(fluxes%lw, isd, ied, jsd, jed)
229  call alloc_if_needed(fluxes%latent, isd, ied, jsd, jed)
230  call alloc_if_needed(fluxes%sens, isd, ied, jsd, jed)
231  else ! This is the buoyancy only mode.
232  call alloc_if_needed(fluxes%buoy, isd, ied, jsd, jed)
233  endif
234 
235 
236  ! MODIFY THE CODE IN THE FOLLOWING LOOPS TO SET THE BUOYANCY FORCING TERMS.
237 
238  if ( cs%use_temperature ) then
239  ! Set whichever fluxes are to be used here. Any fluxes that
240  ! are always zero do not need to be changed here.
241  do j=js,je ; do i=is,ie
242  ! Fluxes of fresh water through the surface are in units of kg m-2 s-1
243  ! and are positive downward - i.e. evaporation should be negative.
244  fluxes%evap(i,j) = -0.0 * g%mask2dT(i,j)
245  fluxes%liq_precip(i,j) = 0.0 * g%mask2dT(i,j)
246 
247  ! virt_precip will be set later, if it is needed for salinity restoring.
248  fluxes%virt_precip(i,j) = 0.0
249 
250  ! Heat fluxes are in units of W m-2 and are positive into the ocean.
251  fluxes%lw(i,j) = 0.0 * g%mask2dT(i,j)
252  fluxes%latent(i,j) = 0.0 * g%mask2dT(i,j)
253  fluxes%sens(i,j) = 0.0 * g%mask2dT(i,j)
254  fluxes%sw(i,j) = 0.0 * g%mask2dT(i,j)
255  enddo ; enddo
256  else ! This is the buoyancy only mode.
257  do j=js,je ; do i=is,ie
258  ! fluxes%buoy is the buoyancy flux into the ocean in m2 s-3. A positive
259  ! buoyancy flux is of the same sign as heating the ocean.
260  fluxes%buoy(i,j) = 0.0 * g%mask2dT(i,j)
261  enddo ; enddo
262  endif
263 
264  if (cs%restorebuoy) then
265  if (cs%use_temperature) then
266  call alloc_if_needed(fluxes%heat_restore, isd, ied, jsd, jed)
267  ! When modifying the code, comment out this error message. It is here
268  ! so that the original (unmodified) version is not accidentally used.
269  call mom_error(fatal, "User_buoyancy_surface_forcing: " // &
270  "Temperature and salinity restoring used without modification." )
271 
272  rhoxcp = cs%Rho0 * fluxes%C_p
273  do j=js,je ; do i=is,ie
274  ! Set Temp_restore and Salin_restore to the temperature (in C) and
275  ! salinity (in PSU) that are being restored toward.
276  temp_restore = 0.0
277  salin_restore = 0.0
278 
279  fluxes%heat_restore(i,j) = (g%mask2dT(i,j) * (rhoxcp * cs%Flux_const)) * &
280  (temp_restore - state%SST(i,j))
281  fluxes%virt_precip(i,j) = - (g%mask2dT(i,j) * (cs%Rho0*cs%Flux_const)) * &
282  ((salin_restore - state%SSS(i,j)) / &
283  (0.5 * (salin_restore + state%SSS(i,j))))
284  enddo ; enddo
285  else
286  ! When modifying the code, comment out this error message. It is here
287  ! so that the original (unmodified) version is not accidentally used.
288  call mom_error(fatal, "User_buoyancy_surface_forcing: " // &
289  "Buoyancy restoring used without modification." )
290 
291  ! The -1 is because density has the opposite sign to buoyancy.
292  buoy_rest_const = -1.0 * (cs%G_Earth * cs%Flux_const) / cs%Rho0
293  do j=js,je ; do i=is,ie
294  ! Set density_restore to an expression for the surface potential
295  ! density in kg m-3 that is being restored toward.
296  density_restore = 1030.0
297 
298  fluxes%buoy(i,j) = g%mask2dT(i,j) * buoy_rest_const * &
299  (density_restore - state%sfc_density(i,j))
300  enddo ; enddo
301  endif
302  endif ! end RESTOREBUOY
303 
304 end subroutine user_buoyancy_forcing

306 subroutine alloc_if_needed(ptr, isd, ied, jsd, jed)
307  ! If ptr is not associated, this routine allocates it with the given size
308  ! and zeros out its contents. This is equivalent to safe_alloc_ptr in

313  allocate(ptr(isd:ied,jsd:jed))
314  ptr(:,:) = 0.0
315  endif
316 end subroutine alloc_if_needed

318 subroutine user_surface_forcing_init(Time, G, param_file, diag, CS)

324 ! Arguments: Time - The current model time.
